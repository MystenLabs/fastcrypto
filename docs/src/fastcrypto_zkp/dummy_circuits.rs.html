14/// A dummy circuit that checks an addition constraint between some of its inputs
15/// as well as a parametrized number of dummy constraints.
16#[derive(Debug, Copy, Clone)]
17pub struct DummyCircuit<F: PrimeField> {
18    /// the first input to the circuit
19    pub a: Option<F>,
20    /// the second input to the circuit
21    pub b: Option<F>,
22    /// The number of variables to the circuit
23    pub num_variables: usize,
24    /// the number of constraints to the circuit
25    pub num_constraints: usize,
26}
27
28impl<F: PrimeField> ConstraintSynthesizer<F> for DummyCircuit<F> {
29    // We'll be proving a relationship involving the product c of a & b.
30    fn generate_constraints(self, cs: ConstraintSystemRef<F>) -> Result<(), SynthesisError> {
31        let a = cs.new_witness_variable(|| self.a.ok_or(SynthesisError::AssignmentMissing))?;
32        let b = cs.new_witness_variable(|| self.b.ok_or(SynthesisError::AssignmentMissing))?;
33        let c = cs.new_input_variable(|| {
34            let a = self.a.ok_or(SynthesisError::AssignmentMissing)?;
35            let b = self.b.ok_or(SynthesisError::AssignmentMissing)?;
36
37            Ok(a * b)
38        })?;
39
40        // a, b, c are above, let's define the rest.
41        for _ in 0..(self.num_variables - 3) {
42            let _ = cs.new_witness_variable(|| self.a.ok_or(SynthesisError::AssignmentMissing))?;
43        }
44
45        for _ in 0..self.num_constraints - 1 {
46            cs.enforce_constraint(lc!() + a, lc!() + b, lc!() + c)?;
47        }
48
49        cs.enforce_constraint(lc!(), lc!(), lc!())?;
50
51        Ok(())
52    }
53}

55/// A circuit that checks a parametrized number of R1CS constraints that verify the computation of the Fibonacci sequence.
56/// It generates constraints in layers, numbered from 1 to num_constraints, specifying the computation of the Fibonacci sequence.
57/// On each layer, it does:
58///
59/// 1. Add two new public inputs a = initial_a and b = initial_b at the first layer, or retrieve them from the previous layer if the layer if larger.
60/// 2. Add a new witness variable c,
61/// 3. Add and check a constraint that c = a + b,
62/// 4. Initialize the values a = b; b = c; and repeat from step 1.
63///
64/// Note: we encode the addition (c = a + b) less efficiently than what R1CS would allow, but this is not a problem for the purpose of this demo.
65#[derive(Debug)]
66pub struct Fibonacci<F: PrimeField> {
67    num_constraints: usize,
68    initial_a: F,
69    initial_b: F,
70    _engine: PhantomData<F>,
71}
72
73impl<F: PrimeField> Fibonacci<F> {
74    /// Create a new instance of the Fibonacci circuit. We have to provide the initial values of a and b, the first and second initial values of
75    /// a sequence verifying the Fibonacci recursion.
76    pub fn new(num_constraints: usize, initial_a: F, initial_b: F) -> Self {
77        Self {
78            num_constraints,
79            initial_a,
80            initial_b,
81            _engine: PhantomData,
82        }
83    }
84}
85
86impl<F: PrimeField> ConstraintSynthesizer<F> for Fibonacci<F> {
87    fn generate_constraints(self, cs: ConstraintSystemRef<F>) -> Result<(), SynthesisError> {
88        let mut assignments = Vec::new();
89        let mut a_val = self.initial_a;
90        let mut a_var = cs.new_input_variable(|| Ok(a_val))?;
91        assignments.push((a_val, a_var));
92
93        let mut b_val = self.initial_b;
94        let mut b_var = cs.new_input_variable(|| Ok(b_val))?;
95        assignments.push((a_val, a_var));
96
97        for _i in 0..self.num_constraints - 1 {
98            let c_val = a_val + b_val;
99            let c_var = cs.new_witness_variable(|| Ok(c_val))?;
100
101            cs.enforce_constraint(lc!() + a_var + b_var, lc!() + Variable::One, lc!() + c_var)?;
102
103            assignments.push((c_val, c_var));
104            a_val = b_val;
105            a_var = b_var;
106            b_val = c_val;
107            b_var = c_var;
108        }
109
110        let mut a_lc = LinearCombination::zero();
111        let mut b_lc = LinearCombination::zero();
112        let mut c_val = F::zero();
113
114        for (val, var) in assignments {
115            a_lc = a_lc + var;
116            b_lc = b_lc + var;
117            c_val += val;
118        }
119        c_val = c_val.square();
120
121        let c_var = cs.new_witness_variable(|| Ok(c_val))?;
122
123        cs.enforce_constraint(lc!() + a_lc, lc!() + b_lc, lc!() + c_var)?;
124
125        Ok(())
126    }
127}
