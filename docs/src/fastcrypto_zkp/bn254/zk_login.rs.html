35/// Key for the modulus hash cache: (modulus bytes, max_rsa_bits).
36type ModulusHashKey = (Vec<u8>, u16);
37
38/// JWKs rotate occasionally, so caching by (modulus bytes, max_rsa_bits) avoids recomputing
39/// bit-packing + poseidon hash on every verification.
40static MODULUS_HASH_CACHE: Lazy<RwLock<HashMap<ModulusHashKey, Bn254Fr>>> =
41    Lazy::new(|| RwLock::new(HashMap::new()));
42
43fn cached_modulus_hash(modulus: &[u8], max_rsa_bits: u16) -> Result<Bn254Fr, FastCryptoError> {
44    if let Some(f) = MODULUS_HASH_CACHE
45        .read()
46        .ok()
47        .and_then(|m| m.get(&(modulus.to_vec(), max_rsa_bits)).copied())
48    {
49        return Ok(f);
50    }
51    let f = hash_to_field(&[BigUint::from_bytes_be(modulus)], max_rsa_bits, PACK_WIDTH)?;
52    if let Ok(mut m) = MODULUS_HASH_CACHE.write() {
53        m.insert((modulus.to_vec(), max_rsa_bits), f);
54    }
55    Ok(f)
56}

58/// Clear the modulus hash cache for testing only benchmark.
59#[cfg(any(test, feature = "test-utils"))]
60pub fn clear_cache_for_testing() {
61    if let Ok(mut m) = MODULUS_HASH_CACHE.write() {
62        m.clear();
63    }
64}

351/// Check if the iss string is formatted as Microsoft's pattern.
352fn match_micrsoft_iss_substring(iss: &str) -> bool {
353    iss.starts_with("https://login.microsoftonline.com/") && iss.ends_with("/v2.0")
354}

356/// Parse the region and tenant_id from the iss string for AWS.
357fn parse_aws_iss_substring(url: &str) -> Result<(&str, &str), FastCryptoError> {
358    let re =
359        Regex::new(r"https://cognito-idp\.(?P<region>[^.]+)\.amazonaws\.com/(?P<tenant_id>[^/]+)")
360            .unwrap();
361
362    if let Some(captures) = re.captures(url) {
363        // Extract the region and tenant_id from the captures
364        let region = captures.name("region").unwrap().as_str();
365        let tenant_id = captures.name("tenant_id").unwrap().as_str();
366
367        Ok((region, tenant_id))
368    } else {
369        Err(FastCryptoError::InvalidInput)
370    }
371}
372/// Struct that contains info for a JWK. A list of them for different kids can
373/// be retrieved from the JWK endpoint (e.g. <https://www.googleapis.com/oauth2/v3/certs>).
374/// The JWK is used to verify the JWT token.
375#[derive(PartialEq, Eq, Hash, Debug, Clone, Serialize, Deserialize, PartialOrd, Ord)]
376pub struct JWK {
377    /// Key type parameter, https://datatracker.ietf.org/doc/html/rfc7517#section-4.1
378    pub kty: String,
379    /// RSA public exponent, https://datatracker.ietf.org/doc/html/rfc7517#section-9.3
380    pub e: String,
381    /// RSA modulus, https://datatracker.ietf.org/doc/html/rfc7517#section-9.3
382    pub n: String,
383    /// Algorithm parameter, https://datatracker.ietf.org/doc/html/rfc7517#section-4.4
384    pub alg: String,
385}
386
387/// Reader struct to parse all fields in a JWK from JSON.
388#[derive(Debug, Serialize, Deserialize, Clone)]
389pub struct JWKReader {
390    e: String,
391    n: String,
392    #[serde(rename = "use", skip_serializing_if = "Option::is_none")]
393    my_use: Option<String>,
394    kid: String,
395    kty: String,
396    #[serde(skip_serializing_if = "Option::is_none")]
397    alg: Option<String>,
398    #[serde(skip_serializing_if = "Option::is_none")]
399    x5c: Option<Vec<String>>,
400    #[serde(skip_serializing_if = "Option::is_none")]
401    x5t: Option<String>,
402    #[serde(skip_serializing_if = "Option::is_none")]
403    issuer: Option<String>,
404}
405
406impl JWK {
407    /// Parse JWK from the reader struct.
408    pub fn from_reader(reader: JWKReader) -> FastCryptoResult<Self> {
409        let trimmed_e = trim(reader.e);
410        // Microsoft does not contain alg field in JWK, so here we only check if it equals to RS256 only if alg field is present.
411        if (reader.alg.is_some() && reader.alg != Some("RS256".to_string()))
412            || reader.kty != "RSA"
413            || trimmed_e != "AQAB"
414        {
415            return Err(FastCryptoError::InvalidInput);
416        }
417        Ok(Self {
418            kty: reader.kty,
419            e: trimmed_e,
420            n: trim(reader.n),
421            alg: "RS256".to_string(),
422        })
423    }
424}

426/// Trim trailing '=' so that it is considered a valid base64 url encoding string by base64ct library.
427fn trim(str: String) -> String {
428    str.trim_end_matches('=').to_owned()
429}

431/// Fetch JWKs from the given provider and return a list of JwkId -> JWK.
432pub async fn fetch_jwks(
433    provider: &OIDCProvider,
434    client: &Client,
435    skip_on_invalid_jwk: bool,
436) -> Result<Vec<(JwkId, JWK)>, FastCryptoError> {
437    let response = client
438        .get(provider.get_config().jwk_endpoint)
439        .send()
440        .await
441        .map_err(|e| {
442            FastCryptoError::GeneralError(format!(
443                "Failed to get JWK {:?} {:?} {:?}",
444                e.source(),
445                e.to_string(),
446                provider
447            ))
448        })?;
449    let bytes = response.bytes().await.map_err(|e| {
450        FastCryptoError::GeneralError(format!(
451            "Failed to get bytes {:?} {:?}",
452            e.to_string(),
453            provider
454        ))
455    })?;
456    parse_jwks(&bytes, provider, skip_on_invalid_jwk)
457}

459/// Parse the JWK bytes received from the given provider and return a list of JwkId -> JWK.
460pub fn parse_jwks(
461    json_bytes: &[u8],
462    provider: &OIDCProvider,
463    skip_on_invalid_jwk: bool,
464) -> Result<Vec<(JwkId, JWK)>, FastCryptoError> {
465    let json_str = String::from_utf8_lossy(json_bytes);
466    let parsed_list: Result<serde_json::Value, serde_json::Error> = serde_json::from_str(&json_str);
467    if let Ok(parsed_list) = parsed_list {
468        if let Some(keys) = parsed_list["keys"].as_array() {
469            let mut ret = Vec::new();
470            for k in keys {
471                if skip_on_invalid_jwk {
472                    // skips if invalid jwk is found, only error if no valid jwk is found for a provider.
473                    if let Ok(parsed) = serde_json::from_value::<JWKReader>(k.clone()) {
474                        if let Ok(jwk) = JWK::from_reader(parsed.clone()) {
475                            ret.push((JwkId::new(provider.get_config().iss, parsed.kid), jwk));
476                        }
477                    }
478                } else {
479                    // legacy handling, if any jwk failed parsing, exit and error.
480                    let parsed: JWKReader = serde_json::from_value(k.clone())
481                        .map_err(|_| FastCryptoError::GeneralError("Parse error".to_string()))?;
482
483                    ret.push((
484                        JwkId::new(provider.get_config().iss, parsed.kid.clone()),
485                        JWK::from_reader(parsed)?,
486                    ));
487                }
488            }
489            if skip_on_invalid_jwk && ret.is_empty() {
490                return Err(FastCryptoError::GeneralError(
491                    "No valid JWKs found".to_string(),
492                ));
493            }
494            return Ok(ret);
495        }
496    }
497    Err(FastCryptoError::GeneralError(
498        "Invalid JWK response".to_string(),
499    ))
500}

604    /// Calculate the poseidon hash from selected fields from inputs, along with the ephemeral pubkey.
605    pub fn calculate_all_inputs_hash(
606        &self,
607        eph_pk_bytes: &[u8],
608        modulus: &[u8],
609        max_epoch: u64,
610        version: CircuitVersion,
611    ) -> Result<Bn254Fr, FastCryptoError> {
612        let config = version.config();
613        if self.header_base64.len() > config.max_header_len_b64 as usize {
614            return Err(FastCryptoError::GeneralError("Header too long".to_string()));
615        }
616
617        let addr_seed = (&self.address_seed).into();
618        let (first, second) = split_to_two_frs(eph_pk_bytes)?;
619        let max_epoch_f = (&Bn254FrElement::from_str(&max_epoch.to_string())?).into();
620        let header_f = hash_ascii_str_to_field(&self.header_base64, config.max_header_len_b64)?;
621        let modulus_f = cached_modulus_hash(modulus, config.max_rsa_bits)?;
622
623        match version {
624            CircuitVersion::V1 => {
625                let index_mod_4_f =
626                    (&Bn254FrElement::from_str(&self.iss_base64_details.index_mod_4.to_string())?)
627                        .into();
628                let iss_base64_f =
629                    hash_ascii_str_to_field(&self.iss_base64_details.value, config.max_iss_len)?;
630                poseidon_zk_login(&[
631                    first,
632                    second,
633                    addr_seed,
634                    max_epoch_f,
635                    iss_base64_f,
636                    index_mod_4_f,
637                    header_f,
638                    modulus_f,
639                ])
640            }
641            CircuitVersion::V2 => {
642                let iss_f = self.hash_iss_decoded(config.max_iss_len)?;
643                let rsa_num_bits = BigUint::from_bytes_be(modulus).bits();
644                let rsa_num_bits_f = (&Bn254FrElement::from_str(&rsa_num_bits.to_string())?).into();
645                poseidon_zk_login(&[
646                    first,
647                    second,
648                    addr_seed,
649                    max_epoch_f,
650                    iss_f,
651                    header_f,
652                    modulus_f,
653                    rsa_num_bits_f,
654                ])
655            }
656        }
657    }
658
659    /// Hash the v2 circuit's `iss_F`: the *decoded* extended iss claim (e.g. `,"iss":"https://...",`),
660    /// obtained by base64-decoding `iss_base64_details` at its `index_mod_4` offset. This differs
661    /// from v1, which hashes the raw base64 value directly.
662    fn hash_iss_decoded(&self, max_len: u16) -> FastCryptoResult<Bn254Fr> {
663        let ext_iss = decode_base64_url(
664            &self.iss_base64_details.value,
665            &self.iss_base64_details.index_mod_4,
666        )?;
667        hash_ascii_str_to_field(&ext_iss, max_len)
668    }
669}

696/// Parse the extended claim json value to its claim value, using the expected claim key.
697fn verify_extended_claim(
698    extended_claim: &str,
699    expected_key: &str,
700) -> Result<String, FastCryptoError> {
701    // Last character of each extracted_claim must be '}' or ','
702    if !(extended_claim.ends_with('}') || extended_claim.ends_with(',')) {
703        return Err(FastCryptoError::GeneralError(
704            "Invalid extended claim".to_string(),
705        ));
706    }
707
708    let json_str = format!("{{{}}}", &extended_claim[..extended_claim.len() - 1]);
709    let json: Value = serde_json::from_str(&json_str).map_err(|_| FastCryptoError::InvalidInput)?;
710    let value = json
711        .as_object()
712        .ok_or(FastCryptoError::InvalidInput)?
713        .get(expected_key)
714        .ok_or(FastCryptoError::InvalidInput)?
715        .as_str()
716        .ok_or(FastCryptoError::InvalidInput)?;
717    Ok(value.to_string())
718}

720/// Parse the base64 string, add paddings based on offset, and convert to a bytearray.
721fn decode_base64_url(s: &str, i: &u8) -> Result<String, FastCryptoError> {
722    if s.len() < 2 {
723        return Err(FastCryptoError::GeneralError(
724            "Base64 string smaller than 2".to_string(),
725        ));
726    }
727    let mut bits = base64_to_bitarray(s)?;
728    match i {
729        0 => {}
730        1 => {
731            bits.drain(..2);
732        }
733        2 => {
734            bits.drain(..4);
735        }
736        _ => {
737            return Err(FastCryptoError::GeneralError(
738                "Invalid first_char_offset".to_string(),
739            ));
740        }
741    }
742
743    let last_char_offset = (i + s.len() as u8 - 1) % 4;
744    match last_char_offset {
745        3 => {}
746        2 => {
747            bits.drain(bits.len() - 2..);
748        }
749        1 => {
750            bits.drain(bits.len() - 4..);
751        }
752        _ => {
753            return Err(FastCryptoError::GeneralError(
754                "Invalid last_char_offset".to_string(),
755            ));
756        }
757    }
758
759    if bits.len() % 8 != 0 {
760        return Err(FastCryptoError::GeneralError(
761            "Invalid bits length".to_string(),
762        ));
763    }
764
765    Ok(std::str::from_utf8(&bitarray_to_bytearray(&bits)?)
766        .map_err(|_| FastCryptoError::GeneralError("Invalid UTF8 string".to_string()))?
767        .to_owned())
768}

770/// Map a base64 string to a bit array by taking each char's index and convert it to binary form with one bit per u8
771/// element in the output. Returns [FastCryptoError::InvalidInput] if one of the characters is not in the base64 charset.
772fn base64_to_bitarray(input: &str) -> FastCryptoResult<Vec<u8>> {
773    input
774        .chars()
775        .map(|c| {
776            BASE64_URL_CHARSET
777                .find(c)
778                .map(|index| index as u8)
779                .map(|index| (0..6).rev().map(move |i| index >> i & 1))
780                .ok_or(FastCryptoError::InvalidInput)
781        })
782        .flatten_ok()
783        .collect()
784}
785
786/// Convert a bitarray (each bit is represented by a u8) to a byte array by taking each 8 bits as a
787/// byte in big-endian format.
788fn bitarray_to_bytearray(bits: &[u8]) -> FastCryptoResult<Vec<u8>> {
789    if !bits.len().is_multiple_of(8) {
790        return Err(FastCryptoError::InvalidInput);
791    }
792    Ok(bits
793        .chunks(8)
794        .map(|chunk| {
795            let mut byte = 0u8;
796            for (i, bit) in chunk.iter().rev().enumerate() {
797                byte |= bit << i;
798            }
799            byte
800        })
801        .collect())
802}

804/// Pads a stream of bytes and maps it to a field element
805pub fn hash_ascii_str_to_field(str: &str, max_size: u16) -> Result<Bn254Fr, FastCryptoError> {
806    let str_padded = str_to_padded_char_codes(str, max_size)?;
807    hash_to_field(&str_padded, 8, PACK_WIDTH)
808}
809
810fn str_to_padded_char_codes(str: &str, max_len: u16) -> Result<Vec<BigUint>, FastCryptoError> {
811    let arr: Vec<BigUint> = str
812        .chars()
813        .map(|c| BigUint::from_slice(&([c as u32])))
814        .collect();
815    pad_with_zeroes(arr, max_len)
816}
817
818fn pad_with_zeroes(in_arr: Vec<BigUint>, out_count: u16) -> Result<Vec<BigUint>, FastCryptoError> {
819    if in_arr.len() > out_count as usize {
820        return Err(FastCryptoError::GeneralError("in_arr too long".to_string()));
821    }
822    let mut padded = in_arr;
823    padded.resize(out_count as usize, BigUint::zero());
824    Ok(padded)
825}

827/// Maps a stream of bigints to a single field element. First we convert the base from
828/// inWidth to packWidth. Then we compute the poseidon hash of the "packed" input.
829/// input is the input vector containing equal-width big ints. inWidth is the width of
830/// each input element.
831fn hash_to_field(
832    input: &[BigUint],
833    in_width: u16,
834    pack_width: u8,
835) -> Result<Bn254Fr, FastCryptoError> {
836    let packed = convert_base(input, in_width, pack_width)?;
837    poseidon_zk_login(&packed)
838}
839
840/// Helper function to pack field elements from big ints.
841fn convert_base(
842    in_arr: &[BigUint],
843    in_width: u16,
844    out_width: u8,
845) -> Result<Vec<Bn254Fr>, FastCryptoError> {
846    if out_width == 0 {
847        return Err(FastCryptoError::InvalidInput);
848    }
849    let bits = big_int_array_to_bits(in_arr, in_width as usize)?;
850    let mut packed: Vec<Bn254Fr> = bits
851        .rchunks(out_width as usize)
852        .map(|chunk| Bn254Fr::from(BigUint::from_radix_be(chunk, 2).unwrap()))
853        .collect();
854    packed.reverse();
855    match packed.len() != (in_arr.len() * in_width as usize).div_ceil(out_width as usize) {
856        true => Err(FastCryptoError::InvalidInput),
857        false => Ok(packed),
858    }
859}

861/// Convert a big int array to a bit array with 0 paddings.
862fn big_int_array_to_bits(integers: &[BigUint], intended_size: usize) -> FastCryptoResult<Vec<u8>> {
863    integers
864        .iter()
865        .map(|integer| {
866            let bits = integer.to_radix_be(2);
867            match bits.len().cmp(&intended_size) {
868                Less => {
869                    let extra_bits = intended_size - bits.len();
870                    let mut padded = vec![0; extra_bits];
871                    padded.extend(bits);
872                    Ok(padded)
873                }
874                Equal => Ok(bits),
875                Greater => Err(FastCryptoError::InvalidInput),
876            }
877        })
878        .flatten_ok()
879        .collect()
880}

882/// Calculate the poseidon hash of the field element inputs. If there are no inputs, return an error.
883/// If input length is <= 16, calculate H(inputs), if it is <= 32, calculate H(H(inputs[0..16]),
884/// H(inputs[16..])), if it is <= 48, calculate H(H(inputs[0..16]), H(inputs[16..32]), H(inputs[32..])),
885/// if it is <= 64, calculate H(H(inputs[0..16]), H(inputs[16..32]), H(inputs[32..48]), H(inputs[48..])),
886/// otherwise return an error.
887///
888/// This functions must be equivalent with the one found in the zk_login circuit.
889pub(crate) fn poseidon_zk_login(inputs: &[Bn254Fr]) -> FastCryptoResult<Bn254Fr> {
890    if inputs.is_empty() || inputs.len() > 64 {
891        return Err(FastCryptoError::InputLengthWrong(inputs.len()));
892    }
893    poseidon_merkle_tree(&inputs.iter().map(|x| FieldElement(*x)).collect_vec()).map(|x| x.0)
894}
