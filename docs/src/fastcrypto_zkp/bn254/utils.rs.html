20const ZK_LOGIN_AUTHENTICATOR_FLAG: u8 = 0x05;
21const MAX_KEY_CLAIM_NAME_LENGTH: u8 = 32;
22const MAX_KEY_CLAIM_VALUE_LENGTH: u8 = 115;
23const MAX_AUD_VALUE_LENGTH: u8 = 145;

25/// Calculate the Sui address based on address seed and address params.
26pub fn get_zk_login_address(
27    address_seed: &Bn254FrElement,
28    iss: &str,
29) -> Result<[u8; 32], FastCryptoError> {
30    let mut hasher = Blake2b256::default();
31    hasher.update([ZK_LOGIN_AUTHENTICATOR_FLAG]);
32    let bytes = iss.as_bytes();
33    hasher.update([bytes.len() as u8]);
34    hasher.update(bytes);
35    hasher.update(address_seed.padded());
36    Ok(hasher.finalize().digest)
37}

39/// Calculate the Sui address based on address seed and address params.
40pub fn gen_address_seed(
41    salt: &str,
42    name: &str,  // i.e. "sub"
43    value: &str, // i.e. the sub value
44    aud: &str,   // i.e. the client ID
45) -> Result<String, FastCryptoError> {
46    let salt_hash = poseidon_zk_login(&[(&Bn254FrElement::from_str(salt)?).into()])?;
47    gen_address_seed_with_salt_hash(&salt_hash.to_string(), name, value, aud)
48}

50/// Same as [`gen_address_seed`] but takes the poseidon hash of the salt as input instead of the salt.
51pub(crate) fn gen_address_seed_with_salt_hash(
52    salt_hash: &str,
53    name: &str,  // i.e. "sub"
54    value: &str, // i.e. the sub value
55    aud: &str,   // i.e. the client ID
56) -> Result<String, FastCryptoError> {
57    Ok(poseidon_zk_login(&[
58        hash_ascii_str_to_field(name, MAX_KEY_CLAIM_NAME_LENGTH)?,
59        hash_ascii_str_to_field(value, MAX_KEY_CLAIM_VALUE_LENGTH)?,
60        hash_ascii_str_to_field(aud, MAX_AUD_VALUE_LENGTH)?,
61        (&Bn254FrElement::from_str(salt_hash)?).into(),
62    ])?
63    .to_string())
64}

96/// Return the token exchange URL for the given auth code.
97pub fn get_token_exchange_url(
98    provider: OIDCProvider,
99    client_id: &str,
100    redirect_url: &str, // not required for Slack, pass in empty string.
101    auth_code: &str,
102    client_secret: &str, // not required for Kakao, pass in empty string.
103) -> Result<String, FastCryptoError> {
104    match provider {
105        OIDCProvider::Kakao => Ok(format!("https://kauth.kakao.com/oauth/token?grant_type=authorization_code&client_id={}&redirect_uri={}&code={}", client_id, redirect_url, auth_code)),
106        OIDCProvider::Slack => Ok(format!("https://slack.com/api/openid.connect.token?code={}&client_id={}&client_secret={}", auth_code, client_id, client_secret)),
107        _ => Err(FastCryptoError::InvalidInput)
108    }
109}

111/// Calculate the nonce for the given parameters. Nonce is defined as the Base64Url encoded of the poseidon hash of 4 inputs:
112/// first half of eph_pk_bytes in BigInt, second half of eph_pk_bytes in BigInt, max_epoch and jwt_randomness.
113pub fn get_nonce(
114    eph_pk_bytes: &[u8],
115    max_epoch: u64,
116    jwt_randomness: &str,
117) -> Result<String, FastCryptoError> {
118    let (first, second) = split_to_two_frs(eph_pk_bytes)?;
119
120    let max_epoch = Bn254Fr::from_str(&max_epoch.to_string())
121        .expect("max_epoch.to_string is always non empty string without trailing zeros");
122    let jwt_randomness =
123        Bn254Fr::from_str(jwt_randomness).map_err(|_| FastCryptoError::InvalidInput)?;
124
125    let hash = poseidon_zk_login(&[first, second, max_epoch, jwt_randomness])
126        .expect("inputs is not too long");
127    let data = BigUint::from(hash).to_bytes_be();
128    let truncated = &data[data.len() - 20..];
129    let mut buf = vec![0; Base64UrlUnpadded::encoded_len(truncated)];
130    Ok(Base64UrlUnpadded::encode(truncated, &mut buf)
131        .unwrap()
132        .to_string())
133}

142/// Call the salt server for the given jwt_token and return the salt.
143pub async fn get_salt(jwt_token: &str, salt_url: &str) -> Result<String, FastCryptoError> {
144    let client = Client::new();
145    let body = json!({ "token": jwt_token });
146    let response = client
147        .post(salt_url)
148        .json(&body)
149        .header("Content-Type", "application/json")
150        .send()
151        .await
152        .map_err(|_| FastCryptoError::InvalidInput)?;
153    let full_bytes = response
154        .bytes()
155        .await
156        .map_err(|_| FastCryptoError::InvalidInput)?;
157    let res: GetSaltResponse =
158        serde_json::from_slice(&full_bytes).map_err(|_| FastCryptoError::InvalidInput)?;
159    Ok(res.salt)
160}

200/// Given a 33-byte public key bytes (flag || pk_bytes), returns the two Bn254Fr split at the 128 bit index.
201pub fn split_to_two_frs(eph_pk_bytes: &[u8]) -> Result<(Bn254Fr, Bn254Fr), FastCryptoError> {
202    // Split the bytes deterministically such that the first element contains the first 128
203    // bits of the hash, and the second element contains the latter ones.
204    let (first_half, second_half) = eph_pk_bytes.split_at(eph_pk_bytes.len() - 16);
205    let first_bigint = BigUint::from_bytes_be(first_half);
206    // TODO: this is not safe if the buffer is large. Can we use a fixed size array for eph_pk_bytes?
207    let second_bigint = BigUint::from_bytes_be(second_half);
208
209    let eph_public_key_0 = Bn254Fr::from(first_bigint);
210    let eph_public_key_1 = Bn254Fr::from(second_bigint);
211    Ok((eph_public_key_0, eph_public_key_1))
212}
