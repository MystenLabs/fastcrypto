20const ZK_LOGIN_AUTHENTICATOR_FLAG: u8 = 0x05;
21const MAX_KEY_CLAIM_NAME_LENGTH: u8 = 32;
22const MAX_KEY_CLAIM_VALUE_LENGTH: u8 = 115;
23const MAX_AUD_VALUE_LENGTH: u8 = 145;

25/// Calculate the Sui address based on address seed and address params.
26pub fn get_zk_login_address(
27    address_seed: &Bn254FrElement,
28    iss: &str,
29) -> Result<[u8; 32], FastCryptoError> {
30    let mut hasher = Blake2b256::default();
31    hasher.update([ZK_LOGIN_AUTHENTICATOR_FLAG]);
32    let bytes = iss.as_bytes();
33    hasher.update([bytes.len() as u8]);
34    hasher.update(bytes);
35    hasher.update(address_seed.padded());
36    Ok(hasher.finalize().digest)
37}

39/// Calculate the Sui address based on address seed and address params.
40pub fn gen_address_seed(
41    salt: &str,
42    name: &str,  // i.e. "sub"
43    value: &str, // i.e. the sub value
44    aud: &str,   // i.e. the client ID
45) -> Result<String, FastCryptoError> {
46    let salt_hash = poseidon_zk_login(&[(&Bn254FrElement::from_str(salt)?).into()])?;
47    gen_address_seed_with_salt_hash(&salt_hash.to_string(), name, value, aud)
48}

50/// Same as [`gen_address_seed`] but takes the poseidon hash of the salt as input instead of the salt.
51pub(crate) fn gen_address_seed_with_salt_hash(
52    salt_hash: &str,
53    name: &str,  // i.e. "sub"
54    value: &str, // i.e. the sub value
55    aud: &str,   // i.e. the client ID
56) -> Result<String, FastCryptoError> {
57    Ok(poseidon_zk_login(&[
58        hash_ascii_str_to_field(name, MAX_KEY_CLAIM_NAME_LENGTH)?,
59        hash_ascii_str_to_field(value, MAX_KEY_CLAIM_VALUE_LENGTH)?,
60        hash_ascii_str_to_field(aud, MAX_AUD_VALUE_LENGTH)?,
61        (&Bn254FrElement::from_str(salt_hash)?).into(),
62    ])?
63    .to_string())
64}

79            OIDCProvider::Facebook => format!("https://www.facebook.com/v17.0/dialog/oauth?client_id={}&redirect_uri={}&scope=openid&nonce={}&response_type=id_token", client_id, redirect_url, nonce),
80            OIDCProvider::Kakao => format!("https://kauth.kakao.com/oauth/authorize?response_type=code&client_id={}&redirect_uri={}&nonce={}", client_id, redirect_url, nonce),
81            OIDCProvider::Apple => format!("https://appleid.apple.com/auth/authorize?client_id={}&redirect_uri={}&scope=email&response_mode=form_post&response_type=code%20id_token&nonce={}", client_id, redirect_url, nonce),
82            OIDCProvider::Slack => format!("https://slack.com/openid/connect/authorize?response_type=code&client_id={}&redirect_uri={}&nonce={}&scope=openid", client_id, redirect_url, nonce),
83            OIDCProvider::Microsoft => format!("https://login.microsoftonline.com/common/oauth2/v2.0/authorize?client_id={}&scope=openid&response_type=id_token&redirect_uri={}&nonce={}", client_id, redirect_url, nonce),
84            OIDCProvider::KarrierOne => format!("https://accounts.karrier.one/Account/PhoneLogin?ReturnUrl=/connect/authorize?nonce={}&redirect_uri={}&response_type=id_token&scope=openid&client_id={}", nonce, redirect_url, client_id),

89            OIDCProvider::AwsTenant((region, tenant_id)) => format!("https://{}.auth.{}.amazoncognito.com/login?response_type=token&client_id={}&redirect_uri={}&nonce={}", tenant_id, region, client_id, redirect_url, nonce),
90            // this URL is only useful if CLI testing from Sui is needed, can ignore if a frontend test plan is in place
91            _ => return Err(FastCryptoError::InvalidInput)
92    })

95/// Return the token exchange URL for the given auth code.
96pub fn get_token_exchange_url(
97    provider: OIDCProvider,
98    client_id: &str,
99    redirect_url: &str, // not required for Slack, pass in empty string.
100    auth_code: &str,
101    client_secret: &str, // not required for Kakao, pass in empty string.
102) -> Result<String, FastCryptoError> {
103    match provider {
104        OIDCProvider::Kakao => Ok(format!("https://kauth.kakao.com/oauth/token?grant_type=authorization_code&client_id={}&redirect_uri={}&code={}", client_id, redirect_url, auth_code)),
105        OIDCProvider::Slack => Ok(format!("https://slack.com/api/openid.connect.token?code={}&client_id={}&client_secret={}", auth_code, client_id, client_secret)),
106        _ => Err(FastCryptoError::InvalidInput)
107    }
108}

110/// Calculate the nonce for the given parameters. Nonce is defined as the Base64Url encoded of the poseidon hash of 4 inputs:
111/// first half of eph_pk_bytes in BigInt, second half of eph_pk_bytes in BigInt, max_epoch and jwt_randomness.
112pub fn get_nonce(
113    eph_pk_bytes: &[u8],
114    max_epoch: u64,
115    jwt_randomness: &str,
116) -> Result<String, FastCryptoError> {
117    let (first, second) = split_to_two_frs(eph_pk_bytes)?;
118
119    let max_epoch = Bn254Fr::from_str(&max_epoch.to_string())
120        .expect("max_epoch.to_string is always non empty string without trailing zeros");
121    let jwt_randomness =
122        Bn254Fr::from_str(jwt_randomness).map_err(|_| FastCryptoError::InvalidInput)?;
123
124    let hash = poseidon_zk_login(&[first, second, max_epoch, jwt_randomness])
125        .expect("inputs is not too long");
126    let data = BigUint::from(hash).to_bytes_be();
127    let truncated = &data[data.len() - 20..];
128    let mut buf = vec![0; Base64UrlUnpadded::encoded_len(truncated)];
129    Ok(Base64UrlUnpadded::encode(truncated, &mut buf)
130        .unwrap()
131        .to_string())
132}

141/// Call the salt server for the given jwt_token and return the salt.
142pub async fn get_salt(jwt_token: &str, salt_url: &str) -> Result<String, FastCryptoError> {
143    let client = Client::new();
144    let body = json!({ "token": jwt_token });
145    let response = client
146        .post(salt_url)
147        .json(&body)
148        .header("Content-Type", "application/json")
149        .send()
150        .await
151        .map_err(|_| FastCryptoError::InvalidInput)?;
152    let full_bytes = response
153        .bytes()
154        .await
155        .map_err(|_| FastCryptoError::InvalidInput)?;
156    let res: GetSaltResponse =
157        serde_json::from_slice(&full_bytes).map_err(|_| FastCryptoError::InvalidInput)?;
158    Ok(res.salt)
159}

199/// Given a 33-byte public key bytes (flag || pk_bytes), returns the two Bn254Fr split at the 128 bit index.
200pub fn split_to_two_frs(eph_pk_bytes: &[u8]) -> Result<(Bn254Fr, Bn254Fr), FastCryptoError> {
201    // Split the bytes deterministically such that the first element contains the first 128
202    // bits of the hash, and the second element contains the latter ones.
203    let (first_half, second_half) = eph_pk_bytes.split_at(eph_pk_bytes.len() - 16);
204    let first_bigint = BigUint::from_bytes_be(first_half);
205    // TODO: this is not safe if the buffer is large. Can we use a fixed size array for eph_pk_bytes?
206    let second_bigint = BigUint::from_bytes_be(second_half);
207
208    let eph_public_key_0 = Bn254Fr::from(first_bigint);
209    let eph_public_key_1 = Bn254Fr::from(second_bigint);
210    Ok((eph_public_key_0, eph_public_key_1))
211}
