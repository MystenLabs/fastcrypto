31/// A field element in the BN254 construction. Thin wrapper around `ark_bn254::fields::fr::Fr`.
32#[derive(Clone, Debug, From, Into, PartialEq, Eq, Display, FromStr)]
33pub struct FieldElement(pub(crate) Fr);
34
35/// A Groth16 proof in the BN254 construction. Thin wrapper around `ark_groth16::Proof::<ark_bn254::Bn254>`.
36#[derive(Debug, From)]
37pub struct Proof(pub(crate) ark_groth16::Proof<Bn254>);
38
39/// A Groth16 verifying key in the BN254 construction. Thin wrapper around `ark_groth16::VerifyingKey::<ark_bn254::Bn254>`.
40#[derive(Debug, From)]
41pub struct VerifyingKey(pub(crate) ark_groth16::VerifyingKey<ark_bn254::Bn254>);
42
43impl Proof {
44    /// Deserialize a serialized Groth16 proof using arkworks' canonical serialisation format: https://docs.rs/ark-serialize/latest/ark_serialize/.
45    pub fn deserialize(proof_points_as_bytes: &[u8]) -> FastCryptoResult<Self> {
46        ark_groth16::Proof::<Bn254>::deserialize_compressed(proof_points_as_bytes)
47            .map_err(|_| FastCryptoError::InvalidInput)
48            .map(Proof)
49    }
50}
51
52impl FieldElement {
53    /// Deserialize 32 bytes into a BN254 field element using little-endian format.
54    pub(crate) fn deserialize(bytes: &[u8]) -> FastCryptoResult<FieldElement> {
55        if bytes.len() != SCALAR_SIZE {
56            return Err(FastCryptoError::InputLengthWrong(bytes.len()));
57        }
58        Fr::deserialize_compressed(bytes)
59            .map_err(|_| FastCryptoError::InvalidInput)
60            .map(FieldElement)
61    }
62
63    /// Deserialize a vector of bytes into a vector of BN254 field elements, assuming that each element
64    /// is serialized as a chunk of 32 bytes. See also [`FieldElement::deserialize`].
65    pub(crate) fn deserialize_vector(
66        field_element_bytes: &[u8],
67    ) -> FastCryptoResult<Vec<FieldElement>> {
68        if field_element_bytes.len() % SCALAR_SIZE != 0 {
69            return Err(FastCryptoError::InputLengthWrong(field_element_bytes.len()));
70        }
71        let mut public_inputs = Vec::new();
72        for chunk in field_element_bytes.chunks(SCALAR_SIZE) {
73            public_inputs.push(FieldElement::deserialize(chunk)?);
74        }
75        Ok(public_inputs)
76    }
