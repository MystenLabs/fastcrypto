16/// The output of the Poseidon hash function is a field element in BN254 which is 254 bits long, so
17/// we need 32 bytes to represent it as an integer.
18pub const FIELD_ELEMENT_SIZE_IN_BYTES: usize = 32;
19
20/// The degree of the Merkle tree used to hash multiple elements.
21pub const MERKLE_TREE_DEGREE: usize = 16;

25/// Define a macro to calculate the poseidon hash of a vector of inputs using the neptune library.
26macro_rules! define_poseidon_hash {
27    ($inputs:expr, $poseidon_constants:expr) => {{
28        let mut poseidon = Poseidon::new(&$poseidon_constants);
29        poseidon.reset();
30        for input in $inputs.iter() {
31            poseidon.input(field_element_to_fr(input)).expect("The number of inputs must be aligned with the constants");
32        }
33        poseidon.hash_in_mode(OptimizedStatic);
34
35        // Neptune returns the state element with index 1 but we want the first element to be aligned
36        // with poseidon-rs and circomlib's implementation which returns the 0'th element.
37        //
38        // See:
39        //  * https://github.com/lurk-lab/neptune/blob/b7a9db1fc6ce096aff52b903f7d228eddea6d4e3/src/poseidon.rs#L698
40        //  * https://github.com/arnaucube/poseidon-rs/blob/f4ba1f7c32905cd2ae5a71e7568564bb150a9862/src/lib.rs#L116
41        //  * https://github.com/iden3/circomlib/blob/cff5ab6288b55ef23602221694a6a38a0239dcc0/circuits/poseidon.circom#L207
42        poseidon.elements[0]
43    }};
44}

46/// Poseidon hash function over BN254. The input vector cannot be empty and must contain at most 16
47/// elements, otherwise an error is returned.
48pub fn poseidon(inputs: &[FieldElement]) -> Result<FieldElement, FastCryptoError> {
49    if inputs.is_empty() || inputs.len() > 16 {
50        return Err(FastCryptoError::InputLengthWrong(inputs.len()));
51    }
52    // Instances of Poseidon and PoseidonConstants from neptune have different types depending on
53    // the number of inputs, so we need to use a macro here.
54    let result = match inputs.len() {
55        1 => define_poseidon_hash!(inputs, POSEIDON_CONSTANTS_U1),
56        2 => define_poseidon_hash!(inputs, POSEIDON_CONSTANTS_U2),
57        3 => define_poseidon_hash!(inputs, POSEIDON_CONSTANTS_U3),
58        4 => define_poseidon_hash!(inputs, POSEIDON_CONSTANTS_U4),
59        5 => define_poseidon_hash!(inputs, POSEIDON_CONSTANTS_U5),
60        6 => define_poseidon_hash!(inputs, POSEIDON_CONSTANTS_U6),
61        7 => define_poseidon_hash!(inputs, POSEIDON_CONSTANTS_U7),
62        8 => define_poseidon_hash!(inputs, POSEIDON_CONSTANTS_U8),
63        9 => define_poseidon_hash!(inputs, POSEIDON_CONSTANTS_U9),
64        10 => define_poseidon_hash!(inputs, POSEIDON_CONSTANTS_U10),
65        11 => define_poseidon_hash!(inputs, POSEIDON_CONSTANTS_U11),
66        12 => define_poseidon_hash!(inputs, POSEIDON_CONSTANTS_U12),
67        13 => define_poseidon_hash!(inputs, POSEIDON_CONSTANTS_U13),
68        14 => define_poseidon_hash!(inputs, POSEIDON_CONSTANTS_U14),
69        15 => define_poseidon_hash!(inputs, POSEIDON_CONSTANTS_U15),
70        16 => define_poseidon_hash!(inputs, POSEIDON_CONSTANTS_U16),
71        _ => return Err(InvalidInput),
72    };
73    Ok(fr_to_field_element(result))
74}

76/// Calculate the poseidon hash of the field element inputs. If the input length is <= MERKLE_TREE_DEGREE, calculate
77/// H(inputs), otherwise chunk the inputs into groups of MERKLE_TREE_DEGREE, hash them and input the results recursively.
78pub fn poseidon_merkle_tree(inputs: &[FieldElement]) -> FastCryptoResult<FieldElement> {
79    if inputs.len() <= MERKLE_TREE_DEGREE {
80        poseidon(inputs)
81    } else {
82        poseidon_merkle_tree(
83            &inputs
84                .chunks(MERKLE_TREE_DEGREE)
85                .map(poseidon)
86                .collect::<FastCryptoResult<Vec<_>>>()?,
87        )
88    }
89}

91/// Calculate the Poseidon hash of an array of inputs. Each input is interpreted as a BN254 field
92/// element assuming a little-endian encoding and must be 32 bytes long.
93///
94/// The field elements are hashed using the poseidon hash function ([poseidon])
95/// and the result is serialized as a little-endian integer (32 bytes).
96///
97/// If one of the inputs is in non-canonical form, e.g., it represents an integer greater than the
98/// field size or is not exactly 32 bytes, an [InvalidInput] error is returned.
99///
100/// This function is used as an interface to the poseidon hash function in the sui-framework.
101///
102/// Note that this returns an error if no inputs are given or if more than 16 inputs are provided.
103pub fn poseidon_bytes(inputs: &[Vec<u8>]) -> FastCryptoResult<[u8; FIELD_ELEMENT_SIZE_IN_BYTES]> {
104    let field_elements = inputs
105        .iter()
106        .map(|b| canonical_le_bytes_to_field_element(b))
107        .collect::<Result<Vec<_>, _>>()?;
108    let output_as_field_element = poseidon(&field_elements)?;
109    Ok(field_element_to_canonical_le_bytes(
110        &output_as_field_element,
111    ))
112}
113
114/// Given a binary representation of a BN254 field element as an integer in little-endian encoding,
115/// this function returns the corresponding field element. If the field element is not canonical (is
116/// larger than the field size as an integer), an [InvalidInput] is returned.
117///
118/// If the input is not exactly 32 bytes long, an [InvalidInput] is returned.
119fn canonical_le_bytes_to_field_element(bytes: &[u8]) -> FastCryptoResult<FieldElement> {
120    if bytes.len() != FIELD_ELEMENT_SIZE_IN_BYTES {
121        return Err(InvalidInput);
122    }
123    let field_element = Fr::from_le_bytes_mod_order(bytes);
124    // Unfortunately, there doesn't seem to be a nice way to check if a modular reduction
125    // happened without doing the extra work of serializing the field element again.
126    let reduced_bytes = field_element.into_bigint().to_bytes_le();
127    if reduced_bytes != bytes {
128        return Err(InvalidInput);
129    }
130    Ok(FieldElement(field_element))
131}
132
133/// Convert a BN254 field element to a byte array as the little-endian representation of the
134/// underlying canonical integer representation of the element.
135fn field_element_to_canonical_le_bytes(
136    field_element: &FieldElement,
137) -> [u8; FIELD_ELEMENT_SIZE_IN_BYTES] {
138    field_element
139        .0
140        .into_bigint()
141        .to_bytes_le()
142        .try_into()
143        .expect("The result is guaranteed to be 32 bytes")
144}

146/// Convert an ff field element to an arkworks-ff field element.
147fn fr_to_field_element(fr: crate::Fr) -> FieldElement {
148    // We use big-endian as in the definition of the BN254 prime field (see fastcrypto-zkp/src/lib.rs).
149    FieldElement(Fr::from_be_bytes_mod_order(fr.to_repr().as_byte_slice()))
150}
151
152/// Convert an arkworks-ff field element to an ff field element.
153fn field_element_to_fr(fr: &FieldElement) -> crate::Fr {
154    let mut bytes = [0u8; 32];
155    // We use big-endian as in the definition of the BN254 prime field (see fastcrypto-zkp/src/lib.rs).
156    bytes.clone_from_slice(&fr.0.into_bigint().to_bytes_be());
157    crate::Fr::from_repr_vartime(FrRepr(bytes))
158        .expect("The bytes of fr are guaranteed to be canonical here")
159}

247    fn test_hash_to_bytes() {
248        let mut one = vec![0; 32];
249        one[0] = 1;
250
251        let inputs: Vec<Vec<u8>> = vec![one.clone()];
252        let hash = poseidon_bytes(&inputs).unwrap();
253        // 18586133768512220936620570745912940619677854269274689475585506675881198879027 in decimal
254        let expected =
255            hex::decode("33018202c57d898b84338b16d1a4960e133c6a4d656cfec1bd62a9ea00611729")
256                .unwrap();
257        assert_eq!(hash.as_slice(), &expected);
258
259        let mut two = vec![0; 32];
260        two[0] = 2;
261
262        // 7853200120776062878684798364095072458815029376092732009249414926327459813530 in decimal
263        let inputs: Vec<Vec<u8>> = vec![one.clone(), two.clone()];
264        let hash = poseidon_bytes(&inputs).unwrap();
265        let expected =
266            hex::decode("9a1817447a60199e51453274f217362acfe962966b4cf63d4190d6e7f5c05c11")
267                .unwrap();
268        assert_eq!(hash.as_slice(), &expected);
269
270        // Input larger than the modulus
271        let inputs = vec![vec![255; 32]];
272        assert!(poseidon_bytes(&inputs).is_err());
