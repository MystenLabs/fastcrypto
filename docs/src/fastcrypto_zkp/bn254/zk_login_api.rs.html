23/// Enum to specify the environment to use for verifying keys.
24#[derive(Serialize, Clone, Deserialize, Debug, Eq, PartialEq, Copy)]
25pub enum ZkLoginEnv {
26    /// Use the secure global verifying key derived from ceremony.
27    Prod,
28    /// Use the insecure global verifying key.
29    Test,
30}
31
32impl Default for ZkLoginEnv {
33    fn default() -> Self {
34        Self::Prod
35    }
36}

322/// Entry point for the ZkLogin API.
323pub fn verify_zk_login(
324    input: &ZkLoginInputs,
325    max_epoch: u64,
326    eph_pubkey_bytes: &[u8],
327    all_jwk: &ImHashMap<JwkId, JWK>,
328    env: &ZkLoginEnv,
329) -> Result<(), FastCryptoError> {
330    // Load the expected JWK based on (iss, kid).
331    let (iss, kid) = (input.get_iss().to_string(), input.get_kid().to_string());
332    let jwk = all_jwk
333        .get(&JwkId::new(iss.clone(), kid.clone()))
334        .ok_or_else(|| {
335            FastCryptoError::GeneralError(format!("JWK not found ({} - {})", iss, kid))
336        })?;
337
338    // Decode modulus to bytes.
339    let modulus = Base64UrlUnpadded::decode_vec(&jwk.n).map_err(|_| {
340        FastCryptoError::GeneralError("Invalid Base64 encoded jwk modulus".to_string())
341    })?;
342
343    // Calculat all inputs hash and passed to the verification function.
344    match verify_zk_login_proof_with_fixed_vk(
345        env,
346        &input.get_proof().as_arkworks()?,
347        &[input.calculate_all_inputs_hash(eph_pubkey_bytes, &modulus, max_epoch)?],
348    ) {
349        Ok(true) => Ok(()),
350        Ok(false) | Err(_) => Err(FastCryptoError::GeneralError(
351            "Groth16 proof verify failed".to_string(),
352        )),
353    }
354}

356/// Verify a proof against its public inputs using the fixed verifying key.
357pub fn verify_zk_login_proof_with_fixed_vk(
358    usage: &ZkLoginEnv,
359    proof: &Proof<Bn254>,
360    public_inputs: &[Bn254Fr],
361) -> Result<bool, FastCryptoError> {
362    let vk = match usage {
363        ZkLoginEnv::Prod => &GLOBAL_VERIFYING_KEY,
364        ZkLoginEnv::Test => &INSECURE_VERIFYING_KEY,
365    };
366    Groth16::<Bn254>::verify_with_processed_vk(vk, public_inputs, proof)
367        .map_err(|e| FastCryptoError::GeneralError(e.to_string()))
368}

370/// Verify that the given parameters (name, value, aud, iss and salt_hash) were used to generate the
371/// given address.
372pub fn verify_zk_login_id(
373    address: &[u8],
374    name: &str,
375    value: &str,
376    aud: &str,
377    iss: &str,
378    salt_hash: &str,
379) -> FastCryptoResult<()> {
380    let address_seed = gen_address_seed_with_salt_hash(salt_hash, name, value, aud)?;
381    verify_zk_login_iss(address, &address_seed, iss)
382}
383
384/// Verify that the given parameters (address_seed and iss) were used to generate the given address.
385pub fn verify_zk_login_iss(address: &[u8], address_seed: &str, iss: &str) -> FastCryptoResult<()> {
386    let reconstructed_address =
387        get_zk_login_address(&Bn254FrElement::from_str(address_seed)?, iss)?;
388    match reconstructed_address == address {
389        true => Ok(()),
390        false => Err(FastCryptoError::InvalidProof),
391    }
