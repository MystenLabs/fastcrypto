469/// Circuit verify mode.
470#[derive(Debug, Copy, Clone, Eq, PartialEq)]
471pub enum ZkLoginCircuitMode {
472    /// Accept proofs against the v1 vk only; v2 proofs are rejected. Currently supported.
473    V1Only,
474    /// Try the v2 circuit first, then fall back to v1. Migration phase.
475    Both,
476    /// Accept proofs against the v2 vk only; v1 proofs are rejected.
477    V2Only,
478}
479
480/// Verify a zkLogin proof against the circuit versions allowed by `mode`.
481pub fn verify_zk_login(
482    input: &ZkLoginInputs,
483    max_epoch: u64,
484    eph_pubkey_bytes: &[u8],
485    all_jwk: &ImHashMap<JwkId, JWK>,
486    env: &ZkLoginEnv,
487    mode: ZkLoginCircuitMode,
488) -> Result<(), FastCryptoError> {
489    // Load the expected JWK based on (iss, kid).
490    let (iss, kid) = (input.get_iss().to_string(), input.get_kid().to_string());
491    let jwk = all_jwk
492        .get(&JwkId::new(iss.clone(), kid.clone()))
493        .ok_or_else(|| {
494            FastCryptoError::GeneralError(format!("JWK not found ({} - {})", iss, kid))
495        })?;
496
497    // Decode modulus to bytes.
498    let modulus = Base64UrlUnpadded::decode_vec(&jwk.n).map_err(|_| {
499        FastCryptoError::GeneralError("Invalid Base64 encoded jwk modulus".to_string())
500    })?;
501
502    let proof = input.get_proof().as_arkworks()?;
503
504    let verify = |version| {
505        verify_with_version(
506            input,
507            max_epoch,
508            eph_pubkey_bytes,
509            &modulus,
510            env,
511            &proof,
512            version,
513        )
514    };
515    match mode {
516        ZkLoginCircuitMode::V1Only => verify(CircuitVersion::V1),
517        ZkLoginCircuitMode::V2Only => verify(CircuitVersion::V2),
518        ZkLoginCircuitMode::Both => verify(CircuitVersion::V2).or_else(|e| {
519            tracing::debug!(
520                "[zkLogin] V2 verify failed (env={:?}, iss={}): {:?}, falling back to V1",
521                env,
522                iss,
523                e
524            );
525            verify(CircuitVersion::V1)
526        }),
527    }
528}

530/// Verify a zkLogin proof against a single circuit version's input hash and verifying key.
531fn verify_with_version(
532    input: &ZkLoginInputs,
533    max_epoch: u64,
534    eph_pubkey_bytes: &[u8],
535    modulus: &[u8],
536    env: &ZkLoginEnv,
537    proof: &Proof<Bn254>,
538    version: CircuitVersion,
539) -> Result<(), FastCryptoError> {
540    let all_inputs_hash =
541        input.calculate_all_inputs_hash(eph_pubkey_bytes, modulus, max_epoch, version)?;
542    match verify_zk_login_proof_with_fixed_vk(
543        env,
544        proof,
545        &[all_inputs_hash],
546        version == CircuitVersion::V2,
547    ) {
548        Ok(true) => Ok(()),
549        Ok(false) | Err(_) => Err(FastCryptoError::GeneralError(
550            "Groth16 proof verify failed".to_string(),
551        )),
552    }
553}

555/// Circuit version, which determines the layout of the `all_inputs_hash` public input.
556#[derive(Debug, Copy, Clone, Eq, PartialEq)]
557pub enum CircuitVersion {
558    /// V1 circuit: the iss claim is folded in as its *base64* value and `index_mod_4` is folded
559    /// into the hash. There is no `rsa_num_bits` field.
560    V1,
561    /// V2 circuit: the iss claim is folded in as its *decoded* extended claim, `index_mod_4` is
562    /// dropped, and the actual RSA modulus bit length (`rsa_num_bits`) is folded in instead.
563    V2,
564}
565
566const V1_MAX_EXT_ISS_LEN: u16 = 165;
567const V2_MAX_EXT_ISS_LEN: u16 = 186;
568
569impl CircuitVersion {
570    /// Circuit-specific parameters (max field lengths) for this version.
571    pub fn config(&self) -> CircuitConfig {
572        match self {
573            CircuitVersion::V1 => CircuitConfig {
574                max_header_len_b64: 248,
575                max_iss_len: 4 * (1 + V1_MAX_EXT_ISS_LEN / 3),
576                max_rsa_bits: 2048,
577            },
578            CircuitVersion::V2 => CircuitConfig {
579                max_header_len_b64: 279,
580                max_iss_len: V2_MAX_EXT_ISS_LEN,
581                max_rsa_bits: 8192,
582            },
583        }
584    }
585}
586
587/// Circuit-specific parameters, obtained via [CircuitVersion::config].
588#[derive(Debug, Copy, Clone)]
589pub struct CircuitConfig {
590    /// Maximum header length in base64.
591    pub max_header_len_b64: u16,
592    /// Maximum length passed to the iss hash. For [CircuitVersion::V1] this is the base64 length
593    /// of the iss claim; for [CircuitVersion::V2] it is the byte length of the *decoded* extended
594    /// iss claim (the circuit's `maxExtIssLength`).
595    pub max_iss_len: u16,
596    /// Maximum RSA key size in bits. The modulus is padded to this size before hashing.
597    pub max_rsa_bits: u16,
598}

600/// Verify a proof against its public inputs using the fixed verifying key.
601pub fn verify_zk_login_proof_with_fixed_vk(
602    usage: &ZkLoginEnv,
603    proof: &Proof<Bn254>,
604    public_inputs: &[Bn254Fr],
605    v2: bool,
606) -> Result<bool, FastCryptoError> {
607    let vk = if v2 {
608        match usage {
609            ZkLoginEnv::Prod => &GLOBAL_VERIFYING_KEY_V2,
610            ZkLoginEnv::Test => &INSECURE_VERIFYING_KEY_V2,
611        }
612    } else {
613        match usage {
614            ZkLoginEnv::Prod => &GLOBAL_VERIFYING_KEY,
615            ZkLoginEnv::Test => &INSECURE_VERIFYING_KEY,
616        }
617    };
618
619    Groth16::<Bn254>::verify_with_processed_vk(vk, public_inputs, proof)
620        .map_err(|e| FastCryptoError::GeneralError(e.to_string()))
621}

623/// Verify that the given parameters (name, value, aud, iss and salt_hash) were used to generate the
624/// given address.
625pub fn verify_zk_login_id(
626    address: &[u8],
627    name: &str,
628    value: &str,
629    aud: &str,
630    iss: &str,
631    salt_hash: &str,
632) -> FastCryptoResult<()> {
633    let address_seed = gen_address_seed_with_salt_hash(salt_hash, name, value, aud)?;
634    verify_zk_login_iss(address, &address_seed, iss)
635}

637/// Verify that the given parameters (address_seed and iss) were used to generate the given address.
638pub fn verify_zk_login_iss(address: &[u8], address_seed: &str, iss: &str) -> FastCryptoResult<()> {
639    let reconstructed_address =
640        get_zk_login_address(&Bn254FrElement::from_str(address_seed)?, iss)?;
641    match reconstructed_address == address {
642        true => Ok(()),
643        false => Err(FastCryptoError::InvalidProof),
644    }
