12/// Size of scalars in the BN254 construction.
13pub const SCALAR_SIZE: usize = 32;
14
15/// Deserialize bytes as an Arkwork representation of a verifying key, and return a vector of the
16/// four components of a prepared verified key (see more at [`PreparedVerifyingKey`]).
17pub fn prepare_pvk_bytes(vk_bytes: &[u8]) -> Result<Vec<Vec<u8>>, FastCryptoError> {
18    PreparedVerifyingKey::from(&VerifyingKey::deserialize(vk_bytes)?).serialize()
19}
20
21/// Verify Groth16 proof using the serialized form of the prepared verifying key (see more at
22/// [`crate::bn254::verifier::PreparedVerifyingKey`]), serialized proof public input and serialized
23/// proof points.
24pub fn verify_groth16_in_bytes(
25    vk_gamma_abc_g1_bytes: &[u8],
26    alpha_g1_beta_g2_bytes: &[u8],
27    gamma_g2_neg_pc_bytes: &[u8],
28    delta_g2_neg_pc_bytes: &[u8],
29    proof_public_inputs_as_bytes: &[u8],
30    proof_points_as_bytes: &[u8],
31) -> Result<bool, FastCryptoError> {
32    if proof_public_inputs_as_bytes.len() % SCALAR_SIZE != 0 {
33        return Err(FastCryptoError::InputLengthWrong(SCALAR_SIZE));
34    }
35
36    let pvk = PreparedVerifyingKey::deserialize(&[
37        vk_gamma_abc_g1_bytes,
38        alpha_g1_beta_g2_bytes,
39        gamma_g2_neg_pc_bytes,
40        delta_g2_neg_pc_bytes,
41    ])?;
42
43    verify_groth16(&pvk, proof_public_inputs_as_bytes, proof_points_as_bytes)
44}
45
46/// Verify proof with a given verifying key in [struct PreparedVerifyingKey], serialized public inputs
47/// and serialized proof points.
48pub fn verify_groth16(
49    pvk: &PreparedVerifyingKey,
50    proof_public_inputs_as_bytes: &[u8],
51    proof_points_as_bytes: &[u8],
52) -> Result<bool, FastCryptoError> {
53    let proof = Proof::deserialize(proof_points_as_bytes)?;
54    let public_inputs = FieldElement::deserialize_vector(proof_public_inputs_as_bytes)?;
55    pvk.verify(&public_inputs, &proof)
56}
