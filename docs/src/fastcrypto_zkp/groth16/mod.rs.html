6use fastcrypto::error::{FastCryptoError, FastCryptoResult};

9use fastcrypto::groups::{GroupElement, MultiScalarMul, Pairing};

11pub mod api;

13#[derive(Debug, Deserialize)]
14pub struct Proof<G1: Pairing>
15where
16    G1::Other: Debug,
17{
18    a: G1,
19    b: G1::Other,
20    c: G1,
21}
22
23#[derive(Debug)]
24pub struct VerifyingKey<G1: Pairing>
25where
26    G1::Other: Debug,
27{
28    alpha: G1,
29    beta: G1::Other,
30    gamma: G1::Other,
31    delta: G1::Other,
32    gamma_abc: Vec<G1>,
33}
34
35/// This is a helper function to store a pre-processed version of the verifying key.
36/// This is roughly homologous to [`ark_groth16::data_structures::PreparedVerifyingKey`].
37/// Note that contrary to Arkworks, we don't store a "prepared" version of the `gamma_neg` and
38/// `delta_neg` fields because they are very large and unpractical to use in the binary API.
39pub struct PreparedVerifyingKey<G1>
40where
41    G1: Pairing,
42{
43    /// The element vk.gamma_abc,
44    /// aka the `[gamma^{-1} * (beta * a_i + alpha * b_i + c_i) * G]`, where i spans the public inputs
45    vk_gamma_abc: Vec<G1>,
46
47    /// The element `e(alpha * G, beta * H)` in `GT`.
48    alpha_beta: <G1 as Pairing>::Output,
49
50    /// The element `- gamma * H` in `G2`, for use in pairings.
51    gamma_neg: <G1 as Pairing>::Other,
52
53    /// The element `- delta * H` in `G2`, for use in pairings.
54    delta_neg: <G1 as Pairing>::Other,
55}
56
57impl<G1> From<&VerifyingKey<G1>> for PreparedVerifyingKey<G1>
58where
59    G1: Pairing,
60{
61    fn from(vk: &VerifyingKey<G1>) -> Self {
62        PreparedVerifyingKey {
63            vk_gamma_abc: vk.gamma_abc.clone(),
64            alpha_beta: vk.alpha.pairing(&vk.beta),
65            gamma_neg: -vk.gamma,
66            delta_neg: -vk.delta,
67        }
68    }
69}
70
71impl<G1: Pairing> PreparedVerifyingKey<G1> {
72    /// Verify Groth16 proof using the prepared verifying key (see more at
73    /// [`crate::bn254::verifier::PreparedVerifyingKey`]), a vector of public inputs and the proof.
74    pub fn verify(
75        &self,
76        public_inputs: &[G1::ScalarType],
77        proof: &Proof<G1>,
78    ) -> FastCryptoResult<()>
79    where
80        G1: MultiScalarMul,
81        <G1 as Pairing>::Output: GroupElement,
82    {
83        let prepared_inputs = self.prepare_inputs(public_inputs)?;
84        self.verify_with_prepared_inputs(&prepared_inputs, proof)
85    }
86
87    /// Verify Groth16 proof using the prepared verifying key (see more at
88    /// [`crate::bn254::verifier::PreparedVerifyingKey`]), a prepared public input (see
89    /// [`prepare_inputs`]) and the proof.
90    pub fn verify_with_prepared_inputs(
91        &self,
92        prepared_inputs: &G1,
93        proof: &Proof<G1>,
94    ) -> FastCryptoResult<()>
95    where
96        <G1 as Pairing>::Output: GroupElement,
97    {
98        let lhs = G1::multi_pairing(
99            &[proof.a, *prepared_inputs, proof.c],
100            &[proof.b, self.gamma_neg, self.delta_neg],
101        )?;
102        if lhs == self.alpha_beta {
103            Ok(())
104        } else {
105            Err(FastCryptoError::InvalidProof)
106        }
107    }
108
109    /// Prepare the public inputs for use in [`verify_with_prepared_inputs`].
110    pub fn prepare_inputs(&self, public_inputs: &[G1::ScalarType]) -> FastCryptoResult<G1>
111    where
112        G1: MultiScalarMul,
113    {
114        if (public_inputs.len() + 1) != self.vk_gamma_abc.len() {
115            return Err(FastCryptoError::InvalidInput);
116        }
117        if public_inputs.is_empty() {
118            return Ok(self.vk_gamma_abc[0]);
119        }
120        G1::multi_scalar_mul(public_inputs, &self.vk_gamma_abc[1..])
121            .map(|x| x + self.vk_gamma_abc[0])
122    }
123}
