13/// Deserialize bytes as an Arkworks representation of a verifying key, and return a vector of the
14/// four components of a prepared verified key (see more at [`PreparedVerifyingKey`]).
15pub fn prepare_pvk_bytes<
16    G1,
17    const G1_SIZE: usize,
18    const G2_SIZE: usize,
19    const GT_SIZE: usize,
20    const FR_SIZE: usize,
21>(
22    vk_bytes: &[u8],
23) -> FastCryptoResult<Vec<Vec<u8>>>
24where
25    G1: Pairing + ToFromByteArray<G1_SIZE>,
26    <G1 as Pairing>::Other: ToFromByteArray<G2_SIZE>,
27    <G1 as Pairing>::Output: GTSerialize<GT_SIZE>,
28{
29    let vk = VerifyingKey::<G1>::from_arkworks_format::<G1_SIZE, G2_SIZE>(vk_bytes)?;
30    Ok(PreparedVerifyingKey::from(&vk).serialize_into_parts())
31}

33/// Verify Groth16 proof using the serialized form of the four components in a prepared verifying key
34/// (see more at [`PreparedVerifyingKey`]), serialized proof public input, which should
35/// be concatenated serialized field elements of the scalar field of [`crate::conversions::SCALAR_SIZE`]
36/// bytes each in little-endian format, and serialized proof points.
37pub fn verify_groth16_in_bytes<
38    G1,
39    const G1_SIZE: usize,
40    const G2_SIZE: usize,
41    const GT_SIZE: usize,
42    const FR_SIZE: usize,
43>(
44    vk_gamma_abc_g1_bytes: &[u8],
45    alpha_g1_beta_g2_bytes: &[u8],
46    gamma_g2_neg_pc_bytes: &[u8],
47    delta_g2_neg_pc_bytes: &[u8],
48    public_inputs_as_bytes: &[u8],
49    proof_points_as_bytes: &[u8],
50) -> Result<bool, FastCryptoError>
51where
52    G1: ToFromByteArray<G1_SIZE> + DeserializeOwned + MultiScalarMul + Pairing,
53    <G1 as Pairing>::Other: ToFromByteArray<G2_SIZE> + DeserializeOwned,
54    <G1 as Pairing>::Output: GroupElement + GTSerialize<GT_SIZE>,
55    G1::ScalarType: FromLittleEndianByteArray<FR_SIZE>,
56{
57    let public_inputs = deserialize_vector(
58        public_inputs_as_bytes,
59        G1::ScalarType::from_little_endian_byte_array,
60    )?;
61    let proof =
62        bcs::from_bytes(proof_points_as_bytes).map_err(|_| FastCryptoError::InvalidInput)?;
63    let prepared_vk = PreparedVerifyingKey::<G1>::deserialize_from_parts(
64        vk_gamma_abc_g1_bytes,
65        alpha_g1_beta_g2_bytes,
66        gamma_g2_neg_pc_bytes,
67        delta_g2_neg_pc_bytes,
68    )?;
69    Ok(prepared_vk.verify(&public_inputs, &proof).is_ok())
70}

72impl<G1: Pairing> VerifyingKey<G1> {
73    pub fn from_arkworks_format<const G1_SIZE: usize, const G2_SIZE: usize>(
74        bytes: &[u8],
75    ) -> FastCryptoResult<Self>
76    where
77        G1: ToFromByteArray<G1_SIZE>,
78        <G1 as Pairing>::Other: ToFromByteArray<G2_SIZE>,
79    {
80        // The verifying key consists of:
81        // - alpha: G1
82        // - beta: G2
83        // - gamma: G2
84        // - delta: G2
85        // - n: u64 lendian (size of gamma_abc)
86        // - gamma_abc: Vec<G1>
87
88        // We can't use bcs because there, the length of the vector is prefixed as a single byte and
89        // not a little-endian u64 as it is here.
90
91        if bytes.len() < G1_SIZE + 3 * G2_SIZE + size_of::<u64>()
92            || !(bytes.len() - (G1_SIZE + 3 * G2_SIZE + size_of::<u64>())).is_multiple_of(G1_SIZE)
93        {
94            return Err(FastCryptoError::InvalidInput);
95        }
96
97        let (alpha, bytes) = bytes.split_at(G1_SIZE);
98        let alpha = G1::from_byte_array(alpha.try_into().expect("Length already checked"))?;
99
100        let (beta, bytes) = bytes.split_at(G2_SIZE);
101        let beta = G1::Other::from_byte_array(beta.try_into().expect("Length already checked"))?;
102
103        let (gamma, bytes) = bytes.split_at(G2_SIZE);
104        let gamma = G1::Other::from_byte_array(gamma.try_into().expect("Length already checked"))?;
105
106        let (delta, bytes) = bytes.split_at(G2_SIZE);
107        let delta = G1::Other::from_byte_array(delta.try_into().expect("Length already checked"))?;
108
109        let (gamma_abc_length, bytes) = bytes.split_at(size_of::<u64>());
110        let gamma_abc_length =
111            u64::from_le_bytes(gamma_abc_length.try_into().expect("Length already checked"));
112
113        // There must be at least one element in gamma_abc, since this should be equal to the number
114        // of public inputs + 1.
115        if gamma_abc_length == 0 {
116            return Err(FastCryptoError::InvalidInput);
117        }
118
119        let gamma_abc = deserialize_vector(bytes, G1::from_byte_array)?;
120
121        if gamma_abc.len() != gamma_abc_length as usize {
122            return Err(FastCryptoError::InvalidInput);
123        }
124
125        Ok(VerifyingKey {
126            alpha,
127            beta,
128            gamma,
129            delta,
130            gamma_abc,
131        })
132    }
133}

135impl<G1: Pairing> PreparedVerifyingKey<G1> {
136    pub fn serialize_into_parts<const G1_SIZE: usize, const G2_SIZE: usize, const GT_SIZE: usize>(
137        &self,
138    ) -> Vec<Vec<u8>>
139    where
140        G1: ToFromByteArray<G1_SIZE>,
141        G1::Other: ToFromByteArray<G2_SIZE>,
142        <G1 as Pairing>::Output: GTSerialize<GT_SIZE>,
143    {
144        vec![
145            serialize_vector(&self.vk_gamma_abc, G1::to_byte_array),
146            self.alpha_beta.to_arkworks_bytes().to_vec(),
147            self.gamma_neg.to_byte_array().to_vec(),
148            self.delta_neg.to_byte_array().to_vec(),
149        ]
150    }
151
152    pub fn deserialize_from_parts<
153        const G1_SIZE: usize,
154        const G2_SIZE: usize,
155        const GT_SIZE: usize,
156    >(
157        vk_gamma_abc_bytes: &[u8],
158        alpha_beta_bytes: &[u8],
159        gamma_neg_bytes: &[u8],
160        delta_neg_bytes: &[u8],
161    ) -> FastCryptoResult<Self>
162    where
163        G1: ToFromByteArray<G1_SIZE>,
164        G1::Other: ToFromByteArray<G2_SIZE>,
165        <G1 as Pairing>::Output: GTSerialize<GT_SIZE>,
166    {
167        let vk_gamma_abc =
168            deserialize_vector::<G1_SIZE, G1>(vk_gamma_abc_bytes, G1::from_byte_array)?;
169
170        let alpha_beta = <G1 as Pairing>::Output::from_arkworks_bytes(
171            alpha_beta_bytes
172                .try_into()
173                .map_err(|_| FastCryptoError::InvalidInput)?,
174        )?;
175
176        let gamma_neg = <G1 as Pairing>::Other::from_byte_array(
177            gamma_neg_bytes
178                .try_into()
179                .map_err(|_| FastCryptoError::InvalidInput)?,
180        )?;
181
182        let delta_neg = <G1 as Pairing>::Other::from_byte_array(
183            delta_neg_bytes
184                .try_into()
185                .map_err(|_| FastCryptoError::InvalidInput)?,
186        )?;
187
188        Ok(Self {
189            vk_gamma_abc,
190            alpha_beta,
191            gamma_neg,
192            delta_neg,
193        })
194    }
195}

197/// Serialization of GT elements is typically not standardized across libraries, so implementations
198/// must specify what implementation to use here to be compatible with the arkworks format (see
199/// [`ark_ec::pairing::PairingOutput`]).
200pub trait GTSerialize<const SIZE_IN_BYTES: usize>: Sized {
201    /// Serialize the element into a byte array.
202    fn to_arkworks_bytes(&self) -> [u8; SIZE_IN_BYTES];
203
204    /// Deserialize the element from a byte array.
205    fn from_arkworks_bytes(bytes: &[u8; SIZE_IN_BYTES]) -> FastCryptoResult<Self>;
206}
207
208/// Scalars given to the API are expected to be in little-endian format.
209pub trait FromLittleEndianByteArray<const SIZE_IN_BYTES: usize>: Sized {
210    fn from_little_endian_byte_array(bytes: &[u8; SIZE_IN_BYTES]) -> FastCryptoResult<Self>;
211}

213/// Given the length of a verifying key in Arkworks format, compute the number of public inputs
214/// for circuits, this verifying key can be used for.
215///
216/// If the length cannot be of a valid verifying key, return an InvalidInput error.
217pub(crate) fn get_public_inputs_num(
218    g1_size: usize,
219    g2_size: usize,
220    vk_length_in_bytes: usize,
221) -> FastCryptoResult<usize> {
222    // The length of the fixed parts of a verifying key: alpha (G1), beta (G2), gamma (G2),
223    // delta (G2) and the first element of gamma_abc (u64 length + G1).
224    let prefix = 2 * g1_size + 3 * g2_size + size_of::<u64>();
225    if vk_length_in_bytes < prefix {
226        return Err(FastCryptoError::InvalidInput);
227    }
228    let per_public_input = vk_length_in_bytes - prefix;
229    if !per_public_input.is_multiple_of(g1_size) {
230        return Err(FastCryptoError::InvalidInput);
231    }
232    // gamma_abc.len() = #public_inputs + 1
233    Ok(per_public_input / g1_size)
234}
