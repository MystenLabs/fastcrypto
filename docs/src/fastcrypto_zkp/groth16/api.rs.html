14/// Deserialize bytes as an Arkworks representation of a verifying key, and return a vector of the
15/// four components of a prepared verified key (see more at [`PreparedVerifyingKey`]).
16pub fn prepare_pvk_bytes<
17    G1,
18    const G1_SIZE: usize,
19    const G2_SIZE: usize,
20    const GT_SIZE: usize,
21    const FR_SIZE: usize,
22>(
23    vk_bytes: &[u8],
24) -> FastCryptoResult<Vec<Vec<u8>>>
25where
26    G1: Pairing + ToFromByteArray<G1_SIZE>,
27    <G1 as Pairing>::Other: ToFromByteArray<G2_SIZE>,
28    <G1 as Pairing>::Output: GTSerialize<GT_SIZE>,
29{
30    let vk = VerifyingKey::<G1>::from_arkworks_format::<G1_SIZE, G2_SIZE>(vk_bytes)?;
31    Ok(PreparedVerifyingKey::from(&vk).serialize_into_parts())
32}

34/// Verify Groth16 proof using the serialized form of the four components in a prepared verifying key
35/// (see more at [`PreparedVerifyingKey`]), serialized proof public input, which should
36/// be concatenated serialized field elements of the scalar field of [`crate::conversions::SCALAR_SIZE`]
37/// bytes each in little-endian format, and serialized proof points.
38pub fn verify_groth16_in_bytes<
39    G1,
40    const G1_SIZE: usize,
41    const G2_SIZE: usize,
42    const GT_SIZE: usize,
43    const FR_SIZE: usize,
44>(
45    vk_gamma_abc_g1_bytes: &[u8],
46    alpha_g1_beta_g2_bytes: &[u8],
47    gamma_g2_neg_pc_bytes: &[u8],
48    delta_g2_neg_pc_bytes: &[u8],
49    public_inputs_as_bytes: &[u8],
50    proof_points_as_bytes: &[u8],
51) -> Result<bool, FastCryptoError>
52where
53    G1: ToFromByteArray<G1_SIZE> + DeserializeOwned + MultiScalarMul + Pairing,
54    <G1 as Pairing>::Other: ToFromByteArray<G2_SIZE> + DeserializeOwned,
55    <G1 as Pairing>::Output: GroupElement + GTSerialize<GT_SIZE>,
56    G1::ScalarType: FromLittleEndianByteArray<FR_SIZE>,
57{
58    let public_inputs = deserialize_vector(
59        public_inputs_as_bytes,
60        G1::ScalarType::from_little_endian_byte_array,
61    )?;
62    let proof =
63        bcs::from_bytes(proof_points_as_bytes).map_err(|_| FastCryptoError::InvalidInput)?;
64    let prepared_vk = PreparedVerifyingKey::<G1>::deserialize_from_parts(
65        vk_gamma_abc_g1_bytes,
66        alpha_g1_beta_g2_bytes,
67        gamma_g2_neg_pc_bytes,
68        delta_g2_neg_pc_bytes,
69    )?;
70    Ok(prepared_vk.verify(&public_inputs, &proof).is_ok())
71}

73impl<G1: Pairing> VerifyingKey<G1> {
74    pub fn from_arkworks_format<const G1_SIZE: usize, const G2_SIZE: usize>(
75        bytes: &[u8],
76    ) -> FastCryptoResult<Self>
77    where
78        G1: ToFromByteArray<G1_SIZE>,
79        <G1 as Pairing>::Other: ToFromByteArray<G2_SIZE>,
80    {
81        // The verifying key consists of:
82        // - alpha: G1
83        // - beta: G2
84        // - gamma: G2
85        // - delta: G2
86        // - n: u64 lendian (size of gamma_abc)
87        // - gamma_abc: Vec<G1>
88
89        // We can't use bcs because there, the length of the vector is prefixed as a single byte and
90        // not a little-endian u64 as it is here.
91
92        if bytes.len() < G1_SIZE + 3 * G2_SIZE + size_of::<u64>()
93            || (bytes.len() - (G1_SIZE + 3 * G2_SIZE + size_of::<u64>())) % G1_SIZE != 0
94        {
95            return Err(FastCryptoError::InvalidInput);
96        }
97
98        let (alpha, bytes) = bytes.split_at(G1_SIZE);
99        let alpha = G1::from_byte_array(alpha.try_into().expect("Length already checked"))?;
100
101        let (beta, bytes) = bytes.split_at(G2_SIZE);
102        let beta = G1::Other::from_byte_array(beta.try_into().expect("Length already checked"))?;
103
104        let (gamma, bytes) = bytes.split_at(G2_SIZE);
105        let gamma = G1::Other::from_byte_array(gamma.try_into().expect("Length already checked"))?;
106
107        let (delta, bytes) = bytes.split_at(G2_SIZE);
108        let delta = G1::Other::from_byte_array(delta.try_into().expect("Length already checked"))?;
109
110        let (gamma_abc_length, bytes) = bytes.split_at(size_of::<u64>());
111        let gamma_abc_length =
112            u64::from_le_bytes(gamma_abc_length.try_into().expect("Length already checked"));
113
114        // There must be at least one element in gamma_abc, since this should be equal to the number
115        // of public inputs + 1.
116        if gamma_abc_length == 0 {
117            return Err(FastCryptoError::InvalidInput);
118        }
119
120        let gamma_abc = deserialize_vector(bytes, G1::from_byte_array)?;
121
122        if gamma_abc.len() != gamma_abc_length as usize {
123            return Err(FastCryptoError::InvalidInput);
124        }
125
126        Ok(VerifyingKey {
127            alpha,
128            beta,
129            gamma,
130            delta,
131            gamma_abc,
132        })
133    }
134}

136impl<G1: Pairing> PreparedVerifyingKey<G1> {
137    pub fn serialize_into_parts<const G1_SIZE: usize, const G2_SIZE: usize, const GT_SIZE: usize>(
138        &self,
139    ) -> Vec<Vec<u8>>
140    where
141        G1: ToFromByteArray<G1_SIZE>,
142        G1::Other: ToFromByteArray<G2_SIZE>,
143        <G1 as Pairing>::Output: GTSerialize<GT_SIZE>,
144    {
145        vec![
146            serialize_vector(&self.vk_gamma_abc, G1::to_byte_array),
147            self.alpha_beta.to_arkworks_bytes().to_vec(),
148            self.gamma_neg.to_byte_array().to_vec(),
149            self.delta_neg.to_byte_array().to_vec(),
150        ]
151    }
152
153    pub fn deserialize_from_parts<
154        const G1_SIZE: usize,
155        const G2_SIZE: usize,
156        const GT_SIZE: usize,
157    >(
158        vk_gamma_abc_bytes: &[u8],
159        alpha_beta_bytes: &[u8],
160        gamma_neg_bytes: &[u8],
161        delta_neg_bytes: &[u8],
162    ) -> FastCryptoResult<Self>
163    where
164        G1: ToFromByteArray<G1_SIZE>,
165        G1::Other: ToFromByteArray<G2_SIZE>,
166        <G1 as Pairing>::Output: GTSerialize<GT_SIZE>,
167    {
168        let vk_gamma_abc =
169            deserialize_vector::<G1_SIZE, G1>(vk_gamma_abc_bytes, G1::from_byte_array)?;
170
171        let alpha_beta = <G1 as Pairing>::Output::from_arkworks_bytes(
172            alpha_beta_bytes
173                .try_into()
174                .map_err(|_| FastCryptoError::InvalidInput)?,
175        )?;
176
177        let gamma_neg = <G1 as Pairing>::Other::from_byte_array(
178            gamma_neg_bytes
179                .try_into()
180                .map_err(|_| FastCryptoError::InvalidInput)?,
181        )?;
182
183        let delta_neg = <G1 as Pairing>::Other::from_byte_array(
184            delta_neg_bytes
185                .try_into()
186                .map_err(|_| FastCryptoError::InvalidInput)?,
187        )?;
188
189        Ok(Self {
190            vk_gamma_abc,
191            alpha_beta,
192            gamma_neg,
193            delta_neg,
194        })
195    }
196}

198/// Serialization of GT elements is typically not standardized across libraries, so implementations
199/// must specify what implementation to use here to be compatible with the arkworks format (see
200/// [`ark_ec::pairing::PairingOutput`]).
201pub trait GTSerialize<const SIZE_IN_BYTES: usize>: Sized {
202    /// Serialize the element into a byte array.
203    fn to_arkworks_bytes(&self) -> [u8; SIZE_IN_BYTES];
204
205    /// Deserialize the element from a byte array.
206    fn from_arkworks_bytes(bytes: &[u8; SIZE_IN_BYTES]) -> FastCryptoResult<Self>;
207}
208
209/// Scalars given to the API are expected to be in little-endian format.
210pub trait FromLittleEndianByteArray<const SIZE_IN_BYTES: usize>: Sized {
211    fn from_little_endian_byte_array(bytes: &[u8; SIZE_IN_BYTES]) -> FastCryptoResult<Self>;
212}
