11/// A G1 point in BN254 serialized as a vector of three strings which is the canonical decimal
12/// representation of the projective coordinates in Fq.
13pub type CircomG1 = Vec<Bn254FqElement>;
14
15/// A G2 point in BN254 serialized as a vector of three vectors each being a vector of two strings
16/// which are the canonical decimal representation of the coefficients of the projective coordinates
17/// in Fq2.
18pub type CircomG2 = Vec<Vec<Bn254FqElement>>;

20/// A struct that stores a Bn254 Fq field element as 32 bytes.
21#[derive(Debug, Clone, JsonSchema, Eq, PartialEq)]
22pub struct Bn254FqElement(#[schemars(with = "String")] [u8; 32]);
23
24impl std::str::FromStr for Bn254FqElement {
25    type Err = FastCryptoError;
26
27    fn from_str(s: &str) -> Result<Self, Self::Err> {
28        let big_int = Fq::from_str(s).map_err(|_| FastCryptoError::InvalidInput)?;
29        let be_bytes = big_int.into_bigint().to_bytes_be();
30        be_bytes
31            .try_into()
32            .map_err(|_| FastCryptoError::InvalidInput)
33            .map(Bn254FqElement)
34    }
35}
36
37impl std::fmt::Display for Bn254FqElement {
38    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
39        let big_int = BigUint::from_bytes_be(&self.0);
40        let radix10 = big_int.to_string();
41        f.write_str(&radix10)
42    }
43}
44
45// Bn254FqElement's serialized format is as a radix10 encoded string
46impl Serialize for Bn254FqElement {
47    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
48    where
49        S: serde::Serializer,
50    {
51        self.to_string().serialize(serializer)
52    }
53}
54
55impl<'de> Deserialize<'de> for Bn254FqElement {
56    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
57    where
58        D: serde::Deserializer<'de>,
59    {
60        let s = std::borrow::Cow::<'de, str>::deserialize(deserializer)?;

65/// A struct that stores a Bn254 Fr field element as 32 bytes.
66#[derive(Debug, Clone, JsonSchema, Eq, PartialEq)]
67pub struct Bn254FrElement(#[schemars(with = "String")] [u8; 32]);
68
69impl Bn254FrElement {
70    /// Returns the unpadded version of the field element. This returns with leading zeros removed.
71    pub fn unpadded(&self) -> &[u8] {
72        let mut buf = self.0.as_slice();
73
74        while !buf.is_empty() && buf[0] == 0 {
75            buf = &buf[1..];
76        }
77
78        // If the value is '0' then just return a slice of length 1 of the final byte
79        if buf.is_empty() {
80            &self.0[31..]
81        } else {
82            buf
83        }
84    }
85
86    /// Returns the padded version of the field element. This returns with leading zeros preserved to 32 bytes.
87    pub fn padded(&self) -> &[u8] {
88        &self.0
89    }
90}

146/// Deserialize a G1 projective point in BN254 serialized as a vector of three strings into an affine
147/// G1 point in arkworks format. Return an error if the input is not a vector of three strings or if
148/// any of the strings cannot be parsed as a field element.
149pub(crate) fn g1_affine_from_str_projective(s: &CircomG1) -> Result<G1Affine, FastCryptoError> {
150    if s.len() != 3 {
151        return Err(FastCryptoError::InvalidInput);
152    }
153
154    let g1: G1Affine =
155        G1Projective::new_unchecked((&s[0]).into(), (&s[1]).into(), (&s[2]).into()).into();
156
157    if !g1.is_on_curve() || !g1.is_in_correct_subgroup_assuming_on_curve() {
158        return Err(FastCryptoError::InvalidInput);
159    }
160
161    Ok(g1)
162}
163
164/// Deserialize a G2 projective point from the BN254 construction serialized as a vector of three
165/// vectors each being a vector of two strings into an affine G2 point in arkworks format. Return an
166/// error if the input is not a vector of the right format or if any of the strings cannot be parsed
167/// as a field element.
168pub(crate) fn g2_affine_from_str_projective(s: &CircomG2) -> Result<G2Affine, FastCryptoError> {
169    if s.len() != 3 || s[0].len() != 2 || s[1].len() != 2 || s[2].len() != 2 {
170        return Err(FastCryptoError::InvalidInput);
171    }
172
173    let g2: G2Affine = G2Projective::new_unchecked(
174        Fq2::new((&s[0][0]).into(), (&s[0][1]).into()),
175        Fq2::new((&s[1][0]).into(), (&s[1][1]).into()),
176        Fq2::new((&s[2][0]).into(), (&s[2][1]).into()),
177    )
178    .into();
179
180    if !g2.is_on_curve() || !g2.is_in_correct_subgroup_assuming_on_curve() {
181        return Err(FastCryptoError::InvalidInput);
182    }
183
184    Ok(g2)
185}

187#[cfg(test)]
188mod test {
189    use crate::zk_login_utils::Bn254FqElement;
190    use std::str::FromStr;
191
192    use super::Bn254FrElement;
193    use num_bigint::BigUint;
194    use proptest::prelude::*;
195    #[test]
196    fn from_str_on_digits_only() {
197        // do not allow non digit results.
198        assert!(Bn254FrElement::from_str("10_________0").is_err());
199        assert!(Bn254FqElement::from_str("10_________0").is_err());
200        // do not allow leading zeros.
201        assert!(Bn254FrElement::from_str("000001").is_err());
202        assert!(Bn254FqElement::from_str("000001").is_err());
203        assert!(Bn254FrElement::from_str("garbage").is_err());
204        assert!(Bn254FqElement::from_str("garbage").is_err());
205    }
206    #[test]
207    fn unpadded_slice() {
208        let seed = Bn254FrElement([0; 32]);
209        let zero: [u8; 1] = [0];
210        assert_eq!(seed.unpadded(), zero.as_slice());
211
212        let mut seed = Bn254FrElement([1; 32]);
213        seed.0[0] = 0;
214        assert_eq!(seed.unpadded(), [1; 31].as_slice());
215    }
216
217    proptest! {
218        #[test]
219        fn dont_crash_on_large_inputs(
220            bytes in proptest::collection::vec(any::<u8>(), 33..1024)
221        ) {
222            let big_int = BigUint::from_bytes_be(&bytes);
223            let radix10 = big_int.to_str_radix(10);
224
225            // doesn't crash
226            let _ = Bn254FrElement::from_str(&radix10);
227            let _ = Bn254FqElement::from_str(&radix10);
228        }
229    }
