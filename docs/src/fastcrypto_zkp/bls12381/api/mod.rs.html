16/// Create a prepared verifying key for Groth16 over the BLS12-381 curve construction. See
17/// [`api::prepare_pvk_bytes`].
18pub fn prepare_pvk_bytes(vk_bytes: &[u8]) -> Result<Vec<Vec<u8>>, FastCryptoError> {
19    api::prepare_pvk_bytes::<
20        G1Element,
21        { G1_ELEMENT_BYTE_LENGTH },
22        { G2_ELEMENT_BYTE_LENGTH },
23        { GT_ELEMENT_BYTE_LENGTH },
24        { SCALAR_LENGTH },
25    >(vk_bytes)
26}

28/// Verify Groth16 proof over the BLS12-381 curve construction. See
29/// [`api::verify_groth16_in_bytes`].
30pub fn verify_groth16_in_bytes(
31    vk_gamma_abc_g1_bytes: &[u8],
32    alpha_g1_beta_g2_bytes: &[u8],
33    gamma_g2_neg_pc_bytes: &[u8],
34    delta_g2_neg_pc_bytes: &[u8],
35    public_inputs_as_bytes: &[u8],
36    proof_points_as_bytes: &[u8],
37) -> Result<bool, FastCryptoError> {
38    api::verify_groth16_in_bytes::<
39        G1Element,
40        { G1_ELEMENT_BYTE_LENGTH },
41        { G2_ELEMENT_BYTE_LENGTH },
42        { GT_ELEMENT_BYTE_LENGTH },
43        { SCALAR_LENGTH },
44    >(
45        vk_gamma_abc_g1_bytes,
46        alpha_g1_beta_g2_bytes,
47        gamma_g2_neg_pc_bytes,
48        delta_g2_neg_pc_bytes,
49        public_inputs_as_bytes,
50        proof_points_as_bytes,
51    )
