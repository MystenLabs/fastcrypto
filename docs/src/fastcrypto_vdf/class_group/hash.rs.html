24/// The security parameter for the hash function in bits. The image will be at least
25/// 2^{2*SECURITY_PARAMETER} large to ensure that the hash function is collision resistant.
26const SECURITY_PARAMETER_IN_BITS: u64 = 128;
27
28/// This lower limit ensures that the default, secure parameters set below give valid results,
29/// namely a reduced quadratic form.
30const MINIMAL_DISCRIMINANT_SIZE: u64 = 600;
31
32/// The image size of the hash function will be "Number of primes of size at most
33/// DEFAULT_PRIME_FACTOR_SIZE_IN_BYTES" * DEFAULT_PRIME_FACTORS, so these have been set such that
34/// the image is ~260 bits. See [n_bit_primes] for the details of this computation.
35const DEFAULT_PRIME_FACTORS: u64 = 2;
36
37/// The default size of the prime factors should be set such that it is not possible for an
38/// adversary to precompute the VDF on all quadratic forms with the first coordinate being the
39/// primes of this size. This is an issue because if an adversary can precompute (a1, _, _)^T and
40/// (a2, _, _)^T then it is possible to compute (a1*a2, _, _)^T as the composition (a1, _, _)^T *
41/// (a2, _, _)^T.
42const DEFAULT_PRIME_FACTOR_SIZE_IN_BYTES: u64 = 17;
43
44impl QuadraticForm {
45    /// Generate a random quadratic form from a seed with the given discriminant. This method is
46    /// deterministic, and it is a random oracle on a large subset of the class group.
47    ///
48    /// This method returns an [InvalidInput] error if the discriminant is so small that there are
49    /// no secure parameters, and it may also happen if the discriminant is not a prime.
50    pub fn hash_to_group_with_default_parameters(
51        seed: &[u8],
52        discriminant: &Discriminant,
53    ) -> FastCryptoResult<Self> {
54        if discriminant.bits() <= MINIMAL_DISCRIMINANT_SIZE {
55            return Err(InvalidInput);
56        }
57        hash_to_group_with_custom_parameters(
58            seed,
59            discriminant,
60            DEFAULT_PRIME_FACTOR_SIZE_IN_BYTES,
61            DEFAULT_PRIME_FACTORS,
62        )
63    }
64}

66/// Generate a random quadratic form from a seed with the given discriminant and custom parameters.
67///
68/// The output will be a uniformly random element from the set of points (a,b,c) where a = p_1 ... p_k
69/// for some primes p_i < 2^lambda.
70///
71/// If the discriminant is not a negative prime, an [InvalidInput] error may be returned.
72///
73/// The parameters must be chosen carefully to ensure that the function is secure and for all
74/// use cases, [hash_to_group] should be used.
75fn hash_to_group_with_custom_parameters(
76    seed: &[u8],
77    discriminant: &Discriminant,
78    prime_factor_size_in_bytes: u64,
79    prime_factors: u64,
80) -> FastCryptoResult<QuadraticForm> {
81    // Ensure that the image is sufficiently large
82    debug_assert!(
83        prime_factors as f64 * n_bit_primes(prime_factor_size_in_bytes * 8)
84            >= 2.0 * SECURITY_PARAMETER_IN_BITS as f64
85    );
86
87    // Ensure that the prime factors are so large that the corresponding quadratic form cannot be precomputed.
88    debug_assert!(
89        n_bit_primes(prime_factor_size_in_bytes * 8) >= SECURITY_PARAMETER_IN_BITS as f64
90    );
91
92    // Ensure that the result will be reduced
93    debug_assert!(
94        discriminant.as_bigint().abs().sqrt().shr(1)
95            > BigInt::one().shl(prime_factors * prime_factor_size_in_bytes)
96    );
97
98    // Sample a and b such that a < sqrt(|discriminant|)/2 has exactly prime_factors prime factors and b is the square root of the discriminant modulo a.
99    let (a, mut b) = sample_modulus(
100        seed,
101        discriminant,
102        prime_factor_size_in_bytes,
103        prime_factors,
104    )?;
105
106    // b must be odd but may be negative
107    if b.is_even() {
108        b -= &a;
109    }
110
111    QuadraticForm::from_a_b_and_discriminant(a, b, discriminant)
112}

114/// Sample a product of `prime_factors` primes each of size `prime_factor_size_in_bytes` and return
115/// this along with the square root of the discriminant modulo `a`. If the discriminant is not a
116/// prime, an [InvalidInput] error may be returned.
117fn sample_modulus(
118    seed: &[u8],
119    discriminant: &Discriminant,
120    prime_factor_size_in_bytes: u64,
121    prime_factors: u64,
122) -> FastCryptoResult<(BigInt, BigInt)> {
123    // Seed a rng with the hash of the seed
124    let mut rng = ChaCha8Rng::from_seed(Sha256::digest(seed).digest);
125    let mut factors = Vec::with_capacity(prime_factors as usize);
126    let mut square_roots = Vec::with_capacity(prime_factors as usize);
127
128    for _ in 0..prime_factors {
129        let mut factor;
130        loop {
131            factor = sample_odd_number(prime_factor_size_in_bytes, &mut rng);
132
133            if factors.contains(&factor) {
134                continue;
135            }
136
137            // The primality check does not try divisions with small primes, so we do it here. This speeds up
138            // the algorithm significantly.
139            if PRIMES.iter().any(|p| factor.is_multiple_of(p)) {
140                continue;
141            }
142
143            if jacobi::jacobi(discriminant.as_bigint(), &factor)
144                .expect("factor is odd and positive")
145                == 1
146                && is_probable_prime(factor.magnitude())
147            {
148                // Found a valid factor
149                break;
150            }
151        }
152        // This only fails if the discriminant is not prime.
153        let square_root = modular_square_root(discriminant.as_bigint(), &factor, false)
154            .map_err(|_| InvalidInput)?;
155        factors.push(factor);
156        square_roots.push(square_root);
157    }
158
159    let result = factors.iter().product();
160    let square_root = solve_congruence_equation_system(&square_roots, &factors)
161        .expect("The factors are distinct primes");
162
163    Ok((result, square_root))
164}

166/// Returns an approximation of the log2 of the number of primes smaller than 2^n.
167fn n_bit_primes(n: u64) -> f64 {
168    // The Prime Number Theorem states that the number of primes smaller than n is close to n / ln(n),
169    // so the number of primes smaller than 2^n is approximately:
170    //
171    // log2(2^n / ln 2^n) = n - log2(ln 2^n)
172    //                    = n - log2(n ln 2)
173    //                    = n - log2(n) - log2(ln 2)
174    n as f64 - (n as f64).log2() - 2f64.ln().log2()
175}

177/// Sample a random odd number smaller than 2^{8*size_in_bytes}.
178fn sample_odd_number<R: Rng>(size_in_bytes: u64, rng: &mut R) -> BigInt {
179    let mut bytes = vec![0u8; size_in_bytes as usize];
180    rng.fill_bytes(&mut bytes);
181    bytes[0] |= 1;
182    BigInt::from_bytes_le(Sign::Plus, &bytes)
183}

210        let mut seed = [0u8; 32];
211        let discriminant = Discriminant::from_seed(&seed, 1024).unwrap();
212
213        for _ in 0..10 {
214            let qf = hash_to_group_with_custom_parameters(&seed, &discriminant, 17, 2).unwrap();

216            assert!(qf.is_in_group(&discriminant));

220        for _ in 0..10 {

223            assert!(qf.is_in_group(&discriminant));
224            seed[0] += 1;
225        }

228    #[test]
229    fn qf_from_seed_sanity_tests() {
230        let discriminant = Discriminant::from_seed(b"discriminant seed", 800).unwrap();
231        let base_qf =
232            hash_to_group_with_custom_parameters(b"qf seed", &discriminant, 17, 3).unwrap();
233        assert!(base_qf.is_in_group(&discriminant));
234
235        // Same seed, same discriminant, same k
236        let other_qf =
237            hash_to_group_with_custom_parameters(b"qf seed", &discriminant, 17, 3).unwrap();
238        assert_eq!(base_qf, other_qf);
239
240        // Smaller k
241        let other_qf =
242            hash_to_group_with_custom_parameters(b"qf seed", &discriminant, 17, 2).unwrap();
243        assert_ne!(base_qf, other_qf);
244
245        // Larger k
246        let other_qf =
247            hash_to_group_with_custom_parameters(b"qf seed", &discriminant, 17, 4).unwrap();
248        assert_ne!(base_qf, other_qf);
249
250        let mut seed = [0u8; 32];
251        for _ in 0..10 {
252            // Different seed
253            thread_rng().fill_bytes(&mut seed);
254            let other_qf =
255                hash_to_group_with_custom_parameters(&seed, &discriminant, 17, 3).unwrap();
256            assert_ne!(base_qf, other_qf);
257        }
258
259        let other_discriminant = Discriminant::from_seed(b"other discriminant seed", 800).unwrap();
260
261        // Same seed, same k, other discriminant
262        let other_qf =
263            hash_to_group_with_custom_parameters(b"qf seed", &other_discriminant, 17, 2).unwrap();
264        assert_ne!(base_qf, other_qf);
265    }
