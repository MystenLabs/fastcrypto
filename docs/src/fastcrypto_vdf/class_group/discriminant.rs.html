17/// A discriminant for an imaginary class group. The discriminant is a negative integer congruent to

38impl TryFrom<BigInt> for Discriminant {
39    type Error = FastCryptoError;
40
41    /// A valid discriminant should be a negative prime congruent to 1 mod 8. For large discriminants,
42    /// this is very slow to check.
43    fn try_from(value: BigInt) -> FastCryptoResult<Self> {
44        if !value.is_negative()
45            || value.mod_floor(&BigInt::from(8)) != BigInt::one()
46            || !is_probable_prime(value.abs().magnitude())
47        {
48            return Err(InvalidInput);
49        }
50        Ok(Self(value))
51    }
52}
53
54impl Discriminant {
55    /// Return the number of bits needed to represent this discriminant, not including the sign bit.
56    pub fn bits(&self) -> u64 {
57        self.0.bits()
58    }
59
60    /// Borrow a reference to the underlying big integer.
61    pub(crate) fn as_bigint(&self) -> &BigInt {
62        &self.0
63    }
64
65    /// Create a discriminant from the given value. It is assumed that the discriminant is a negative
66    /// prime which is 1 mod 8. If this is not the case, some functions may panic, so this should only
67    /// be used in tests.
68    #[cfg(test)]
69    pub(crate) fn from_trusted_bigint(value: BigInt) -> Self {
70        Self(value)
71    }
72
73    /// Compute a valid discriminant (aka a negative prime equal to 1 mod 8) based on the given seed.
74    /// The size_in_bits must be divisible by 8.
75    pub fn from_seed(seed: &[u8], size_in_bits: usize) -> FastCryptoResult<Discriminant> {
76        if size_in_bits % 8 != 0 {
77            return Err(InvalidInput);
78        }
79        // Set the lower three bits to ensure that the prime is 7 mod 8 which makes the discriminant 1 mod 8.
80        Ok(Self(
81            hash_prime::hash_prime(seed, size_in_bits / 8, &[0, 1, 2, size_in_bits - 1])
82                .to_bigint()
83                .expect("Never fails")
84                .neg(),
85        ))
86    }
87}

93    #[test]
94    fn test_discriminant() {
95        let discriminant = Discriminant(-BigInt::from(223));
96        assert_eq!(discriminant.bits(), 8);
97        assert_eq!(discriminant.as_bigint(), &-BigInt::from(223));
98
99        // Invalid modulus
100        let candidate = BigInt::from(-29);
101        assert!(candidate.is_negative());
102        assert!(Discriminant::try_from(candidate).is_err());
103
104        // Invalid sign
105        let candidate = BigInt::from(17);
106        assert_eq!(candidate.mod_floor(&BigInt::from(8)), BigInt::from(1));
107        assert!(Discriminant::try_from(candidate).is_err());
108
109        // Not prime
110        let candidate = BigInt::from(-231);
111        assert!(Discriminant::try_from(candidate).is_err());
112    }

125    #[test]
126    fn test_discriminant_to_from_bytes() {
127        let discriminant = Discriminant(BigInt::from(-223));
128        let bytes = bcs::to_bytes(&discriminant).unwrap();
129        let discriminant2 = bcs::from_bytes(&bytes).unwrap();
130        assert_eq!(discriminant, discriminant2);
131
132        let discriminant = Discriminant::from_seed(&[0x01, 0x02, 0x03], 512).unwrap();
133        let bytes = bcs::to_bytes(&discriminant).unwrap();
134        let discriminant2 = bcs::from_bytes(&bytes).unwrap();
135        assert_eq!(discriminant, discriminant2);
136
137        // Test serde on invalid discriminants
138        let invalid_discriminant_bytes =
139            bcs::to_bytes(&BigInt::from(-221).to_signed_bytes_be()).unwrap();
140        assert!(bcs::from_bytes::<Discriminant>(&invalid_discriminant_bytes).is_err());
141
142        let invalid_discriminant_bytes =
143            bcs::to_bytes(&BigInt::from(17).to_signed_bytes_be()).unwrap();
144        assert!(bcs::from_bytes::<Discriminant>(&invalid_discriminant_bytes).is_err());
