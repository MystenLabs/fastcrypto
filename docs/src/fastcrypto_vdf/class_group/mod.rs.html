26/// This module implements a hash function for imaginary class groups which takes an arbitrary binary input and returns
27/// an element in the class group. The output of the hash function is uniformly random on a large subset of the class
28/// group but not the entire class group.
29pub(crate) mod hash;

31/// Two quadratic forms may represent the same element in the class group, but each equivalence class contains exactly
32/// one reduced form. This module contains methods to reduce quadratic forms, which besides uniqueness also ensures that
33/// the coefficients do not become too large.
34pub(crate) mod reduction;

36/// Discriminants of quadratic forms are negative primes which is 1 mod 8. This module contains a type to represent
37/// discriminants and methods to create them.

46/// A binary quadratic form, (a, b, c) for arbitrary integers a, b, and c.
47#[derive(Eq, Debug, Clone, Serialize, Deserialize)]
48pub struct QuadraticForm {
49    #[serde(with = "bigint_serde")]
50    pub a: BigInt,
51    #[serde(with = "bigint_serde")]
52    pub b: BigInt,
53    #[serde(with = "bigint_serde")]
54    pub c: BigInt,
55    #[serde(skip)]
56    partial_gcd_limit: OnceCell<BigInt>,
57}
58
59impl QuadraticForm {
60    /// Create a new quadratic form given only the a and b coefficients and the discriminant.
61    pub fn from_a_b_and_discriminant(
62        a: BigInt,
63        b: BigInt,
64        discriminant: &Discriminant,
65    ) -> FastCryptoResult<Self> {
66        if !a.is_positive() {
67            return Err(InvalidInput);
68        }
69
70        let numerator = b.pow(2) - discriminant.as_bigint();
71        let denominator = &a << 2;
72        if !numerator.is_multiple_of(&denominator) {
73            return Err(InvalidInput);
74        }
75
76        let c = numerator / denominator;
77        Ok(Self {
78            a,
79            b,
80            c,
81            partial_gcd_limit: OnceCell::new(),
82        })
83    }
84
85    /// The GCD computation in composition may finish early when coefficients are below this limit
86    /// which is set lazily to `L = |discriminant|^{1/4}`. Algorithm 1 from Jacobson, Jr, Michael &
87    /// Poorten, Alfred (2002). "Computational aspects of NUCOMP", Lecture Notes in Computer Science.
88    /// (https://www.researchgate.net/publication/221451638_Computational_aspects_of_NUCOMP)
89    fn partial_gcd_limit(&self) -> &BigInt {
90        self.partial_gcd_limit
91            .get_or_init(|| self.discriminant().abs().nth_root(4))
92    }
93
94    /// Compute the discriminant of this quadratic form, eg. b^2 - 4ac.
95    pub fn discriminant(&self) -> BigInt {
96        self.b.pow(2) - ((&self.a * &self.c) << 2)
97    }
98
99    /// Return a generator (or, more precisely, an element with a presumed large order) in a class group with a given
100    /// discriminant which is 1 mod 8. We use the element `(2, 1, c)` where `c` is determined from the discriminant.
101    pub fn generator(discriminant: &Discriminant) -> Self {
102        Self::from_a_b_and_discriminant(BigInt::from(2), BigInt::one(), discriminant)
103            .expect("Always succeeds when the discriminant is 1 mod 8")
104    }
105
106    /// Compute the composition of this quadratic form with another quadratic form.
107    pub fn compose(&self, rhs: &QuadraticForm) -> QuadraticForm {
108        // Slightly optimised version of Algorithm 1 from Jacobson, Jr, Michael & Poorten, Alfred
109        // (2002). "Computational aspects of NUCOMP", Lecture Notes in Computer Science.
110        // (https://www.researchgate.net/publication/221451638_Computational_aspects_of_NUCOMP)
111        // The paragraph numbers and variable names follow the paper.
112
113        let u1 = &self.a;
114        let v1 = &self.b;
115        let w1 = &self.c;
116        let u2 = &rhs.a;
117        let v2 = &rhs.b;
118        let w2 = &rhs.c;
119
120        // 1.
121        if w1 < w2 {
122            swap(&mut (u1, v1, w1), &mut (u2, v2, w2));
123        }
124        let s: BigInt = (v1 + v2) >> 1;
125        let m = v2 - &s;
126
127        // 2.
128        let EuclideanAlgorithmOutput {
129            gcd: f,
130            x: b,
131            y: c,
132            a_divided_by_gcd: mut capital_cy,
133            b_divided_by_gcd: mut capital_by,
134        } = extended_euclidean_algorithm(u2, u1, true);
135        let b = b.expect("Was computed because compute_x is true above");
136
137        let (q, r) = s.div_rem(&f);
138        let (g, capital_bx, capital_dy) = if r.is_zero() {
139            (f, &m * &b, q)
140        } else {
141            // 3.
142            let EuclideanAlgorithmOutput {
143                gcd: g,
144                x: _,
145                y,
146                a_divided_by_gcd: h,
147                b_divided_by_gcd,
148            } = extended_euclidean_algorithm(&f, &s, false);
149
150            // 4.
151            let l = (&y * (&b * (w1.mod_floor(&h)) + &c * (w2.mod_floor(&h)))).mod_floor(&h);
152            let capital_bx = &b * (&m / &h) + &l * &capital_by;
153            capital_by *= &h;
154            capital_cy *= &h;
155            (g, capital_bx, b_divided_by_gcd)
156        };
157
158        // 5. (partial xgcd)
159        let (bx, x, by, y, iterated) = partial_xgcd(
160            capital_bx.mod_floor(&capital_by),
161            capital_by.clone(),
162            self.partial_gcd_limit(),
163        );
164
165        let u3: BigInt;
166        let w3: BigInt;
167        let v3: BigInt;
168
169        if !iterated {
170            // 6.
171            let q = &capital_cy * &bx;
172            let cx = (&q - &m) / &capital_by;
173            let dx = (&bx * &capital_dy - w2) / &capital_by;
174            u3 = &by * &capital_cy;
175            v3 = v2 - (&q << 1);
176            w3 = &bx * &cx - &g * &dx;
177        } else {
178            // 7.
179            let cx = (&capital_cy * &bx - &m * &x) / &capital_by;
180            let q1 = &by * &cx;
181            let q2 = &q1 + &m;
182            let dx = (&capital_dy * &bx - w2 * &x) / &capital_by;
183            let q3 = &y * &dx;
184            let q4 = &q3 + &capital_dy;
185            let dy = &q4 / &x;
186            let cy = if !b.is_zero() {
187                &q2 / &bx
188            } else {
189                (&cx * &dy - w1) / &dx
190            };
191
192            u3 = &by * &cy - &g * &y * &dy;
193            v3 = &g * (&q3 + &q4) - &q1 - &q2;
194            w3 = &bx * &cx - &g * &x * &dx;
195        }
196
197        QuadraticForm {
198            a: u3,
199            b: v3,
200            c: w3,
201            partial_gcd_limit: self.partial_gcd_limit.clone(),
202        }
203        .reduce()
204    }
205}

207impl Doubling for QuadraticForm {
208    fn double(self) -> Self {
209        // Slightly optimised version of Algorithm 2 from Jacobson, Jr, Michael & Poorten, Alfred
210        // (2002). "Computational aspects of NUCOMP", Lecture Notes in Computer Science.
211        // (https://www.researchgate.net/publication/221451638_Computational_aspects_of_NUCOMP)
212        // The paragraph numbers and variable names follow the paper.
213
214        let EuclideanAlgorithmOutput {
215            gcd: g,
216            x: _,
217            y,
218            a_divided_by_gcd: capital_by,
219            b_divided_by_gcd: capital_dy,
220        } = extended_euclidean_algorithm(&self.a, &self.b, false);
221
222        let (bx, x, by, y, iterated) = partial_xgcd(
223            (&y * &self.c).mod_floor(&capital_by),
224            capital_by.clone(),
225            self.partial_gcd_limit(),
226        );
227
228        let mut u3 = &by * &by;
229        let mut w3 = &bx * &bx;
230        let mut v3 = -(by * &bx) << 1;
231
232        if !iterated {
233            let dx = (&bx * &capital_dy - &self.c) / &capital_by;
234            v3 += &self.b;
235            w3 -= &g * &dx;
236        } else {
237            let dx = (&bx * &capital_dy - &self.c * &x) / &capital_by;
238            let q1 = &dx * &y;
239            let mut dy = &q1 + &capital_dy;
240            v3 += &g * (&dy + &q1);
241            dy /= &x;
242
243            u3 -= &g * &y * &dy;
244            w3 -= &g * &x * &dx;
245        }
246
247        QuadraticForm {
248            a: u3,
249            b: v3,
250            c: w3,
251            partial_gcd_limit: self.partial_gcd_limit.clone(),
252        }
253        .reduce()
254    }
255}

257/// Compute the xgcd of bx and by with a partial limit. When by is below the limit, the computation
258/// stops early and returns the result. The result is a tuple (bx, x, by, y, iterated) where bx and
259/// by are the now reduced coefficients, x and y are the Bezout coefficients for bx and by respectively,
260/// and iterated is true if the there were any iterations.
261fn partial_xgcd(
262    mut bx: BigInt,
263    mut by: BigInt,
264    limit: &BigInt,
265) -> (BigInt, BigInt, BigInt, BigInt, bool) {
266    let mut x = BigInt::one();
267    let mut y = BigInt::zero();
268    let mut iterated = false;
269    let mut odd = false;
270
271    while by.abs() > *limit && !bx.is_zero() {
272        let (q, r) = by.div_rem(&bx);
273        by = bx;
274        bx = r;
275        swap(&mut x, &mut y);
276        x -= &q * &y;
277
278        odd = !odd;
279        iterated = true;
280    }
281
282    if odd {
283        by = -by;
284        y = -y;
285    }
286
287    (bx, x, by, y, iterated)
288}

290impl ParameterizedGroupElement for QuadraticForm {
291    /// The discriminant of a quadratic form defines the class group.
292    type ParameterType = Discriminant;
293
294    fn zero(discriminant: &Self::ParameterType) -> Self {
295        Self::from_a_b_and_discriminant(BigInt::one(), BigInt::one(), discriminant)
296            .expect("Doesn't fail")
297    }
298
299    fn is_in_group(&self, discriminant: &Discriminant) -> bool {
300        discriminant.as_bigint().eq(&self.discriminant())
301    }
302}

341impl PartialEq for QuadraticForm {
342    fn eq(&self, other: &Self) -> bool {
343        // Ignore the partial_gcd_limit field. This field is computed lazily and is not part of the
344        // actual value of the QuadraticForm.
345        self.a == other.a && self.b == other.b && self.c == other.c
346    }
