14/// Default size in bytes of the Fiat-Shamir challenge used in proving and verification.
15///
16/// This is based on Pietrzak (2018), "Simple Verifiable Delay Functions" (https://eprint.iacr.org/2018/627.pdf)
17/// which states that the challenge should be 2^l bits (see section 6), where l is the security
18/// parameter. Soundness is proven in section 6.3 in this paper.
19pub const DEFAULT_CHALLENGE_SIZE_IN_BYTES: usize = 16;
20
21/// This implements Pietrzak's VDF construction from https://eprint.iacr.org/2018/627.pdf.
22///
23/// The VDF is, as in [crate::vdf::wesolowski::WesolowskisVDF], based on the repeated squaring of an
24/// element in a group of unknown order. However, in this construction, proofs are larger and
25/// verification is slower than in Wesolowski's construction, but the output of a VDF is unique,
26/// assuming that the used group have no small subgroups, and proving is faster for the same number
27/// of iterations.
28pub struct PietrzaksVDF<G: ParameterizedGroupElement> {
29    group_parameter: G::ParameterType,
30    iterations: u64,
31}
32
33impl<G: ParameterizedGroupElement> PietrzaksVDF<G> {
34    /// Create a new VDF using the group defined by the given group parameter. Evaluating this VDF
35    /// will require computing `2^iterations * input` which requires `iterations` group operations.
36    pub fn new(group_parameter: G::ParameterType, iterations: u64) -> Self {
37        Self {
38            group_parameter,
39            iterations,
40        }
41    }
42
43    /// Compute the Fiat-Shamir challenge used in Pietrzak's VDF construction.
44    fn compute_challenge(&self, input: &G, output: &G, mu: &G) -> BigUint {
45        let seed = bcs::to_bytes(&(input, output, mu, self.iterations, &self.group_parameter))
46            .expect("Failed to serialize Fiat-Shamir input.");
47        let hash = Keccak256::digest(seed);
48        BigUint::from_bytes_be(&hash.digest[..DEFAULT_CHALLENGE_SIZE_IN_BYTES])
49    }
50}
51
52impl<G: ParameterizedGroupElement + Serialize> VDF for PietrzaksVDF<G>
53where
54    G::ParameterType: Serialize,
55{
56    type InputType = G;
57    type OutputType = G;
58    type ProofType = Vec<G>;
59
60    fn evaluate(&self, input: &G) -> FastCryptoResult<(G, Vec<G>)> {
61        // Proof generation works but is not optimised.
62
63        if !input.is_in_group(&self.group_parameter) || self.iterations == 0 {
64            return Err(InvalidInput);
65        }
66
67        // Compute output = 2^iterations * input
68        let output = input.clone().repeated_doubling(self.iterations);
69
70        let mut x = input.clone();
71        let mut y = output.clone();
72        let mut t = self.iterations;
73
74        let mut proof = Vec::new();
75
76        // Compute the full proof. This loop may stop at any time which will give a shorter proof
77        // that is computationally harder to verify.
78        while t != 1 {
79            if check_parity_and_iterate(&mut t) {
80                y = y.double();
81            }
82
83            // TODO: Precompute some of the mu's to speed up the proof generation.
84            let mu = x.clone().repeated_doubling(t);
85
86            let r = self.compute_challenge(&x, &y, &mu);
87            x = x.multiply(&r, &self.group_parameter) + &mu;
88            y = mu.multiply(&r, &self.group_parameter) + &y;
89
90            proof.push(mu);
91        }
92
93        Ok((output, proof))
94    }
95
96    fn verify(&self, input: &G, output: &G, proof: &Vec<G>) -> FastCryptoResult<()> {
97        if !input.is_in_group(&self.group_parameter)
98            || !output.is_in_group(&self.group_parameter)
99            || !proof.iter().all(|mu| mu.is_in_group(&self.group_parameter))
100            || self.iterations == 0
101        {
102            return Err(InvalidInput);
103        }
104
105        let mut x = input.clone();
106        let mut y = output.clone();
107        let mut t = self.iterations;
108
109        for mu in proof {
110            if check_parity_and_iterate(&mut t) {
111                y = y.double();
112            }
113
114            let r = self.compute_challenge(&x, &y, mu);
115            x = x.multiply(&r, &self.group_parameter) + mu;
116            y = mu.multiply(&r, &self.group_parameter) + &y;
117        }
118
119        // In case the proof is shorter than the full proof, we need to compute the remaining powers.
120        x = x.repeated_doubling(t);
121        if x != y {
122            return Err(InvalidProof);
123        }
124        Ok(())
125    }
126}

128/// Replace t with (t+1) >> 1 and return true iff the input was odd.
129#[inline]
130fn check_parity_and_iterate(t: &mut u64) -> bool {
131    mem::replace(t, (*t >> 1) + (*t & 1)).is_odd()
132}

167        .unwrap();
