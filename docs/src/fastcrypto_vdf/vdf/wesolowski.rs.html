19/// Default size in bytes of the Fiat-Shamir challenge used in proving and verification.
20///
21/// From Wesolowski (2018), "Efficient verifiable delay functions" (https://eprint.iacr.org/2018/623),
22/// we get that the challenge must be a random prime among the first 2^{2k} primes where k is the
23/// security parameter in bits. Setting k = 128, and recalling that the prime number theorem states
24/// that the n-th prime number is approximately n * ln(n), we can estimate the number of bits required
25/// to represent the n-th prime as log2(n * ln(n)). For n = 2^{2*128}, this is approximately 264 bits
26/// = 33 bytes. This is also the challenge size used by chiavdf.
27pub const DEFAULT_CHALLENGE_SIZE_IN_BYTES: usize = 33;
28
29/// An implementation of Wesolowski's VDF construction (https://eprint.iacr.org/2018/623) over a
30/// group of unknown order.
31pub struct WesolowskisVDF<G: ParameterizedGroupElement, M: ScalarMultiplier<G, BigUint>> {
32    group_parameter: G::ParameterType,
33    iterations: u64,
34    challenge: fn(&Self, &G, &G) -> BigUint,
35    _scalar_multiplier: PhantomData<M>,
36}
37
38impl<G: ParameterizedGroupElement, M: ScalarMultiplier<G, BigUint>> WesolowskisVDF<G, M> {
39    /// Create a new VDF using the group defined by the given group parameter. Evaluating this VDF
40    /// will require computing `2^iterations * input` which requires `iterations` group operations.
41    pub fn new(group_parameter: G::ParameterType, iterations: u64) -> Self {
42        Self {
43            group_parameter,
44            iterations,
45            challenge: compute_challenge,
46            _scalar_multiplier: PhantomData::<M>,
47        }
48    }
49
50    /// Create a new VDF with a custom challenge function. This is useful for testing.
51    #[cfg(test)]
52    fn new_with_custom_challenge(
53        group_parameter: G::ParameterType,
54        iterations: u64,
55        challenge: fn(&Self, &G, &G) -> BigUint,
56    ) -> Self {
57        Self {
58            group_parameter,
59            iterations,
60            challenge,
61            _scalar_multiplier: PhantomData::<M>,
62        }
63    }
64}
65
66impl<G: ParameterizedGroupElement, M: ScalarMultiplier<G, BigUint>> VDF for WesolowskisVDF<G, M> {
67    type InputType = G;
68    type OutputType = G;
69    type ProofType = G;
70
71    fn evaluate(&self, input: &G) -> FastCryptoResult<(G, G)> {
72        if !input.is_in_group(&self.group_parameter) || self.iterations == 0 {
73            return Err(InvalidInput);
74        }
75
76        // Compute output = 2^iterations * input
77        let output = input.clone().repeated_doubling(self.iterations);
78
79        let multiplier = M::new(input.clone(), G::zero(&self.group_parameter));
80
81        // Algorithm from page 3 on https://crypto.stanford.edu/~dabo/pubs/papers/VDFsurvey.pdf
82        let challenge = (self.challenge)(self, input, &output);
83        let mut quotient_remainder = (BigUint::from(0u8), BigUint::from(2u8));
84        let mut proof = G::zero(&self.group_parameter);
85        for _ in 1..self.iterations {
86            quotient_remainder.1.shl_assign(1);
87            quotient_remainder = quotient_remainder.1.div_mod_floor(&challenge);
88            proof = proof.double() + &multiplier.mul(&quotient_remainder.0);
89        }
90
91        Ok((output, proof))
92    }
93
94    fn verify(&self, input: &G, output: &G, proof: &G) -> FastCryptoResult<()> {
95        if !input.is_in_group(&self.group_parameter)
96            || !output.is_in_group(&self.group_parameter)
97            || !proof.is_in_group(&self.group_parameter)
98        {
99            return Err(InvalidInput);
100        }
101
102        let challenge = (self.challenge)(self, input, output);
103        let r = BigUint::modpow(
104            &BigUint::from(2u8),
105            &BigUint::from(self.iterations),
106            &challenge,
107        );
108        let multiplier = M::new(input.clone(), G::zero(&self.group_parameter));
109
110        if multiplier.two_scalar_mul(&r, proof, &challenge) != *output {
111            return Err(InvalidProof);
112        }
113        Ok(())
114    }
115}

117/// Implementation of Wesolowski's VDF construction over an imaginary class group using a strong
118/// Fiat-Shamir implementation.
119pub type DefaultVDF =
120    WesolowskisVDF<QuadraticForm, WindowedScalarMultiplier<QuadraticForm, BigUint, 256, 5>>;
121
122fn compute_challenge<G: ParameterizedGroupElement, M: ScalarMultiplier<G, BigUint>>(
123    vdf: &WesolowskisVDF<G, M>,
124    input: &G,
125    output: &G,
126) -> BigUint {
127    let seed = bcs::to_bytes(&(input, output, vdf.iterations, &vdf.group_parameter))
128        .expect("Failed to serialize Fiat-Shamir input");
129    hash_prime(
130        &seed,
131        DEFAULT_CHALLENGE_SIZE_IN_BYTES,
132        &[0, 8 * DEFAULT_CHALLENGE_SIZE_IN_BYTES - 1],
133    )
134}

149    fn test_prove_and_verify() {
150        let challenge = hex::decode("99c9e5e3a4449a4b4e15").unwrap();
151        let iterations = 1000u64;
152        let discriminant = Discriminant::from_seed(&challenge, 512).unwrap();
153
154        let input = QuadraticForm::generator(&discriminant);

159        // Regression tests

161        assert_eq!(bcs::to_bytes(&proof).unwrap(), hex::decode("2051242499ff4467ba4d9bd858d6609b3a29736a80e09aa3edc6fa4ca49292daae20c066a1b52a8c5a0ff230224fc403dec11ee156e523f22b78ab8abc9706ae38c52100b4c56b5d7815425677bbe0ddc924d6277a237e4c5317b10a1a36039148d1d19c").unwrap());
162
163        assert!(vdf.verify(&input, &output, &proof).is_ok());
164
165        // A modified output or proof fails to verify
166        let modified_output = &output + &QuadraticForm::generator(&discriminant);
167        let modified_proof = &proof + &QuadraticForm::generator(&discriminant);
168        assert!(vdf.verify(&input, &modified_output, &proof).is_err());
169        assert!(vdf.verify(&input, &output, &modified_proof).is_err());
170    }
