4//! This module contains an implementation of the extended Euclidean algorithm for [BigInt]`s.
5//! Besides the gcd and the Bezout coefficients, it also returns the quotients of the two inputs
6//! divided by the GCD since these are often used, for example in the NUCOMP and NUDPL algorithms,
7//! and come out for free while computing the Bezout coefficients.
8
9use num_bigint::{BigInt, Sign};
10use num_integer::Integer;
11use num_traits::{One, Signed, Zero};
12use std::mem;
13
14/// The output of the extended Euclidean algorithm on inputs `a` and `b`: The Bezout coefficients `x`
15/// and `y` such that `ax + by = gcd`. The quotients `a / gcd` and `b / gcd` are also returned.
16/// Note that `x` is optional and only computed in [extended_euclidean_algorithm] if `compute_x` is true.
17pub struct EuclideanAlgorithmOutput {
18    pub gcd: BigInt,
19    pub x: Option<BigInt>,
20    pub y: BigInt,
21    pub a_divided_by_gcd: BigInt,
22    pub b_divided_by_gcd: BigInt,
23}
24
25/// Compute the greatest common divisor gcd of a and b. The output also returns the Bezout coefficients
26/// x and y such that ax + by = gcd and also the quotients a / gcd and b / gcd.
27/// Note that `x` is only computed if `compute_x` is true.
28pub(crate) fn extended_euclidean_algorithm(
29    a: &BigInt,
30    b: &BigInt,
31    compute_x: bool,
32) -> EuclideanAlgorithmOutput {
33    let mut s = (BigInt::zero(), BigInt::one());
34    let mut t = (BigInt::one(), BigInt::zero());
35    let mut r = (a.clone(), b.clone());
36
37    while !r.0.is_zero() {
38        let (q, r_prime) = r.1.div_rem(&r.0);
39        r.1 = mem::replace(&mut r.0, r_prime);
40
41        mem::swap(&mut s.0, &mut s.1);
42        s.0 -= &q * &s.1;
43
44        if compute_x {
45            mem::swap(&mut t.0, &mut t.1);
46            t.0 -= &q * &t.1;
47        }
48    }
49
50    // The last coefficients are equal to +/- a / gcd(a,b) and b / gcd(a,b) respectively.
51    let a_divided_by_gcd = with_sign(s.0, a.sign());
52
53    let negate = r.1.is_negative();
54    let gcd = conditional_negate(negate, r.1);
55    let y = conditional_negate(negate, s.1);
56
57    let (x, b_divided_by_gcd) = if compute_x {
58        (
59            Some(conditional_negate(negate, t.1)),
60            with_sign(t.0, b.sign()),
61        )
62    } else {
63        // If the t coefficients have not been computed, we can compute b_divided_by_gcd directly.
64        (None, b / &gcd)
65    };
66
67    EuclideanAlgorithmOutput {
68        gcd,
69        x,
70        y,
71        a_divided_by_gcd,
72        b_divided_by_gcd,
73    }
74}
75
76/// Return a number with the same magnitude as `value` but with the given sign.
77#[inline]
78fn with_sign(value: BigInt, sign: Sign) -> BigInt {
79    BigInt::from_biguint(sign, value.into_parts().1)
80}
81
82/// Return `-value` if `negate` is true, otherwise return `value`.
83#[inline]
84fn conditional_negate(negate: bool, value: BigInt) -> BigInt {
85    if negate {
86        -value
87    } else {
88        value
89    }
90}

92#[test]
93fn test_xgcd() {
94    test_xgcd_single(BigInt::from(240), BigInt::from(46));
95    test_xgcd_single(BigInt::from(-240), BigInt::from(46));
96    test_xgcd_single(BigInt::from(240), BigInt::from(-46));
97    test_xgcd_single(BigInt::from(-240), BigInt::from(-46));
98}
99
100#[cfg(test)]
101fn test_xgcd_single(a: BigInt, b: BigInt) {
102    let output = extended_euclidean_algorithm(&a, &b, true);
103    assert_eq!(output.gcd, a.gcd(&b));
104    assert_eq!(&output.x.unwrap() * &a + &output.y * &b, output.gcd);
105    assert_eq!(output.a_divided_by_gcd, &a / &output.gcd);
106    assert_eq!(output.b_divided_by_gcd, &b / &output.gcd);
107
108    let output = extended_euclidean_algorithm(&a, &b, false);
109    assert_eq!(output.gcd, a.gcd(&b));
110    assert!(output.x.is_none());
111    assert_eq!(output.a_divided_by_gcd, &a / &output.gcd);
112    assert_eq!(output.b_divided_by_gcd, &b / &output.gcd);
113}
