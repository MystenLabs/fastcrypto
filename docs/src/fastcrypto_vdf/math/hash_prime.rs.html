4//! This module contains an implementation of a hash-to-prime function identical to the HashPrime
5//! function from [chiavdf](https://github.com/Chia-Network/chiavdf/blob/bcc36af3a8de4d2fcafa571602040a4ebd4bdd56/src/proof_common.h#L14-L43).
6use fastcrypto::hash::{HashFunction, Sha256};
7use num_bigint::BigUint;
8use num_prime::nt_funcs::is_prime;
9use num_prime::PrimalityTestConfig;
10use std::cmp::min;
11
12struct HashPrimeIterator {
13    seed: Vec<u8>,
14    length_in_bytes: usize,
15    bitmask: Vec<usize>,
16}
17
18impl Iterator for HashPrimeIterator {
19    type Item = BigUint;
20
21    fn next(&mut self) -> Option<Self::Item> {
22        let mut blob = vec![];
23        while blob.len() < self.length_in_bytes {
24            for i in (0..self.seed.len()).rev() {
25                self.seed[i] = self.seed[i].wrapping_add(1);
26                if self.seed[i] != 0 {
27                    break;
28                }
29            }
30            let hash = Sha256::digest(&self.seed).digest;
31            blob.extend_from_slice(&hash[..min(hash.len(), self.length_in_bytes - blob.len())]);
32        }
33        let mut x = BigUint::from_bytes_be(&blob);
34        for b in &self.bitmask {
35            x.set_bit(*b as u64, true);
36        }
37        Some(x)
38    }
39}
40
41/// Implementation of HashPrime from chiavdf ():
42/// Generates a random pseudo-prime using the hash and check method:
43/// Randomly chooses x with bit-length `length`, then applies a mask
44///   (for b in bitmask) { x |= (1 << b) }.
45/// Then return x if it is a pseudo-prime, otherwise repeat.
46pub(crate) fn hash_prime(seed: &[u8], length_in_bytes: usize, bitmask: &[usize]) -> BigUint {
47    let mut iterator = HashPrimeIterator {
48        seed: seed.to_vec(),
49        length_in_bytes,
50        bitmask: bitmask.to_vec(),
51    };
52    iterator.find(is_probable_prime).unwrap()
53}
54
55/// Check if the input is a probable prime.
56///
57/// We use the Baillie-PSW primality test here. This is in accordance with the recommendations of "Prime and
58/// Prejudice: Primality Testing Under Adversarial Conditions" by Albrecht et al. (https://eprint.iacr.org/2018/749)
59/// because this test is also used in use cases where an adversary could influence the input.
60pub fn is_probable_prime(x: &BigUint) -> bool {
61    is_prime(x, Some(PrimalityTestConfig::bpsw())).probably()
62}

64#[cfg(test)]
65mod tests {
66    use super::*;
67    use num_bigint::BigUint;
68    use num_integer::Integer;
69    use num_prime::PrimalityTestConfig;
70    use std::str::FromStr;
71
72    #[test]
73    fn test_hash_prime() {
74        let seed = [0u8; 32];
75        let length = 64;
76        let bitmask: [usize; 3] = [0, 1, 8 * length - 1];
77
78        let prime = hash_prime(&seed, length, &bitmask);
79
80        // Prime has right length
81        assert_eq!((length * 8) as u64, prime.bits());
82
83        // The last two bits are set (see bitmask)
84        assert_eq!(BigUint::from(3u64), prime.mod_floor(&BigUint::from(4u64)));
85
86        // The result is a prime, even when checking with a stricter test
87        assert!(
88            num_prime::nt_funcs::is_prime(&prime, Some(PrimalityTestConfig::strict())).probably()
89        );
90
91        // Regression test
92        assert_eq!(prime, BigUint::from_str("7904272817142338150419757415334055106926417574777773392214522399425467199262039794276651240832053626391864792937889238336287002167559810128294881253078163").unwrap());
93    }
