12/// Compute a modular square root of a modulo p with p prime if this exists. This function does not
13/// check that p is prime and if it is not, the result is undefined. If check_legendre is set to
14/// true, the function verifies that a is a quadratic residue modulo p and returns None otherwise.
15/// If check_legendre is set to false, the function assumes that a is a quadratic residue modulo p
16/// and if this is not the case, the result is undefined.
17pub(crate) fn modular_square_root(
18    a: &BigInt,
19    p: &BigInt,
20    check_legendre: bool,
21) -> FastCryptoResult<BigInt> {
22    // Algorithm 2.3.8 in Crandall & Pomerance, "Prime Numbers: A Computational Perspective"
23
24    // Handle special cases
25    if !p.is_positive() || !p.is_odd() || p.is_one() {
26        return Err(InvalidInput);
27    }
28
29    if a.is_zero() {
30        return Ok(BigInt::zero());
31    }
32
33    // Check that a is a quadratic residue modulo p
34    if check_legendre && jacobi(a, p).unwrap() != 1 {
35        return Err(InvalidInput);
36    }
37
38    let two = BigInt::from(2);
39
40    let a = a.mod_floor(p);
41    match mod8(p) {
42        3 | 7 => Ok(a.modpow(&((p + 1) >> 2), p)),
43        5 => {
44            let mut x = a.modpow(&((p + 3) >> 3), p);
45            let c = x.modpow(&two, p);
46            if c != a {
47                x *= two.modpow(&((p - 1) >> 2), p) % p;
48            }
49            Ok(x)
50        }
51        1 => {
52            let mut d: BigInt = two.clone();
53            while jacobi::jacobi(&d, p).expect("p is positive and odd") != -1 {
54                d += 1;
55                if &d >= p {
56                    return Err(InvalidInput);
57                }
58            }
59            let p_minus_1: BigInt = p - 1;
60            let s = p_minus_1.trailing_zeros().expect("p is verified to be > 1");
61            let t = &p_minus_1 >> s;
62
63            let a_t = a.modpow(&t, p);
64            let d_t = d.modpow(&t, p);
65            let mut m = 0.into();
66
67            for i in 0..s {
68                let lhs = (&a_t * d_t.modpow(&m, p)).modpow(&(1 << (s - 1 - i)).into(), p);
69                if lhs == p_minus_1 {
70                    m += 1 << i;
71                }
72            }
73            let x = a.modpow(&((t + 1) >> 1), p) * d_t.modpow(&(m >> 1), p) % p;
74            Ok(x)
75        }
76        _ => Err(InvalidInput),
77    }
78}

80/// Compute a mod 8.
81fn mod8(a: &BigInt) -> u8 {
82    (a & &7.into()).to_u8().expect("Is smaller than 8")
83}

85#[cfg(test)]
86mod tests {
87    use crate::math::modular_sqrt::jacobi;
88    use num_bigint::BigInt;
89
90    #[test]
91    fn test_sqrt() {
92        // 1, 3, 5, 7 mod 8
93        let moduli: [usize; 4] = [257, 163, 197, 127];
94        for p in moduli.iter() {
95            for a in 2..*p {
96                let p = BigInt::from(*p);
97                let a = BigInt::from(a);
98                match super::modular_square_root(&a, &p, true) {
99                    Ok(x) => {
100                        assert_eq!(jacobi(&a, &p).unwrap(), 1);
101                        assert_eq!(x.modpow(&BigInt::from(2), &p), a);
102                    }
103                    Err(_) => assert_eq!(jacobi(&a, &p).unwrap(), -1),
104                }
105            }
106        }
107    }
