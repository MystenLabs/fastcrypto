4use fastcrypto::groups::Doubling;
5use num_bigint::BigUint;
6use serde::Serialize;
7use std::ops::Add;
8
9/// Trait implemented by elements of an additive group where the group is parameterized, for example
10/// by the modulus in case of the group being Z mod N or the discriminant in case of class groups.
11pub trait ParameterizedGroupElement:
12    Sized + Clone + for<'a> Add<&'a Self, Output = Self> + Eq + Doubling + Serialize
13{
14    /// The type of the parameter which uniquely defines this group.
15    type ParameterType: Serialize;
16
17    /// Return an instance of the identity element in this group.
18    fn zero(parameter: &Self::ParameterType) -> Self;
19
20    /// Returns true if this is an element of the group defined by `parameter`.
21    fn is_in_group(&self, parameter: &Self::ParameterType) -> bool;
22
23    /// Compute self * scalar.
24    fn multiply(&self, scalar: &BigUint, parameter: &Self::ParameterType) -> Self {
25        // Generic double-and-add algorithm.
26        (0..scalar.bits())
27            .rev()
28            .map(|i| scalar.bit(i))
29            .fold(Self::zero(parameter), |acc, bit| match bit {
30                true => acc.double() + self,
31                false => acc.double(),
32            })
33    }
34}

36#[cfg(test)]
37mod tests {
38    use crate::class_group::discriminant::Discriminant;
39    use crate::class_group::QuadraticForm;
40    use crate::math::parameterized_group::ParameterizedGroupElement;
41    use num_bigint::BigUint;
42    use num_traits::{One, Zero};
43
44    #[test]
45    fn test_scalar_multiplication() {
46        let discriminant = Discriminant::from_seed(b"test", 256).unwrap();
47        let input = QuadraticForm::generator(&discriminant);
48
49        // Edge cases
50        assert_eq!(
51            QuadraticForm::zero(&discriminant),
52            input.multiply(&BigUint::zero(), &discriminant)
53        );
54        assert_eq!(input, input.multiply(&BigUint::one(), &discriminant));
55
56        let exponent = 12345u64;
57        let output = input.multiply(&BigUint::from(exponent), &discriminant);
58
59        // Check alignment with repeated addition.
60        let mut expected_output = input.clone();
61        for _ in 1..exponent {
62            expected_output = expected_output + &input;
63        }
64        assert_eq!(output, expected_output);
65    }
66}
