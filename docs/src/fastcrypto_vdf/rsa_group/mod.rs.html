16/// This represents an element of the subgroup of an RSA group <i>Z<sub>N</sub><sup>*</sup> / <±1></i>
17/// where <i>N</i> is the product of two large primes. See also [RSAModulus].
18#[derive(Clone, Debug, Eq, PartialEq, Serialize)]
19pub struct RSAGroupElement<'a> {
20    value: BigUint,
21
22    #[serde(skip)]
23    modulus: &'a RSAModulus,
24}
25
26impl<'a> RSAGroupElement<'a> {
27    /// Create a new RSA group element with the given value and modulus. The value will be reduced to
28    /// the subgroup <i>Z<sub>N</sub><sup>*</sup> / <±1></i>, so it does not need to be in canonical
29    /// representation.
30    pub fn new(value: BigUint, modulus: &'a RSAModulus) -> Self {
31        Self {
32            value: modulus.reduce(value),
33            modulus,
34        }
35    }
36
37    /// Return the canonical representation of this group element.
38    pub fn value(&self) -> &BigUint {
39        &self.value
40    }
41
42    /// Generate a random element of the subgroup <i>Z<sub>N</sub><sup>*</sup> / <±1></i>
43    /// using the given seed.
44    pub fn from_seed(seed: [u8; 32], modulus: &'a RSAModulus) -> Self {
45        let mut rng = ChaCha20Rng::from_seed(seed);
46        Self::new(rng.gen_biguint_below(&modulus.half), modulus)
47    }
48}
49
50impl Add<&Self> for RSAGroupElement<'_> {
51    type Output = Self;
52
53    fn add(self, rhs: &Self) -> Self::Output {
54        assert_eq!(self.modulus, rhs.modulus);
55        Self::new(self.value.mul(&rhs.value), self.modulus)
56    }
57}
58
59impl Doubling for RSAGroupElement<'_> {
60    fn double(self) -> Self {
61        Self::new(self.value.pow(2), self.modulus)
62    }
63}
64
65impl<'a> ParameterizedGroupElement for RSAGroupElement<'a> {
66    type ParameterType = &'a RSAModulus;
67
68    fn zero(modulus: &Self::ParameterType) -> Self {
69        Self::new(BigUint::one(), modulus)
70    }
71
72    fn is_in_group(&self, modulus: &Self::ParameterType) -> bool {
73        &self.modulus == modulus
74    }
75
76    fn multiply(&self, scalar: &BigUint, modulus: &Self::ParameterType) -> Self {
77        let value = self
78            .modulus
79            .ensure_in_subgroup(self.value.modpow(scalar, &modulus.value));
80        Self { value, modulus }
81    }
82}
