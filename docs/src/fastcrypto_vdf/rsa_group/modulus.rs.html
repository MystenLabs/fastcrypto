4use fastcrypto::error::FastCryptoError;
5use num_bigint::BigUint;
6use num_integer::Integer;
7use serde::Serialize;
8use std::ops::Shr;
9use std::str::FromStr;
10
11#[derive(PartialEq, Eq, Debug, Serialize)]
12pub struct RSAModulus {
13    pub(super) value: BigUint,
14
15    /// Precomputed value of `modulus / 2` for faster reduction.
16    #[serde(skip)]
17    pub(super) half: BigUint,
18}
19
20impl FromStr for RSAModulus {
21    type Err = FastCryptoError;
22
23    /// Parse an RSA modulus from a decimal string. The modulus is not validated, so it is the caller's
24    /// responsibility to ensure that it is a valid RSA modulus.
25    fn from_str(s: &str) -> Result<Self, Self::Err> {
26        BigUint::from_str(s)
27            .map(Self::from)
28            .map_err(|_| FastCryptoError::InvalidInput)
29    }
30}
31
32impl From<BigUint> for RSAModulus {
33    /// Create an RSA modulus from a [BigUint]. The modulus is not validated, so it is the caller's
34    /// responsibility to ensure that it is a valid RSA modulus.
35    fn from(value: BigUint) -> Self {
36        let half = (&value).shr(1);
37        Self { value, half }
38    }
39}
40
41impl RSAModulus {
42    /// Reduce the given value modulo this modulus. Further, if the value is greater than half of the
43    /// modulus, the result is negated. This is to ensure that the result is in the subgroup
44    /// <i>Z<sub>N</sub><sup>*</sup> / <±1></i>.
45    pub(super) fn reduce(&self, value: BigUint) -> BigUint {
46        self.ensure_in_subgroup(value.mod_floor(&self.value))
47    }
48
49    /// Assuming that `value < N`, this ensures that the given value is in the subgroup <i>Z<sub>N</sub><sup>*</sup> / <±1></i>.
50    /// Panics if `value` is greater than or equal to `N`.
51    pub(super) fn ensure_in_subgroup(&self, value: BigUint) -> BigUint {
52        if value < self.half {
53            value
54        } else {
55            &self.value - value
56        }
57    }
58}
