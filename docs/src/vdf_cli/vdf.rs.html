4use clap::Parser;
5use fastcrypto_vdf::class_group::discriminant::{Discriminant, DISCRIMINANT_3072};
6use fastcrypto_vdf::class_group::QuadraticForm;
7use fastcrypto_vdf::math::parameterized_group::ParameterizedGroupElement;
8use fastcrypto_vdf::vdf::wesolowski::DefaultVDF;
9use fastcrypto_vdf::vdf::VDF;
10use std::io::{Error, ErrorKind};

12/// This discriminant size is based on a lower bound from "Trustless unkown-order groups" by Dobson et al.
13/// (https://inria.hal.science/hal-02882161/file/unknown-order.pdf)
14///
15/// A discriminant size of 3072 bits ensures that the computational hardness of computing the group order of a group
16/// with a randomly chosen discriminant is at least 128 bits with probability at least 1 - 2^{-40}.
17const DEFAULT_DISCRIMINANT_BIT_LENGTH: u64 = 3072;
18
19#[derive(Parser)]
20#[command(name = "vdf-cli")]
21#[command(about = "Verifiable delay function using Wesolowski's construction over imaginary class groups", long_about = None)]
22enum Command {
23    /// Sample a random discriminant from a seed.
24    Discriminant(DiscriminantArguments),
25
26    /// Compute VDF output and proof.
27    Evaluate(EvaluateArguments),
28
29    /// Verify an output .
30    Verify(VerifyArguments),
31
32    /// Hash a binary message to a quadratic form.
33    Hash(HashArguments),
34}
35
36#[derive(Parser, Clone)]
37struct DiscriminantArguments {
38    /// The hex encoded discriminant.
39    #[clap(short, long)]
40    seed: String,
41
42    /// Bit length of the discriminant (default is 3072).
43    #[clap(short, long, default_value_t = DEFAULT_DISCRIMINANT_BIT_LENGTH)]
44    bit_length: u64,
45}
46
47#[derive(Parser, Clone)]
48struct EvaluateArguments {
49    /// The hex encoded input to the VDF.
50    #[clap(long)]
51    input: String,
52
53    /// The number of iterations.
54    #[clap(long)]
55    iterations: u64,
56}
57
58#[derive(Parser, Clone)]
59struct VerifyArguments {
60    /// Iterations
61    #[clap(long)]
62    iterations: u64,
63
64    /// The input to the VDF.
65    #[clap(long)]
66    input: String,
67
68    /// The output of the VDF.
69    #[clap(short, long)]
70    output: String,
71
72    /// The proof of the correctness of the VDF output.
73    #[clap(short, long)]
74    proof: String,
75}
76
77#[derive(Parser, Clone)]
78struct HashArguments {
79    /// The hex encoded input to the hash function.
80    #[clap(short, long)]
81    message: String,
82}

84fn main() {
85    match execute(Command::parse()) {
86        Ok(res) => {
87            println!("{}", res);
88            std::process::exit(exitcode::OK);
89        }
90        Err(e) => {
91            println!("Error: {}", e);
92            std::process::exit(exitcode::DATAERR);
93        }
94    }
95}

97fn execute(cmd: Command) -> Result<String, Error> {
98    match cmd {
99        Command::Discriminant(arguments) => {
100            let seed = hex::decode(arguments.seed)
101                .map_err(|_| Error::new(ErrorKind::InvalidInput, "Invalid seed."))?;
102            let discriminant =
103                Discriminant::from_seed(&seed, arguments.bit_length as usize).unwrap();
104            let discriminant_string = hex::encode(bcs::to_bytes(&discriminant).unwrap());
105            let mut result = "Discriminant: ".to_string();
106            result.push_str(&discriminant_string);
107            Ok(result)
108        }
109
110        Command::Evaluate(arguments) => {
111            let input_bytes = hex::decode(arguments.input)
112                .map_err(|_| Error::new(ErrorKind::InvalidInput, "Invalid input point."))?;
113            let g = bcs::from_bytes::<QuadraticForm>(&input_bytes).map_err(|_| {
114                Error::new(
115                    ErrorKind::InvalidInput,
116                    "Invalid input point or discriminant.",
117                )
118            })?;
119            if !g.is_in_group(&DISCRIMINANT_3072) {
120                return Err(Error::new(
121                    ErrorKind::InvalidInput,
122                    "Input point does not match discriminant.",
123                ));
124            }
125
126            let vdf = DefaultVDF::new(DISCRIMINANT_3072.clone(), arguments.iterations);
127            let (output, proof) = vdf
128                .evaluate(&g)
129                .map_err(|_| Error::new(ErrorKind::Other, "VDF evaluation failed"))?;
130
131            let output_string = hex::encode(bcs::to_bytes(&output).unwrap());
132            let proof_string = hex::encode(bcs::to_bytes(&proof).unwrap());
133
134            let mut result = "Output: ".to_string();
135            result.push_str(&output_string);
136            result.push_str("\nProof:  ");
137            result.push_str(&proof_string);
138            Ok(result)
139        }
140
141        Command::Verify(arguments) => {
142            let input =
143                bcs::from_bytes::<QuadraticForm>(&hex::decode(arguments.input).map_err(|_| {
144                    Error::new(ErrorKind::InvalidInput, "Invalid output hex string.")
145                })?)
146                .map_err(|_| Error::new(ErrorKind::InvalidInput, "Invalid input."))?;
147            if !input.is_in_group(&DISCRIMINANT_3072) {
148                return Err(Error::new(
149                    ErrorKind::InvalidInput,
150                    "Input has wrong discriminant.",
151                ));
152            }
153
154            let output =
155                bcs::from_bytes::<QuadraticForm>(&hex::decode(arguments.output).map_err(|_| {
156                    Error::new(ErrorKind::InvalidInput, "Invalid output hex string.")
157                })?)
158                .map_err(|_| Error::new(ErrorKind::InvalidInput, "Invalid output."))?;
159            if !output.is_in_group(&DISCRIMINANT_3072) {
160                return Err(Error::new(
161                    ErrorKind::InvalidInput,
162                    "Output has wrong discriminant.",
163                ));
164            }
165
166            let proof =
167                bcs::from_bytes::<QuadraticForm>(&hex::decode(arguments.proof).map_err(|_| {
168                    Error::new(ErrorKind::InvalidInput, "Invalid proof hex string.")
169                })?)
170                .map_err(|_| Error::new(ErrorKind::InvalidInput, "Invalid proof."))?;
171            if !proof.is_in_group(&DISCRIMINANT_3072) {
172                return Err(Error::new(
173                    ErrorKind::InvalidInput,
174                    "Proof has wrong discriminant.",
175                ));
176            }
177
178            let vdf = DefaultVDF::new(DISCRIMINANT_3072.clone(), arguments.iterations);
179            let verifies = vdf.verify(&input, &output, &proof).is_ok();
180
181            let mut result = "Verified: ".to_string();
182            result.push_str(&verifies.to_string());
183            Ok(result)
184        }
185        Command::Hash(arguments) => {
186            let input = hex::decode(arguments.message)
187                .map_err(|_| Error::new(ErrorKind::InvalidInput, "Invalid message."))?;
188            let output =
189                QuadraticForm::hash_to_group_with_default_parameters(&input, &DISCRIMINANT_3072)
190                    .map_err(|_| {
191                        Error::new(ErrorKind::InvalidInput, "The discriminant is invalid.")
192                    })?;
193
194            let output_bytes = hex::encode(bcs::to_bytes(&output).unwrap());
195
196            let mut result = "Output: ".to_string();
197            result.push_str(&output_bytes);
198            Ok(result)
199        }
200    }
201}
