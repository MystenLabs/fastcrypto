4use clap::Parser;
5use fastcrypto::vrf::ecvrf::{ECVRFKeyPair, ECVRFPrivateKey, ECVRFProof, ECVRFPublicKey};
6use fastcrypto::vrf::{VRFKeyPair, VRFProof};
7use rand::thread_rng;
8use std::io::{Error, ErrorKind};
9
10#[derive(Parser)]
11#[command(name = "ecvrf-cli")]
12#[command(about = "Elliptic Curve Verifiable Random Function (ECVRF) over Ristretto255 according to draft-irtf-cfrg-vrf-15.", long_about = None)]
13enum Command {
14    /// Generate a key pair for proving and verification.
15    Keygen,
16
17    /// Create an output/hash and a proof.
18    Prove(ProveArguments),
19
20    /// Verify an output/hash and a proof.
21    Verify(VerifyArguments),
22}
23
24#[derive(Parser, Clone)]
25struct ProveArguments {
26    /// The hex encoded input string.
27    #[clap(short, long)]
28    input: String,
29
30    /// A hex encoding of the secret key. Corresponds to a scalar in Ristretto255 and must be 32 bytes.
31    #[clap(short, long)]
32    secret_key: String,
33}
34
35#[derive(Parser, Clone)]
36struct VerifyArguments {
37    /// Hex-encoded Sha512 hash of the proof. Must be 64 bytes.
38    #[clap(short, long)]
39    output: String,
40
41    /// Encoding of the proof to verify. Must be 80 bytes.
42    #[clap(short, long)]
43    proof: String,
44
45    /// Hex encoding of the input string used to generate the proof.
46    #[clap(short, long)]
47    input: String,
48
49    /// The public key corresponding to the secret key used to generate the proof.
50    #[clap(short = 'k', long)]
51    public_key: String,
52}
53
54fn main() {
55    match execute(Command::parse()) {
56        Ok(res) => {
57            println!("{}", res);
58            std::process::exit(exitcode::OK);
59        }
60        Err(e) => {
61            println!("Error: {}", e);
62            std::process::exit(exitcode::DATAERR);
63        }
64    }
65}
66
67fn execute(cmd: Command) -> Result<String, std::io::Error> {
68    match cmd {
69        Command::Keygen => {
70            let keypair = ECVRFKeyPair::generate(&mut thread_rng());
71            let sk_string =
72                hex::encode(bcs::to_bytes(&keypair.sk).map_err(|_| {
73                    Error::new(ErrorKind::Other, "Failed to serialize secret key.")
74                })?);
75            let pk_string =
76                hex::encode(bcs::to_bytes(&keypair.pk).map_err(|_| {
77                    Error::new(ErrorKind::Other, "Failed to serialize public key.")
78                })?);
79
80            let mut result = "Secret key: ".to_string();
81            result.push_str(&sk_string);
82            result.push_str("\nPublic key: ");
83            result.push_str(&pk_string);
84            Ok(result)
85        }
86
87        Command::Prove(arguments) => {
88            // Parse inputs
89            let secret_key_bytes = hex::decode(arguments.secret_key)
90                .map_err(|_| Error::new(ErrorKind::InvalidInput, "Invalid private key."))?;
91            let alpha_string = hex::decode(arguments.input)
92                .map_err(|_| Error::new(ErrorKind::InvalidInput, "Invalid input string."))?;
93
94            // Create keypair from the secret key bytes
95            let secret_key = bcs::from_bytes::<ECVRFPrivateKey>(&secret_key_bytes)
96                .map_err(|_| Error::new(ErrorKind::InvalidInput, "Failed to parse private key."))?;
97            let kp = ECVRFKeyPair::from(secret_key);
98
99            // Generate proof
100            let proof = kp.prove(&alpha_string);
101            let proof_string = hex::encode(bcs::to_bytes(&proof).unwrap());
102            let proof_hash = hex::encode(proof.to_hash());
103
104            let mut result = "Proof:  ".to_string();
105            result.push_str(&proof_string);
106            result.push_str("\nOutput: ");
107            result.push_str(&proof_hash);
108            Ok(result)
109        }
110
111        Command::Verify(arguments) => {
112            // Parse inputs
113            let public_key_bytes = hex::decode(arguments.public_key)
114                .map_err(|_| Error::new(ErrorKind::InvalidInput, "Invalid public key."))?;
115            let alpha_string = hex::decode(arguments.input)
116                .map_err(|_| Error::new(ErrorKind::InvalidInput, "Invalid input string."))?;
117            let proof_bytes = hex::decode(arguments.proof)
118                .map_err(|_| Error::new(ErrorKind::InvalidInput, "Invalid proof string."))?;
119            let output_bytes = hex::decode(arguments.output)
120                .map_err(|_| Error::new(ErrorKind::InvalidInput, "Invalid output string."))?;
121            let output: [u8; 64] = output_bytes
122                .try_into()
123                .map_err(|_| Error::new(ErrorKind::InvalidInput, "Output must be 64 bytes."))?;
124
125            // Create public key and proof from parsed bytes
126            let public_key: ECVRFPublicKey =
127                bcs::from_bytes::<ECVRFPublicKey>(&public_key_bytes)
128                    .map_err(|_| Error::new(ErrorKind::InvalidInput, "Invalid public key."))?;
129            let proof: ECVRFProof = bcs::from_bytes::<ECVRFProof>(&proof_bytes)
130                .map_err(|_| Error::new(ErrorKind::InvalidInput, "Unable to parse proof."))?;
131
132            if proof
133                .verify_output(&alpha_string, &public_key, &output)
134                .is_ok()
135            {
136                return Ok("Proof verified correctly!".to_string());
137            }
138            Err(Error::new(ErrorKind::Other, "Proof is not correct."))
139        }
140    }
141}

149    #[test]
150    fn test_keygen() {
151        let result = execute(Command::Keygen).unwrap();
152        let expected =
153            Regex::new(r"Secret key: ([0-9a-fA-F]{64})\nPublic key: ([0-9a-fA-F]{64})").unwrap();
154        assert!(expected.is_match(&result));
155    }

162            input: input.to_string(),

169        let invalid_input = "InvalidInput";
