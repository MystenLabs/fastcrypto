4use clap::Parser;
5use fastcrypto::encoding::{Base64, Encoding, Hex};
6use std::io::{Error, ErrorKind};
7
8#[derive(Parser)]
9#[command(name = "encode-cli")]
10#[command(about = "Convert between base64 and hex encoding of a string", long_about = None)]
11enum Command {
12    /// Decode a base64 string into hex string.
13    Base64ToHex(Arg),
14
15    /// Decode a hex string into base64 string.
16    HexToBase64(Arg),
17}
18
19#[derive(Parser, Clone)]
20struct Arg {
21    #[clap(short, long)]
22    value: String,
23}
24
25fn main() {
26    match execute(Command::parse()) {
27        Ok(_) => {
28            std::process::exit(exitcode::OK);
29        }
30        Err(e) => {
31            println!("Error: {}", e);
32            std::process::exit(exitcode::DATAERR);
33        }
34    }
35}
36
37fn execute(cmd: Command) -> Result<(), std::io::Error> {
38    match cmd {
39        Command::Base64ToHex(args) => {
40            let val = Base64::decode(&args.value)
41                .map_err(|_| Error::new(ErrorKind::InvalidInput, "Invalid base64 string"))?;
42            println!("Decoded bytes: {:?}", val);
43            println!("Hex: {:?}", Hex::encode(val));
44            Ok(())
45        }
46        Command::HexToBase64(args) => {
47            let val = Hex::decode(&args.value)
48                .map_err(|_| Error::new(ErrorKind::InvalidInput, "Invalid hex string"))?;
49            println!("Decoded bytes: {:?}", val);
50            println!("Base64: {:?}", Base64::encode(val));
51            Ok(())
52        }
53    }
54}

56#[cfg(test)]
57mod tests {
58    use crate::{execute, Arg, Command};
59
60    #[test]
61    fn test_encode_base64_to_hex() {
62        // The correctness of the output is tested in the integration tests in fastcrypto-cli/tests/encode_cli.rs.
63        let base64 = "SGVsbG8gV29ybGQh";
64        assert!(execute(Command::Base64ToHex(Arg {
65            value: base64.to_string(),
66        }))
67        .is_ok());
68
69        let invalid_base64 = "SGVsbG8gV29ybGQ";
70        assert!(execute(Command::Base64ToHex(Arg {
71            value: invalid_base64.to_string(),
72        }))
73        .is_err());
74    }
75
76    #[test]
77    fn test_encode_hex_to_base64() {
78        // The correctness of the output is tested in the integration tests in fastcrypto-cli/tests/encode_cli.rs.
79        let hex = "48656c6c6f20576f726c6421";
80        assert!(execute(Command::HexToBase64(Arg {
81            value: hex.to_string(),
82        }))
83        .is_ok());
84
85        let invalid_hex = "48656c6c6f20576f726c642";
86        assert!(execute(Command::HexToBase64(Arg {
87            value: invalid_hex.to_string(),
88        }))
89        .is_err());
90    }
