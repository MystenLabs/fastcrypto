4use crate::ecies_v1;
5use crate::types::ShareIndex;
6use fastcrypto::error::{FastCryptoError, FastCryptoResult};
7use fastcrypto::groups::GroupElement;
8use fastcrypto::hash::{Blake2b256, Digest, HashFunction};
9use serde::{Deserialize, Serialize};
10use tracing::debug;
11
12pub type PartyId = u16;
13
14/// Public parameters of a party.
15#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
16pub struct Node<G: GroupElement> {
17    pub id: PartyId,
18    pub pk: ecies_v1::PublicKey<G>,
19    pub weight: u16, // May be zero after reduce()
20}
21
22/// Wrapper for a set of nodes.
23#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
24pub struct Nodes<G: GroupElement> {
25    nodes: Vec<Node<G>>, // Party ids are 0..len(nodes)-1 (inclusive)
26    total_weight: u16,   // Share ids are 1..total_weight (inclusive)
27    // Next two fields are used to map share ids to party ids.
28    accumulated_weights: Vec<u16>, // Accumulated sum of all nodes' weights. Used to map share ids to party ids.
29    nodes_with_nonzero_weight: Vec<u16>, // Indexes of nodes with non-zero weight
30}
31
32impl<G: GroupElement + Serialize> Nodes<G> {
33    // We don't expect to have more than 1000 nodes (simplifies some checks).
34    const MAX_NODES: usize = 1000;
35
36    /// Create a new set of nodes. Nodes must have consecutive ids starting from 0.
37    pub fn new(nodes: Vec<Node<G>>) -> FastCryptoResult<Self> {
38        let mut nodes = nodes;
39        nodes.sort_by_key(|n| n.id);
40        // Check all ids are consecutive and start from 0
41        if (0..nodes.len()).any(|i| (nodes[i].id as usize) != i) {
42            return Err(FastCryptoError::InvalidInput);
43        }
44        // Make sure we never overflow in the functions below.
45        if nodes.is_empty() || nodes.len() > Self::MAX_NODES {
46            return Err(FastCryptoError::InvalidInput);
47        }
48        // Make sure we never overflow in the functions below, as we don't expect to have more than u16::MAX total weight.
49        let total_weight = nodes.iter().map(|n| n.weight as u32).sum::<u32>();
50        if total_weight > u16::MAX as u32 || total_weight == 0 {
51            return Err(FastCryptoError::InvalidInput);
52        }
53        let total_weight = total_weight as u16;
54
55        // We use the next two to map share ids to party ids.
56        let accumulated_weights = Self::get_accumulated_weights(&nodes);
57        let nodes_with_nonzero_weight = Self::filter_nonzero_weights(&nodes);
58
59        Ok(Self {
60            nodes,
61            total_weight,
62            accumulated_weights,
63            nodes_with_nonzero_weight,
64        })
65    }
66
67    fn filter_nonzero_weights(nodes: &[Node<G>]) -> Vec<u16> {
68        nodes
69            .iter()
70            .enumerate()
71            .filter_map(|(i, n)| if n.weight > 0 { Some(i as u16) } else { None })
72            .collect::<Vec<_>>()
73    }
74
75    fn get_accumulated_weights(nodes: &[Node<G>]) -> Vec<u16> {
76        nodes
77            .iter()
78            .filter_map(|n| if n.weight > 0 { Some(n.weight) } else { None })
79            .scan(0, |accumulated_weight, weight| {
80                *accumulated_weight += weight;
81                Some(*accumulated_weight)
82            })
83            .collect::<Vec<_>>()
84    }
85
86    /// Total weight of the nodes.
87    pub fn total_weight(&self) -> u16 {
88        self.total_weight
89    }
90
91    /// Total weight of a subset of the parties. Returns error if any party does not exist.
92    pub fn total_weight_of<'a>(
93        &self,
94        ids: impl Iterator<Item = &'a PartyId>,
95    ) -> FastCryptoResult<u16> {
96        ids.map(|id| self.weight_of(*id)).sum()
97    }
98
99    pub fn weight_of(&self, id: PartyId) -> FastCryptoResult<u16> {
100        Ok(self.node_id_to_node(id)?.weight)
101    }
102
103    /// Number of nodes.
104    pub fn num_nodes(&self) -> usize {
105        self.nodes.len()
106    }
107
108    /// Get an iterator on the share ids.
109    pub fn share_ids_iter(&self) -> impl Iterator<Item = ShareIndex> {
110        (1..=self.total_weight).map(|i| ShareIndex::new(i).expect("nonzero"))
111    }
112
113    /// Get the node corresponding to a share id.
114    pub fn share_id_to_node(&self, share_id: &ShareIndex) -> FastCryptoResult<&Node<G>> {
115        let nonzero_node_id = self
116            .accumulated_weights
117            .binary_search(&share_id.get())
118            .unwrap_or_else(|i| i);
119        match self.nodes_with_nonzero_weight.get(nonzero_node_id) {
120            Some(node_id) => self.node_id_to_node(*node_id),
121            None => Err(FastCryptoError::InvalidInput),
122        }
123    }
124
125    pub fn node_id_to_node(&self, party_id: PartyId) -> FastCryptoResult<&Node<G>> {
126        self.nodes
127            .get(party_id as usize)
128            .ok_or(FastCryptoError::InvalidInput)
129    }
130
131    /// Get the share ids of a node (ordered). Returns error if the node does not exist.
132    pub fn share_ids_of(&self, id: PartyId) -> FastCryptoResult<Vec<ShareIndex>> {
133        // Check if the input is valid.
134        self.node_id_to_node(id)?;
135
136        // TODO: [perf opt] Cache this or impl differently.
137        Ok(self
138            .share_ids_iter()
139            .filter(|share_id| self.share_id_to_node(share_id).expect("valid share id").id == id)
140            .collect::<Vec<_>>())
141    }
142
143    /// Get an iterator on the nodes.
144    pub fn iter(&self) -> impl Iterator<Item = &Node<G>> {
145        self.nodes.iter()
146    }
147
148    // Used for logging.
149    pub fn hash(&self) -> Digest<32> {
150        let mut hash = Blake2b256::default();
151        hash.update(bcs::to_bytes(&self.nodes).expect("should serialize"));
152        hash.finalize()
153    }
154
155    /// Create a new set of nodes. Nodes must have consecutive ids starting from 0.
156    /// Reduces weights up to an allowed delta in the original total weight.
157    /// Finds the largest d such that:
158    /// - The new threshold is ceil(t / d)
159    /// - The new weights are all divided by d (floor division)
160    /// - The precision loss, counted as the sum of the remainders of the division by d, is at most
161    ///   the allowed delta
162    ///
163    /// In practice, allowed delta will be the extra liveness we would assume above 2f+1.
164    ///
165    /// total_weight_lower_bound allows limiting the level of reduction (e.g., in benchmarks). To
166    /// get the best results, set it to 1.
167    pub fn new_reduced(
168        nodes_vec: Vec<Node<G>>,
169        t: u16,
170        allowed_delta: u16,
171        total_weight_lower_bound: u16,
172    ) -> FastCryptoResult<(Self, u16)> {
173        let n = Self::new(nodes_vec)?; // checks the input, etc
174        assert!(total_weight_lower_bound <= n.total_weight && total_weight_lower_bound > 0);
175        let mut max_d = 1;
176        for d in 2..=40 {
177            // Break if we reached the lower bound.
178            // U16 is safe here since total_weight is u16.
179            let new_total_weight = n.nodes.iter().map(|n| n.weight / d).sum::<u16>();
180            if new_total_weight < total_weight_lower_bound {
181                break;
182            }
183            // Compute the precision loss.
184            // U16 is safe here since total_weight is u16.
185            let delta = n.nodes.iter().map(|n| n.weight % d).sum::<u16>();
186            if delta <= allowed_delta {
187                max_d = d;
188            }
189        }
190        debug!(
191            "Nodes::reduce reducing from {} with max_d {}, allowed_delta {}, total_weight_lower_bound {}",
192            n.total_weight, max_d, allowed_delta, total_weight_lower_bound
193        );
194
195        let nodes = n
196            .nodes
197            .iter()
198            .map(|n| Node {
199                id: n.id,
200                pk: n.pk.clone(),
201                weight: n.weight / max_d,
202            })
203            .collect::<Vec<_>>();
204        let accumulated_weights = Self::get_accumulated_weights(&nodes);
205        let nodes_with_nonzero_weight = Self::filter_nonzero_weights(&nodes);
206        // U16 is safe here since the original total_weight is u16.
207        let total_weight = nodes.iter().map(|n| n.weight).sum::<u16>();
208        let new_t = t / max_d + (t % max_d != 0) as u16;
209        Ok((
210            Self {
211                nodes,
212                total_weight,
213                accumulated_weights,
214                nodes_with_nonzero_weight,
215            },
216            new_t,
217        ))
218    }
