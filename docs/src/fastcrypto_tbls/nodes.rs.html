4use crate::ecies_v1;
5use crate::types::ShareIndex;
6use fastcrypto::error::{FastCryptoError, FastCryptoResult};
7use fastcrypto::groups::GroupElement;
8use fastcrypto::hash::{Blake2b256, Digest, HashFunction};
9use serde::{Deserialize, Serialize};
10use tracing::debug;
11
12pub type PartyId = u16;
13
14/// Public parameters of a party.
15#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
16pub struct Node<G: GroupElement> {
17    pub id: PartyId,
18    pub pk: ecies_v1::PublicKey<G>,
19    pub weight: u16, // May be zero after reduce()
20}
21
22/// Wrapper for a set of nodes.
23#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
24pub struct Nodes<G: GroupElement> {
25    nodes: Vec<Node<G>>, // Party ids are 0..len(nodes)-1 (inclusive)
26    total_weight: u16,   // Share ids are 1..total_weight (inclusive)
27    // Next two fields are used to map share ids to party ids.
28    accumulated_weights: Vec<u16>, // Accumulated sum of all nodes' weights. Used to map share ids to party ids.
29    nodes_with_nonzero_weight: Vec<u16>, // Indexes of nodes with non-zero weight
30}
31
32impl<G: GroupElement + Serialize> Nodes<G> {
33    // We don't expect to have more than 1000 nodes (simplifies some checks).
34    const MAX_NODES: usize = 1000;
35
36    /// Create a new set of nodes. Nodes must have consecutive ids starting from 0.
37    pub fn new(nodes: Vec<Node<G>>) -> FastCryptoResult<Self> {
38        let mut nodes = nodes;
39        nodes.sort_by_key(|n| n.id);
40        // Check all ids are consecutive and start from 0
41        if (0..nodes.len()).any(|i| (nodes[i].id as usize) != i) {
42            return Err(FastCryptoError::InvalidInput);
43        }
44        // Make sure we never overflow in the functions below.
45        if nodes.is_empty() || nodes.len() > Self::MAX_NODES {
46            return Err(FastCryptoError::InvalidInput);
47        }
48        // Make sure we never overflow in the functions below, as we don't expect to have more than u16::MAX total weight.
49        let total_weight = nodes.iter().map(|n| n.weight as u32).sum::<u32>();
50        if total_weight > u16::MAX as u32 || total_weight == 0 {
51            return Err(FastCryptoError::InvalidInput);
52        }
53        let total_weight = total_weight as u16;
54
55        // We use the next two to map share ids to party ids.
56        let accumulated_weights = Self::get_accumulated_weights(&nodes);
57        let nodes_with_nonzero_weight = Self::filter_nonzero_weights(&nodes);
58
59        Ok(Self {
60            nodes,
61            total_weight,
62            accumulated_weights,
63            nodes_with_nonzero_weight,
64        })
65    }
66
67    fn filter_nonzero_weights(nodes: &[Node<G>]) -> Vec<u16> {
68        nodes
69            .iter()
70            .enumerate()
71            .filter_map(|(i, n)| if n.weight > 0 { Some(i as u16) } else { None })
72            .collect::<Vec<_>>()
73    }
74
75    fn get_accumulated_weights(nodes: &[Node<G>]) -> Vec<u16> {
76        nodes
77            .iter()
78            .filter_map(|n| if n.weight > 0 { Some(n.weight) } else { None })
79            .scan(0, |accumulated_weight, weight| {
80                *accumulated_weight += weight;
81                Some(*accumulated_weight)
82            })
83            .collect::<Vec<_>>()
84    }
85
86    /// Total weight of the nodes.
87    pub fn total_weight(&self) -> u16 {
88        self.total_weight
89    }
90
91    /// Total weight of a subset of the parties. Returns error if any party does not exist.
92    pub fn total_weight_of<'a>(
93        &self,
94        ids: impl Iterator<Item = &'a PartyId>,
95    ) -> FastCryptoResult<u16> {
96        ids.map(|id| self.weight_of(*id)).sum()
97    }
98
99    pub fn weight_of(&self, id: PartyId) -> FastCryptoResult<u16> {
100        Ok(self.node_id_to_node(id)?.weight)
101    }
102
103    /// Number of nodes.
104    pub fn num_nodes(&self) -> usize {
105        self.nodes.len()
106    }
107
108    /// Get an iterator on the share ids.
109    pub fn share_ids_iter(&self) -> impl Iterator<Item = ShareIndex> {
110        (1..=self.total_weight).map(|i| ShareIndex::new(i).expect("nonzero"))
111    }
112
113    /// Get an iterator on the node ids.
114    pub fn node_ids_iter(&self) -> impl Iterator<Item = PartyId> + '_ {
115        self.nodes.iter().map(|n| n.id)
116    }
117
118    /// Get the node corresponding to a share id.
119    pub fn share_id_to_node(&self, share_id: &ShareIndex) -> FastCryptoResult<&Node<G>> {
120        let nonzero_node_id = self
121            .accumulated_weights
122            .binary_search(&share_id.get())
123            .unwrap_or_else(|i| i);
124        match self.nodes_with_nonzero_weight.get(nonzero_node_id) {
125            Some(node_id) => self.node_id_to_node(*node_id),
126            None => Err(FastCryptoError::InvalidInput),
127        }
128    }
129
130    pub fn node_id_to_node(&self, party_id: PartyId) -> FastCryptoResult<&Node<G>> {
131        self.nodes
132            .get(party_id as usize)
133            .ok_or(FastCryptoError::InvalidInput)
134    }
135
136    /// Get the share ids of a node (ordered). Returns error if the node does not exist.
137    pub fn share_ids_of(&self, id: PartyId) -> FastCryptoResult<Vec<ShareIndex>> {
138        // Check if the input is valid.
139        self.node_id_to_node(id)?;
140
141        // TODO: [perf opt] Cache this or impl differently.
142        Ok(self
143            .share_ids_iter()
144            .filter(|share_id| self.share_id_to_node(share_id).expect("valid share id").id == id)
145            .collect::<Vec<_>>())
146    }
147
148    /// Get an iterator on the nodes.
149    pub fn iter(&self) -> impl Iterator<Item = &Node<G>> {
150        self.nodes.iter()
151    }
152
153    // Used for logging.
154    pub fn hash(&self) -> Digest<32> {
155        let mut hash = Blake2b256::default();
156        hash.update(bcs::to_bytes(&self.nodes).expect("should serialize"));
157        hash.finalize()
158    }
159
160    /// Create a new set of nodes. Nodes must have consecutive ids starting from 0.
161    /// Reduces weights up to an allowed delta in the original total weight.
162    /// Finds the largest d such that:
163    /// - The new threshold is ceil(t / d)
164    /// - The new weights are all divided by d (floor division)
165    /// - The precision loss, counted as the sum of the remainders of the division by d, is at most
166    ///   the allowed delta
167    ///
168    /// In practice, allowed delta will be the extra liveness we would assume above 2f+1.
169    ///
170    /// total_weight_lower_bound allows limiting the level of reduction (e.g., in benchmarks). To
171    /// get the best results, set it to 1.
172    pub fn new_reduced(
173        nodes_vec: Vec<Node<G>>,
174        t: u16,
175        allowed_delta: u16,
176        total_weight_lower_bound: u16,
177    ) -> FastCryptoResult<(Self, u16)> {
178        let n = Self::new(nodes_vec)?; // checks the input, etc
179        assert!(total_weight_lower_bound <= n.total_weight && total_weight_lower_bound > 0);
180        let mut max_d = 1;
181        for d in 2..=40 {
182            // Break if we reached the lower bound.
183            // U16 is safe here since total_weight is u16.
184            let new_total_weight = n.nodes.iter().map(|n| n.weight / d).sum::<u16>();
185            if new_total_weight < total_weight_lower_bound {
186                break;
187            }
188            // Compute the precision loss.
189            // U16 is safe here since total_weight is u16.
190            let delta = n.nodes.iter().map(|n| n.weight % d).sum::<u16>();
191            if delta <= allowed_delta {
192                max_d = d;
193            }
194        }
195        debug!(
196            "Nodes::reduce reducing from {} with max_d {}, allowed_delta {}, total_weight_lower_bound {}",
197            n.total_weight, max_d, allowed_delta, total_weight_lower_bound
198        );
199
200        let nodes = n
201            .nodes
202            .iter()
203            .map(|n| Node {
204                id: n.id,
205                pk: n.pk.clone(),
206                weight: n.weight / max_d,
207            })
208            .collect::<Vec<_>>();
209        let accumulated_weights = Self::get_accumulated_weights(&nodes);
210        let nodes_with_nonzero_weight = Self::filter_nonzero_weights(&nodes);
211        // U16 is safe here since the original total_weight is u16.
212        let total_weight = nodes.iter().map(|n| n.weight).sum::<u16>();
213        let new_t = t.div_ceil(max_d);
214        Ok((
215            Self {
216                nodes,
217                total_weight,
218                accumulated_weights,
219                nodes_with_nonzero_weight,
220            },
221            new_t,
222        ))
223    }
