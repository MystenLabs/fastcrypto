7use std::borrow::Borrow;
8
9use crate::dl_verification::{batch_coefficients, get_random_scalars};
10use crate::polynomial::Poly;
11use crate::types::IndexedValue;
12use fastcrypto::error::{FastCryptoError, FastCryptoResult};
13use fastcrypto::groups::{GroupElement, HashToGroupElement, MultiScalarMul, Scalar};
14use fastcrypto::traits::AllowedRng;
15use itertools::Itertools;
16
17pub type Share<S> = IndexedValue<S>;
18pub type PartialSignature<S> = IndexedValue<S>;
19
20/// Trait [ThresholdBls] provides sign & verify functions for standard and partial BLS signatures.
21pub trait ThresholdBls {
22    type Private: Scalar;
23    /// `Public` represents the group over which the public keys are represented.
24    type Public: GroupElement<ScalarType = Self::Private> + MultiScalarMul;
25    /// `Signature` represents the group over which the signatures are represented.
26    type Signature: GroupElement<ScalarType = Self::Private> + HashToGroupElement + MultiScalarMul;
27
28    /// Verify a signature on a given message. This is standard BLS signature verification
29    /// over the used curve construction.
30    fn verify(public: &Self::Public, msg: &[u8], sig: &Self::Signature) -> FastCryptoResult<()>;
31
32    /// Sign a message using the private share/partial key.
33    fn partial_sign(share: &Share<Self::Private>, msg: &[u8]) -> PartialSignature<Self::Signature> {
34        Self::partial_sign_batch(std::iter::once(share), msg)[0].clone()
35    }
36
37    /// Sign a message using one of more private share/partial keys.
38    fn partial_sign_batch(
39        shares: impl Iterator<Item = impl Borrow<Share<Self::Private>>>,
40        msg: &[u8],
41    ) -> Vec<PartialSignature<Self::Signature>> {
42        let h = Self::Signature::hash_to_group_element(msg);
43        shares
44            .map(|share| {
45                let share = share.borrow();
46                PartialSignature {
47                    index: share.index,
48                    value: h * share.value,
49                }
50            })
51            .collect()
52    }
53
54    /// Verify a signature done by a partial key holder.
55    fn partial_verify(
56        vss_pk: &Poly<Self::Public>,
57        msg: &[u8],
58        partial_sig: &PartialSignature<Self::Signature>,
59    ) -> FastCryptoResult<()> {
60        let pk_i = vss_pk.eval(partial_sig.index);
61        Self::verify(&pk_i.value, msg, &partial_sig.value)
62    }
63
64    /// Verify a set of signatures done by a partial key holder.
65    /// Randomly check if \sum r_i sig_i is a valid signature with public key \sum r_i p(i) G
66    /// where r_i are random scalars, and p(i) are points on the polynomial.
67    fn partial_verify_batch<R: AllowedRng>(
68        vss_pk: &Poly<Self::Public>,
69        msg: &[u8],
70        partial_sigs: impl Iterator<Item = impl Borrow<PartialSignature<Self::Signature>>>,
71        rng: &mut R,
72    ) -> FastCryptoResult<()> {
73        assert!(vss_pk.degree_bound() > 0 || !msg.is_empty());
74        let (evals_as_scalars, points): (Vec<_>, Vec<_>) = partial_sigs
75            .map(|sig| {
76                let sig = sig.borrow();
77                (Self::Private::from(sig.index.get().into()), sig.value)
78            })
79            .unzip();
80        if points.is_empty() {
81            return Ok(());
82        }
83        let rs = get_random_scalars::<Self::Private, R>(points.len(), rng);
84        // TODO: should we cache it instead? that would replace t-wide msm with w-wide msm.
85        let coeffs = batch_coefficients(&rs, &evals_as_scalars, vss_pk.degree_bound());
86        let pk = Self::Public::multi_scalar_mul(&coeffs, vss_pk.as_vec()).expect("sizes match");
87        let aggregated_sig = Self::Signature::multi_scalar_mul(&rs, &points).expect("sizes match");
88
89        Self::verify(&pk, msg, &aggregated_sig)
90    }
91
92    /// Interpolate partial signatures to recover the full signature.
93    /// Uses the first `threshold` unique partials to interpolate the signature and the rest are
94    /// ignored (thus if those partials are invalid, the resulting signature can still be valid).
95    fn aggregate(
96        threshold: u16,
97        partials: impl Iterator<Item = impl Borrow<PartialSignature<Self::Signature>>> + Clone,
98    ) -> FastCryptoResult<Self::Signature> {
99        let unique_partials = partials
100            .unique_by(|p| p.borrow().index)
101            .take(threshold as usize);
102        if unique_partials.clone().count() != threshold as usize {
103            return Err(FastCryptoError::NotEnoughInputs);
104        }
105        // No conversion is required since PartialSignature<S> and Eval<S> are different aliases to
106        // IndexedValue<S>.
107        Poly::<Self::Signature>::recover_c0_msm(threshold, unique_partials)
108    }
109}
