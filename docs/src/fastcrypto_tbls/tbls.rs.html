7use std::borrow::Borrow;

9use crate::dl_verification::{batch_coefficients, get_random_scalars};
10use crate::polynomial::Poly;
11use crate::types::{IndexedValue, UnindexedValues};
12use fastcrypto::error::{FastCryptoError, FastCryptoResult};
13use fastcrypto::groups::{GroupElement, HashToGroupElement, MultiScalarMul, Scalar};
14use fastcrypto::traits::AllowedRng;
15use itertools::Itertools;

17pub type Share<S> = IndexedValue<S>;
18pub type PartialSignature<S> = IndexedValue<S>;
19pub type UnindexedPartialSignatures<S> = UnindexedValues<S>;
20
21/// Trait [ThresholdBls] provides sign & verify functions for standard and partial BLS signatures.
22pub trait ThresholdBls {
23    type Private: Scalar;
24    /// `Public` represents the group over which the public keys are represented.
25    type Public: GroupElement<ScalarType = Self::Private> + MultiScalarMul;
26    /// `Signature` represents the group over which the signatures are represented.
27    type Signature: GroupElement<ScalarType = Self::Private> + HashToGroupElement + MultiScalarMul;
28
29    /// Verify a signature on a given message. This is standard BLS signature verification
30    /// over the used curve construction.
31    fn verify(public: &Self::Public, msg: &[u8], sig: &Self::Signature) -> FastCryptoResult<()>;
32
33    /// Sign a message using the private share/partial key.
34    fn partial_sign(share: &Share<Self::Private>, msg: &[u8]) -> PartialSignature<Self::Signature> {
35        Self::partial_sign_batch(std::iter::once(share), msg)[0].clone()
36    }
37
38    /// Sign a message using one of more private share/partial keys.
39    fn partial_sign_batch(
40        shares: impl Iterator<Item = impl Borrow<Share<Self::Private>>>,
41        msg: &[u8],
42    ) -> Vec<PartialSignature<Self::Signature>> {
43        let h = Self::Signature::hash_to_group_element(msg);
44        shares
45            .map(|share| {
46                let share = share.borrow();
47                PartialSignature {
48                    index: share.index,
49                    value: h * share.value,
50                }
51            })
52            .collect()
53    }
54
55    /// Verify a signature done by a partial key holder.
56    fn partial_verify(
57        vss_pk: &Poly<Self::Public>,
58        msg: &[u8],
59        partial_sig: &PartialSignature<Self::Signature>,
60    ) -> FastCryptoResult<()> {
61        let pk_i = vss_pk.eval(partial_sig.index);
62        Self::verify(&pk_i.value, msg, &partial_sig.value)
63    }
64
65    /// Verify a set of signatures done by a partial key holder.
66    /// Randomly check if \sum r_i sig_i is a valid signature with public key \sum r_i p(i) G
67    /// where r_i are random scalars, and p(i) are points on the polynomial.
68    fn partial_verify_batch<R: AllowedRng>(
69        vss_pk: &Poly<Self::Public>,
70        msg: &[u8],
71        partial_sigs: impl Iterator<Item = impl Borrow<PartialSignature<Self::Signature>>>,
72        rng: &mut R,
73    ) -> FastCryptoResult<()> {
74        assert!(vss_pk.degree() > 0 || !msg.is_empty());
75        let (evals_as_scalars, points): (Vec<_>, Vec<_>) = partial_sigs
76            .map(|sig| {
77                let sig = sig.borrow();
78                (Self::Private::from(sig.index.get().into()), sig.value)
79            })
80            .unzip();
81        if points.is_empty() {
82            return Ok(());
83        }
84        let rs = get_random_scalars::<Self::Private, R>(points.len(), rng);
85        // TODO: should we cache it instead? that would replace t-wide msm with w-wide msm.
86        let coeffs = batch_coefficients(&rs, &evals_as_scalars, vss_pk.degree());
87        let pk = Self::Public::multi_scalar_mul(&coeffs, vss_pk.as_vec()).expect("sizes match");
88        let aggregated_sig = Self::Signature::multi_scalar_mul(&rs, &points).expect("sizes match");
89
90        Self::verify(&pk, msg, &aggregated_sig)
91    }
92
93    /// Interpolate partial signatures to recover the full signature.
94    /// Uses the first `threshold` unique partials to interpolate the signature and the rest are
95    /// ignored (thus if those partials are invalid, the resulting signature can still be valid).
96    fn aggregate(
97        threshold: u16,
98        partials: impl Iterator<Item = impl Borrow<PartialSignature<Self::Signature>>> + Clone,
99    ) -> FastCryptoResult<Self::Signature> {
100        let unique_partials = partials
101            .unique_by(|p| p.borrow().index)
102            .take(threshold as usize);
103        if unique_partials.clone().count() != threshold as usize {
104            return Err(FastCryptoError::NotEnoughInputs);
105        }
106        // No conversion is required since PartialSignature<S> and Eval<S> are different aliases to
107        // IndexedValue<S>.
108        Poly::<Self::Signature>::recover_c0_msm(threshold, unique_partials)
109    }
110}
