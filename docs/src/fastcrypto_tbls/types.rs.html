5use crate::tbls;
6use fastcrypto::error::{FastCryptoError, FastCryptoResult};
7use fastcrypto::groups::{bls12381, GroupElement, HashToGroupElement, Pairing};
8use serde::{Deserialize, Serialize};
9use std::num::NonZeroU16;

11/// Implementation of [ThresholdBls] for BLS12-381-min-sig. A variant for BLS12-381-min-pk can be
12/// defined in a similar way if needed in the future.
13pub struct ThresholdBls12381MinSig {}
14
15impl tbls::ThresholdBls for ThresholdBls12381MinSig {
16    type Private = bls12381::Scalar;
17    type Public = bls12381::G2Element;
18    type Signature = bls12381::G1Element;
19
20    fn verify(pk: &Self::Public, msg: &[u8], sig: &Self::Signature) -> FastCryptoResult<()> {
21        let hashed_message = Self::Signature::hash_to_group_element(msg);
22        // e(sig, g2)
23        let left = sig.pairing(&Self::Public::generator());
24        // e(H(m), pk)
25        let right = hashed_message.pairing(pk);
26        match left == right {
27            true => Ok(()),
28            false => Err(FastCryptoError::InvalidSignature),
29        }
30    }
31}
32
33/// tBLS with ThresholdBls12381MinSig types.
34///
35pub type Share = Eval<<ThresholdBls12381MinSig as tbls::ThresholdBls>::Private>;
36pub type PrivateBlsKey = <ThresholdBls12381MinSig as tbls::ThresholdBls>::Private;
37pub type PublicBlsKey = <ThresholdBls12381MinSig as tbls::ThresholdBls>::Public;
38pub type PublicVssKey = PublicPoly<<ThresholdBls12381MinSig as tbls::ThresholdBls>::Public>;
39pub type Signature = <ThresholdBls12381MinSig as tbls::ThresholdBls>::Signature;
40pub type RawSignature = bls12381::G1ElementAsBytes;
41
42/// Indexes of shares/private keys (0 is reserved for the secret itself).
43pub type ShareIndex = NonZeroU16;
44
45/// Wrapper of a value that is associated with a specific index.
46#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
47pub struct IndexedValue<A> {
48    pub index: ShareIndex,
49    pub value: A,
50}
51
52/// Basic wrapper of a set of values that are not associated with indexes, assuming the indexes are known to all
53/// parties. Used to reduce the size of the messages in the protocol.
54#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
55pub struct UnindexedValues<A> {
56    pub values: Vec<A>,
57}
58
59impl<A> From<Vec<IndexedValue<A>>> for UnindexedValues<A> {
60    fn from(index_values: Vec<IndexedValue<A>>) -> Self {
61        let mut values: Vec<A> = Vec::with_capacity(index_values.len());
62        for v in index_values {
63            values.push(v.value);
64        }
65        Self { values }
66    }
67}
68
69impl<A> UnindexedValues<A> {
70    pub fn add_indexes(self, indexes: &[ShareIndex]) -> FastCryptoResult<Vec<IndexedValue<A>>> {
71        if self.values.len() != indexes.len() {
72            return Err(FastCryptoError::InvalidInput);
73        }
74        let values = self
75            .values
76            .into_iter()
77            .zip(indexes)
78            .map(|(value, index)| IndexedValue {
79                index: *index,
80                value,
81            })
82            .collect();
83        Ok(values)
84    }
85}
