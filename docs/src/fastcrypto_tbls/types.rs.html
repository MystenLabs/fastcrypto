13/// Implementation of [ThresholdBls] for BLS12-381-min-sig. A variant for BLS12-381-min-pk can be
14/// defined in a similar way if needed in the future.
15pub struct ThresholdBls12381MinSig {}
16
17impl tbls::ThresholdBls for ThresholdBls12381MinSig {
18    type Private = bls12381::Scalar;
19    type Public = bls12381::G2Element;
20    type Signature = bls12381::G1Element;
21
22    fn verify(pk: &Self::Public, msg: &[u8], sig: &Self::Signature) -> FastCryptoResult<()> {
23        let hashed_message = Self::Signature::hash_to_group_element(msg);
24        // e(sig, g2)
25        let left = sig.pairing(&Self::Public::generator());
26        // e(H(m), pk)
27        let right = hashed_message.pairing(pk);
28        match left == right {
29            true => Ok(()),
30            false => Err(FastCryptoError::InvalidSignature),
31        }
32    }
33}

35/// tBLS with ThresholdBls12381MinSig types.
36///
37pub type Share = Eval<<ThresholdBls12381MinSig as tbls::ThresholdBls>::Private>;
38pub type PrivateBlsKey = <ThresholdBls12381MinSig as tbls::ThresholdBls>::Private;
39pub type PublicBlsKey = <ThresholdBls12381MinSig as tbls::ThresholdBls>::Public;
40pub type PublicVssKey = PublicPoly<<ThresholdBls12381MinSig as tbls::ThresholdBls>::Public>;
41pub type Signature = <ThresholdBls12381MinSig as tbls::ThresholdBls>::Signature;
42pub type RawSignature = bls12381::G1ElementAsBytes;

44/// Indexes of shares/private keys (0 is reserved for the secret itself).
45pub type ShareIndex = NonZeroU16;
46
47/// Wrapper of a value that is associated with a specific index.
48#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
49pub struct IndexedValue<A> {
50    pub index: ShareIndex,
51    pub value: A,
52}
53
54#[inline]
55pub(crate) fn to_scalar<C: Scalar>(index: impl Borrow<ShareIndex>) -> C {
56    C::from(index.borrow().get() as u128)
57}

59/// Helper function to add two evaluations.
60/// Panics if the indices are not equal.
61pub fn sum<S>((a, b): (&Eval<S>, &Eval<S>)) -> Eval<S>
62where
63    for<'b> &'b S: Add<&'b S, Output = S>,
64{
65    assert_eq!(a.index, b.index);
66    Eval {
67        value: &a.value + &b.value,
68        index: a.index,
69    }
70}

72/// If all elements in this iterator have the same value, return that value.
73/// Otherwise, or if the iterator is empty, return `None`.
74pub fn get_uniform_value<T: PartialEq>(items: impl IntoIterator<Item = T>) -> Option<T> {
75    let mut iterator = items.into_iter();
76    let uniform_value = iterator.next()?;
77    iterator
78        .all(|x| uniform_value == x)
79        .then_some(uniform_value)
80}
