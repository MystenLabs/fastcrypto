26/// Generics below use `G: GroupElement' for the group of the VSS public key, and `EG: GroupElement'
27/// for the group of the ECIES public key.
28
29/// Party in the DKG protocol.
30#[derive(Clone, PartialEq, Eq, Serialize, Deserialize)]
31pub struct Party<G: GroupElement, EG: GroupElement>
32where
33    EG::ScalarType: Zeroize,
34{
35    pub id: PartyId,
36    pub(crate) nodes: Nodes<EG>,
37    pub t: u16,
38    pub random_oracle: RandomOracle,
39    pub(crate) enc_sk: ecies_v1::PrivateKey<EG>,
40    pub(crate) vss_sk: PrivatePoly<G>,
41    pub old_t: Option<u16>, // Used only for key rotation
42}
43
44/// Assumptions:
45/// - The high-level protocol is responsible for verifying that the 'sender' is correct in the
46///   following messages (based on the chain's authentication).
47/// - The high-level protocol is responsible that all parties see the same order of messages.
48///
49/// A complaint/fraud claim against a dealer that created invalid encrypted share.
50#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
51pub struct Complaint<EG: GroupElement> {
52    pub(crate) accused_sender: PartyId,
53    pub(crate) proof: RecoveryPackage<EG>,
54}
55
56/// A [Confirmation] is sent during the second phase of the protocol. It includes complaints
57/// created by receiver of invalid encrypted shares (if any).
58#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
59pub struct Confirmation<EG: GroupElement> {
60    pub sender: PartyId,
61    /// List of complaints against other parties. Empty if there are none.
62    pub complaints: Vec<Complaint<EG>>,
63}

65// Upper bound on the size of binary serialized incoming messages assuming <=3333 shares, <=400
66// parties, and using G2Element for encryption. This is a safe upper bound since:
67// - Message is O(96*t + 32*n) bytes.
68// - Confirmation is O((96*3 + 32) * k) bytes.
69// Could be used as a sanity safety check before deserializing an incoming message.
70pub const DKG_MESSAGES_MAX_SIZE: usize = 400_000; // 400 KB

72/// [Output] is the final output of the DKG protocol in case it runs
73/// successfully. It can be used later with [ThresholdBls], see examples in tests.
74///
75/// If shares is None, the object can only be used for verifying (partial and full) signatures.
76#[derive(Clone, Debug, Serialize, Deserialize)]
77pub struct Output<G: GroupElement, EG: GroupElement> {
78    pub nodes: Nodes<EG>,
79    pub vss_pk: Poly<G>,
80    pub shares: Option<Vec<Share<G::ScalarType>>>, // None if some shares are missing or weight is zero.
81}

83/// [Message] holds all encrypted shares a dealer sends during the first phase of the
84/// protocol.
85#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
86pub struct Message<G: GroupElement, EG: GroupElement> {
87    pub sender: PartyId,
88    /// The commitment of the secret polynomial created by the sender.
89    pub vss_pk: PublicPoly<G>,
90    /// The encrypted shares created by the sender. Sorted according to the receivers.
91    pub encrypted_shares: ecies_v1::MultiRecipientEncryption<EG>,
92}

94/// Wrapper for collecting everything related to a processed message.
95#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
96pub struct ProcessedMessage<G: GroupElement, EG: GroupElement> {
97    pub message: Message<G, EG>,
98    /// Possibly empty
99    pub shares: Vec<Share<G::ScalarType>>,
100    pub complaint: Option<Complaint<EG>>,
101}

103/// Unique processed messages that are being used in the protocol.
104#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
105pub struct UsedProcessedMessages<G: GroupElement, EG: GroupElement>(
106    pub Vec<ProcessedMessage<G, EG>>,
107);
108
109impl<G: GroupElement, EG: GroupElement> From<&[ProcessedMessage<G, EG>]>
110    for UsedProcessedMessages<G, EG>
111{
112    // Assumes all parties see the same order of messages.
113    fn from(msgs: &[ProcessedMessage<G, EG>]) -> Self {
114        let filtered = msgs
115            .iter()
116            .unique_by(|&m| m.message.sender) // stable
117            .cloned()
118            .collect::<Vec<_>>();
119        Self(filtered)
120    }
121}

123/// Processed messages that were not excluded after the third phase of the protocol.
124pub struct VerifiedProcessedMessages<G: GroupElement, EG: GroupElement>(
125    Vec<ProcessedMessage<G, EG>>,
126);
127
128impl<G: GroupElement, EG: GroupElement> VerifiedProcessedMessages<G, EG> {
129    fn filter_from(msgs: &UsedProcessedMessages<G, EG>, to_exclude: &[PartyId]) -> Self {
130        let filtered = msgs
131            .0
132            .iter()
133            .filter(|m| !to_exclude.contains(&m.message.sender))
134            .cloned()
135            .collect::<Vec<_>>();
136        Self(filtered)
137    }
138
139    pub fn len(&self) -> usize {
140        self.0.len()
141    }
142
143    pub fn is_empty(&self) -> bool {
144        self.0.is_empty()
145    }
146
147    #[cfg(test)]
148    pub fn data(&self) -> &[ProcessedMessage<G, EG>] {
149        &self.0
150    }
151}

153/// A dealer in the DKG ceremony.
154///
155/// Can be instantiated with G1Curve or G2Curve.
156impl<G, EG> Party<G, EG>
157where
158    G: GroupElement + MultiScalarMul + Serialize + DeserializeOwned,
159    EG: GroupElement + Serialize + HashToGroupElement + DeserializeOwned,
160    EG::ScalarType: FiatShamirChallenge + Zeroize,
161{
162    /// 1. Create a new ECIES private key and send the public key to all parties.
163    /// 2. After *all* parties have sent their ECIES public keys, create the (same) set of nodes.
164
165    /// 3. Create a new Party instance with the ECIES private key and the set of nodes.
166    pub fn new<R: AllowedRng>(
167        enc_sk: ecies_v1::PrivateKey<EG>,
168        nodes: Nodes<EG>,
169        t: u16, // The number of parties that are needed to reconstruct the full key/signature (f+1).
170        random_oracle: RandomOracle, // Should be unique for each invocation, but the same for all parties.
171        rng: &mut R,
172    ) -> FastCryptoResult<Self> {
173        // Sanity check that the threshold makes sense (t <= n/2 since we later wait for 2t-1).
174        if t > (nodes.total_weight() / 2) {
175            return Err(FastCryptoError::InvalidInput);
176        }
177        Self::new_advanced(enc_sk, nodes, t, random_oracle, None, None, rng)
178    }
179
180    /// Alternative to new(), to be used for:
181    /// - Any threshold t, useful in the sync setting.
182    /// - Passing a secret/share for key rotation (should be set to None for DKG or by new parties).
183    /// - Passing old threshold for key rotation (should be set to None for DKG).
184    pub fn new_advanced<R: AllowedRng>(
185        enc_sk: ecies_v1::PrivateKey<EG>,
186        nodes: Nodes<EG>,
187        t: u16, // The number of parties that are needed to reconstruct the full key/signature (f+1).
188        random_oracle: RandomOracle, // Should be unique for each invocation, but the same for all parties.
189        secret: Option<G::ScalarType>, // Should be used only for key rotation.
190        old_t: Option<u16>,          // Used only for key rotation
191        rng: &mut R,
192    ) -> FastCryptoResult<Self> {
193        // Sanity check that the threshold makes sense.
194        if t > nodes.total_weight() || t == 0 || old_t.is_some_and(|ot| ot == 0) {
195            return Err(FastCryptoError::InvalidInput);
196        }
197
198        // Confirm that my ecies pk is in the nodes.
199        let enc_pk = ecies_v1::PublicKey::<EG>::from_private_key(&enc_sk);
200        let my_node = nodes
201            .iter()
202            .find(|n| n.pk == enc_pk)
203            .ok_or(FastCryptoError::InvalidInput)?;
204        // TODO: [comm opt] Instead of generating the polynomial at random, use PRF generated values
205        // to reduce communication.
206        let vss_sk = if let Some(secret) = secret {
207            PrivatePoly::<G>::rand_fixed_c0(t - 1, secret, rng)
208        } else {
209            PrivatePoly::<G>::rand(t - 1, rng)
210        };
211
212        // TODO: remove once the protocol is stable since it's a non negligible computation.
213        let vss_pk = vss_sk.commit::<G>();
214        info!(
215            "DKG: Creating party {} with weight {}, nodes hash {:?}, t {}, n {}, ro {:?}, enc pk {:?}, vss pk c0 {:?}",
216            my_node.id,
217            my_node.weight,
218            nodes.hash(),
219            t,
220            nodes.total_weight(),
221            random_oracle,
222            enc_pk,
223            vss_pk.c0(),
224        );
225
226        Ok(Self {
227            id: my_node.id,
228            nodes,
229            t,
230            random_oracle,
231            enc_sk,
232            vss_sk,
233            old_t,
234        })
235    }

237    /// The threshold needed to reconstruct the full key/signature.
238    pub fn t(&self) -> u16 {
239        self.t
240    }

242    /// 4. Create the first message to be broadcasted.
243    ///
244    ///    Returns IgnoredMessage if the party has zero weight (so no need to create a message).
245    pub fn create_message<R: AllowedRng>(&self, rng: &mut R) -> FastCryptoResult<Message<G, EG>> {
246        let node = self.nodes.node_id_to_node(self.id).expect("my id is valid");
247        if node.weight == 0 {
248            return Err(FastCryptoError::IgnoredMessage);
249        }
250
251        let vss_pk = self.vss_sk.commit();
252        let encryption_ro = self.encryption_random_oracle(self.id);
253        info!(
254            "DKG: Creating message for party {} with vss pk c0 {:?}, ro {:?}",
255            self.id,
256            vss_pk.c0(),
257            encryption_ro,
258        );
259        // Create a vector of a public key and shares per receiver.
260        let pk_and_shares = self
261            .nodes
262            .iter()
263            .map(|node| {
264                let share_ids = self
265                    .nodes
266                    .share_ids_of(node.id)
267                    .expect("iterating on valid nodes");
268                let shares = share_ids
269                    .iter()
270                    .map(|share_id| self.vss_sk.eval(*share_id).value)
271                    .collect::<Vec<_>>();
272                // Works even with empty shares_ids (will result in [0]).
273                let buff = bcs::to_bytes(&shares).expect("serialize of shares should never fail");
274                (node.pk.clone(), buff)
275            })
276            .collect::<Vec<_>>();
277        // Encrypt everything.
278        let encrypted_shares =
279            ecies_v1::MultiRecipientEncryption::encrypt(&pk_and_shares, &encryption_ro, rng);
280
281        debug!(
282            "DKG: Created message using {:?}, with eph key {:?} nizk {:?}",
283            encryption_ro,
284            encrypted_shares.ephemeral_key(),
285            encrypted_shares.proof(),
286        );
287
288        Ok(Message {
289            sender: self.id,
290            vss_pk,
291            encrypted_shares,
292        })
293    }

295    // Sanity checks that can be done by any party on a received message.
296    fn sanity_check_message(&self, msg: &Message<G, EG>) -> FastCryptoResult<()> {
297        let node = self
298            .nodes
299            .node_id_to_node(msg.sender)
300            .tap_err(|_| {
301                warn!(
302                    "DKG: Message sanity check failed, invalid id {}",
303                    msg.sender
304                )
305            })
306            .map_err(|_| FastCryptoError::InvalidMessage)?;
307        if node.weight == 0 {
308            warn!(
309                "DKG: Message sanity check failed for id {}, zero weight",
310                msg.sender
311            );
312            return Err(FastCryptoError::InvalidMessage);
313        };
314
315        if self.t as usize != msg.vss_pk.degree_bound() + 1 {
316            warn!(
317                "DKG: Message sanity check failed for id {}, expected degree={}, got {}",
318                msg.sender,
319                self.t - 1,
320                msg.vss_pk.degree_bound()
321            );
322            return Err(FastCryptoError::InvalidMessage);
323        }
324
325        if *msg.vss_pk.c0() == G::zero() {
326            warn!(
327                "DKG: Message sanity check failed for id {}, zero c0",
328                msg.sender,
329            );
330            return Err(FastCryptoError::InvalidMessage);
331        }
332
333        if self.nodes.num_nodes() != msg.encrypted_shares.len() {
334            warn!(
335                "DKG: Message sanity check failed for id {}, expected encrypted_shares.len={}, got {}",
336                msg.sender,
337                self.nodes.num_nodes(),
338                msg.encrypted_shares.len()
339            );
340            return Err(FastCryptoError::InvalidMessage);
341        }
342
343        let encryption_ro = self.encryption_random_oracle(msg.sender);
344        msg.encrypted_shares
345            .verify(&encryption_ro)
346            .tap_err(|e| {
347                warn!("DKG: Message sanity check failed for id {}, verify with RO {:?}, eph key {:?} and proof {:?}, returned err: {:?}",
348                    msg.sender,
349                    encryption_ro,
350                    msg.encrypted_shares.ephemeral_key(),
351                    msg.encrypted_shares.proof(),
352                    e)
353            })
354            .map_err(|_| FastCryptoError::InvalidMessage)
355    }

357    /// 5. Process a message and create the second message to be broadcasted.
358    ///    The second message contains the list of complaints on invalid shares. In addition, it
359    ///    returns a set of valid shares (so far).
360    ///
361    ///    We split this function into two parts: process_message and merge, so that the caller can
362    ///    process messages concurrently and then merge the results.
363
364    ///    [process_message] processes a message and returns an intermediate object ProcessedMessage.
365    ///
366    ///    Returns error InvalidMessage if the message is invalid and should be ignored (note that we
367    ///    could count it as part of the f+1 messages we wait for, but it's also safe to ignore it
368    ///    and just wait for f+1 valid messages).
369    ///
370    ///    Assumptions: Called only once per sender (the high level protocol is responsible for deduplication).
371    pub fn process_message<R: AllowedRng>(
372        &self,
373        message: Message<G, EG>,
374        rng: &mut R,
375    ) -> FastCryptoResult<ProcessedMessage<G, EG>> {
376        debug!(
377            "DKG: Processing message from party {} with vss pk c0 {:?}",
378            message.sender,
379            message.vss_pk.c0()
380        );
381        // Ignore if invalid (and other honest parties will ignore as well).
382        self.sanity_check_message(&message)?;
383
384        let my_share_ids = self.nodes.share_ids_of(self.id).expect("my id is valid");
385        let encryption_ro = self.encryption_random_oracle(message.sender);
386        let buffer =
387            message
388                .encrypted_shares
389                .decrypt(&self.enc_sk, &encryption_ro, self.id as usize);
390        let decrypted_shares: Option<Vec<G::ScalarType>> = bcs::from_bytes(buffer.as_slice())
391            .map_err(|_| FastCryptoError::InvalidInput)
392            .ok();
393
394        if decrypted_shares.is_none()
395            || decrypted_shares.as_ref().unwrap().len() != my_share_ids.len()
396        {
397            warn!(
398                "DKG: Processing message from party {} failed, invalid number of decrypted shares",
399                message.sender
400            );
401            let complaint = Complaint {
402                accused_sender: message.sender,
403                proof: message.encrypted_shares.create_recovery_package(
404                    &self.enc_sk,
405                    &self.recovery_random_oracle(self.id, message.sender),
406                    rng,
407                ),
408            };
409            return Ok(ProcessedMessage {
410                message,
411                shares: vec![],
412                complaint: Some(complaint),
413            });
414        }
415
416        let decrypted_shares = decrypted_shares
417            .expect("checked above")
418            .iter()
419            .zip(my_share_ids)
420            .map(|(s, i)| Eval {
421                index: i,
422                value: *s,
423            })
424            .collect::<Vec<_>>();
425        debug!(
426            "DKG: Successfully decrypted shares from party {}",
427            message.sender
428        );
429        // Verify all shares in a batch.
430        if verify_poly_evals(&decrypted_shares, &message.vss_pk, rng).is_err() {
431            warn!(
432                "DKG: Processing message from party {} failed, invalid shares",
433                message.sender
434            );
435            let complaint = Complaint {
436                accused_sender: message.sender,
437                proof: message.encrypted_shares.create_recovery_package(
438                    &self.enc_sk,
439                    &self.recovery_random_oracle(self.id, message.sender),
440                    rng,
441                ),
442            };
443            return Ok(ProcessedMessage {
444                message,
445                shares: vec![],
446                complaint: Some(complaint),
447            });
448        }
449
450        info!(
451            "DKG: Successfully processed message from party {}",
452            message.sender
453        );
454        Ok(ProcessedMessage {
455            message,
456            shares: decrypted_shares,
457            complaint: None,
458        })
459    }

461    /// Alternative to process_message for additional checks:
462    /// - Expected vss.c0 public key, useful for key rotation.
463    /// - Knowledge of the shared secret, useful for high thresholds protocols.
464    pub fn process_message_with_checks<R: AllowedRng>(
465        &self,
466        message: Message<G, EG>,
467        partial_pk: &Option<G>,
468        secret_nizk: &Option<DLNizk<G>>,
469        rng: &mut R,
470    ) -> FastCryptoResult<ProcessedMessage<G, EG>>
471    where
472        G::ScalarType: FiatShamirChallenge,
473    {
474        if let Some(partial_pk) = partial_pk {
475            if message.vss_pk.c0() != partial_pk {
476                warn!(
477                    "DKG: Processing message from party {} failed, invalid vss pk c0 {:?}, expected {:?}",
478                    message.sender,
479                    message.vss_pk.c0(),
480                    partial_pk
481                );
482                return Err(FastCryptoError::InvalidMessage);
483            }
484        };
485        if let Some(secret_nizk) = secret_nizk {
486            secret_nizk
487                .verify(
488                    message.vss_pk.c0(),
489                    &self.nizk_pop_of_secret_random_oracle(message.sender),
490                )
491                .tap_err(|_| {
492                    warn!(
493                        "DKG: Processing message from party {} failed, invalid secret's nizk proof",
494                        message.sender
495                    )
496                })?;
497        }
498        self.process_message(message, rng)
499    }

501    /// 6. Merge results from multiple ProcessedMessages so only one message needs to be sent.
502    ///
503    ///    Returns NotEnoughInputs if the threshold t is not met.
504    ///
505    ///    Assumptions: processed_messages is the result of process_message on the same set of messages
506    ///    on all parties.
507    pub fn merge(
508        &self,
509        processed_messages: &[ProcessedMessage<G, EG>],
510    ) -> FastCryptoResult<(Confirmation<EG>, UsedProcessedMessages<G, EG>)> {
511        debug!("DKG: Trying to merge {} messages", processed_messages.len());
512        let filtered_messages = UsedProcessedMessages::from(processed_messages);
513        // Verify we have enough messages
514        let total_weight = filtered_messages
515            .0
516            .iter()
517            .map(|m| {
518                self.nodes
519                    .node_id_to_node(m.message.sender)
520                    .expect("checked in process_message")
521                    .weight as u32
522            })
523            .sum::<u32>();
524        let required_t = self.old_t.unwrap_or(self.t);
525        if total_weight < (required_t as u32) {
526            debug!("Merge failed with total weight {total_weight}");
527            return Err(FastCryptoError::NotEnoughInputs);
528        }
529
530        info!("DKG: Merging messages with total weight {total_weight}");
531
532        // Log used parties.
533        let used_parties = filtered_messages
534            .0
535            .iter()
536            .map(|m| m.message.sender.to_string())
537            .collect::<Vec<String>>()
538            .join(",");
539        debug!("DKG: Using messages from parties: {}", used_parties);
540
541        let mut conf = Confirmation {
542            sender: self.id,
543            complaints: Vec::new(),
544        };
545        for m in &filtered_messages.0 {
546            if let Some(complaint) = &m.complaint {
547                info!("DKG: Including a complaint on party {}", m.message.sender);
548                conf.complaints.push(complaint.clone());
549            }
550        }
551
552        if filtered_messages.0.iter().all(|m| m.complaint.is_some()) {
553            error!("DKG: All processed messages resulted in complaints, this should never happen");
554            return Err(FastCryptoError::GeneralError(
555                "All processed messages resulted in complaints".to_string(),
556            ));
557        }
558
559        Ok((conf, filtered_messages))
560    }

562    /// 7. Process all confirmations, check all complaints, and update the local set of
563    ///    valid shares accordingly.
564    ///
565    ///    Returns NotEnoughInputs if the threshold minimal_threshold is not met.
566    ///
567    ///    Assumptions: All parties use the same set of confirmations (and outputs from merge).
568    pub(crate) fn process_confirmations<R: AllowedRng>(
569        &self,
570        messages: &UsedProcessedMessages<G, EG>,
571        confirmations: &[Confirmation<EG>],
572        rng: &mut R,
573    ) -> FastCryptoResult<VerifiedProcessedMessages<G, EG>> {
574        debug!("Processing {} confirmations", confirmations.len());
575        let required_threshold = 2 * (self.t as u32) - 1; // guarantee that at least t honest nodes have valid shares.
576
577        // Ignore confirmations with invalid sender or zero weights
578        let confirmations = confirmations
579            .iter()
580            .filter(|c| {
581                self.nodes
582                    .node_id_to_node(c.sender)
583                    .is_ok_and(|n| n.weight > 0)
584            })
585            .unique_by(|m| m.sender)
586            .collect::<Vec<_>>();
587        // Verify we have enough confirmations
588        let total_weight = confirmations
589            .iter()
590            .map(|c| {
591                self.nodes
592                    .node_id_to_node(c.sender)
593                    .expect("checked above")
594                    .weight as u32
595            })
596            .sum::<u32>();
597        if total_weight < required_threshold {
598            debug!("Processing confirmations failed with total weight {total_weight}");
599            return Err(FastCryptoError::NotEnoughInputs);
600        }
601
602        info!("DKG: Processing confirmations with total weight {total_weight}, expected {required_threshold}");
603
604        // Two hash maps for faster access in the main loop below.
605        let id_to_pk = self
606            .nodes
607            .iter()
608            .map(|n| (n.id, &n.pk))
609            .collect::<HashMap<_, _>>();
610        let id_to_m1 = messages
611            .0
612            .iter()
613            .map(|m| (m.message.sender, &m.message))
614            .collect::<HashMap<_, _>>();
615
616        let mut to_exclude = HashSet::new();
617        'outer: for m2 in confirmations {
618            'inner: for complaint in &m2.complaints {
619                let accused = complaint.accused_sender;
620                let accuser = m2.sender;
621                debug!("DKG: Checking complaint from {accuser} on {accused}");
622                let accuser_pk = id_to_pk
623                    .get(&accuser)
624                    .expect("checked above that accuser is valid id");
625                // If the claim refers to a non existing message, it's an invalid complaint.
626                let valid_complaint = match id_to_m1.get(&accused) {
627                    Some(related_m1) => Self::check_complaint_proof(
628                        &complaint.proof,
629                        accuser_pk,
630                        accuser,
631                        &self
632                            .nodes
633                            .share_ids_of(accuser)
634                            .expect("checked above the accuser is valid id"),
635                        &related_m1.vss_pk,
636                        &related_m1.encrypted_shares,
637                        &self.recovery_random_oracle(accuser, accused),
638                        &self.encryption_random_oracle(accused),
639                        rng,
640                    )
641                    .is_ok(),
642                    None => false,
643                };
644                match valid_complaint {
645                    // Ignore accused from now on, and continue processing complaints from the
646                    // current accuser.
647                    true => {
648                        warn!("DKG: Processing confirmations excluded accused party {accused}");
649                        to_exclude.insert(accused);
650                        continue 'inner;
651                    }
652                    // Ignore the accuser from now on, including its other complaints (not critical
653                    // for security, just saves some work).
654                    false => {
655                        warn!("DKG: Processing confirmations excluded accuser {accuser}");
656                        to_exclude.insert(accuser);
657                        continue 'outer;
658                    }
659                }
660            }
661        }
662
663        let verified_messages = VerifiedProcessedMessages::filter_from(
664            messages,
665            &to_exclude.into_iter().collect::<Vec<_>>(),
666        );
667
668        if verified_messages.is_empty() {
669            error!(
670                "DKG: No verified messages after processing complaints, this should never happen"
671            );
672            return Err(FastCryptoError::GeneralError(
673                "No verified messages after processing complaints".to_string(),
674            ));
675        }
676
677        // Log verified messages parties.
678        let used_parties = verified_messages
679            .0
680            .iter()
681            .map(|m| m.message.sender.to_string())
682            .collect::<Vec<String>>()
683            .join(",");
684        debug!(
685            "DKG: Using verified messages from parties: {}",
686            used_parties
687        );
688
689        Ok(verified_messages)
690    }

692    /// 8. Aggregate the valid shares (as returned from the previous step) and the public key.
693    pub(crate) fn aggregate(&self, messages: &VerifiedProcessedMessages<G, EG>) -> Output<G, EG> {
694        debug!(
695            "Aggregating shares from {} verified messages",
696            messages.0.len()
697        );
698        let id_to_m1 = messages
699            .0
700            .iter()
701            .map(|m| (m.message.sender, &m.message))
702            .collect::<HashMap<_, _>>();
703        let mut vss_pk = PublicPoly::<G>::zero();
704        let my_share_ids = self.nodes.share_ids_of(self.id).expect("my id is valid");
705
706        let mut final_shares = my_share_ids
707            .iter()
708            .map(|share_id| {
709                (
710                    share_id,
711                    Share {
712                        index: *share_id,
713                        value: G::ScalarType::zero(),
714                    },
715                )
716            })
717            .collect::<HashMap<_, _>>();
718
719        for m in &messages.0 {
720            vss_pk += &id_to_m1
721                .get(&m.message.sender)
722                .expect("shares only includes shares from valid first messages")
723                .vss_pk;
724            for share in &m.shares {
725                final_shares
726                    .get_mut(&share.index)
727                    .expect("created above")
728                    .value += share.value;
729            }
730        }
731
732        // If I didn't receive a valid share for one of the verified messages (i.e., my complaint
733        // was not processed), then I don't have a valid share for the final key.
734        let has_invalid_share = messages.0.iter().any(|m| m.complaint.is_some());
735        let has_zero_shares = final_shares.is_empty(); // if my weight is zero
736        info!(
737            "DKG: Aggregating my shares completed with has_invalid_share={}, has_zero_shares={}",
738            has_invalid_share, has_zero_shares
739        );
740        if has_invalid_share {
741            warn!("DKG: Aggregating my shares failed");
742        }
743
744        let shares = if !has_invalid_share && !has_zero_shares {
745            Some(
746                final_shares
747                    .values()
748                    .cloned()
749                    .sorted_by_key(|s| s.index)
750                    .collect(),
751            )
752        } else {
753            None
754        };
755
756        Output {
757            nodes: self.nodes.clone(),
758            vss_pk,
759            shares,
760        }
761    }

763    /// Execute the previous two steps together.
764    pub fn complete<R: AllowedRng>(
765        &self,
766        messages: &UsedProcessedMessages<G, EG>,
767        confirmations: &[Confirmation<EG>],
768        rng: &mut R,
769    ) -> FastCryptoResult<Output<G, EG>> {
770        let verified_messages = self.process_confirmations(messages, confirmations, rng)?;
771        Ok(self.aggregate(&verified_messages))
772    }

774    /// Alternative to complete() - Optimistic version that assumes all messages are valid and if
775    /// other parties received invalid shares, the higher level protocol will handle it.
776    pub fn complete_optimistic(
777        &self,
778        messages: &UsedProcessedMessages<G, EG>,
779    ) -> FastCryptoResult<Output<G, EG>> {
780        // Do not filter out any messages, assume all are valid.
781        let verified_messages = VerifiedProcessedMessages::filter_from(messages, &[]);
782        Ok(self.aggregate(&verified_messages))
783    }

785    /// Alternative to complete() - Optimistic variant that interpolates the shares instead of
786    /// summing them, to be used for key rotation. Works only if all weights are 1.
787    pub fn complete_optimistic_key_rotation(
788        &self,
789        messages: &UsedProcessedMessages<G, EG>,
790        // Mapping party id from new committee to its id in the previous committee
791        new_to_old_party_ids: &HashMap<PartyId, PartyId>,
792    ) -> FastCryptoResult<Output<G, EG>> {
793        // Check that all weights are 1.
794        if self.nodes.iter().any(|n| n.weight != 1) {
795            return Err(FastCryptoError::InvalidInput);
796        }
797        // Check the mapping is valid: unique and all parties exist in the new committee.
798        if new_to_old_party_ids.values().collect::<HashSet<_>>().len() != new_to_old_party_ids.len()
799            || new_to_old_party_ids
800                .keys()
801                .any(|id| self.nodes.node_id_to_node(*id).is_err())
802        {
803            return Err(FastCryptoError::InvalidInput);
804        }
805
806        // Do not filter out any message, assume all are valid.
807        let messages = VerifiedProcessedMessages::filter_from(messages, &[]);
808        let old_t = self.old_t.ok_or(FastCryptoError::InvalidInput)?;
809        if messages.len() != old_t as usize {
810            return Err(FastCryptoError::InvalidInput);
811        }
812
813        let mut new_sender_to_old_share_id = HashMap::new();
814        for m in messages.0.iter() {
815            if !new_to_old_party_ids.contains_key(&m.message.sender) {
816                return Err(FastCryptoError::InvalidInput);
817            }
818            let old_share_id = new_to_old_party_ids
819                .get(&m.message.sender)
820                .expect("checked above that the sender is valid")
821                + 1; // +1 since the old party ids are 0-based.
822            new_sender_to_old_share_id.insert(
823                m.message.sender,
824                ShareIndex::new(old_share_id).expect("nonzero"),
825            );
826        }
827
828        // Interpolate my shares.
829        let shares = messages
830            .0
831            .iter()
832            .map(|m| Share {
833                index: *new_sender_to_old_share_id
834                    .get(&m.message.sender)
835                    .expect("checked above"),
836                value: m
837                    .shares
838                    .iter()
839                    .map(|s| s.value)
840                    .exactly_one()
841                    .expect("weight is 1"),
842            })
843            .collect::<Vec<_>>();
844        let share = PrivatePoly::<G>::recover_c0(old_t, shares.iter())
845            .expect("checked we have exactly t shares");
846        let my_share_id = self
847            .nodes
848            .share_ids_of(self.id)
849            .expect("my id is valid")
850            .into_iter()
851            .exactly_one()
852            .expect("tested above that weight is 1");
853        let shares = vec![Share {
854            index: my_share_id,
855            value: share,
856        }];
857
858        let vss_pk = (0..self.t)
859            .map(|i| {
860                let mut partial_keys = Vec::new();
861                for m in messages.0.iter() {
862                    let share = Share {
863                        index: *new_sender_to_old_share_id
864                            .get(&m.message.sender)
865                            .expect("checked above that the sender is valid"),
866                        value: *m.message.vss_pk.coefficient(i as usize),
867                    };
868                    partial_keys.push(share);
869                }
870                PublicPoly::<G>::recover_c0_msm(old_t, partial_keys.iter())
871            })
872            .collect::<Result<Vec<_>, _>>()?;
873
874        Ok(Output {
875            nodes: self.nodes.clone(),
876            vss_pk: vss_pk.into(),
877            shares: Some(shares),
878        })
879    }

881    // Returns an error if the *complaint* is invalid (counterintuitive).
882    #[allow(clippy::too_many_arguments)]
883    fn check_complaint_proof<R: AllowedRng>(
884        recovery_pkg: &ecies_v1::RecoveryPackage<EG>,
885        receiver_pk: &ecies_v1::PublicKey<EG>,
886        receiver_id: PartyId,
887        share_ids: &[ShareIndex],
888        vss_pk: &PublicPoly<G>,
889        encryption: &ecies_v1::MultiRecipientEncryption<EG>,
890        recovery_random_oracle: &RandomOracle,
891        encryption_random_oracle: &RandomOracle,
892        rng: &mut R,
893    ) -> FastCryptoResult<()> {
894        // Check that the recovery package is valid, and if not, return an error since the complaint
895        // is invalid.
896        let buffer = encryption.decrypt_with_recovery_package(
897            recovery_pkg,
898            recovery_random_oracle,
899            encryption_random_oracle,
900            receiver_pk,
901            receiver_id as usize,
902        )?;
903
904        let decrypted_shares: Vec<G::ScalarType> = match bcs::from_bytes(buffer.as_slice()) {
905            Ok(s) => s,
906            Err(_) => {
907                debug!("DKG: check_complaint_proof failed to deserialize shares");
908                return Ok(());
909            }
910        };
911
912        if decrypted_shares.len() != share_ids.len() {
913            debug!("DKG: check_complaint_proof recovered invalid number of shares");
914            return Ok(());
915        }
916
917        let decrypted_shares = decrypted_shares
918            .into_iter()
919            .zip(share_ids)
920            .map(|(s, i)| Eval {
921                index: *i,
922                value: s,
923            })
924            .collect::<Vec<_>>();
925
926        match verify_poly_evals(&decrypted_shares, vss_pk, rng) {
927            Ok(()) => Err(FastCryptoError::InvalidProof),
928            Err(_) => {
929                debug!("DKG: check_complaint_proof failed to verify shares");
930                Ok(())
931            }
932        }
933    }

935    fn encryption_random_oracle(&self, sender: PartyId) -> RandomOracle {
936        self.random_oracle.extend(&format!("encs {}", sender))
937    }

939    fn recovery_random_oracle(&self, accuser: PartyId, accused: PartyId) -> RandomOracle {
940        self.random_oracle.extend(&format!(

942            accuser, accused

951    /// When the protocol is used in the sync setting with high thresholds, we need a dealer
952    /// to prove that it knows the shared secret.
953    /// The next function can be used to create such a proof.
954    pub fn nizk_pop_of_secret<R: AllowedRng>(&self, rng: &mut R) -> DLNizk<G>
955    where
956        G::ScalarType: FiatShamirChallenge,
957    {
958        let vss_pk = self.vss_sk.commit::<G>();
959        let ro = self.nizk_pop_of_secret_random_oracle(self.id);
960        DLNizk::create(self.vss_sk.c0(), vss_pk.c0(), &ro, rng)
961    }
