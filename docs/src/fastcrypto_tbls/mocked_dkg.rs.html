4use crate::dkg_v1::Output;
5use crate::nodes::{Nodes, PartyId};
6use crate::polynomial::PrivatePoly;
7use fastcrypto::groups::GroupElement;
8use serde::Serialize;
9
10/// Emulates the output of an insecure DKG protocol (to be used in tests).
11pub fn generate_mocked_output<G: GroupElement + Serialize, EG: GroupElement + Serialize>(
12    nodes: Nodes<EG>,
13    t: u32,
14    full_private_key: u128,
15    party: PartyId,
16) -> Output<G, EG> {
17    let mut coefficients: Vec<G::ScalarType> = (0..t)
18        .map(|i| G::ScalarType::from((i + 1) as u128))
19        .collect();
20    *coefficients.get_mut(0).unwrap() = G::ScalarType::from(full_private_key);
21
22    let poly = PrivatePoly::<G::ScalarType>::from(coefficients);
23    let vss_pk = poly.commit();
24
25    let shares = nodes
26        .share_ids_of(party)
27        .unwrap()
28        .iter()
29        .map(|sid| poly.eval(*sid))
30        .collect();
31
32    Output {
33        nodes,
34        vss_pk,
35        shares: Some(shares),
36    }
37}
