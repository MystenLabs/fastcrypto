15/// Generate partial threshold Schnorr signatures for a given message using a presigning tuple.
16/// The presigning tuple must be taken from a [Presignatures] iterator, the other parties should use the same tuple and one tuple may only be used once.
17/// Returns also the public nonce R.
18///
19/// The signatures produced follow the BIP-0340 standard (https://github.com/bitcoin/bips/blob/master/bip-0340.mediawiki).
20///
21/// If a derivation index is provided, a new verifying key is derived for this index (see
22/// [derive_verifying_key]), and the signature is adjusted accordingly.
23/// The signature will be valid for the derived verifying key.
24///
25/// `GeneralOpaqueError` is returned if the generated nonce R is the identity element (should happen only with negligible probability).
26/// `InvalidInput` is returned if the verifying key is the identity element.
27pub fn generate_partial_signatures(
28    message: &[u8],
29    (mut secret_presigs, public_presig): (Vec<S>, G),
30    beacon_value: &S,
31    my_signing_key_shares: &avss::SharesForNode,
32    verifying_key: &G,
33    derivation_address: Option<&Address>,
34) -> FastCryptoResult<(G, Vec<Eval<S>>)> {
35    let r_g = compute_nonce(&public_presig, beacon_value)?;
36
37    // In BIP-340, the nonce R must have an even Y coordinate.
38    // If it doesn't, we negate the secret nonce to get a new nonce R' = -R with an even Y.
39    // Since only the X coordinate of R is included in the signature, we don't need to change R, but we must negate the presigs.
40    if !r_g.has_even_y()? {
41        for presig in &mut secret_presigs {
42            *presig = -*presig;
43        }
44    }
45
46    // If a derivation index is provided, derive a new verifying key (and implicitly also signing key) for this index.
47    let verifying_key = if let Some(address) = derivation_address {
48        derive_verifying_key_internal(verifying_key, address)?
49    } else {
50        *verifying_key
51    };
52
53    // The verifying key must also have an even Y coordinate.
54    // If this is not the case, we must negate the verifying key (and hence also the signing key).
55    // Since the signing key shares are multiplied with the challenge, we just change the sign of the challenge instead.
56    let mut h = bip0340_hash(&r_g, &verifying_key, message)?;
57    if !verifying_key.has_even_y()? {
58        h = -h;
59    }
60
61    Ok((
62        public_presig,
63        my_signing_key_shares
64            .shares
65            .iter()
66            .zip_eq(secret_presigs)
67            .map(
68                |(
69                    Eval {
70                        index,
71                        value: sk_share,
72                    },
73                    presig,
74                )| Eval {
75                    index: *index,
76                    value: presig + h * sk_share,
77                },
78            )
79            .collect(),
80    ))
81}

83/// Given enough partial signatures, aggregate them into a full signature and verify it.
84/// The signature produced follows the BIP-0340 standard.
85///
86/// If a derivation index is provided, a new verifying key is derived for this index (see
87/// [derive_verifying_key]), and the signature is adjusted accordingly.
88/// The signature will be valid for the derived verifying key.
89///
90/// Returns an `InputTooShort` error if not enough partial signatures are provided.
91/// `GeneralOpaqueError` is returned if the computed nonce R is the identity element.
92/// `InvalidSignature` is returned if the aggregated signature does not verify.
93/// `InvalidInput` is returned if the provided verifying key is the identity element.
94pub fn aggregate_signatures(
95    message: &[u8],
96    public_presig: &G,
97    beacon_value: &S,
98    partial_signatures: &[Eval<S>],
99    threshold: u16,
100    verifying_key: &G,
101    derivation_address: Option<&Address>,
102) -> FastCryptoResult<SchnorrSignature> {
103    if partial_signatures.len() < threshold as usize {
104        return Err(InputTooShort(threshold as usize));
105    }
106
107    if !partial_signatures.iter().map(|s| s.index).all_unique() {
108        return Err(FastCryptoError::InvalidInput);
109    }
110
111    // Interpolate the partial signatures to get the full signature.
112    let s = Poly::recover_c0(
113        threshold,
114        partial_signatures.iter().take(threshold as usize),
115    )?;
116
117    finalize_schnorr_signature(
118        message,
119        public_presig,
120        beacon_value,
121        s,
122        verifying_key,
123        derivation_address,
124    )
125}

127/// Wrap an already-recovered signing scalar `s = f(0)` into a BIP-0340 Schnorr signature.
128///
129/// This is the second half of [aggregate_signatures], split out so callers that recover `s`
130/// through a different path (e.g. Reed–Solomon decoding, which yields `s` as the constant
131/// coefficient of the message polynomial) can reuse the BIP-0340 finalization without
132/// re-running Lagrange interpolation.
133///
134/// If a derivation index is provided, a new verifying key is derived for this index (see
135/// [derive_verifying_key]), and the signature is adjusted accordingly. The signature will
136/// be valid for the derived verifying key.
137///
138/// `GeneralOpaqueError` is returned if the computed nonce R is the identity element.
139/// `InvalidSignature` is returned if the aggregated signature does not verify.
140/// `InvalidInput` is returned if the provided verifying key is the identity element.
141pub fn finalize_schnorr_signature(
142    message: &[u8],
143    public_presig: &G,
144    beacon_value: &S,
145    mut s: S,
146    verifying_key: &G,
147    derivation_address: Option<&Address>,
148) -> FastCryptoResult<SchnorrSignature> {
149    // Compute the nonce R for the signature.
150    let r_g = compute_nonce(public_presig, beacon_value)?;
151
152    // In acc. with BIP-0340, we need to ensure the nonce R has an even Y coordinate.
153    // If it doesn't, we subtract the beacon value instead of adding it like it is done for the secret shares.
154    // We don't need to change R itself since only the X coordinate of this is used in the hash and signature below.
155    if r_g.has_even_y()? {
156        s += beacon_value
157    } else {
158        s -= beacon_value
159    };
160
161    // If a derivation index is provided, compute the derived verifying key and adjust the signature accordingly.
162    let verifying_key = if let Some(address) = derivation_address {
163        let tweak = compute_tweak(verifying_key, address)?;
164        let derived_vk = derive_verifying_key_internal(verifying_key, address)?;
165        let h = tweak * bip0340_hash(&r_g, &derived_vk, message)?;
166        if derived_vk.has_even_y()? {
167            s += h;
168        } else {
169            s -= h;
170        }
171        derived_vk
172    } else {
173        *verifying_key
174    };
175
176    let signature = SchnorrSignature::try_from((r_g, s))?;
177
178    // TODO: Handle invalid signatures
179    SchnorrPublicKey::try_from(&verifying_key)?.verify(message, &signature)?;
180
181    Ok(signature)
182}

184/// Compute the signature nonce `R = public_presig + G * beacon_value`. Since both inputs are
185/// random, the identity element occurs only with negligible probability and is rejected with
186/// [`FastCryptoError::GeneralOpaqueError`].
187fn compute_nonce(public_presig: &G, beacon_value: &S) -> FastCryptoResult<G> {
188    let r_g = *public_presig + G::generator() * beacon_value;
189    if r_g == G::zero() {
190        return Err(FastCryptoError::GeneralOpaqueError);
191    }
192    Ok(r_g)
193}
194
195fn bip0340_hash(r_g: &G, vk: &G, message: &[u8]) -> FastCryptoResult<S> {
196    Ok(bip0340_hash_to_scalar(
197        Tag::Challenge,
198        [&r_g.x_as_be_bytes()?, &vk.x_as_be_bytes()?, message],
199    ))
