4//! Generic Asynchronous Verifiable Information Dispersal (AVID).
5//!
6//! A dealer disperses one payload per recipient across weighted parties such that any `≥ W-2f`
7//! weight of authenticated shards can reconstruct it, while a Merkle commitment binds every shard
8//! to the dealer's broadcast. The set of recipients does not have to be all nodes.
9

12use crate::threshold_schnorr::reed_solomon::{ErasureCoder, Shard};
13use crate::threshold_schnorr::EG;
14use crate::types::get_uniform_value;

25/// AVID over a fixed node set and Byzantine bound `f`.
26pub struct Avid {
27    nodes: Arc<Nodes<EG>>,
28    coder: ErasureCoder,
29    f: u16,
30}

32/// The dealer's per-party dispersal message. One [AuthenticatedShards] per recipient.
33pub type Dispersal = BTreeMap<PartyId, AuthenticatedShards>;
34
35/// One disperser's shards for a recipient's payload with a two-level Merkle proof against the
36/// dispersal's `top_root` (row proof to the recipient's row root, top proof binding that row root
37/// to `top_root`).
38#[derive(Clone, Debug, Serialize, Deserialize)]
39pub struct AuthenticatedShards {
40    pub(crate) shards: Vec<Shard>,
41    pub(crate) proof: NestedMerkleProof,
42}
43
44/// An endorsement of a dispersal's `top_root`.
45#[derive(Clone, Debug, Serialize, Deserialize)]
46pub struct Vote {
47    pub top_root: merkle::Node,
48}

50/// A precomputed dispersal-side cache produced by [Avid::process_dispersal] to build [Echo]s.
51pub struct EchoBuilder {
52    dispersal: Dispersal,
53    pub top_root: merkle::Node,
54}

56/// One disperser's echo to a single recipient.
57#[derive(Clone, Debug, Serialize, Deserialize)]
58pub struct Echo {
59    authenticated_shards: AuthenticatedShards,
60}

62/// An [Echo] verified by [Avid::verify_echo].
63#[derive(Clone, Debug)]
64pub struct VerifiedEcho {
65    sender: PartyId,
66    echo: Echo,
67}

69/// A complaint that a dispersal is inconsistent, carrying the shards the accuser collected so that
70/// others can re-run the check.
71#[derive(Clone, Debug, Serialize, Deserialize)]
72pub struct Complaint {
73    pub shards: BTreeMap<PartyId, AuthenticatedShards>,
74}

76impl Avid {
77    /// Build an AVID instance over `nodes` with Byzantine bound `f`, constructing the
78    /// `(W, W − 2f)` Reed-Solomon coder once. Fails if `f == 0`, `W ≤ 2f`, or the RS parameters
79    /// are otherwise invalid.
80    pub fn new(nodes: Arc<Nodes<EG>>, f: u16) -> FastCryptoResult<Self> {
81        let total_weight = nodes.total_weight();
82        if f == 0 {
83            return Err(InvalidInput);
84        }
85        let k = total_weight.checked_sub(2 * f).ok_or(InvalidInput)?;
86        let coder = ErasureCoder::new(total_weight as usize, k as usize)?;
87        Ok(Self { nodes, coder, f })
88    }
89
90    /// 1. Disperse one payload per recipient. Returns one per-party [Dispersal]. Runs `mutate`
91    ///    over the per-recipient, per-disperser shards before the Merkle trees are built — for
92    ///    production callers pass `|_| {}`.
93    ///    Fails if any of the payloads are empty.
94    #[cfg_attr(not(test), allow(unused_variables, unused_mut))]
95    pub fn disperse_with_mutation(
96        &self,
97        payloads_by_recipient: &BTreeMap<PartyId, Vec<u8>>,
98        mutate: impl FnOnce(&mut BTreeMap<PartyId, Vec<Vec<Shard>>>),
99    ) -> FastCryptoResult<BTreeMap<PartyId, Dispersal>> {
100        // RS-encode each recipient's payload and bucket the shards by disperser.
101        let mut shards_by_recipient: BTreeMap<PartyId, Vec<Vec<Shard>>> = payloads_by_recipient
102            .iter()
103            .map(|(&i, payload)| {
104                let shards = self.coder.encode(payload)?;
105                let by_disperser = self.nodes.collect_to_nodes(shards.into_iter())?;
106                Ok((i, by_disperser))
107            })
108            .collect::<FastCryptoResult<_>>()?;
109
110        #[cfg(test)]
111        mutate(&mut shards_by_recipient);
112
113        // Two-level Merkle commitment: per-recipient row trees + a top tree over the row roots.
114        let tree = NestedMerkleTree::new(shards_by_recipient.values().cloned())?;
115
116        Ok(self
117            .nodes
118            .node_ids_iter()
119            .map(|j| {
120                let dispersal: Dispersal = shards_by_recipient
121                    .iter()
122                    .enumerate()
123                    .map(|(recipient_idx, (&i, by_disperser))| {
124                        (
125                            i,
126                            AuthenticatedShards {
127                                shards: by_disperser[j as usize].clone(),
128                                proof: tree
129                                    .get_proof(recipient_idx, j as usize)
130                                    .expect("valid leaf index"),
131                            },
132                        )
133                    })
134                    .collect();
135                (j, dispersal)
136            })
137            .collect())
138    }
139
140    /// 2. Verify a [Dispersal] and return an [EchoBuilder] that can produce individual [Echo]s on demand via
141    ///    [EchoBuilder::create_echo] and a [Vote] to return to the disperser. When the disperser
142    ///    has `W-f` weight of signed [Vote]s, he can publish the certified `top_root`.
143    pub fn process_dispersal(
144        &self,
145        my_id: PartyId,
146        dispersal: Dispersal,
147    ) -> FastCryptoResult<(EchoBuilder, Vote)> {
148        if dispersal.keys().any(|i| !self.nodes.is_valid_id(*i)) {
149            warn!("avid echo: dispersal contains an invalid recipient id");
150            return Err(InvalidMessage);
151        }
152
153        let implied_roots = dispersal
154            .iter()
155            .enumerate()
156            .map(|(recipient_idx, (_, shards))| {
157                shards
158                    .proof
159                    .derive_top_root(&shards.shards, recipient_idx, my_id as usize)
160                    .tap_err(|err| warn!("avid echo: implied root failed at leaf {my_id}: {err:?}"))
161            })
162            .collect::<FastCryptoResult<Vec<_>>>()?;
163        let top_root = get_uniform_value(implied_roots).ok_or(InvalidMessage)?;
164        Ok((
165            EchoBuilder {
166                dispersal,
167                top_root: top_root.clone(),
168            },
169            Vote { top_root },
170        ))
171    }

173    /// 3a. Verify an [Echo] addressed to `receiver`. Use the published `certified_top_root`.
174    pub fn verify_echo(
175        &self,
176        echo: Echo,
177        sender: PartyId,
178        certified_top_root: &merkle::Node,
179        pending_recipients: &BTreeSet<PartyId>,
180        receiver: PartyId,
181    ) -> FastCryptoResult<VerifiedEcho> {
182        let auth = &echo.authenticated_shards;
183        if auth.shards.len() != self.nodes.weight_of(sender)? as usize {
184            return Err(InvalidMessage);
185        }
186        let receiver_idx = pending_recipients
187            .iter()
188            .position(|&id| id == receiver)
189            .ok_or(InvalidInput)?;
190        auth.proof.verify(
191            certified_top_root,
192            &auth.shards,
193            receiver_idx,
194            sender as usize,
195        )?;
196        Ok(VerifiedEcho { echo, sender })
197    }
198
199    /// 3b. Reconstruct the caller's payload from a quorum of [VerifiedEcho]s, or raise a
200    ///     [Complaint]. Rejects duplicate dispersers, requires `≥ W − 2f` weight (the RS-decode
201    ///     minimum). With well-formed inputs returns `Ok(Ok(payload))` iff the shards decode to a
202    ///     payload that passes `payload_ok` and re-encoding it rebuilds a row tree whose root
203    ///     matches the dispersal's `recipient_root` (so the dealer's cells form a valid
204    ///     codeword). Otherwise `Ok(Err(Complaint))` over the shards.
205    pub fn decode_or_complain(
206        &self,
207        echoes: &[VerifiedEcho],
208        payload_ok: impl Fn(&[u8]) -> bool,
209    ) -> FastCryptoResult<Result<Vec<u8>, Complaint>> {
210        if echoes.is_empty() || !echoes.iter().map(|e| e.sender).all_unique() {
211            return Err(InvalidInput);
212        }
213        let required = self.required_weight();
214        if self
215            .nodes
216            .total_weight_of(echoes.iter().map(|e| &e.sender))?
217            < required
218        {
219            return Err(NotEnoughWeight(required as usize));
220        }
221        let shards: BTreeMap<PartyId, AuthenticatedShards> = echoes
222            .iter()
223            .cloned()
224            .map(|e| (e.sender, e.echo.authenticated_shards))
225            .collect();
226        let payload = match self.decode(&shards) {
227            Ok(p) if payload_ok(&p) => p,
228            _ => return Ok(Err(Complaint { shards })),
229        };
230
231        // Re-encode the payload and rebuild the row's Merkle tree.
232        let re_encoded: Vec<Vec<Shard>> = self
233            .nodes
234            .collect_to_nodes(self.coder.encode(&payload)?.into_iter())?;
235
236        // Take the expected recipient root from any verified echo's proof, since they all share the same top root and row index.
237        let (sender, authenticated_shards) =
238            shards.iter().next().expect("non-empty by check above");
239        let expected_recipient_root = authenticated_shards
240            .proof
241            .derive_row_root(&authenticated_shards.shards, *sender as usize)?;
242        if NestedMerkleTree::compute_row_root(&re_encoded)? != expected_recipient_root {
243            return Ok(Err(Complaint { shards }));
244        }
245
246        Ok(Ok(payload))
247    }

249    /// Check if `complaint` is a valid complaint against the dispersal.
250    pub fn complaint_is_valid(
251        &self,
252        complaint: &Complaint,
253        accuser_id: PartyId,
254        top_root: &merkle::Node,
255        pending_recipients: &BTreeSet<PartyId>,
256        payload_ok: impl Fn(&[u8]) -> bool,
257    ) -> FastCryptoResult<bool> {
258        let Some(accuser_idx) = pending_recipients.iter().position(|&id| id == accuser_id) else {
259            return Ok(false);
260        };
261        let shards_verify = complaint.shards.iter().all(|(&disperser, auth)| {
262            auth.proof
263                .verify(top_root, &auth.shards, accuser_idx, disperser as usize)
264                .is_ok()
265        });
266        if !shards_verify
267            || self.nodes.total_weight_of(complaint.shards.keys())? < self.required_weight()
268        {
269            return Ok(false);
270        }
271        Ok(!self
272            .decode(&complaint.shards)
273            .is_ok_and(|payload| payload_ok(&payload)))
274    }

276    /// RS-decode a payload from authenticated shard contributions keyed by disperser. Missing dispersers
277    /// and dispersers whose shard count doesn't match their weight are treated as erasures, so
278    /// decoding fails if those exceed `2f` weight.
279    fn decode(&self, shards: &BTreeMap<PartyId, AuthenticatedShards>) -> FastCryptoResult<Vec<u8>> {
280        let matrix = self
281            .nodes
282            .node_ids_iter()
283            .flat_map(|id| -> Vec<Option<Shard>> {
284                let weight = self.nodes.weight_of(id).expect("valid party id") as usize;
285                match shards.get(&id) {
286                    Some(auth) if auth.shards.len() == weight => {
287                        auth.shards.iter().cloned().map(Some).collect_vec()
288                    }
289                    _ => vec![None; weight],
290                }
291            })
292            .collect_vec();
293        self.coder.decode(matrix)
294    }
295
296    fn required_weight(&self) -> u16 {
297        self.nodes.total_weight() - 2 * self.f
298    }
299}

301impl EchoBuilder {
302    /// The recipients of the dispersal (sorted by [PartyId]).
303    pub fn recipients(&self) -> BTreeSet<PartyId> {
304        self.dispersal.keys().copied().collect()
305    }
306
307    /// Build an [Echo] for `recipient`. Returns [InvalidInput] if `recipient` isn't in the
308    /// prepared dispersal.
309    pub fn create_echo(&self, recipient: PartyId) -> FastCryptoResult<Echo> {
310        let authenticated_shards = self.dispersal.get(&recipient).ok_or(InvalidInput)?.clone();
311        Ok(Echo {
312            authenticated_shards,
313        })
314    }

324    fn nodes_with_weights(weights: &[u16]) -> Nodes<EG> {
325        let mut rng = thread_rng();
326        let nodes = weights
327            .iter()
328            .enumerate()
329            .map(|(id, &weight)| {
330                let sk = ecies_v1::PrivateKey::<EG>::new(&mut rng);
331                Node::<EG> {
332                    id: id as PartyId,
333                    pk: ecies_v1::PublicKey::<EG>::from_private_key(&sk),
334                    weight,
335                }
336            })
337            .collect();
338        Nodes::new(nodes).unwrap()
339    }

341    /// End-to-end happy path test
342    #[test]
343    fn disperse_and_reconstruct_random_bytes() {
344        let weights = [1u16, 2, 1, 1]; // total weight W = 5
345        let f = 1u16; // any W − 2f = 3 weight reconstructs
346        let nodes = nodes_with_weights(&weights);
347        let nodes = Arc::new(nodes);
348        let avid = Avid::new(Arc::clone(&nodes), f).unwrap();
349
350        // Random bytes to disperse to recipient 0.
351        let recipient = 0u16;
352        let mut payload = vec![0u8; 200];
353        thread_rng().fill_bytes(&mut payload);
354        let payloads: BTreeMap<PartyId, Vec<u8>> =
355            std::iter::once((recipient, payload.clone())).collect();
356
357        // 1. Dealer disperses: one message per party.
358        let messages = avid.disperse_with_mutation(&payloads, |_| {}).unwrap();
359        assert_eq!(messages.len(), weights.len());
360
361        // 2. Each party verifies its dispersal and emits the echoes it will send to others.
362        let mut top_root = None;
363        let mut recipients = BTreeSet::new();
364        let party_echoes: Vec<(PartyId, BTreeMap<PartyId, Echo>)> = messages
365            .into_iter()
366            .map(|(j, m)| {
367                let (builder, vote) = avid.process_dispersal(j, m).unwrap();
368                if top_root.is_none() {
369                    top_root = Some(vote.top_root);
370                    recipients = builder.recipients();
371                }
372                let echoes = builder
373                    .recipients()
374                    .iter()
375                    .map(|&r| (r, builder.create_echo(r).unwrap()))
376                    .collect();
377                (j, echoes)
378            })
379            .collect();
380        let top_root = top_root.unwrap();
381
382        // The recipient verifies the echoes addressed to it.
383        let echoes: Vec<VerifiedEcho> = party_echoes
384            .into_iter()
385            .map(|(sender, mut echoes)| {
386                let echo = echoes.remove(&recipient).unwrap();
387                avid.verify_echo(echo, sender, &top_root, &recipients, recipient)
388                    .unwrap()
389            })
390            .collect();
391
392        // 3. The recipient reconstructs its payload from the quorum of echoes.
393        let recovered = avid.decode_or_complain(&echoes, |_| true).unwrap().unwrap();
394        assert_eq!(recovered, payload);
395    }

397    #[test]
398    fn cheating_dealer_complaint() {
399        let weights = [1u16; 5]; // total weight W = 5
400        let f = 1u16; // any W − 2f = 3 weight reconstructs
401        let nodes = nodes_with_weights(&weights);
402        let nodes = Arc::new(nodes);
403        let avid = Avid::new(Arc::clone(&nodes), f).unwrap();
404
405        let recipient = 0u16;
406        let cheater = 4u16;
407        let mut payload = vec![0u8; 200];
408        thread_rng().fill_bytes(&mut payload);
409        let payloads: BTreeMap<PartyId, Vec<u8>> =
410            std::iter::once((recipient, payload.clone())).collect();
411
412        let messages = avid
413            .disperse_with_mutation(&payloads, |shards| {
414                shards.get_mut(&recipient).unwrap()[cheater as usize][0].0[0] ^= 1;
415            })
416            .unwrap();
417
418        let mut top_root = None;
419        let mut recipients = BTreeSet::new();
420        let party_echoes: Vec<(PartyId, BTreeMap<PartyId, Echo>)> = messages
421            .into_iter()
422            .map(|(j, m)| {
423                let (builder, vote) = avid.process_dispersal(j, m).unwrap();
424                if top_root.is_none() {
425                    top_root = Some(vote.top_root);
426                    recipients = builder.recipients();
427                }
428                let echoes = builder
429                    .recipients()
430                    .iter()
431                    .map(|&r| (r, builder.create_echo(r).unwrap()))
432                    .collect();
433                (j, echoes)
434            })
435            .collect();
436        let top_root = top_root.unwrap();
437
438        // The recipient gathers a quorum of echoes including the cheater's ...
439        let _ = cheater;
440        let echoes: Vec<VerifiedEcho> = party_echoes
441            .into_iter()
442            .map(|(sender, mut echoes)| {
443                let echo = echoes.remove(&recipient).unwrap();
444                avid.verify_echo(echo, sender, &top_root, &recipients, recipient)
445                    .unwrap()
446            })
447            .collect();
448
449        // ... but the cheater's shards don't fit any codeword with the rest, so it raises a
450        // Complaint.
451        let complaint = avid
452            .decode_or_complain(&echoes, |_| true)
453            .unwrap()
454            .unwrap_err();
455
456        // Another party validates the complaint.
457        assert!(avid
458            .complaint_is_valid(&complaint, recipient, &top_root, &recipients, |_| true)
459            .unwrap());
460    }
