4//! Implementation of an asynchronous verifiable secret sharing (AVSS) protocol to distribute shares for a secret to a set of receivers.
5//! A receiver can verify that the secret being shared is the same as a share from a previous round (e.g., the secret key share of a threshold signature).
6//!
7//! Before the protocol starts, the following setup is needed:
8//! * Each receiver has an encryption key pair (ECIES) and these public keys are known to all parties.
9//! * The public keys along with the weights of each receiver are known to all parties and defined in the [Nodes] structure.
10//! * Define a new [crate::threshold_schnorr::Dealer] with the secrets, who begins by calling [crate::threshold_schnorr::Dealer::create_message].
11

30/// This represents a Dealer in the AVSS. There is exactly one dealer, who creates the shares and broadcasts the encrypted shares.
31#[allow(dead_code)]
32pub struct Dealer {
33    t: u16,
34    nodes: Nodes<EG>,
35    sid: Vec<u8>,
36    secret: Option<S>,
37}

39#[allow(dead_code)]
40pub struct Receiver {

43    nodes: Nodes<EG>,
44    commitment: Option<G>,
45    sid: Vec<u8>,

49/// The message broadcast by the dealer, containing the encrypted shares and the public keys of the nonces.
50#[derive(Clone, Debug, Serialize, Deserialize)]
51pub struct Message {
52    ciphertext: MultiRecipientEncryption<EG>,
53    feldman_commitment: Poly<G>,
54}
55
56/// The result of a [Receiver] processing a [Message]: Either valid shares or a complaint.
57#[allow(clippy::large_enum_variant)] // Clippy complains because ReceiverOutput can be very small if BATCH_SIZE is small.
58pub enum ProcessedMessage {
59    Valid(ReceiverOutput),
60    Complaint(Complaint),
61}
62
63/// The output of a receiver: The shares for each nonce + commitments for the next round.
64#[derive(Debug, Clone, Serialize, Deserialize)]
65pub struct ReceiverOutput {
66    pub my_shares: SharesForNode,
67
68    /// The commitments to the polynomials will be used for key rotation.
69    pub commitments: Vec<Eval<G>>,
70}
71
72/// All the shares given to a node. One share per the node's weight.
73/// These can be created either by decrypting the shares from the dealer (see [Receiver::process_message]) or by recovering them from complaint responses.
74#[derive(Clone, Debug, Serialize, Deserialize)]
75pub struct SharesForNode {
76    pub shares: Vec<Eval<S>>,
77}
78
79impl SharesForNode {
80    /// Get the weight of this node (number of shares it has).
81    pub fn weight(&self) -> usize {
82        self.shares.len()
83    }
84
85    fn verify(&self, message: &Message) -> FastCryptoResult<()> {
86        for share in &self.shares {
87            message
88                .feldman_commitment
89                .verify_share(share.index, &share.value)?
90        }
91        Ok(())
92    }
93
94    /// Assuming that enough shares are given, recover the shares for this node.
95    fn recover(
96        indices: Vec<ShareIndex>,
97        threshold: u16,
98        other_shares: &[Self],
99    ) -> FastCryptoResult<Self> {
100        // Compute the total weight of the valid responses
101        let response_weight = other_shares
102            .iter()
103            .map(SharesForNode::weight)
104            .sum::<usize>();
105        if response_weight < threshold as usize {
106            return Err(FastCryptoError::GeneralError(
107                "Not enough valid responses".to_string(),
108            ));
109        }
110
111        let shares = indices
112            .into_iter()
113            .map(|index| {
114                let evaluations = other_shares
115                    .iter()
116                    .flat_map(|share| share.shares.clone())
117                    .collect_vec();
118                Poly::interpolate_at_index(index, &evaluations).unwrap()
119            })
120            .collect_vec();
121
122        Ok(Self { shares })
123    }
124}

128impl Dealer {
129    /// Create a new dealer.
130    ///
131    /// * `secret`: The secret to share. If None, a random secret will be generated.
132    /// * `nodes`: The set of nodes (parties) participating in the protocol, including their public keys and weights.
133    /// * `t`: The threshold number of shares required to reconstruct the secret. One party can have multiple shares according to its weight.
134    /// * `f`: An upper bound on the number of Byzantine parties counted by weight.
135    /// * `sid`: A session identifier that should be unique for each invocation of the protocol but the same for all parties in a single invocation.
136    pub fn new(
137        secret: Option<S>,
138        nodes: Nodes<EG>,
139        t: u16,
140        f: u16,
141        sid: Vec<u8>,
142    ) -> FastCryptoResult<Self> {
143        // We need to collect t+f confirmations to make sure that at least t honest parties have confirmed.
144        if t <= f || t + 2 * f > nodes.total_weight() {
145            return Err(InvalidInput);
146        }
147
148        Ok(Self {
149            secret,
150            t,
151            nodes,
152            sid,
153        })
154    }
155
156    /// 1. The Dealer samples nonces, generates shares and broadcasts the encrypted shares.
157    pub fn create_message<Rng: AllowedRng>(&self, rng: &mut Rng) -> FastCryptoResult<Message> {
158        let secret = self.secret.unwrap_or(S::rand(rng));
159        let polynomial = Poly::rand_fixed_c0(self.t - 1, secret, rng);
160
161        // Encrypt all shares to the receivers
162        let pk_and_msgs = self
163            .nodes
164            .iter()
165            .map(|node| (node.pk.clone(), self.nodes.share_ids_of(node.id).unwrap()))
166            .map(|(public_key, share_ids)| {
167                (
168                    public_key,
169                    SharesForNode {
170                        shares: share_ids
171                            .into_iter()
172                            .map(|index| polynomial.eval(index))
173                            .collect_vec(),
174                    }
175                    .to_bytes(),
176                )
177            })
178            .collect_vec();
179
180        let ciphertext = MultiRecipientEncryption::encrypt(
181            &pk_and_msgs,
182            &self.random_oracle().extend(&Encryption.to_string()),
183            rng,
184        );
185
186        Ok(Message {
187            ciphertext,
188            feldman_commitment: polynomial.commit(),
189        })
190    }
191
192    fn random_oracle(&self) -> RandomOracle {
193        random_oracle_from_sid(&self.sid)
194    }
195}

197impl Receiver {
198    /// Create a new receiver.
199    ///
200    /// * `nodes`: The set of nodes (parties) participating in the protocol, including their public keys and weights.
201    /// * `id`: The unique identifier of this receiver. Should match one of the party ids in `nodes`.
202    /// * `t`: The threshold number of shares required to reconstruct the secret. One party can have multiple shares according to its weight.
203    /// * `sid`: A session identifier that should be unique for each invocation of the protocol but the same for all parties in a single invocation.
204    /// * `commitment`: A commitment to the secret being shared. This should be equal to `secret * G` and is typically found as the commitment from a previous round (see [ReceiverOutput]). If None, no consistency check will be performed.
205    /// * `enc_secret_key`: The private key used to decrypt the shares sent to this receiver.
206    pub fn new(
207        nodes: Nodes<EG>,
208        id: PartyId,
209        t: u16,
210        sid: Vec<u8>,
211        commitment: Option<G>,
212        enc_secret_key: PrivateKey<EG>,
213    ) -> Self {
214        Self {
215            id,
216            enc_secret_key,
217            commitment,
218            sid,
219            t,
220            nodes,
221        }
222    }
223
224    /// 2. Each receiver processes the message, verifies and decrypts its shares.
225    ///
226    /// If this works, the receiver can store the shares and contribute a signature on the message to a certificate.
227    ///
228    /// This returns an [InvalidMessage] error if the ciphertext cannot be verified, if the commitments are invalid or do not match the commitments from a previous round.
229    /// All honest receivers will reject such a message with the same error, and such a message should be ignored.
230    ///
231    /// If the message is valid but contains invalid shares for this receiver, the call will succeed but will return a [Complaint].
232    ///
233    /// 3. When t+f signatures have been collected in the certificate, the receivers can now verify the certificate and finish the protocol.
234    pub fn process_message(&self, message: &Message) -> FastCryptoResult<ProcessedMessage> {
235        if message.feldman_commitment.degree() != self.t as usize - 1 {
236            return Err(InvalidMessage);
237        }
238
239        // If a commitment is given, verify that the secret the dealer is distributing is consistent
240        if let Some(c) = &self.commitment {
241            if message.feldman_commitment.c0() != c {
242                return Err(InvalidMessage);
243            }
244        }
245
246        let random_oracle_encryption = self.random_oracle().extend(&Encryption.to_string());
247        message
248            .ciphertext
249            .verify(&random_oracle_encryption)
250            .map_err(|_| InvalidMessage)?;
251
252        let plaintext = message.ciphertext.decrypt(
253            &self.enc_secret_key,
254            &random_oracle_encryption,
255            self.id as usize,
256        );
257
258        match SharesForNode::from_bytes(&plaintext).and_then(|my_shares| {
259            if my_shares.weight() != self.my_weight() {
260                return Err(InvalidInput);
261            }
262            my_shares.verify(message)?;
263            Ok(my_shares)
264        }) {
265            Ok(my_shares) => Ok(ProcessedMessage::Valid(ReceiverOutput {
266                my_shares,
267                commitments: self.compute_commitments(message),
268            })),
269            Err(_) => Ok(ProcessedMessage::Complaint(Complaint::create(
270                self.id,
271                &message.ciphertext,
272                &self.enc_secret_key,
273                &self.random_oracle(),
274                &mut rand::thread_rng(),
275            ))),
276        }
277    }
278
279    /// 4. Upon receiving a complaint, a receiver verifies it and responds with a recovery package for the shares of the accuser.
280    pub fn handle_complaint(
281        &self,
282        message: &Message,
283        complaint: &Complaint,
284        rng: &mut impl AllowedRng,
285    ) -> FastCryptoResult<ComplaintResponse> {
286        complaint.check(
287            &self.nodes.node_id_to_node(complaint.accuser_id)?.pk,
288            &message.ciphertext,
289            &self.random_oracle(),
290            |shares: &SharesForNode| shares.verify(message),
291        )?;
292        Ok(ComplaintResponse::create(
293            self.id,
294            &message.ciphertext,
295            &self.enc_secret_key,
296            &self.random_oracle(),
297            rng,
298        ))
299    }
300
301    /// 5. Upon receiving t valid responses to a complaint, the accuser can recover its shares.
302    ///    Fails if there are not enough valid responses to recover the shares or if any of the responses come from an invalid party.
303    pub fn recover(
304        &self,
305        message: &Message,
306        responses: &[ComplaintResponse],
307    ) -> FastCryptoResult<ReceiverOutput> {
308        // TODO: This fails if one of the responses has an invalid responder_id. We could probably just ignore those instead.
309
310        // Sanity check that we have enough responses (by weight) to recover the shares.
311        let total_response_weight = self
312            .nodes
313            .total_weight_of(responses.iter().map(|response| &response.responder_id))?;
314        if total_response_weight < self.t {
315            return Err(FastCryptoError::InputTooShort(self.t as usize));
316        }
317
318        let response_shares = responses
319            .iter()
320            .filter_map(|response| {
321                self.nodes
322                    .node_id_to_node(response.responder_id)
323                    .and_then(|node| {
324                        response.decrypt_with_response(
325                            &self.random_oracle(),
326                            &node.pk,
327                            &message.ciphertext,
328                        )
329                    })
330                    .and_then(|shares: SharesForNode| {
331                        // Verify the shares are valid
332                        shares.verify(message).map(|_| shares)
333                    })
334                    .tap_err(|_| {
335                        warn!(
336                            "Ignoring invalid recovery package from {}",
337                            response.responder_id
338                        )
339                    })
340                    .ok()
341            })
342            .collect_vec();
343
344        let my_shares = SharesForNode::recover(self.my_indices(), self.t, &response_shares)?;
345        my_shares.verify(message)?;
346
347        Ok(ReceiverOutput {
348            my_shares,
349            commitments: self.compute_commitments(message),
350        })
351    }
352
353    fn compute_commitments(&self, message: &Message) -> Vec<Eval<G>> {
354        self.nodes
355            .share_ids_iter()
356            .map(|index| message.feldman_commitment.eval(index))
357            .collect()
358    }
359
360    pub fn my_indices(&self) -> Vec<ShareIndex> {
361        self.nodes.share_ids_of(self.id).unwrap()
362    }
363
364    pub fn my_weight(&self) -> usize {
365        self.nodes
366            .total_weight_of(std::iter::once(&self.id))
367            .unwrap() as usize
368    }
369
370    fn random_oracle(&self) -> RandomOracle {
371        random_oracle_from_sid(&self.sid)
372    }
373}

573        let n = 7;

651    impl Dealer {
652        pub fn create_message_cheating<Rng: AllowedRng>(
653            &self,
654            rng: &mut Rng,
655        ) -> FastCryptoResult<Message> {
656            let secret = self.secret.unwrap_or(S::rand(rng));
657            let polynomial = Poly::rand_fixed_c0(self.t - 1, secret, rng);
658            let commitment = polynomial.commit();
659
660            // Encrypt all shares to the receivers
661            let mut pk_and_msgs = self
662                .nodes
663                .iter()
664                .map(|node| (node.pk.clone(), self.nodes.share_ids_of(node.id).unwrap()))
665                .map(|(public_key, share_ids)| {
666                    (
667                        public_key,
668                        SharesForNode {
669                            shares: share_ids
670                                .into_iter()
671                                .map(|index| polynomial.eval(index))
672                                .collect_vec(),
673                        }
674                        .to_bytes(),
675                    )
676                })
677                .collect_vec();
678
679            // Modify the first share of the first receiver to simulate a cheating dealer
680            pk_and_msgs[0].1[7] += 1;
681
682            let ciphertext = MultiRecipientEncryption::encrypt(
683                &pk_and_msgs,
684                &self.random_oracle().extend(&Encryption.to_string()),
685                rng,
686            );
687
688            Ok(Message {
689                ciphertext,
690                feldman_commitment: commitment,
691            })
692        }
693    }
