4//! Implementation of an asynchronous verifiable secret sharing (AVSS) protocol to distribute shares for a secret to a set of receivers.
5//! A receiver can verify that the secret being shared is the same as a share from a previous round (e.g., the secret key share of a threshold signature).
6//!
7//! Before the protocol starts, the following setup is needed:
8//! * Each receiver has an encryption key pair (ECIES) and these public keys are known to all parties.
9//! * The public keys along with the weights of each receiver are known to all parties and defined in the [Nodes] structure.
10//! * Define a new [crate::threshold_schnorr::Dealer] with the secrets, who begins by calling [crate::threshold_schnorr::Dealer::create_message].
11

56/// The message broadcast by the dealer, containing the encrypted shares and the public keys of the nonces.
57#[derive(Clone, Debug, Serialize, Deserialize)]
58pub struct Message {
59    feldman_commitment: Poly<G>,
60    ciphertext: MultiRecipientEncryption<EG>,
61}

63/// The result of a [Receiver] processing a [Message]: Either valid shares or a complaint.
64#[allow(clippy::large_enum_variant)] // Clippy complains because DkOutput can be very small if BATCH_SIZE is small.
65pub enum ProcessedMessage {
66    Valid(AvssOutput),
67    Complaint(Complaint),
68}

70/// A complaint by a receiver who could not decrypt or verify its shares from the dealer's
71/// broadcast. Given enough responses, the accuser can recover its shares.
72///
73/// The accuser's id is not carried here; the higher-level protocol tracks which party a complaint
74/// came from and passes it to [Receiver::handle_complaint].
75#[derive(Clone, Debug, Serialize, Deserialize)]
76pub struct Complaint {
77    pub proof: RecoveryProof,
78}

80/// A response to a [Complaint], containing the responder's shares so the accuser can
81/// Lagrange-interpolate their own.
82///
83/// The responder's id is not carried here; the higher-level protocol tracks which party a response
84/// came from and passes it to [Receiver::verify_complaint_response].
85#[derive(Debug, Clone, Serialize, Deserialize)]
86pub struct ComplaintResponse {
87    pub shares: SharesForNode,
88}
89
90/// A [ComplaintResponse] whose shares have been verified against the dealer's [Message] and bound
91/// to the responder that sent it. Created only by [Receiver::verify_complaint_response] and
92/// consumed by [Receiver::recover], which can therefore trust its contents.
93#[derive(Debug, Clone)]
94pub struct VerifiedComplaintResponse {
95    responder_id: PartyId,
96    shares: SharesForNode,
97}

120/// A single share: an evaluation of the secret-sharing polynomial at a share index.
121pub type Share = Eval<S>;
122
123/// All the shares given to a node. One share per the node's weight.
124/// These can be created either by decrypting the shares from the dealer (see [Receiver::process_message]) or by recovering them from complaint responses.
125#[derive(Clone, Debug, Serialize, Deserialize)]
126pub struct SharesForNode {
127    pub shares: Vec<Share>,
128}
129
130impl SharesForNode {
131    /// Get the weight of this node (number of shares it has).
132    pub fn weight(&self) -> usize {
133        self.shares.len()
134    }
135
136    /// Get this node's share at `index`, if it has one.
137    pub fn share_for_index(&self, index: ShareIndex) -> Option<&Eval<S>> {
138        self.shares.iter().find(|s| s.index == index)
139    }
140
141    fn verify(&self, message: &Message) -> FastCryptoResult<()> {
142        for share in &self.shares {
143            // TODO: possible optimization - all shares get be verified at once
144            message
145                .feldman_commitment
146                .verify_share(share.index, &share.value)?
147        }
148        Ok(())
149    }
150
151    /// Assuming that enough shares are given, recover the shares for this node.
152    fn recover(
153        indices: Vec<ShareIndex>,
154        threshold: u16,
155        other_shares: &[Self],
156    ) -> FastCryptoResult<Self> {
157        if !indices.iter().all_unique() {
158            return Err(InvalidInput);
159        }
160
161        let evaluations = other_shares
162            .iter()
163            .flat_map(|share| share.shares.iter().cloned())
164            .collect_vec();
165        if !evaluations.iter().map(|e| e.index).all_unique() {
166            return Err(InvalidInput);
167        }
168        if evaluations.len() < threshold as usize {
169            return Err(FastCryptoError::GeneralError(
170                "Not enough valid responses".to_string(),
171            ));
172        }
173        let evaluations = evaluations
174            .into_iter()
175            .take(threshold as usize)
176            .collect_vec();
177
178        let shares = indices
179            .into_iter()
180            .map(|index| Poly::recover_at(threshold, index, &evaluations))
181            .collect::<FastCryptoResult<Vec<_>>>()?;
182
183        Ok(Self { shares })
184    }
185}

189impl Dealer {
190    /// Create a new dealer.
191    ///
192    /// * `secret`: The secret to share. If None, a random secret is sampled from `rng`.
193    /// * `nodes`: The set of nodes (parties) participating in the protocol, including their public keys and weights.
194    /// * `params`: The threshold parameters.
195    /// * `sid`: A session identifier that should be unique for each invocation of the protocol (for each dealer).
196    /// * `rng`: Used to sample the secret when `secret` is None.
197    ///
198    /// Returns an error if `t` is zero or larger than the total weight of the nodes.
199    pub fn new<R: AllowedRng>(
200        secret: Option<S>,
201        nodes: Nodes<EG>,
202        params: Parameters,
203        sid: Vec<u8>,
204        rng: &mut R,
205    ) -> FastCryptoResult<Self> {
206        params.validate(nodes.total_weight())?;
207        Ok(Self {
208            secret: secret.unwrap_or_else(|| S::rand(rng)),
209            params,
210            nodes,
211            sid,
212        })
213    }
214
215    /// 1. The Dealer samples nonces, generates shares and broadcasts the encrypted shares.
216    pub fn create_message<Rng: AllowedRng>(&self, rng: &mut Rng) -> Message {
217        let polynomial = Poly::rand_fixed_c0(self.params.t - 1, self.secret, rng);
218
219        // Compute all shares
220        let all_shares = polynomial.eval_range(self.nodes.total_weight());
221
222        // Encrypt all shares to the receivers
223        let pk_and_msgs = self
224            .nodes
225            .iter()
226            .map(|node| (node.pk.clone(), self.nodes.share_ids_of(node.id).unwrap()))
227            .map(|(public_key, share_ids)| {
228                (
229                    public_key,
230                    SharesForNode {
231                        shares: share_ids
232                            .into_iter()
233                            .map(|index| all_shares.get_eval(index))
234                            .collect_vec(),
235                    }
236                    .to_bytes(),
237                )
238            })
239            .collect_vec();
240
241        let ciphertext = MultiRecipientEncryption::encrypt(
242            &pk_and_msgs,
243            &self.random_oracle().extend(&Encryption.to_string()),
244            rng,
245        );
246
247        Message {
248            ciphertext,
249            feldman_commitment: polynomial.commit(),
250        }
251    }
252
253    fn random_oracle(&self) -> RandomOracle {
254        random_oracle_from_sid(&self.sid)
255    }
256}

258impl Receiver {
259    /// Create a new receiver.
260    ///
261    /// * `nodes`: The set of nodes (parties) participating in the protocol, including their public keys and weights.
262    /// * `id`: The unique identifier of this receiver. Should match one of the party ids in `nodes`.
263    /// * `params`: The threshold parameters. Only `params.t` is used by the AVSS; `params.f` is carried for consistency with the other protocols.
264    /// * `sid`: A session identifier that should be unique for each invocation of the protocol but the same for all parties in a single invocation.
265    /// * `commitment`: A commitment to the secret being shared. This should be equal to `secret * G` and is typically found as the commitment from a previous round (see [DkOutput]). If None, no consistency check will be performed.
266    /// * `enc_secret_key`: The private key used to decrypt the shares sent to this receiver.
267    pub fn new(
268        nodes: Nodes<EG>,
269        id: PartyId,
270        params: Parameters,
271        sid: Vec<u8>,
272        commitment: Option<G>,
273        enc_secret_key: PrivateKey<EG>,
274    ) -> FastCryptoResult<Self> {
275        // Validate that `id` is one of the parties in `nodes`.
276        nodes.node_id_to_node(id)?;
277        Ok(Self {
278            id,
279            enc_secret_key,
280            commitment,
281            sid,
282            params,
283            nodes,
284        })
285    }
286
287    pub fn id(&self) -> PartyId {
288        self.id
289    }
290
291    /// 2. Each receiver processes the message, verifies and decrypts its shares.
292    ///
293    /// If this works, the receiver can store the shares and contribute a signature on the message to a certificate.
294    ///
295    /// This returns an [InvalidMessage] error if the ciphertext cannot be verified, if the commitments are invalid or do not match the commitments from a previous round.
296    /// All honest receivers will reject such a message with the same error, and such a message should be ignored.
297    ///
298    /// If the message is valid but contains invalid shares for this receiver, the call will succeed but will return a [Complaint].
299    /// `rng` is used to create the recovery package attached to such a complaint.
300    ///
301    /// 3. When t+f signatures have been collected in the certificate, the receivers can now verify the certificate and finish the protocol.
302    pub fn process_message<R: AllowedRng>(
303        &self,
304        message: &Message,
305        rng: &mut R,
306    ) -> FastCryptoResult<ProcessedMessage> {
307        if message.feldman_commitment.degree() + 1 != self.params.t as usize {
308            warn!(
309                "AVSS process_message: invalid feldman commitment degree {} (expected {})",
310                message.feldman_commitment.degree(),
311                self.params.t as usize - 1,
312            );
313            return Err(InvalidMessage);
314        }
315
316        // If a commitment is given, verify that the secret the dealer is distributing is consistent
317        if let Some(c) = &self.commitment {
318            if message.feldman_commitment.c0() != *c {
319                warn!(
320                    "AVSS process_message: feldman commitment c0 does not match the expected commitment from a previous round"
321                );
322                return Err(InvalidMessage);
323            }
324        }
325
326        if message.ciphertext.len() != self.nodes.num_nodes() {
327            warn!("AVSS process_message: ciphertext has the wrong number of recipients");
328            return Err(InvalidMessage);
329        }
330
331        let random_oracle_encryption = self.random_oracle().extend(&Encryption.to_string());
332        message
333            .ciphertext
334            .verify(&random_oracle_encryption)
335            .map_err(|e| {
336                warn!("AVSS process_message: ciphertext verification failed: {e:?}");
337                InvalidMessage
338            })?;
339
340        let plaintext = message.ciphertext.decrypt(
341            &self.enc_secret_key,
342            &random_oracle_encryption,
343            self.id as usize,
344        );
345
346        match SharesForNode::from_bytes(&plaintext).and_then(|my_shares| {
347            verify_shares(
348                &my_shares,
349                &self.nodes.share_ids_of(self.id)?,
350                self.id,
351                message,
352            )?;
353            Ok(my_shares)
354        }) {
355            Ok(my_shares) => Ok(ProcessedMessage::Valid(AvssOutput {
356                my_shares,
357                feldman_commitment: message.feldman_commitment.clone(),
358            })),
359            Err(_) => Ok(ProcessedMessage::Complaint(Complaint {
360                proof: RecoveryProof::create(
361                    self.id,
362                    &message.ciphertext.shared(),
363                    &self.enc_secret_key,
364                    &self.random_oracle(),
365                    rng,
366                ),
367            })),
368        }
369    }
370
371    /// 4. Upon receiving a complaint, a receiver verifies it and responds with its shares.
372    ///
373    /// `accuser_id` is the party that raised the complaint (tracked by the caller, see [Complaint]).
374    /// Assumes the dealer's `message` has already been validated via [Receiver::process_message].
375    pub fn handle_complaint(
376        &self,
377        message: &Message,
378        accuser_id: PartyId,
379        complaint: &Complaint,
380        my_output: &AvssOutput,
381    ) -> FastCryptoResult<ComplaintResponse> {
382        let accuser_share_ids = self.nodes.share_ids_of(accuser_id)?;
383        complaint.proof.check(
384            accuser_id,
385            &self.nodes.node_id_to_node(accuser_id)?.pk,
386            message
387                .ciphertext
388                .encs
389                .get(accuser_id as usize)
390                .ok_or(InvalidInput)?,
391            &message.ciphertext.shared(),
392            &self.random_oracle(),
393            |shares: &SharesForNode| verify_shares(shares, &accuser_share_ids, accuser_id, message),
394        )?;
395        Ok(ComplaintResponse {
396            shares: my_output.my_shares.clone(),
397        })
398    }
399
400    /// Verify a [ComplaintResponse] received from `responder_id` against the dealer's `message`,
401    /// binding the verified shares to the responder. The resulting [VerifiedComplaintResponse]s are
402    /// the input to [Receiver::recover].
403    pub fn verify_complaint_response(
404        &self,
405        message: &Message,
406        responder_id: PartyId,
407        response: ComplaintResponse,
408    ) -> FastCryptoResult<VerifiedComplaintResponse> {
409        verify_shares(
410            &response.shares,
411            &self.nodes.share_ids_of(responder_id)?,
412            responder_id,
413            message,
414        )?;
415        Ok(VerifiedComplaintResponse {
416            responder_id,
417            shares: response.shares,
418        })
419    }
420
421    /// 5. Upon receiving enough verified responses (by weight) to a complaint, the accuser can
422    ///    recover its shares. Each response must first be verified with
423    ///    [Receiver::verify_complaint_response].
424    ///
425    ///    Fails if the responses do not come from distinct parties or if their combined weight is
426    ///    below the threshold `t`.
427    pub fn recover(
428        &self,
429        message: &Message,
430        responses: Vec<VerifiedComplaintResponse>,
431    ) -> FastCryptoResult<AvssOutput> {
432        // Responses must come from distinct parties, otherwise a single responder could be counted
433        // multiple times towards the threshold.
434        if !responses.iter().map(|r| r.responder_id).all_unique() {
435            return Err(InvalidInput);
436        }
437
438        // The responses are already verified, so we only need enough weight to interpolate.
439        let total_response_weight = self
440            .nodes
441            .total_weight_of(responses.iter().map(|r| &r.responder_id))?;
442        if total_response_weight < self.params.t {
443            return Err(FastCryptoError::InputTooShort(self.params.t as usize));
444        }
445
446        let valid_shares = responses.into_iter().map(|r| r.shares).collect_vec();
447        let my_shares = SharesForNode::recover(self.my_indices(), self.params.t, &valid_shares)?;
448
449        // The recovered shares are interpolated from already-verified shares, so this should never
450        // fail; if it does, something is seriously wrong.
451        my_shares.verify(message).tap_err(|e| {
452            warn!(
453                "AVSS recover: recovered shares failed verification, this should never happen: {e:?}"
454            );
455        })?;
456
457        Ok(AvssOutput {
458            my_shares,
459            feldman_commitment: message.feldman_commitment.clone(),
460        })
461    }
462
463    pub fn my_indices(&self) -> Vec<ShareIndex> {
464        self.nodes.share_ids_of(self.id).unwrap()
465    }
466
467    pub fn my_weight(&self) -> usize {
468        self.nodes
469            .total_weight_of(std::iter::once(&self.id))
470            .unwrap() as usize
471    }
472
473    fn random_oracle(&self) -> RandomOracle {
474        random_oracle_from_sid(&self.sid)
475    }
476}

478/// Verify a set of shares received from a Dealer: that the share indices are exactly
479/// `expected_share_ids` and that each share is consistent with the dealer's commitment.
480fn verify_shares(
481    shares: &SharesForNode,
482    expected_share_ids: &[ShareIndex],
483    receiver: PartyId,
484    message: &Message,
485) -> FastCryptoResult<()> {
486    // TODO: this function returs err both in case verify failed and in case there is a bug in the impl.
487    // We should decide on a consistent error handling strategy here
488
489    if !shares
490        .shares
491        .iter()
492        .map(|s| s.index)
493        .eq(expected_share_ids.iter().copied())
494    {
495        warn!(
496            "AVSS verify_shares: share indices do not match the receiver's assigned indices for receiver {}",
497            receiver,
498        );
499        return Err(InvalidMessage);
500    }
501    shares.verify(message).tap_err(|e| {
502        warn!(
503            "AVSS verify_shares: cryptographic share verification failed for receiver {}: {e:?}",
504            receiver,
505        );
506    })
507}

509impl DkOutput {

512    }

515        self.commitments.iter().find(|c| c.index == index)
516    }

518    /// Combine multiple outputs from different dealers into a single output by summing.
519    /// This is used after a successful AVSS used for DKG to combine the shares from multiple dealers into a single share for each party.
520    /// Panics if the given `DkOutput`s are not compatible (same weight, same indices, same number of commitments)
521    /// Returns the combined output, including the joint verifying key
522    pub fn complete_dkg(
523        t: u16,
524        nodes: &Nodes<EG>,
525        outputs: HashMap<PartyId, AvssOutput>,
526    ) -> FastCryptoResult<Self> {
527        if nodes.total_weight_of(outputs.keys())? < t {
528            return Err(NotEnoughWeight(t as usize));
529        }
530
531        let outputs = outputs.into_values().collect_vec();
532
533        // Sanity check: Outputs cannot be empty and all outputs must have the same weight.
534        if outputs.is_empty() || !outputs.iter().map(|output| output.weight()).all_equal() {
535            return Err(InvalidInput);
536        }
537
538        outputs
539            .into_iter()
540            .map(Ok)
541            .reduce(|acc, output| acc?.try_add(output?))
542            .expect("outputs is non-empty, as checked above")
543            .map(|o| o.into_dk_output(nodes))
544    }
545
546    /// Interpolate shares from multiple outputs to create new shares for the given indices.
547    /// This is used after key rotation where each party shares their shares from the previous round as the new secret.
548    /// After collecting t such shares from different parties, new shares for the given indices can be created using this function.
549    ///
550    /// The `outputs` parameter is a list of `IndexedValue`, where each `value` is the output of an
551    /// AVSS instance and the corresponding `index` indicates which share from the previous round
552    /// the AVSS instance was sharing.
553    pub fn complete_key_rotation(
554        t: u16,
555        my_id: PartyId,
556        nodes: &Nodes<EG>,
557        outputs: &[IndexedValue<AvssOutput>],
558    ) -> FastCryptoResult<Self> {
559        if outputs.len() != t as usize {
560            return Err(InputLengthWrong(t as usize));
561        }
562        if outputs.is_empty() {
563            return Err(InvalidInput);
564        }
565
566        let my_indices = nodes.share_ids_of(my_id)?;
567
568        let lagrange_coefficients: Vec<S> = Poly::<G>::get_lagrange_coefficients_for_c0(
569            t,
570            outputs.iter().map(|output| output.index),
571        )
572        .map(|c| c.1.iter().map(|s| s * c.0).collect_vec())?;
573
574        let feldman_commitment = Poly::multi_scalar_mul(
575            &outputs
576                .iter()
577                .map(|output| output.value.feldman_commitment.clone())
578                .collect_vec(),
579            &lagrange_coefficients,
580        )?;
581
582        let commitments = feldman_commitment.eval_range(nodes.total_weight()).to_vec();
583
584        let shares = my_indices
585            .iter()
586            .map(|&index| {
587                let terms = outputs
588                    .iter()
589                    .zip(&lagrange_coefficients)
590                    .map(|(output, coeff)| {
591                        Ok(output
592                            .value
593                            .share_for_index(index)
594                            .ok_or(InvalidInput)?
595                            .value
596                            * coeff)
597                    })
598                    .collect::<FastCryptoResult<Vec<_>>>()?;
599                Ok(Eval {
600                    index,
601                    value: S::sum(terms.into_iter()),
602                })
603            })
604            .collect::<FastCryptoResult<Vec<_>>>()?;
605
606        let vk = G::multi_scalar_mul(
607            &lagrange_coefficients,
608            outputs
609                .iter()
610                .map(|o| o.value.feldman_commitment.c0())
611                .collect_vec()
612                .as_slice(),
613        )?;
614
615        Ok(Self {
616            my_shares: SharesForNode { shares },
617            commitments,
618            vk,
619        })
620    }
621}

652    /// Combine this output with another by summing shares that share the same index.
653    /// Returns [InvalidInput] if the two outputs hold a different number of shares or if their
654    /// indices do not line up positionally.
655    fn try_add(self, rhs: Self) -> FastCryptoResult<Self> {
656        if self.my_shares.shares.len() != rhs.my_shares.shares.len() {
657            return Err(InvalidInput);
658        }
659        let shares = self
660            .my_shares
661            .shares
662            .iter()
663            .zip(&rhs.my_shares.shares)
664            .map(|(a, b)| {
665                if a.index != b.index {
666                    return Err(InvalidInput);
667                }
668                Ok(Eval {
669                    index: a.index,
670                    value: a.value + b.value,
671                })
672            })
673            .collect::<FastCryptoResult<Vec<_>>>()?;
674        Ok(Self {
675            my_shares: SharesForNode { shares },
676            feldman_commitment: self.feldman_commitment + &rhs.feldman_commitment,
677        })
678    }
679}

1031    impl Dealer {
1032        pub fn create_message_cheating<Rng: AllowedRng>(
1033            &self,
1034            rng: &mut Rng,
1035        ) -> FastCryptoResult<Message> {
1036            let polynomial = Poly::rand_fixed_c0(self.params.t - 1, self.secret, rng);
1037            let commitment = polynomial.commit();
1038
1039            // Encrypt all shares to the receivers
1040            let mut pk_and_msgs = self
1041                .nodes
1042                .iter()
1043                .map(|node| (node.pk.clone(), self.nodes.share_ids_of(node.id).unwrap()))
1044                .map(|(public_key, share_ids)| {
1045                    (
1046                        public_key,
1047                        SharesForNode {
1048                            shares: share_ids
1049                                .into_iter()
1050                                .map(|index| polynomial.eval(index))
1051                                .collect_vec(),
1052                        }
1053                        .to_bytes(),
1054                    )
1055                })
1056                .collect_vec();
1057
1058            // Modify the first share of the first receiver to simulate a cheating dealer
1059            pk_and_msgs[0].1[7] ^= 1;
1060
1061            let ciphertext = MultiRecipientEncryption::encrypt(
1062                &pk_and_msgs,
1063                &self.random_oracle().extend(&Encryption.to_string()),
1064                rng,
1065            );
1066
1067            Ok(Message {
1068                ciphertext,
1069                feldman_commitment: commitment,
1070            })
1071        }
1072    }

1222                )
1223            })

1227        let mut rotated = HashMap::<(PartyId, ShareIndex), AvssOutput>::new();
