4//! Implementation of an asynchronous verifiable secret sharing (AVSS) protocol to distribute shares for a secret to a set of receivers.
5//! A receiver can verify that the secret being shared is the same as a share from a previous round (e.g., the secret key share of a threshold signature).
6//!
7//! Before the protocol starts, the following setup is needed:
8//! * Each receiver has an encryption key pair (ECIES) and these public keys are known to all parties.
9//! * The public keys along with the weights of each receiver are known to all parties and defined in the [Nodes] structure.
10//! * Define a new [crate::threshold_schnorr::Dealer] with the secrets, who begins by calling [crate::threshold_schnorr::Dealer::create_message].
11

50/// The message broadcast by the dealer, containing the encrypted shares and the public keys of the nonces.
51#[derive(Clone, Debug, Serialize, Deserialize)]
52pub struct Message {
53    ciphertext: MultiRecipientEncryption<EG>,
54    feldman_commitment: Poly<G>,
55}

57/// The result of a [Receiver] processing a [Message]: Either valid shares or a complaint.
58#[allow(clippy::large_enum_variant)] // Clippy complains because ReceiverOutput can be very small if BATCH_SIZE is small.
59pub enum ProcessedMessage {
60    Valid(ReceiverOutput),
61    Complaint(Complaint),
62}

64/// The output of a receiver: The shares for each nonce + commitments for the next round.
65#[derive(Debug, Clone, Serialize, Deserialize)]
66pub struct ReceiverOutput {
67    pub my_shares: SharesForNode,
68
69    /// The commitments to the polynomials will be used for key rotation.
70    pub commitments: Vec<Eval<G>>,
71
72    /// The public key corresponding to the secret the dealer is sharing.
73    pub vk: G,
74}

76/// All the shares given to a node. One share per the node's weight.
77/// These can be created either by decrypting the shares from the dealer (see [Receiver::process_message]) or by recovering them from complaint responses.
78#[derive(Clone, Debug, Serialize, Deserialize)]
79pub struct SharesForNode {
80    pub shares: Vec<Eval<S>>,
81}
82
83impl SharesForNode {
84    /// Get the weight of this node (number of shares it has).
85    pub fn weight(&self) -> usize {
86        self.shares.len()
87    }
88
89    fn verify(&self, message: &Message) -> FastCryptoResult<()> {
90        for share in &self.shares {
91            message
92                .feldman_commitment
93                .verify_share(share.index, &share.value)?
94        }
95        Ok(())
96    }
97
98    /// Assuming that enough shares are given, recover the shares for this node.
99    fn recover(
100        indices: Vec<ShareIndex>,
101        threshold: u16,
102        other_shares: &[Self],
103    ) -> FastCryptoResult<Self> {
104        // Compute the total weight of the valid responses
105        let response_weight = other_shares
106            .iter()
107            .map(SharesForNode::weight)
108            .sum::<usize>();
109        if response_weight < threshold as usize {
110            return Err(FastCryptoError::GeneralError(
111                "Not enough valid responses".to_string(),
112            ));
113        }
114
115        let shares = indices
116            .into_iter()
117            .map(|index| {
118                let evaluations = other_shares
119                    .iter()
120                    .flat_map(|share| share.shares.clone())
121                    .collect_vec();
122                Poly::interpolate_at_index(index, &evaluations).unwrap()
123            })
124            .collect_vec();
125
126        Ok(Self { shares })
127    }
128}

132impl Dealer {
133    /// Create a new dealer.
134    ///
135    /// * `secret`: The secret to share. If None, a random secret will be generated.
136    /// * `nodes`: The set of nodes (parties) participating in the protocol, including their public keys and weights.
137    /// * `t`: The threshold number of shares required to reconstruct the secret. One party can have multiple shares according to its weight.
138    /// * `f`: An upper bound on the number of Byzantine parties counted by weight.
139    /// * `sid`: A session identifier that should be unique for each invocation of the protocol but the same for all parties in a single invocation.
140    pub fn new(
141        secret: Option<S>,
142        nodes: Nodes<EG>,
143        t: u16,
144        f: u16,
145        sid: Vec<u8>,
146    ) -> FastCryptoResult<Self> {
147        // We need to collect t+f confirmations to make sure that at least t honest parties have confirmed.
148        if t <= f || t + 2 * f > nodes.total_weight() {
149            return Err(InvalidInput);
150        }
151
152        Ok(Self {
153            secret,
154            t,
155            nodes,
156            sid,
157        })
158    }
159
160    /// 1. The Dealer samples nonces, generates shares and broadcasts the encrypted shares.
161    pub fn create_message<Rng: AllowedRng>(&self, rng: &mut Rng) -> FastCryptoResult<Message> {
162        let secret = self.secret.unwrap_or(S::rand(rng));
163        let polynomial = Poly::rand_fixed_c0(self.t - 1, secret, rng);
164
165        // Encrypt all shares to the receivers
166        let pk_and_msgs = self
167            .nodes
168            .iter()
169            .map(|node| (node.pk.clone(), self.nodes.share_ids_of(node.id).unwrap()))
170            .map(|(public_key, share_ids)| {
171                (
172                    public_key,
173                    SharesForNode {
174                        shares: share_ids
175                            .into_iter()
176                            .map(|index| polynomial.eval(index))
177                            .collect_vec(),
178                    }
179                    .to_bytes(),
180                )
181            })
182            .collect_vec();
183
184        let ciphertext = MultiRecipientEncryption::encrypt(
185            &pk_and_msgs,
186            &self.random_oracle().extend(&Encryption.to_string()),
187            rng,
188        );
189
190        Ok(Message {
191            ciphertext,
192            feldman_commitment: polynomial.commit(),
193        })
194    }
195
196    fn random_oracle(&self) -> RandomOracle {
197        random_oracle_from_sid(&self.sid)
198    }
199}

201impl Receiver {
202    /// Create a new receiver.
203    ///
204    /// * `nodes`: The set of nodes (parties) participating in the protocol, including their public keys and weights.
205    /// * `id`: The unique identifier of this receiver. Should match one of the party ids in `nodes`.
206    /// * `t`: The threshold number of shares required to reconstruct the secret. One party can have multiple shares according to its weight.
207    /// * `sid`: A session identifier that should be unique for each invocation of the protocol but the same for all parties in a single invocation.
208    /// * `commitment`: A commitment to the secret being shared. This should be equal to `secret * G` and is typically found as the commitment from a previous round (see [ReceiverOutput]). If None, no consistency check will be performed.
209    /// * `enc_secret_key`: The private key used to decrypt the shares sent to this receiver.
210    pub fn new(
211        nodes: Nodes<EG>,
212        id: PartyId,
213        t: u16,
214        sid: Vec<u8>,
215        commitment: Option<G>,
216        enc_secret_key: PrivateKey<EG>,
217    ) -> Self {
218        Self {
219            id,
220            enc_secret_key,
221            commitment,
222            sid,
223            t,
224            nodes,
225        }
226    }
227
228    pub fn id(&self) -> PartyId {
229        self.id
230    }
231
232    /// 2. Each receiver processes the message, verifies and decrypts its shares.
233    ///
234    /// If this works, the receiver can store the shares and contribute a signature on the message to a certificate.
235    ///
236    /// This returns an [InvalidMessage] error if the ciphertext cannot be verified, if the commitments are invalid or do not match the commitments from a previous round.
237    /// All honest receivers will reject such a message with the same error, and such a message should be ignored.
238    ///
239    /// If the message is valid but contains invalid shares for this receiver, the call will succeed but will return a [Complaint].
240    ///
241    /// 3. When t+f signatures have been collected in the certificate, the receivers can now verify the certificate and finish the protocol.
242    pub fn process_message(&self, message: &Message) -> FastCryptoResult<ProcessedMessage> {
243        if message.feldman_commitment.degree() != self.t as usize - 1 {
244            return Err(InvalidMessage);
245        }
246
247        // If a commitment is given, verify that the secret the dealer is distributing is consistent
248        if let Some(c) = &self.commitment {
249            if message.feldman_commitment.c0() != c {
250                return Err(InvalidMessage);
251            }
252        }
253
254        let random_oracle_encryption = self.random_oracle().extend(&Encryption.to_string());
255        message
256            .ciphertext
257            .verify(&random_oracle_encryption)
258            .map_err(|_| InvalidMessage)?;
259
260        let plaintext = message.ciphertext.decrypt(
261            &self.enc_secret_key,
262            &random_oracle_encryption,
263            self.id as usize,
264        );
265
266        match SharesForNode::from_bytes(&plaintext).and_then(|my_shares| {
267            if my_shares.weight() != self.my_weight() {
268                return Err(InvalidInput);
269            }
270            my_shares.verify(message)?;
271            Ok(my_shares)
272        }) {
273            Ok(my_shares) => Ok(ProcessedMessage::Valid(ReceiverOutput {
274                my_shares,
275                commitments: self.compute_commitments(message),
276                vk: *message.feldman_commitment.c0(),
277            })),
278            Err(_) => Ok(ProcessedMessage::Complaint(Complaint::create(
279                self.id,
280                &message.ciphertext,
281                &self.enc_secret_key,
282                &self.random_oracle(),
283                &mut rand::thread_rng(),
284            ))),
285        }
286    }
287
288    /// 4. Upon receiving a complaint, a receiver verifies it and responds with a recovery package for the shares of the accuser.
289    pub fn handle_complaint(
290        &self,
291        message: &Message,
292        complaint: &Complaint,
293        rng: &mut impl AllowedRng,
294    ) -> FastCryptoResult<ComplaintResponse> {
295        complaint.check(
296            &self.nodes.node_id_to_node(complaint.accuser_id)?.pk,
297            &message.ciphertext,
298            &self.random_oracle(),
299            |shares: &SharesForNode| shares.verify(message),
300        )?;
301        Ok(ComplaintResponse::create(
302            self.id,
303            &message.ciphertext,
304            &self.enc_secret_key,
305            &self.random_oracle(),
306            rng,
307        ))
308    }
309
310    /// 5. Upon receiving t valid responses to a complaint, the accuser can recover its shares.
311    ///    Fails if there are not enough valid responses to recover the shares or if any of the responses come from an invalid party.
312    pub fn recover(
313        &self,
314        message: &Message,
315        responses: &[ComplaintResponse],
316    ) -> FastCryptoResult<ReceiverOutput> {
317        // TODO: This fails if one of the responses has an invalid responder_id. We could probably just ignore those instead.
318
319        // Sanity check that we have enough responses (by weight) to recover the shares.
320        let total_response_weight = self
321            .nodes
322            .total_weight_of(responses.iter().map(|response| &response.responder_id))?;
323        if total_response_weight < self.t {
324            return Err(FastCryptoError::InputTooShort(self.t as usize));
325        }
326
327        let response_shares = responses
328            .iter()
329            .filter_map(|response| {
330                self.nodes
331                    .node_id_to_node(response.responder_id)
332                    .and_then(|node| {
333                        response.decrypt_with_response(
334                            &self.random_oracle(),
335                            &node.pk,
336                            &message.ciphertext,
337                        )
338                    })
339                    .and_then(|shares: SharesForNode| {
340                        // Verify the shares are valid
341                        shares.verify(message).map(|_| shares)
342                    })
343                    .tap_err(|_| {
344                        warn!(
345                            "Ignoring invalid recovery package from {}",
346                            response.responder_id
347                        )
348                    })
349                    .ok()
350            })
351            .collect_vec();
352
353        let my_shares = SharesForNode::recover(self.my_indices(), self.t, &response_shares)?;
354        my_shares.verify(message)?;
355
356        Ok(ReceiverOutput {
357            my_shares,
358            commitments: self.compute_commitments(message),
359            vk: *message.feldman_commitment.c0(),
360        })
361    }
362
363    fn compute_commitments(&self, message: &Message) -> Vec<Eval<G>> {
364        self.nodes
365            .share_ids_iter()
366            .map(|index| message.feldman_commitment.eval(index))
367            .collect()
368    }
369
370    pub fn my_indices(&self) -> Vec<ShareIndex> {
371        self.nodes.share_ids_of(self.id).unwrap()
372    }
373
374    pub fn my_weight(&self) -> usize {
375        self.nodes
376            .total_weight_of(std::iter::once(&self.id))
377            .unwrap() as usize
378    }
379
380    fn random_oracle(&self) -> RandomOracle {
381        random_oracle_from_sid(&self.sid)
382    }

398    /// Combine multiple outputs from different dealers into a single output by summing.
399    /// This is used after a successful AVSS used for DKG to combine the shares from multiple dealers into a single share for each party.
400    /// Panics if the given `ReceiverOutput`s are not compatible (same weight, same indices, same number of commitments)
401    /// Returns the combined output + the joint verifying key
402    pub fn complete_dkg(t: u16, outputs: Vec<Self>) -> FastCryptoResult<Self> {
403        // The same t dealers are needed for all parties to ensure uniqueness and that at least one honest dealers secret is included in the key.
404        if outputs.len() != t as usize {
405            return Err(InputLengthWrong(t as usize));
406        }
407
408        // Sanity check: Threshold cannot be zero and all outputs must have the same weight.
409        if outputs.is_empty() || !outputs.iter().map(|output| output.weight()).all_equal() {
410            return Err(InvalidInput);
411        }
412
413        Ok(outputs
414            .into_iter()
415            .reduce(|acc, output| {
416                let shares = acc
417                    .my_shares
418                    .shares
419                    .iter()
420                    .zip_eq(&output.my_shares.shares)
421                    .map(types::sum)
422                    .collect_vec();
423                let commitments = acc
424                    .commitments
425                    .iter()
426                    .zip_eq(&output.commitments)
427                    .map(types::sum)
428                    .collect_vec();
429                ReceiverOutput {
430                    my_shares: SharesForNode { shares },
431                    commitments,
432                    vk: acc.vk + output.vk,
433                }
434            })
435            .expect("Should not be empty"))
436    }
437
438    /// Interpolate shares from multiple outputs to create new shares for the given indices.
439    /// This is used after key rotation where each party shares their shares from the previous round as the new secret.
440    /// After collecting t such shares from different parties, new shares for the given indices can be created using this function.
441    ///
442    /// The `outputs` parameter is a list of `IndexedValue`, where each `value` is the output of an
443    /// AVSS instance and the corresponding `index` indicates which share from the previous round
444    /// the AVSS instance was sharing.
445    pub fn complete_key_rotation(
446        t: u16,
447        my_indices: &[ShareIndex],
448        all_indices: impl Iterator<Item = ShareIndex>,
449        outputs: &[IndexedValue<Self>],
450    ) -> FastCryptoResult<Self> {
451        if outputs.len() != t as usize {
452            return Err(InputLengthWrong(t as usize));
453        }
454        if outputs.is_empty() {
455            return Err(InvalidInput);
456        }
457
458        let shares = my_indices
459            .iter()
460            .map(|&index| Eval {
461                index,
462                value: Poly::recover_c0(
463                    t,
464                    outputs.iter().map(|output| Eval {
465                        index: output.index,
466                        value: output.value.share_for_index(index).unwrap().clone().value,
467                    }),
468                )
469                .unwrap(),
470            })
471            .collect();
472
473        let commitments = all_indices
474            .map(|index| Eval {
475                index,
476                value: Poly::recover_c0_msm(
477                    t,
478                    outputs.iter().map(|output| Eval {
479                        index: output.index,
480                        value: output.value.commitment_for_index(index).unwrap().value,
481                    }),
482                )
483                .unwrap(),
484            })
485            .collect_vec();
486
487        // TODO: This will not change, so perhaps it's not meaningful to compute it again, except for a sanity check?
488        let vk = Poly::recover_c0_msm(
489            t,
490            outputs.iter().map(|output| Eval {
491                index: output.index,
492                value: output.value.vk,
493            }),
494        )?;
495
496        Ok(Self {
497            my_shares: SharesForNode { shares },
498            commitments,
499            vk,
500        })
501    }

780    impl Dealer {
781        pub fn create_message_cheating<Rng: AllowedRng>(
782            &self,
783            rng: &mut Rng,
784        ) -> FastCryptoResult<Message> {
785            let secret = self.secret.unwrap_or(S::rand(rng));
786            let polynomial = Poly::rand_fixed_c0(self.t - 1, secret, rng);
787            let commitment = polynomial.commit();
788
789            // Encrypt all shares to the receivers
790            let mut pk_and_msgs = self
791                .nodes
792                .iter()
793                .map(|node| (node.pk.clone(), self.nodes.share_ids_of(node.id).unwrap()))
794                .map(|(public_key, share_ids)| {
795                    (
796                        public_key,
797                        SharesForNode {
798                            shares: share_ids
799                                .into_iter()
800                                .map(|index| polynomial.eval(index))
801                                .collect_vec(),
802                        }
803                        .to_bytes(),
804                    )
805                })
806                .collect_vec();
807
808            // Modify the first share of the first receiver to simulate a cheating dealer
809            pk_and_msgs[0].1[7] ^= 1;
810
811            let ciphertext = MultiRecipientEncryption::encrypt(
812                &pk_and_msgs,
813                &self.random_oracle().extend(&Encryption.to_string()),
814                rng,
815            );
816
817            Ok(Message {
818                ciphertext,
819                feldman_commitment: commitment,
820            })
821        }
822    }

825    fn test_dkg_simple() {
826        // No complaints, all honest. All have weight 1
827        let t = 3;
828        let f = 2;
829        let n = 7;
830
831        let mut rng = rand::thread_rng();
832        let sks = (0..n)
833            .map(|_| ecies_v1::PrivateKey::<EG>::new(&mut rng))
834            .collect::<Vec<_>>();
835        let nodes = Nodes::new(
836            sks.iter()
837                .enumerate()
838                .map(|(id, sk)| Node {
839                    id: id as u16,
840                    pk: PublicKey::from_private_key(sk),
841                    weight: 1,
842                })
843                .collect::<Vec<_>>(),
844        )
845        .unwrap();
846
847        // Map from each party to the ordered list of outputs it has received
848        let mut outputs = HashMap::<PartyId, Vec<ReceiverOutput>>::new();
849        for node in nodes.iter() {
850            outputs.insert(node.id, Vec::new());
851        }
852
853        let mut messages = Vec::new();
854
855        // Each node acts as dealer in the DKG
856        for node in nodes.iter() {
857            let sid = format!("dkg-test-session-{}", node.id).into_bytes();
858            let dealer: Dealer = Dealer::new(None, nodes.clone(), t, f, sid.clone()).unwrap();
859            let receivers = sks
860                .iter()
861                .enumerate()
862                .map(|(id, enc_secret_key)| {
863                    Receiver::new(
864                        nodes.clone(),
865                        id as u16,
866                        t,
867                        sid.clone(),
868                        None,
869                        enc_secret_key.clone(),
870                    )
871                })
872                .collect::<Vec<_>>();
873
874            // Each dealer creates a message
875            let message = dealer.create_message(&mut rng).unwrap();
876            messages.push(message.clone());
877
878            // Each receiver processes the message. In this case, we assume all are honest and there are no complaints.
879            receivers.iter().for_each(|receiver| {
880                let output = assert_valid(receiver.process_message(&message).unwrap());
881                outputs.get_mut(&receiver.id()).unwrap().push(output);
882            });
883
884            // TODO: Create certificate and post it on TOB
885        }
886
887        // Now, each party has collected their outputs from all dealers.
888        // We use the first f + 1 outputs seen on-chain to create the final shares.
889        let mut final_shares = HashMap::<PartyId, ReceiverOutput>::new();
890        for node in nodes.iter() {
891            let my_outputs = outputs.get(&node.id).unwrap();
892            let final_share =
893                ReceiverOutput::complete_dkg(t, my_outputs[..t as usize].to_vec()).unwrap();
894            final_shares.insert(node.id, final_share.clone());
895
896            // Each party now has their final shares
897            println!("Node {} final shares: {:?}", node.id, final_share);
898        }
899
900        // We may now compute the joint verification key from the commitments of the first t dealers.
901        let vk = final_shares.get(&0).unwrap().vk;
902
903        // For testing, we can recover the secret key from t shares and check that the secret key matches the verification key.
904        let shares = final_shares
905            .values()
906            .flat_map(|output| output.my_shares.shares.clone())
907            .collect_vec();
908        let sk = Poly::recover_c0(t, shares[..t as usize].iter()).unwrap();
909        assert_eq!(G::generator() * sk, vk);
910    }
