4use crate::polynomial::{Eval, MonicLinear, Poly};
5use crate::threshold_schnorr::S;
6use crate::types::{to_scalar, ShareIndex};
7use fastcrypto::error::{FastCryptoError, FastCryptoResult};
8use itertools::Itertools;
9
10/// Decoder for Reed-Solomon codes.
11/// This can correct up to (d-1)/2 errors, where d is the distance of the code.
12/// The code is defined by the evaluation points `a` and the message length `k`.
13/// The distance is given by `n - k + 1`, where `n` is the length of `a`.
14///
15/// The implementation follows the Gao decoding algorithm (see https://www.math.clemson.edu/~sgao/papers/RS.pdf).
16pub struct RSDecoder {
17    g0: Poly<S>,
18    a: Vec<ShareIndex>,
19    k: usize,
20}
21
22impl RSDecoder {
23    /// Create a new Gao decoder with the given evaluation points `a` and message length `k`.
24    pub fn new(a: Vec<ShareIndex>, k: usize) -> Self {
25        assert!(k < a.len(), "Message length must be less than block length");
26        let mut g0 = Poly::one();
27        for ai in &a {
28            g0 *= MonicLinear(-to_scalar::<S>(ai));
29        }
30        Self { g0, a, k }
31    }
32
33    /// The length of the code words.
34    fn block_length(&self) -> usize {
35        self.a.len()
36    }
37
38    /// The length of the messages.
39    fn message_length(&self) -> usize {
40        self.k
41    }
42
43    /// The distance of the code.
44    fn distance(&self) -> usize {
45        self.block_length() - self.message_length() + 1
46    }
47
48    /// Compute the message polynomial.
49    /// Returns an error if the input length is wrong or if there are too many errors to correct.
50    pub fn compute_message_polynomial(&self, code_word: &[S]) -> FastCryptoResult<Poly<S>> {
51        // The implementation follows Algorithm 1 in Gao's paper.
52
53        if code_word.len() != self.block_length() {
54            return Err(FastCryptoError::InputLengthWrong(self.block_length()));
55        }
56
57        // Step 1: Interpolation
58        let g1 = Poly::interpolate(
59            &self
60                .a
61                .iter()
62                .zip(code_word)
63                .map(|(&index, &value)| Eval { index, value })
64                .collect_vec(),
65        )?;
66
67        // Step 2: Partial GCD
68        let (g, _, v) = Poly::partial_extended_gcd(
69            &self.g0,
70            &g1,
71            (self.message_length() + self.block_length()) / 2,
72        )?;
73
74        // Step 3: Long division
75        let (f1, r) = g.div_rem(&v)?;
76        if !r.is_zero() || f1.degree() >= self.k {
77            return Err(FastCryptoError::TooManyErrors((self.distance() - 1) / 2));
78        }
79        Ok(f1)
80    }
81
82    /// Encode the message using the Reed-Solomon code defined by the evaluation points `a`.
83    /// Returns an error if the message length is wrong.
84    pub fn encode(&self, message: Vec<S>) -> FastCryptoResult<Vec<S>> {
85        if message.len() != self.message_length() {
86            return Err(FastCryptoError::InputLengthWrong(self.message_length()));
87        }
88        let f = Poly::from(message);
89        Ok(self.a.iter().map(|&ai| f.eval(ai).value).collect_vec())
90    }
91
92    /// Try to correct the input and return the decoded message.
93    /// Returns an error if the input length is wrong or if there are too many errors to correct.
94    pub fn decode(&self, input: &[S]) -> FastCryptoResult<Vec<S>> {
95        let mut f1 = self.compute_message_polynomial(input)?.to_vec();
96        f1.truncate(self.k);
97        Ok(f1)
98    }
99}

101#[cfg(test)]
102mod tests {
103    use super::*;
104
105    #[test]
106    fn test_gao_decoder() {
107        let a = (1..=7).map(|i| ShareIndex::new(i).unwrap()).collect_vec();
108        let k = 3;
109        let decoder = RSDecoder::new(a.clone(), k);
110
111        let message = vec![S::from(11u128), S::from(22u128), S::from(33u128)];
112        let code_word = decoder.encode(message.clone()).unwrap();
113
114        // Introduce errors
115        let mut received = code_word.clone();
116        received[4] = S::from(20u128); // Error at position 4
117        received[2] = S::from(200u128); // Error at position 2
118
119        let decoded_message = decoder.decode(&received).unwrap();
120        assert_eq!(decoded_message, message);
121
122        // Test with too many errors
123        let mut received = code_word.clone();
124        received[4] = S::from(20u128); // Error at position 4
125        received[3] = S::from(2000u128); // Error at position 3
126        received[2] = S::from(200u128); // Error at position 2
127        assert!(decoder.decode(&received).is_err());
128    }
129}
