227impl Dealer {
228    /// Create a new dealer.
229    ///
230    /// * `nodes` defines the set of receivers and their weights.
231    /// * `dealer_id` is the id of this dealer as a node.
232    /// * `params` carries the reconstruction threshold `t` and Byzantine bound `f`.
233    /// * `sid` is a session identifier that should be unique for each invocation, but the same
234    ///   for all parties.
235    /// * `batch_size_per_weight` is the number of secrets a dealer should deal per weight it has.
236    ///
237    /// Returns an `InvalidInput` error if
238    /// * t is larger than the total weight of the nodes.
239    /// * the `dealer_id` is invalid (not part of `nodes`).
240    pub fn new(
241        nodes: Nodes<EG>,
242        dealer_id: PartyId,
243        params: Parameters,
244        sid: Vec<u8>,
245        batch_size_per_weight: u16,
246    ) -> FastCryptoResult<Self> {
247        let total_weight = nodes.total_weight();
248        params.validate(total_weight)?;
249        params.check_erasure_code_params(total_weight)?;
250        let nodes = Arc::new(nodes);
251        let avid = avid::Avid::new(Arc::clone(&nodes), params.f)?;
252        // Each dealer deals a number of nonces proportional to their weight.
253        let batch_size = nodes.weight_of(dealer_id)? as usize * batch_size_per_weight as usize;
254        Ok(Self {
255            params,
256            nodes,
257            sid,
258            batch_size,
259            avid,
260        })
261    }
262
263    /// 1. Build the [AvssMessageBuilder]. This encrypts the shares for every receiver and holds
264    ///    everything the dealer needs to create the per-receiver [AvssMessage]s.
265    ///
266    ///    The dealer must send all [AvssMessage]s until it has collected an AVID certificate, even
267    ///    after an AVSS certificate has been created.
268    ///
269    ///    To survive a crash, the caller should persist the returned [AvssMessageBuilder] before
270    ///    sending any messages, so that even after a restart it can reproduce both the optimistic
271    ///    [AvssMessage]s and use the same ciphertexts in the pessimistic AVID dispersal.
272    pub fn create_avss_messages(
273        &self,
274        rng: &mut impl AllowedRng,
275    ) -> FastCryptoResult<AvssMessageBuilder> {
276        self.create_avss_messages_with_mutation(rng, |_| {})
277    }
278
279    /// Encrypt shares, build `v`, and return the dealer state. Test mutation hook runs after the
280    /// plaintexts are constructed and before encryption.
281    #[cfg_attr(not(test), allow(unused_variables, unused_mut))]
282    fn create_avss_messages_with_mutation(
283        &self,
284        rng: &mut impl AllowedRng,
285        mutate_plaintexts: impl FnOnce(&mut [(crate::ecies_v1::PublicKey<EG>, Vec<u8>)]),
286    ) -> FastCryptoResult<AvssMessageBuilder> {
287        let secrets = repeat_with(|| S::rand(rng))
288            .take(self.batch_size)
289            .collect_vec();
290
291        // Compute the (full) public keys for all secrets
292        let full_public_keys = secrets.iter().map(|s| G::generator() * s).collect_vec();
293
294        // "blinding" polynomial as defined in https://eprint.iacr.org/2023/536.pdf.
295        let total_weight = self.nodes.total_weight();
296        let blinding_secret = S::rand(rng);
297        let blinding_poly_evaluations =
298            create_secret_sharing(rng, blinding_secret, self.params.t, total_weight);
299        let blinding_commit = G::generator() * blinding_secret;
300
301        // Compute all evaluations of all polynomials
302        let share_batches = secrets
303            .iter()
304            .map(|&s| create_secret_sharing(rng, s, self.params.t, total_weight))
305            .collect_vec();
306
307        // Encrypt all shares to the receivers
308        let mut pk_and_msgs = self
309            .nodes
310            .iter()
311            .map(|node| (node.pk.clone(), self.nodes.share_ids_of(node.id).unwrap()))
312            .map(|(pk, share_ids)| {
313                (
314                    pk,
315                    SharesForNode {
316                        shares: share_ids
317                            .into_iter()
318                            .map(|index| ShareBatch {
319                                batch: share_batches.iter().map(|shares| shares[index]).collect(),
320                                blinding_share: blinding_poly_evaluations[index],
321                            })
322                            .collect_vec(),
323                    }
324                    .to_bytes(),
325                )
326            })
327            .collect_vec();
328
329        #[cfg(test)]
330        mutate_plaintexts(&mut pk_and_msgs);
331
332        let random_oracle = self.random_oracle();
333
334        let encryption = MultiRecipientEncryption::encrypt(
335            &pk_and_msgs,
336            &random_oracle.extend(&Encryption.to_string()),
337            rng,
338        );
339        let (ciphertext_shared, ciphertexts) = encryption.into_parts();
340
341        let ciphertext_hashes = ciphertexts
342            .iter()
343            .map(|c| {
344                let bytes: &[u8] = &c.0;
345                Blake2b256::digest(bytes)
346            })
347            .collect_vec();
348
349        // "response" polynomial from https://eprint.iacr.org/2023/536.pdf
350        let challenge = compute_challenge(
351            &random_oracle,
352            &full_public_keys,
353            &blinding_commit,
354            &ciphertext_shared,
355            &ciphertext_hashes,
356        );
357
358        // Get the first t evaluations for the response polynomial and use these to compute the coefficients.
359        let response_polynomial = Poly::interpolate(
360            &share_batches
361                .into_iter()
362                .map(|s| s.take(self.params.t))
363                .zip_eq(&challenge)
364                .fold(
365                    blinding_poly_evaluations.take(self.params.t),
366                    |acc, (p_l, gamma_l)| acc + p_l * gamma_l,
367                )
368                .to_vec(),
369        )?;
370
371        let common = AvssCommonMessage {
372            full_public_keys,
373            blinding_commit,
374            ciphertext_shared,
375            ciphertext_hashes,
376            response_polynomial,
377        };
378
379        Ok(AvssMessageBuilder {
380            common,
381            ciphertexts,
382        })
383    }
384
385    /// 3. RS-encode and commit the pending recipients' ciphertexts via AVID, returning an
386    ///    [AvidMessageBuilder] that can create per-receiver [AvidMessage]s on demand via
387    ///    [AvidMessageBuilder::message_for]. The pending recipients are derived as the relative
388    ///    complement of `avss_cert.signers()` within the node set.
389    ///
390    ///    Every receiver that sent an [AvssVote] (including late voters not in `avss_cert`) should
391    ///    get an [AvidMessage] until the dealer has collected an AVID certificate.
392    ///
393    ///    Forming the input `avss_cert`: the dealer should collect `≥ t + f` weight of [AvssVote]s,
394    ///    then wait a short grace period to pick up any additional votes, and only then form the
395    ///    certificate. This keeps the pending-recipient set (and hence the dispersal) as small as
396    ///    possible.
397    ///
398    ///    Once `≥ W − f` weight of [AvidVote]s has been collected, the dealer can form and
399    ///    publish a certificate over those votes on the TOB.
400    ///
401    ///    This phase is only needed if any receiver failed to confirm in the optimistic phase.
402    ///    If every receiver confirmed, the pessimistic phase can be skipped entirely.
403    ///
404    ///    The pending recipients (the non-confirmers) may carry at most `f` weight: the confirmers
405    ///    then hold `≥ W − f` weight, so `≥ W − 2f` of them are honest — exactly the AVID decode
406    ///    quorum needed to reconstruct each pending recipient's ciphertext. This returns
407    ///    [InvalidInput] if the pending weight exceeds `f`.
408    pub fn create_avid_messages<C: Certificate<Payload = AvssVote>>(
409        &self,
410        avss_message_builder: &AvssMessageBuilder,
411        avss_cert: C,
412    ) -> FastCryptoResult<AvidMessageBuilder<C>> {
413        self.create_avid_messages_with_mutation(avss_message_builder, avss_cert, |_| {})
414    }
415
416    /// Test-only variant of [Self::create_avid_messages] that runs `mutate_shards` over the
417    /// per-recipient, per-disperser shards before they are committed, to simulate a cheating dealer.
418    #[cfg(test)]
419    fn create_avid_messages_for_testing<C: Certificate<Payload = AvssVote>>(
420        &self,
421        avss_message_builder: &AvssMessageBuilder,
422        avss_cert: C,
423        mutate_shards: impl FnOnce(&mut BTreeMap<PartyId, Vec<Vec<Shard>>>),
424    ) -> FastCryptoResult<AvidMessageBuilder<C>> {
425        self.create_avid_messages_with_mutation(avss_message_builder, avss_cert, mutate_shards)
426    }
427
428    fn create_avid_messages_with_mutation<C: Certificate<Payload = AvssVote>>(
429        &self,
430        avss_message_builder: &AvssMessageBuilder,
431        avss_cert: C,
432        mutate_shards: impl FnOnce(&mut BTreeMap<PartyId, Vec<Vec<Shard>>>),
433    ) -> FastCryptoResult<AvidMessageBuilder<C>> {
434        if avss_message_builder.common.hash() != avss_cert.payload().common_message_hash {
435            warn!("batch_avss create_avid_messages_with_mutation: AVSS Cert binds a different common message");
436            return Err(InvalidInput);
437        }
438        let pending_recipients: BTreeSet<PartyId> = self
439            .nodes
440            .node_ids_iter()
441            .filter(|id| !avss_cert.signers().contains(id))
442            .collect();
443        if self.nodes.total_weight_of(pending_recipients.iter())? > self.params.f {
444            warn!("batch_avss create_avid_messages_with_mutation: too many pending recipients");
445            return Err(InvalidInput);
446        }
447        let payloads: BTreeMap<PartyId, Vec<u8>> = pending_recipients
448            .iter()
449            .map(|&i| (i, avss_message_builder.ciphertexts[i as usize].0.clone()))
450            .collect();
451        self.avid
452            .disperse_with_mutation(&payloads, mutate_shards)
453            .map(|inner| AvidMessageBuilder { inner, avss_cert })
454    }
455
456    fn random_oracle(&self) -> RandomOracle {
457        random_oracle_from_sid(&self.sid)
458    }
459}

462    /// Create a new receiver.
463    ///
464    /// * `nodes` defines the set of receivers and what shares they should receive.
465    /// * `id` is the id of this receiver.
466    /// * `dealer_id` is the id of the dealer.
467    /// * `params` carries the reconstruction threshold `t` and Byzantine bound `f`.
468    /// * `sid` is a session identifier that should be unique for each invocation, but the same
469    ///   for all parties.
470    /// * `enc_secret_key` is this Receivers' secret key for the distribution of nonces. The
471    ///   corresponding public key is defined in `nodes`.
472    /// * `batch_size_per_weight` is the number of secrets a dealer should deal per weight it has.
473    ///
474    /// Returns an `InvalidInput` error if the `id` or `dealer_id` is invalid.
475    #[allow(clippy::too_many_arguments)]
476    pub fn new(
477        nodes: Nodes<EG>,
478        id: PartyId,
479        dealer_id: PartyId,
480        params: Parameters,
481        sid: Vec<u8>,
482        enc_secret_key: PrivateKey<EG>,
483        batch_size_per_weight: u16,
484    ) -> FastCryptoResult<Self> {
485        let _ = nodes.node_id_to_node(id)?;
486
487        // The dealer is expected to deal a number of nonces proportional to it's weight
488        let batch_size = nodes.weight_of(dealer_id)? as usize * batch_size_per_weight as usize;
489
490        let total_weight = nodes.total_weight();
491        params.validate(total_weight)?;
492        params.check_erasure_code_params(total_weight)?;
493        let nodes = Arc::new(nodes);
494        let avid = avid::Avid::new(nodes.clone(), params.f)?;
495
496        Ok(Self {
497            id,
498            enc_secret_key,
499            nodes,
500            sid,
501            params,
502            batch_size,
503            avid,
504        })
505    }

507    /// 2. Process an [AvssMessage] sent by the dealer in the optimistic phase.
508    ///
509    ///    The receiver should provide a `verified_avss_common_message` if it has already received
510    ///    and verified the [AvssCommonMessage] for this AVSS round. If the common provided here
511    ///    disagrees with the one in `message`, e.g. an equivocating dealer sent a different common
512    ///    after the receiver already committed to one, the message is rejected with [InvalidInput].
513    ///
514    ///    On success, returns
515    ///     * a [ReceiverOutput],
516    ///     * an [AvssVote] to be returned to the dealer,
517    ///     * a [VerifiedAvssCommonMessage] to be retained for the rest of the session.
518    ///
519    ///    On any failure the receiver silently ignores the message and falls through to the
520    ///    pessimistic phase.
521    ///
522    ///    A voter should also retain its own ciphertext (`message.ciphertext`) so it can answer
523    ///    complaints later via [Self::handle_avss_complaint] / [Self::handle_avid_complaint], and
524    ///    should persist the returned outputs before sending its [AvssVote].
525    pub fn process_avss_message(
526        &self,
527        message: &AvssMessage,
528        verified_avss_common_message: Option<&VerifiedAvssCommonMessage>,
529    ) -> FastCryptoResult<(ReceiverOutput, AvssVote, VerifiedAvssCommonMessage)> {
530        let verified_common = match verified_avss_common_message {
531            Some(verified) if verified.0.hash() == message.common.hash() => verified.clone(),
532            Some(_) => {
533                warn!("batch_avss process_avss_message: provided verified common message does not match the message's common message");
534                return Err(InvalidInput);
535            }
536            None => self.verify_common_message(message.common.clone())?,
537        };
538        self.check_ciphertext_hash(&message.ciphertext, &verified_common)?;
539        let output = self.decrypt_and_verify_shares(&message.ciphertext, &verified_common)?;
540        Ok((
541            output,
542            AvssVote {
543                common_message_hash: verified_common.0.hash(),
544            },
545            verified_common,
546        ))
547    }
548
549    /// Verify a [AvssCommonMessage] (see [AvssCommonMessage::verify]) and return the resulting
550    /// [VerifiedAvssCommonMessage].
551    ///
552    /// Called in the pessimistic phase by a receiver that needs a [VerifiedAvssCommonMessage] but
553    /// did not obtain one from its own [AvssMessage] since the result is required by
554    /// [Self::process_avid_message] and [Self::decode_and_decrypt].
555    pub fn verify_common_message(
556        &self,
557        common_message: AvssCommonMessage,
558    ) -> FastCryptoResult<VerifiedAvssCommonMessage> {
559        common_message.verify(
560            self.params.t,
561            self.batch_size,
562            self.nodes.num_nodes(),
563            &self.random_oracle(),
564        )
565    }

567    /// 4. Verify a pessimistic-phase [AvidMessage] and emit an [EchoBuilder] which can build
568    ///    [Echo]es on demand. Returns also an [AvidVote] to be signed and returned to the dealer.
569    ///
570    ///    The caller must already hold a `verified_avss_common_message` for this AVSS round before
571    ///    processing the [AvidMessage], obtained either when verifying its shares in the AVSS
572    ///    phase ([Self::process_avss_message]) or by fetching the [AvssCommonMessage] from a peer
573    ///    and verify it using [Self::verify_common_message].
574    ///
575    ///    Receivers who did not receive their shares through an [AvssMessage] should wait for a
576    ///    published [Certificate] over [AvidVote]s that confirms they are a pending recipient and
577    ///    then get the [AvssCommonMessage] and [Echo]es from the signers.
578    ///
579    ///    The caller should persist the outputs before sending its [AvidVote]. Returns
580    ///    [InvalidMessage] if the `avss_cert` binds a different common than the one provided.
581    pub fn process_avid_message<C: Certificate<Payload = AvssVote>>(
582        &self,
583        verified_avss_common_message: &VerifiedAvssCommonMessage,
584        message: AvidMessage<C>,
585    ) -> FastCryptoResult<(EchoBuilder, AvidVote)> {
586        let AvidMessage {
587            dispersal,
588            avss_cert,
589        } = message;
590
591        if avss_cert.payload().common_message_hash != verified_avss_common_message.0.hash() {
592            warn!("batch_avss process_avid_message: AVSS Cert binds a different common message");
593            return Err(InvalidMessage);
594        }
595
596        let required_weight_of_voters = self.params.t + self.params.f;
597        if self.nodes.total_weight_of(avss_cert.signers().iter())? < required_weight_of_voters {
598            warn!("batch_avss echo: not enough voters");
599            return Err(NotEnoughWeight(required_weight_of_voters as usize));
600        }
601        avss_cert.verify()?;
602        // Dispersal recipients and voters must partition the node set (AVSS policy).
603        if !dispersal
604            .keys()
605            .eq(self.nodes.relative_complement(avss_cert.signers()).iter())
606        {
607            warn!("batch_avss echo: dispersal recipients and voters do not partition the node set");
608            return Err(InvalidMessage);
609        }
610        let (builder, vote) = self.avid.process_dispersal(self.id, dispersal)?;
611        let avid_vote = AvidVote {
612            vote,
613            common_message_hash: avss_cert.payload().common_message_hash,
614        };
615        Ok((builder, avid_vote))
616    }

618    /// Verify an [Echo] addressed to this receiver. Returns a [VerifiedEcho] suitable for
619    /// [Self::decode_and_decrypt].
620    ///
621    /// Precondition: `self.id` is one of the pending recipients. Echoes are only
622    /// meaningful for pending recipients, so voters calling this for themselves get
623    /// [InvalidInput].
624    pub fn verify_avid_echo_message<C: Certificate<Payload = AvidVote>>(
625        &self,
626        echo: Echo,
627        sender: PartyId,
628        avid_cert: &VerifiedCertificate<C>,
629    ) -> FastCryptoResult<VerifiedEcho> {
630        self.avid
631            .verify_echo(echo, sender, &avid_cert.payload().vote, self.id)
632    }

634    /// 5. Reconstruct this receiver's ciphertext from at least `W − 2f` weight of
635    ///    [VerifiedEcho]s and, when the dispersal is consistent, decrypt and verify its shares.
636    ///
637    ///    A pending recipient should only pull echoes from the signers after the [AvidVote]
638    ///    certificate is published and the cert confirms it is a pending recipient.
639    ///
640    ///    The caller should get an [AvssCommonMessage] from one of the signers of the cert and
641    ///    verify it before calling this method.
642    ///
643    ///    Returns [DecodeAndDecryptOutcome::InvalidDispersal] (an [AvidComplaint]) when the collected
644    ///    shards fail to RS-decode, or decode to a ciphertext whose re-encoding disagrees with the
645    ///    hash pinned in [VerifiedAvssCommonMessage] — a dispersal-layer fault, after which
646    ///    decryption is skipped. Otherwise the ciphertext decoded and this receiver's shares are
647    ///    verified: [DecodeAndDecryptOutcome::Valid] with the decoded [Ciphertext] and [ReceiverOutput]
648    ///    when they verify, or [DecodeAndDecryptOutcome::InvalidDecryption] (an [AvssComplaint]) when they
649    ///    don't.
650    ///
651    ///    An [AvidComplaint] is a dispersal-layer fault. Hold it until the matching cert on
652    ///    [AvidVote]s is certified on the TOB, or discard it if a different one wins. An
653    ///    [AvssComplaint] is an encryption-layer fault carrying the accuser's ciphertext and an
654    ///    ECIES recovery package. Broadcast it only after seeing a TOB certificate for the
655    ///    corresponding `common_message_hash`.
656    ///
657    ///    The caller should persist its decoded ciphertext for the session to answer complaints.
658    pub fn decode_and_decrypt<C: Certificate<Payload = AvidVote>>(
659        &self,
660        echoes: &[VerifiedEcho],
661        verified_common: &VerifiedAvssCommonMessage,
662        avid_cert: &VerifiedCertificate<C>,
663        rng: &mut impl AllowedRng,
664    ) -> FastCryptoResult<DecodeAndDecryptOutcome> {
665        if avid_cert.payload().common_message_hash != verified_common.0.hash() {
666            warn!("batch_avss decode_and_decrypt: AvidCert binds a different common message");
667            return Err(InvalidMessage);
668        }
669        let expected_hash = verified_common
670            .0
671            .ciphertext_hashes
672            .get(self.id as usize)
673            .ok_or(InvalidProof)?;
674
675        let ciphertext = match self.avid.decode_or_complain(echoes, |payload| {
676            Blake2b256::digest(payload) == *expected_hash
677        })? {
678            Ok(bytes) => Ciphertext(bytes),
679            Err(complaint) => {
680                warn!(
681                    "batch_avss decode_and_decrypt: receiver {} raising AvidComplaint",
682                    self.id,
683                );
684                return Ok(DecodeAndDecryptOutcome::InvalidDispersal(complaint));
685            }
686        };
687
688        Ok(
689            match self.decrypt_and_verify_shares(&ciphertext, verified_common) {
690                Ok(output) => DecodeAndDecryptOutcome::Valid(ciphertext, output),
691                Err(e) => {
692                    warn!(
693                    "batch_avss decode_and_decrypt: receiver {} raising AvssComplaint after share decode/verify failed: {e:?}",
694                    self.id,
695                );
696                    DecodeAndDecryptOutcome::InvalidDecryption(AvssComplaint {
697                        ciphertext,
698                        proof: recovery_proof::RecoveryProof::create(
699                            self.id,
700                            &verified_common.0.ciphertext_shared,
701                            &self.enc_secret_key,
702                            &self.random_oracle(),
703                            rng,
704                        ),
705                    })
706                }
707            },
708        )
709    }

711    /// Reject a ciphertext whose hash doesn't match the one pinned for this receiver in
712    /// [VerifiedAvssCommonMessage]. This stops a dealer from dispersing a different ciphertext
713    /// via AVID.
714    fn check_ciphertext_hash(
715        &self,
716        ciphertext: &Ciphertext,
717        verified_common: &VerifiedAvssCommonMessage,
718    ) -> FastCryptoResult<()> {
719        if Blake2b256::digest(&ciphertext.0)
720            != *verified_common
721                .0

723                .get(self.id as usize)
724                .ok_or(InvalidMessage)?
725        {
726            warn!(
727                "batch_avss check_ciphertext_hash: ciphertext hash does not match ciphertext_hashes[{}]",

733    }

735    /// Decrypt and verify this receiver's own shares against the common message, returning the
736    /// [ReceiverOutput] on success or the underlying error.
737    fn decrypt_and_verify_shares(
738        &self,
739        ciphertext: &Ciphertext,
740        common_message: &VerifiedAvssCommonMessage,
741    ) -> FastCryptoResult<ReceiverOutput> {
742        let AvssCommonMessage {
743            full_public_keys,
744            ciphertext_shared,
745            ..
746        } = &&common_message.0;
747        let random_oracle = self.random_oracle();
748
749        let random_oracle_encryption = self.random_oracle().extend(&Encryption.to_string());
750        let plaintext = ciphertext_shared.decrypt(
751            ciphertext,
752            &self.enc_secret_key,
753            &random_oracle_encryption,
754            self.id as usize,
755        );
756
757        let challenge = compute_challenge_from_common_message(&random_oracle, &common_message.0);
758        let my_shares = SharesForNode::from_bytes(plaintext)?;
759        my_shares.verify(
760            &common_message.0,
761            &challenge,
762            &self.nodes.share_ids_of(self.id)?,
763            self.batch_size,
764        )?;
765        Ok(ReceiverOutput {
766            my_shares,
767            public_keys: full_public_keys.clone(),
768        })
769    }

771    /// 6a. Validate a [AvssComplaint] and respond with this party's own shares.
772    pub fn handle_avss_complaint(
773        &self,
774        reveal: &AvssComplaint,
775        accuser_id: PartyId,
776        verified_common: &VerifiedAvssCommonMessage,
777        own_ciphertext: Ciphertext,
778        rng: &mut impl AllowedRng,
779    ) -> FastCryptoResult<ComplaintResponse> {
780        let challenge =
781            compute_challenge_from_common_message(&self.random_oracle(), &verified_common.0);
782
783        let AvssComplaint { proof, ciphertext } = reveal;
784
785        if Blake2b256::digest(&ciphertext.0)
786            != *verified_common
787                .0
788                .ciphertext_hashes
789                .get(accuser_id as usize)
790                .ok_or(InvalidProof)?
791        {
792            return Err(InvalidProof);
793        }
794        let accuser = self.nodes.node_id_to_node(accuser_id)?;
795        let accuser_indices = self.nodes.share_ids_of(accuser_id)?;
796        proof.check(
797            accuser_id,
798            &accuser.pk,
799            ciphertext,
800            &verified_common.0.ciphertext_shared,
801            &self.random_oracle(),
802            |shares: &SharesForNode| {
803                shares.verify(
804                    &verified_common.0,
805                    &challenge,
806                    &accuser_indices,
807                    self.batch_size,
808                )
809            },
810        )?;
811
812        Ok(self.build_complaint_response(&verified_common.0, own_ciphertext, rng))
813    }

815    /// 6b. Validate a [AvidComplaint] and respond with this party's own shares.
816    pub fn handle_avid_complaint<C: Certificate<Payload = AvidVote>>(
817        &self,
818        blame: &AvidComplaint,
819        accuser_id: PartyId,
820        verified_common: &VerifiedAvssCommonMessage,
821        avid_cert: &VerifiedCertificate<C>,
822        own_ciphertext: Ciphertext,
823        rng: &mut impl AllowedRng,
824    ) -> FastCryptoResult<ComplaintResponse> {
825        let expected_hash = verified_common
826            .0
827            .ciphertext_hashes
828            .get(accuser_id as usize)
829            .ok_or(InvalidProof)?;
830
831        let vote = &avid_cert.payload().vote;
832        self.avid
833            .complaint_is_valid(blame, accuser_id, vote, |payload| {
834                Blake2b256::digest(payload) == *expected_hash
835            })?;
836
837        Ok(self.build_complaint_response(&verified_common.0, own_ciphertext, rng))
838    }

841    fn build_complaint_response(
842        &self,
843        common_message: &AvssCommonMessage,
844        ciphertext: Ciphertext,
845        rng: &mut impl AllowedRng,
846    ) -> ComplaintResponse {

848            &self.enc_secret_key,
849            &self.random_oracle().extend(&Recovery(self.id).to_string()),
850            rng,

858    /// Verify a [ComplaintResponse] against a [VerifiedAvssCommonMessage].
859    pub fn verify_complaint_response(
860        &self,
861        responder_id: PartyId,
862        response: ComplaintResponse,
863        verified_common: &VerifiedAvssCommonMessage,
864    ) -> FastCryptoResult<VerifiedComplaintResponse> {
865        let ComplaintResponse {
866            ciphertext,
867            recovery_package,
868        } = response;
869        if Blake2b256::digest(&ciphertext.0)
870            != *verified_common
871                .0
872                .ciphertext_hashes
873                .get(responder_id as usize)
874                .ok_or(InvalidProof)?
875        {
876            return Err(InvalidProof);
877        }
878        let shares = verified_common
879            .0
880            .ciphertext_shared
881            .decrypt_with_recovery_package(
882                &ciphertext,
883                &recovery_package,
884                &self
885                    .random_oracle()
886                    .extend(&Recovery(responder_id).to_string()),
887                &self.random_oracle().extend(&Encryption.to_string()),
888                &self.nodes.node_id_to_node(responder_id)?.pk,
889                responder_id as usize,
890            )
891            .and_then(SharesForNode::from_bytes)?;
892
893        shares.verify(
894            &verified_common.0,
895            &compute_challenge_from_common_message(&self.random_oracle(), &verified_common.0),
896            &self.nodes.share_ids_of(responder_id)?,
897            self.batch_size,
898        )?;
899
900        Ok(VerifiedComplaintResponse {
901            responder_id,
902            shares,
903        })
904    }

906    /// 7. Recover the accuser's own shares from a quorum (`≥ t`) of [VerifiedComplaintResponse]s.
907    ///    Responses must already be validated via [Self::verify_complaint_response]. Fails if the
908    ///    responses contribute `< t` weight, or the interpolated shares fail final verification.
909    pub fn recover(
910        &self,
911        verified_common: &VerifiedAvssCommonMessage,
912        responses: Vec<VerifiedComplaintResponse>,
913    ) -> FastCryptoResult<ReceiverOutput> {
914        if !responses.iter().map(|r| r.responder_id).all_unique() {
915            return Err(InvalidInput);
916        }
917
918        if responses.iter().any(|r| {
919            self.nodes
920                .weight_of(r.responder_id)
921                .ok()
922                .is_none_or(|w| w != r.shares.weight())
923        }) {
924            return Err(InvalidInput);
925        }
926
927        if self
928            .nodes
929            .total_weight_of(responses.iter().map(|r| &r.responder_id))?
930            < self.params.t
931        {
932            return Err(NotEnoughWeight(self.params.t as usize));
933        }
934
935        let response_shares: Vec<(PartyId, SharesForNode)> = responses
936            .into_iter()
937            .map(|v| (v.responder_id, v.shares))
938            .collect();
939
940        let my_shares = SharesForNode::recover(self, &response_shares)?;
941
942        // Each response was already checked by verify_complaint_response, and interpolating valid
943        // shares yields valid shares, so this final verification is defense-in-depth and should be
944        // unreachable as a failure. Warn loudly if it ever does fail, since that signals a logic
945        // error rather than a malicious input.
946        let challenge =
947            compute_challenge_from_common_message(&self.random_oracle(), &verified_common.0);
948        my_shares
949            .verify(
950                &verified_common.0,
951                &challenge,
952                &self.nodes.share_ids_of(self.id)?,
953                self.batch_size,
954            )
955            .tap_err(|e| {
956                warn!("batch_avss recover: recovered shares failed final verification, which should be unreachable with verified responses: {e:?}")
957            })?;
958
959        Ok(ReceiverOutput {
960            my_shares,
961            public_keys: verified_common.0.full_public_keys.clone(),
962        })
963    }

999impl AvssCommonMessage {
1000    /// Verify the dealer's commitments: lengths/degree are well-formed, the encryption NIZK in
1001    /// `ciphertext_shared` checks, and `g^{p''(0)} = c' · ∏ c_l^{γ_l}`. Consumes `self` and
1002    /// returns a [VerifiedAvssCommonMessage] on success.
1003    fn verify(
1004        self,
1005        t: u16,
1006        batch_size: usize,
1007        num_nodes: usize,
1008        random_oracle: &RandomOracle,
1009    ) -> FastCryptoResult<VerifiedAvssCommonMessage> {
1010        if t == 0
1011            || self.full_public_keys.len() != batch_size
1012            || self.response_polynomial.degree() != t as usize - 1
1013            || self.ciphertext_hashes.len() != num_nodes
1014        {
1015            warn!(
1016                "batch_avss AvssCommonMessage::verify: invalid sizes (t = {}, full_public_keys.len() = {}, expected {}; response_polynomial.degree() = {}, expected {}; ciphertext_hashes.len() = {}, expected {})",
1017                t,
1018                self.full_public_keys.len(),
1019                batch_size,
1020                self.response_polynomial.degree(),
1021                t as usize - 1,
1022                self.ciphertext_hashes.len(),
1023                num_nodes,
1024            );
1025            return Err(InvalidMessage);
1026        }
1027        self.ciphertext_shared
1028            .verify(&random_oracle.extend(&Encryption.to_string()))
1029            .tap_err(|e| warn!("batch_avss AvssCommonMessage::verify: ciphertext_shared NIZK verification failed: {e:?}"))
1030            .map_err(|_| InvalidMessage)?;
1031        let challenge = compute_challenge_from_common_message(random_oracle, &self);
1032        if G::generator() * self.response_polynomial.c0()
1033            != self.blinding_commit
1034                + G::multi_scalar_mul(&challenge, &self.full_public_keys)
1035                    .expect("Inputs have constant lengths")
1036        {
1037            warn!(
1038                "batch_avss AvssCommonMessage::verify: response polynomial does not match the blinding commitment and public keys"
1039            );
1040            return Err(InvalidMessage);
1041        }
1042        Ok(VerifiedAvssCommonMessage(self))
1043    }
1044
1045    /// Canonical Blake2b-256 hash of this [AvssCommonMessage]. Used to bind echoes and complaints
1046    /// to a specific dealer broadcast.
1047    pub fn hash(&self) -> Digest {
1048        Blake2b256::digest(bcs::to_bytes(self).expect("serialize should never fail"))
1049    }
1050}

1062impl ShareBatch {
1063    /// Verify a batch of shares at share index `index` using the given challenge.
1064    fn verify(
1065        &self,
1066        index: ShareIndex,
1067        message: &AvssCommonMessage,
1068        challenge: &[S],
1069    ) -> FastCryptoResult<()> {
1070        if challenge.len() != self.batch.len() {
1071            return Err(InvalidInput);
1072        }
1073
1074        // Verify that r' + sum_l r_l * gamma_l == p''(i)
1075        if self
1076            .batch
1077            .iter()
1078            .zip_eq(challenge)
1079            .fold(self.blinding_share, |acc, (r_l, gamma_l)| {
1080                acc + r_l * gamma_l
1081            })
1082            != message.response_polynomial.eval(index).value
1083        {
1084            return Err(InvalidInput);
1085        }
1086        Ok(())
1087    }
1088}

1090impl SharesForNode {
1091    /// Get the weight of this node (number of shares it has).
1092    pub fn weight(&self) -> u16 {
1093        self.shares.len() as u16
1094    }
1095
1096    /// If all shares have the same batch size, return that.
1097    /// Otherwise, return an InvalidInput error.
1098    pub fn try_uniform_batch_size(&self) -> FastCryptoResult<usize> {
1099        get_uniform_value(self.shares.iter().map(|b| b.batch.len())).ok_or(InvalidInput)
1100    }
1101
1102    /// Get all shares this node has for the <i>i</i>-th secret/nonce in the batch, paired with
1103    /// the share indices the node was assigned (`indices.len()` must equal the number of share
1104    /// batches). Panics if `i` is larger than or equal to the batch size.
1105    pub fn shares_for_secret<'a>(
1106        &'a self,
1107        indices: &'a [ShareIndex],
1108        i: usize,
1109    ) -> impl Iterator<Item = Eval<S>> + 'a {
1110        self.shares
1111            .iter()
1112            .zip(indices)
1113            .map(move |(s, &index)| Eval {
1114                index,
1115                value: s.batch[i],
1116            })
1117    }
1118
1119    fn verify(
1120        &self,
1121        message: &AvssCommonMessage,
1122        challenge: &[S],
1123        expected_indices: &[ShareIndex],
1124        expected_batch_size: usize,
1125    ) -> FastCryptoResult<()> {
1126        let actual_batch_size = self.try_uniform_batch_size()?;
1127        if actual_batch_size != expected_batch_size {
1128            warn!(
1129                "batch_avss SharesForNode::verify: batch_size {} does not match expected {}",
1130                actual_batch_size, expected_batch_size,
1131            );
1132            return Err(InvalidMessage);
1133        }
1134        if self.shares.len() != expected_indices.len() {
1135            warn!(
1136                "batch_avss SharesForNode::verify: share count {} does not match expected weight {}",
1137                self.shares.len(),
1138                expected_indices.len(),
1139            );
1140            return Err(InvalidMessage);
1141        }
1142        for (shares, &index) in self.shares.iter().zip(expected_indices) {
1143            shares.verify(index, message, challenge).map_err(|e| {
1144                warn!(
1145                    "batch_avss SharesForNode::verify: cryptographic share verification failed at index {:?}: {e:?}",
1146                    index,
1147                );
1148                e
1149            })?;
1150        }
1151        Ok(())
1152    }
1153
1154    /// Recover the shares for this node.
1155    ///
1156    /// Fails if `other_shares` is empty.
1157    fn recover(
1158        receiver: &Receiver,
1159        other_shares: &[(PartyId, SharesForNode)],
1160    ) -> FastCryptoResult<Self> {
1161        if other_shares.is_empty() {
1162            return Err(InvalidInput);
1163        }
1164
1165        // Pre-compute each responder's share indices once.
1166        let responders: Vec<(Vec<ShareIndex>, &SharesForNode)> = other_shares
1167            .iter()
1168            .map(|(id, s)| Ok((receiver.nodes.share_ids_of(*id)?, s)))
1169            .collect::<FastCryptoResult<_>>()?;
1170
1171        let shares = receiver
1172            .my_indices()
1173            .into_iter()
1174            .map(|index| {
1175                let batch = (0..receiver.batch_size)
1176                    .map(|i| {
1177                        let evaluations = responders
1178                            .iter()
1179                            .flat_map(|(ids, s)| s.shares_for_secret(ids, i))
1180                            .take(receiver.params.t as usize)
1181                            .collect_vec();
1182                        Ok(Poly::recover_at(receiver.params.t, index, &evaluations)?.value)
1183                    })
1184                    .collect::<FastCryptoResult<Vec<_>>>()?;
1185
1186                let blinding_share = Poly::recover_at(
1187                    receiver.params.t,
1188                    index,
1189                    &responders
1190                        .iter()
1191                        .flat_map(|(ids, s)| ids.iter().copied().zip(s.shares.iter()))
1192                        .map(|(index, share)| Eval {
1193                            index,
1194                            value: share.blinding_share,
1195                        })
1196                        .take(receiver.params.t as usize)
1197                        .collect_vec(),
1198                )?
1199                .value;
1200
1201                Ok(ShareBatch {
1202                    batch,
1203                    blinding_share,
1204                })
1205            })
1206            .collect::<FastCryptoResult<Vec<_>>>()?;
1207        Ok(Self { shares })
1208    }
1209}

1220    random_oracle: &RandomOracle,
1221    c: &[G],
1222    c_prime: &G,
1223    shared: &SharedComponents<EG>,
1224    ciphertext_hashes: &[Digest],
