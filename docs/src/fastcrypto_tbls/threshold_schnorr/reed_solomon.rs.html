13/// Decoder for Reed-Solomon codes.
14/// This can correct up to (d-1)/2 errors, where d is the distance of the code.
15/// The code is defined by the evaluation points `a` and the message length `k`.
16/// The distance is given by `n - k + 1`, where `n` is the length of `a`.
17///
18/// The implementation follows the Gao decoding algorithm
19/// (see https://www.math.clemson.edu/~sgao/papers/RS.pdf).
20pub struct RSDecoder {
21    g0: Poly<S>,
22    a: Vec<ShareIndex>,
23    k: usize,
24}
25
26impl RSDecoder {
27    /// Create a new Gao decoder with the given evaluation points `a` and message length `k`.
28    pub fn new(a: Vec<ShareIndex>, k: usize) -> Self {
29        assert!(k < a.len(), "Message length must be less than block length");
30        let mut g0 = Poly::one();
31        for ai in &a {
32            g0 *= MonicLinear(-to_scalar::<S>(ai));
33        }
34        Self { g0, a, k }
35    }
36
37    /// The length of the code words.
38    fn block_length(&self) -> usize {
39        self.a.len()
40    }
41
42    /// The length of the messages.
43    fn message_length(&self) -> usize {
44        self.k
45    }
46
47    /// The distance of the code.
48    fn distance(&self) -> usize {
49        self.block_length() - self.message_length() + 1
50    }
51
52    /// Compute the message polynomial.
53    /// Returns an error if the input length is wrong or if there are too many errors to correct.
54    pub fn compute_message_polynomial(&self, code_word: &[S]) -> FastCryptoResult<Poly<S>> {
55        // The implementation follows Algorithm 1 in Gao's paper.
56
57        if code_word.len() != self.block_length() {
58            return Err(InputLengthWrong(self.block_length()));
59        }
60
61        // Step 1: Interpolation
62        let g1 = Poly::interpolate(
63            &self
64                .a
65                .iter()
66                .zip(code_word)
67                .map(|(&index, &value)| Eval { index, value })
68                .collect_vec(),
69        )?;
70
71        // Step 2: Partial GCD
72        let (g, _, v) = Poly::partial_extended_gcd(
73            &self.g0,
74            &g1,
75            (self.message_length() + self.block_length()) / 2,
76        )?;
77
78        // Step 3: Long division
79        let (f1, r) = g.div_rem(&v)?;
80        if !r.is_zero() || f1.degree() >= self.k {
81            return Err(TooManyErrors((self.distance() - 1) / 2));
82        }
83        Ok(f1)
84    }
85
86    /// Encode the message using the Reed-Solomon code defined by the evaluation points `a`.
87    /// Returns an error if the message length is wrong.
88    pub fn encode(&self, message: Vec<S>) -> FastCryptoResult<Vec<S>> {
89        if message.len() != self.message_length() {
90            return Err(InputLengthWrong(self.message_length()));
91        }
92        let f = Poly::from(message);
93        Ok(self.a.iter().map(|&ai| f.eval(ai).value).collect_vec())
94    }
95
96    /// Try to correct the input and return the decoded message.
97    /// Returns an error if the input length is wrong or if there are too many errors to correct.
98    pub fn decode(&self, input: &[S]) -> FastCryptoResult<Vec<S>> {
99        let mut f1 = self.compute_message_polynomial(input)?.to_vec();
100        f1.truncate(self.k);
101        Ok(f1)
102    }
103
104    /// Create a new decoder that can correct the given erasures.
105    /// Returns an InvalidInput error if the given erasures are not unique or not a subset of
106    /// self.a.
107    pub fn with_erasures(&self, erasures: &[ShareIndex]) -> FastCryptoResult<RSDecoder> {
108        // This follows section 4 in Gao's paper
109        let erasures = erasures.iter().sorted().collect_vec();
110        let a = self
111            .a
112            .iter()
113            .filter(|ai| erasures.binary_search(ai).is_err())
114            .cloned()
115            .collect_vec();
116
117        // Check if the erasures is a subset of a, e.g., that we have removed one a_i per erasure.
118        if a.len() + erasures.len() != self.block_length() {
119            return Err(InvalidInput);
120        }
121
122        let g0 = erasures.iter().fold(self.g0.clone(), |g0, ai| {
123            &g0 / MonicLinear(-to_scalar::<S>(*ai))
124        });
125
126        Ok(RSDecoder { g0, a, k: self.k })
127    }
128}

130/// A wrapper struct for the Reed-Solomon erasure coding library.
131pub struct ErasureCoder(ReedSolomon);
132
133/// An element of `GF(2^16)` as represented by the underlying coder.
134type Element = [u8; ELEMENT_SIZE_IN_BYTES];
135
136/// Size in bytes of one `GF(2^16)` element.
137const ELEMENT_SIZE_IN_BYTES: usize = 2;
138
139/// Size in bytes of the `u32` length prefix framing an encoded payload.
140const LEN_PREFIX_SIZE: usize = std::mem::size_of::<u32>();
141
142#[derive(Clone, Debug, Eq, PartialEq, Serialize, Deserialize)]
143#[serde(transparent)]
144pub struct Shard(pub(crate) Vec<u8>);
145
146impl ErasureCoder {
147    /// Create a new erasure encoder/decoder.
148    ///
149    /// # Parameters
150    /// - `n`: Total number of shards.
151    /// - `k`: Number of data shards.
152    ///
153    /// # Errors
154    /// Returns [`FastCryptoError::InvalidInput`] if `k == 0`, `n <= k` or `n > 65536`.
155    pub fn new(n: usize, k: usize) -> FastCryptoResult<Self> {
156        // The code is defined over GF(2^16), which has 2^16 = 65536 elements; n cannot exceed
157        // that or the evaluation points would collide.
158        Self::check_parameters(n, k)?;
159        ReedSolomon::new(k, n - k)
160            .map_err(|_| InvalidInput)
161            .map(Self)
162    }
163
164    pub fn check_parameters(n: usize, k: usize) -> FastCryptoResult<()> {
165        if k == 0 || n <= k || n > 65536 {
166            return Err(InvalidInput);
167        }
168        Ok(())
169    }
170
171    /// Encode `data` into `n` shards of equal size, the first `k` of which hold the (zero-padded)
172    /// data and the remaining `n - k` parity. Any `k` shards suffice to reconstruct the data.
173    pub fn encode(&self, data: &[u8]) -> FastCryptoResult<Vec<Shard>> {
174        if data.is_empty() {
175            return Err(InvalidInput);
176        }
177        let len = u32::try_from(data.len()).map_err(|_| InvalidInput)?;
178        let framed_len = LEN_PREFIX_SIZE + data.len();
179        let shard_size = framed_len.div_ceil(ELEMENT_SIZE_IN_BYTES * self.0.data_shard_count());
180        let bytes_per_shard = ELEMENT_SIZE_IN_BYTES * shard_size;
181        let mut framed = Vec::with_capacity(bytes_per_shard * self.0.total_shard_count());
182        framed.extend_from_slice(&len.to_le_bytes());
183        framed.extend_from_slice(data);
184        framed.resize(bytes_per_shard * self.0.total_shard_count(), 0);
185        let mut shards: Vec<Vec<Element>> = framed
186            .chunks_exact(bytes_per_shard)
187            .map(bytes_to_elements)
188            .collect::<FastCryptoResult<_>>()?;
189        self.0.encode(&mut shards).map_err(|_| InvalidInput)?;
190        Ok(shards.into_iter().map(|s| Shard(s.concat())).collect_vec())
191    }
192
193    /// Reconstruct the original data from `n` (possibly missing) shards. Fails if more than
194    /// `n - k` shards are missing, if the present shards are inconsistent with any single
195    /// codeword, or if the recovered length prefix doesn't fit the recovered bytes.
196    pub fn decode(&self, shards: Vec<Option<Shard>>) -> FastCryptoResult<Vec<u8>> {
197        if shards.len() != self.0.total_shard_count() {
198            return Err(InputLengthWrong(self.0.total_shard_count()));
199        }
200
201        let mut shards: Vec<Option<Vec<Element>>> = shards
202            .into_iter()
203            .map(|opt| {
204                opt.map(|Shard(bytes)| bytes_to_elements(&bytes))
205                    .transpose()
206            })
207            .collect::<FastCryptoResult<_>>()?;
208        self.0.reconstruct(&mut shards).map_err(|_| InvalidInput)?;
209        let shards = shards
210            .into_iter()
211            .map(|s| s.ok_or(InvalidInput))
212            .collect::<FastCryptoResult<Vec<_>>>()?;
213
214        // Ensure the reconstructed shards are consistent
215        if !self.0.verify(&shards).map_err(|_| InvalidInput)? {
216            return Err(TooManyErrors(0)); // This is just an erasure code, so we can't correct errors.
217        }
218
219        let framed: Vec<u8> = shards
220            .into_iter()
221            .take(self.0.data_shard_count())
222            .flatten()
223            .flatten()
224            .collect();
225        if framed.len() < LEN_PREFIX_SIZE {
226            return Err(InvalidInput);
227        }
228        let len = u32::from_le_bytes(framed[..LEN_PREFIX_SIZE].try_into().unwrap()) as usize;
229        let end = LEN_PREFIX_SIZE.checked_add(len).ok_or(InvalidInput)?;
230        if end > framed.len() {
231            return Err(InvalidInput);
232        }
233        Ok(framed[LEN_PREFIX_SIZE..end].to_vec())
234    }
235}
236
237/// Reinterpret `bytes` as a sequence of [Element]s. Fails with [`InvalidInput`] if the input
238/// length is not a multiple of [`ELEMENT_SIZE_IN_BYTES`].
239fn bytes_to_elements(bytes: &[u8]) -> FastCryptoResult<Vec<Element>> {
240    if !bytes.len().is_multiple_of(ELEMENT_SIZE_IN_BYTES) {
241        return Err(InvalidInput);
242    }
243    Ok(bytes
244        .chunks_exact(ELEMENT_SIZE_IN_BYTES)
245        .map(|p| p.try_into().expect("chunk has ELEMENT_SIZE_IN_BYTES bytes"))
246        .collect())
247}

253    #[test]
254    fn test_gao_decoder() {
255        let a = (1..=7).map(|i| ShareIndex::new(i).unwrap()).collect_vec();
256        let k = 3;
257        let decoder = RSDecoder::new(a.clone(), k);
258
259        let message = vec![S::from(11u128), S::from(22u128), S::from(33u128)];
260        let code_word = decoder.encode(message.clone()).unwrap();
261
262        // Introduce errors
263        let mut received = code_word.clone();
264        received[4] = S::from(20u128); // Error at position 4
265        received[2] = S::from(200u128); // Error at position 2
266
267        let decoded_message = decoder.decode(&received).unwrap();
268        assert_eq!(decoded_message, message);
269
270        // Test with too many errors
271        let mut received = code_word.clone();
272        received[4] = S::from(20u128); // Error at position 4
273        received[3] = S::from(2000u128); // Error at position 3
274        received[2] = S::from(200u128); // Error at position 2
275        assert!(decoder.decode(&received).is_err());
276
277        // But with erasure coding, it works!
278        let mut received = code_word.clone();
279        let erasures = vec![a[3], a[2], a[4]];
280        received.remove(4);
281        received.remove(3);
282        received.remove(2);
283        let decoded_message = decoder
284            .with_erasures(&erasures)
285            .unwrap()
286            .decode(&received)
287            .unwrap();
288        assert_eq!(decoded_message, message);
289    }

299    #[test]
300    fn test_erasure_coder_roundtrip() {
301        let n = 10;
302        let k = 6;
303        let coder = ErasureCoder::new(n, k).unwrap();
304
305        for len in [1usize, 2, 3, 7, 8, 31, 32, 33, 100, 255] {
306            let data: Vec<u8> = (0..len)
307                .map(|i| (i as u8).wrapping_mul(31).wrapping_add(7))
308                .collect();
309            let shards = coder.encode(&data).unwrap();
310            assert_eq!(shards.len(), n);
311
312            // Remove up to `parity` shards (erasures) and reconstruct.
313            let mut opt_shards: Vec<Option<Shard>> = shards.into_iter().map(Some).collect();
314            for shard in opt_shards.iter_mut().take(n - k) {
315                *shard = None;
316            }
317
318            let coder = ErasureCoder::new(n, k).unwrap();
319            let recovered = coder.decode(opt_shards).unwrap();
320            assert_eq!(recovered, data);
321        }
322    }

357    #[test]
358    fn test_erasure_coder_encode_shard_lengths() {
359        // Each GF(2^16) element is 2 bytes; the framed payload is `u32_len || data` so the
360        // pre-padding length is `4 + data_len`. Shards are sized to a whole number of pairs with
361        // pair count ⌈(4 + data_len) / (2 · k)⌉.
362        for &(n, k, data_len, expected_shard_bytes) in &[
363            (10, 6, 1, 2),       // ⌈  5 /  12⌉ = 1 pair
364            (10, 6, 8, 2),       // ⌈ 12 /  12⌉ = 1 pair
365            (10, 6, 9, 4),       // ⌈ 13 /  12⌉ = 2 pairs
366            (10, 6, 12, 4),      // ⌈ 16 /  12⌉ = 2 pairs
367            (10, 6, 100, 18),    // ⌈104 /  12⌉ = 9 pairs
368            (800, 268, 2028, 8), // ⌈2032 / 536⌉ = 4 pairs
369        ] {
370            let coder = ErasureCoder::new(n, k).unwrap();
371            let data: Vec<u8> = (0..data_len).map(|i| i as u8).collect();
372            let shards = coder.encode(&data).unwrap();
373            assert_eq!(shards.len(), n, "shard count");
374            for shard in &shards {
375                assert_eq!(
376                    shard.0.len(),
377                    expected_shard_bytes,
378                    "shard byte length for n={n}, k={k}, data_len={data_len}"
379                );
380            }
381        }
382    }
