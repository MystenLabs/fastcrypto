12/// An iterator that yields presigning tuples (t_i, p_i).
13pub struct Presignatures {
14    secret: Vec<LazyPascalMatrixMultiplier<S>>,
15    public: LazyPascalMatrixMultiplier<G>,
16}
17
18impl Iterator for Presignatures {
19    type Item = (Vec<S>, G);
20
21    fn next(&mut self) -> Option<Self::Item> {
22        // `public` drives the length; `secret` is empty for a zero-weight party.
23        let public = self.public.next()?;
24        let secret = self
25            .secret
26            .iter_mut()
27            .map(Iterator::next)
28            .collect::<Option<Vec<_>>>()
29            .expect("secret and public multipliers have equal length");
30        Some((secret, public))
31    }
32
33    fn size_hint(&self) -> (usize, Option<usize>) {
34        let remaining = self.public.len();
35        (remaining, Some(remaining))
36    }
37}
38
39impl ExactSizeIterator for Presignatures {}
40
41impl Presignatures {
42    /// Based on the output of a batched AVSS from multiple dealers, create a presignature generator.
43    ///
44    /// All parties must use the same outputs in the same order, and the output from a dealer with weight `w` should be equal to `batch_size_per_weight * w`.
45    ///
46    /// More parties contributing outputs gives more presignatures, so include as many as possible but at least `params.t` (by weight).
47    ///
48    /// `params.t` is the reconstruction threshold and `params.f` is the Byzantine bound used to size
49    /// the Pascal matrix, which produces `total_weight - params.f` presignatures per nonce position.
50    /// Requires `params.t > params.f`.
51    ///
52    /// An InvalidInput error will be returned if:
53    /// * The total weight of the dealers for the outputs is not at least `params.t`,
54    /// * The batch size of one of the outputs is not divisible by `batch_size_per_weight`,
55    /// * or if batch_size_per_weight is zero.
56    pub fn new(
57        outputs: Vec<ReceiverOutput>,
58        batch_size_per_weight: u16,
59        params: Parameters,
60    ) -> FastCryptoResult<Self> {
61        if batch_size_per_weight == 0 {
62            return Err(InvalidInput);
63        }
64        let batch_size_per_weight = batch_size_per_weight as usize;
65
66        // Recover each dealer's weight from its public key count, which works even for a zero-weight party.
67        let weights = outputs
68            .iter()
69            .map(|o| {
70                let batch_size = o.public_keys.len();
71                (batch_size % batch_size_per_weight == 0)
72                    .then_some(batch_size / batch_size_per_weight)
73                    .ok_or(InvalidInput)
74            })
75            .collect::<FastCryptoResult<Vec<_>>>()?;
76        let total_weight_of_outputs: usize = weights.iter().sum();
77        if total_weight_of_outputs < params.t as usize {
78            return Err(InvalidInput);
79        }
80
81        // The number of rows the Pascal multiplier folds the dealers' shares into.
82        let height = total_weight_of_outputs - params.f as usize;
83
84        // This party's weight, aka it's number of shares
85        let my_weight =
86            get_uniform_value(outputs.iter().map(|o| o.my_shares.weight())).ok_or(InvalidInput)?;
87
88        // Each share's batch must cover exactly the nonces dealt by that dealer.
89        // The zero-weight party holds no shares, so there is nothing to check.
90        if outputs.iter().zip(weights.iter()).any(|(o, w)| {
91            !o.my_shares.shares.is_empty()
92                && o.my_shares
93                    .try_uniform_batch_size()
94                    .ok()
95                    .is_none_or(|bs| bs != *w * batch_size_per_weight)
96        }) {
97            return Err(InvalidInput);
98        }
99
100        // There is one secret presigning output per shares for this party
101        let secret = (0..my_weight as usize)
102            .map(|i| {
103                LazyPascalMatrixMultiplier::new(
104                    height,
105                    (0..batch_size_per_weight)
106                        .map(|j| {
107                            outputs
108                                .iter()
109                                .zip(weights.iter())
110                                .flat_map(|(o, w)| {
111                                    o.my_shares.shares[i].batch[j * w..(j + 1) * w].to_vec()
112                                })
113                                .collect()
114                        })
115                        .collect(),
116                )
117            })
118            .collect_vec();
119
120        let public = LazyPascalMatrixMultiplier::new(
121            height,
122            (0..batch_size_per_weight)
123                .map(|j| {
124                    outputs
125                        .iter()
126                        .zip(weights.iter())
127                        .flat_map(|(o, w)| o.public_keys[j * w..(j + 1) * w].to_vec())
128                        .collect()
129                })
130                .collect_vec(),
131        );
132
133        // Sanity check that the multiplier sizes match the expected nonce count.
134        let expected_len = height * batch_size_per_weight;
135        assert!(secret.iter().all(|s| s.len() == expected_len));
136        assert_eq!(public.len(), expected_len);
137
138        Ok(Self { secret, public })
139    }
140}

142#[cfg(test)]
143mod tests {
144    use super::Presignatures;
145    use crate::threshold_schnorr::batch_avss::{ReceiverOutput, ShareBatch, SharesForNode};
146    use crate::threshold_schnorr::{Parameters, G, S};
147    use crate::types::ShareIndex;
148    use fastcrypto::groups::GroupElement;
149
150    #[test]
151    fn test_new_with_zero_weight_party() {
152        // A zero-weight party gets ReceiverOutputs with empty shares; this must not panic.
153        let batch_size_per_weight: u16 = 2;
154        let params = Parameters { t: 2, f: 1 }; // total weight is 2; requires t > f
155
156        // Two weight-1 dealers: each output has batch_size_per_weight public keys, no shares.
157        let outputs = (0..2)
158            .map(|_| ReceiverOutput {
159                my_shares: SharesForNode { shares: vec![] },
160                public_keys: vec![G::generator(); batch_size_per_weight as usize],
161            })
162            .collect::<Vec<_>>();
163
164        let presignatures = Presignatures::new(outputs, batch_size_per_weight, params).unwrap();
165
166        let total_weight_of_outputs = 2;
167        let expected_len =
168            (total_weight_of_outputs - params.f as usize) * batch_size_per_weight as usize;
169        assert_eq!(presignatures.len(), expected_len);
170
171        let tuples = presignatures.collect::<Vec<_>>();
172        assert_eq!(tuples.len(), expected_len);
173        assert!(tuples.iter().all(|(secret, _public)| secret.is_empty()));
174    }
175
176    #[test]
177    fn test_new_rejects_too_short_batch() {
178        // Each dealer deals batch_size_per_weight nonces per weight, so a weight-1 dealer's share
179        // batch must have batch_size_per_weight entries. A shorter batch must be rejected, not panic.
180        let batch_size_per_weight: u16 = 2;
181        let params = Parameters { t: 2, f: 1 }; // total weight is 2; requires t > f
182
183        let outputs = (0..2)
184            .map(|_| ReceiverOutput {
185                my_shares: SharesForNode {
186                    shares: vec![ShareBatch {
187                        index: ShareIndex::new(1).unwrap(),
188                        batch: vec![S::generator()], // length 1 < expected 2
189                        blinding_share: S::generator(),
190                    }],
191                },
192                public_keys: vec![G::generator(); batch_size_per_weight as usize],
193            })
194            .collect::<Vec<_>>();
195
196        assert!(Presignatures::new(outputs, batch_size_per_weight, params).is_err());
197    }
