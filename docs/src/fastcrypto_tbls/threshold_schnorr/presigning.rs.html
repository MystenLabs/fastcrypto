4use crate::threshold_schnorr::batch_avss::ReceiverOutput;
5use crate::threshold_schnorr::pascal_matrix::LazyPascalMatrixMultiplier;
6use crate::threshold_schnorr::{G, S};
7use crate::types::get_uniform_value;
8use fastcrypto::error::FastCryptoError::InvalidInput;
9use fastcrypto::error::FastCryptoResult;
10use itertools::Itertools;
11
12/// An iterator that yields presigning tuples (i, t_i, p_i).
13pub struct Presignatures {
14    secret: Vec<LazyPascalMatrixMultiplier<S>>,
15    public: LazyPascalMatrixMultiplier<G>,
16    index: usize,
17}
18
19impl Iterator for Presignatures {
20    type Item = (usize, Vec<S>, G);
21
22    fn next(&mut self) -> Option<Self::Item> {
23        let secret = self
24            .secret
25            .iter_mut()
26            .map(Iterator::next)
27            .collect::<Option<Vec<_>>>();
28        let public = self.public.next();
29
30        match (secret, public) {
31            (Some(s), Some(p)) => {
32                self.index += 1;
33                Some((self.index, s, p))
34            }
35            (None, None) => None,
36            _ => unreachable!(),
37        }
38    }
39
40    fn size_hint(&self) -> (usize, Option<usize>) {
41        let remaining = self.public.len();
42        (remaining, Some(remaining))
43    }
44}
45
46impl ExactSizeIterator for Presignatures {}
47
48impl Presignatures {
49    /// Based on the output of a batched AVSS from multiple dealers, create a presignature generator.
50    ///
51    /// All parties must use the same outputs in the same order, and the output from a dealer with weight `w` should be equal to `batch_size_per_weight * w`.
52    ///
53    /// An InvalidInput error will be returned if:
54    /// * The total weight of the dealers for the outputs is not at least 2f+1,
55    /// * The batch size of one of the outputs is not divisible by `batch_size_per_weight`,
56    /// * or if batch_size_per_weight is zero.
57    pub fn new(
58        outputs: Vec<ReceiverOutput>,
59        batch_size_per_weight: u16,
60        f: usize,
61    ) -> FastCryptoResult<Self> {
62        if batch_size_per_weight == 0 {
63            return Err(InvalidInput);
64        }
65
66        // Each node should deal a batch sized proportional to their weight, and the total weight of the outputs should be at least t + f
67        let weights = outputs
68            .iter()
69            .map(|o| {
70                let batch_size = o
71                    .my_shares
72                    .try_uniform_batch_size()
73                    .expect("Checked in batch_avss");
74                (batch_size % batch_size_per_weight as usize == 0)
75                    .then_some(batch_size / batch_size_per_weight as usize)
76                    .ok_or(InvalidInput)
77            })
78            .collect::<FastCryptoResult<Vec<_>>>()?;
79        let total_weight_of_outputs: usize = weights.iter().sum();
80        if total_weight_of_outputs < 2 * f + 1 {
81            return Err(InvalidInput);
82        }
83
84        // This party's weight, aka it's number of shares
85        let my_weight = get_uniform_value(outputs.iter().map(|o| o.my_shares.weight()))
86            .expect("Checked in batch_avss");
87
88        // There is one secret presigning output per shares for this party
89        let secret = (0..my_weight as usize)
90            .map(|i| {
91                LazyPascalMatrixMultiplier::new(
92                    total_weight_of_outputs - f,
93                    (0..batch_size_per_weight as usize)
94                        .map(|j| {
95                            outputs
96                                .iter()
97                                .zip(weights.iter())
98                                .flat_map(|(o, w)| {
99                                    o.my_shares.shares[i].batch[j * w..(j + 1) * w].to_vec()
100                                })
101                                .collect()
102                        })
103                        .collect(),
104                )
105            })
106            .collect_vec();
107
108        let public = LazyPascalMatrixMultiplier::new(
109            total_weight_of_outputs - f,
110            (0..batch_size_per_weight as usize)
111                .map(|j| {
112                    outputs
113                        .iter()
114                        .zip(weights.iter())
115                        .flat_map(|(o, w)| o.public_keys[j * w..(j + 1) * w].to_vec())
116                        .collect()
117                })
118                .collect_vec(),
119        );
120
121        // Sanity checks that the size of the multipliers matches the expected number of nonces that this presigning will give
122        let expected_len = (total_weight_of_outputs - f) * batch_size_per_weight as usize;
123        assert_eq!(
124            get_uniform_value(secret.iter().map(|s| s.len())).unwrap(),
125            expected_len
126        );
127        assert_eq!(public.len(), expected_len);
128
129        Ok(Self {
130            secret,
131            public,
132            index: 0,
133        })
134    }
135}
