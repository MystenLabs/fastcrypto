18/// A complaint by an accuser that it could not decrypt or verify its shares.
19/// Given enough responses to the complaint, the accuser can recover its shares.
20#[derive(Clone, Debug, Serialize, Deserialize)]
21pub struct Complaint {
22    pub(crate) accuser_id: PartyId,
23    pub(crate) proof: RecoveryPackage<EG>,
24}
25
26impl Complaint {
27    /// Try to decrypt the shares for the accuser.
28    pub fn check<S: BCSSerialized>(
29        &self,
30        enc_pk: &ecies_v1::PublicKey<EG>,
31        ciphertext: &ecies_v1::MultiRecipientEncryption<EG>,
32        random_oracle: &RandomOracle,
33        verifier: impl Fn(&S) -> FastCryptoResult<()>,
34    ) -> FastCryptoResult<()> {
35        // Check that the recovery package is valid, and if not, return an error since the complaint is invalid.
36        ciphertext
37            .decrypt_with_recovery_package(
38                &self.proof,
39                &random_oracle.extend(&Recovery(self.accuser_id).to_string()),
40                &random_oracle.extend(&Encryption.to_string()),
41                enc_pk,
42                self.accuser_id as usize,
43            )
44            .and_then(|buffer| check_recovered_shares(&buffer, verifier))
45    }
46
47    pub fn create(
48        accuser_id: PartyId,
49        ciphertext: &ecies_v1::MultiRecipientEncryption<EG>,
50        enc_sk: &ecies_v1::PrivateKey<EG>,
51        random_oracle: &RandomOracle,
52        rng: &mut impl AllowedRng,
53    ) -> Self {
54        Self {
55            accuser_id,
56            proof: ciphertext.create_recovery_package(
57                enc_sk,
58                &random_oracle.extend(&Recovery(accuser_id).to_string()),
59                rng,
60            ),
61        }
62    }
63}

65/// A response to a complaint, containing the responder's shares. Constructed only via
66/// `Receiver::handle_complaint`, which gates on `Complaint::check`.
67#[derive(Debug, Clone, Serialize, Deserialize)]
68pub struct ComplaintResponse<S> {
69    pub(crate) responder_id: PartyId,
70    pub(crate) shares: S,
71}
