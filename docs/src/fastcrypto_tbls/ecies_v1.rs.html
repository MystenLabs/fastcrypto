15/// Simple ECIES encryption using a generic group and AES-256-counter.
16///
17/// Random oracles are extended from two oracles provided by the caller, one for
18/// encryptions/decryptions and one for recovery packages. We assume that the
19/// caller extended the random oracles with the relevant tags and omit them here.
20///
21/// The encryption uses AES Counter mode and is not CCA secure as is.
22///
23
24#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize, ZeroizeOnDrop)]
25pub struct PrivateKey<G: GroupElement>(pub(crate) G::ScalarType)
26where
27    G::ScalarType: Zeroize;
28
29#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
30pub struct PublicKey<G: GroupElement>(pub(crate) G);
31
32/// A recovery package that allows decrypting a *specific* ECIES Encryption.
33/// It also includes a NIZK proof of correctness (DDH-NIZK).
34#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
35pub struct RecoveryPackage<G: GroupElement> {
36    pub(crate) ephemeral_key: G,
37    pub(crate) proof: DdhTupleNizk<G>,
38}
39
40pub const AES_KEY_LENGTH: usize = 32;
41
42/// Multi-recipient encryption with a proof-of-possession of the ephemeral key.
43/// (rG, r RO1(rG), {AES(k=RO2(rPK_i), m_i)}_i, DDH-NIZK(G, RO1(rG), rG, r RO1(rG)) )
44#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
45pub struct MultiRecipientEncryption<G: GroupElement> {
46    c: G,
47    c_hat: G,
48    encs: Vec<Vec<u8>>,
49    proof: DdhTupleNizk<G>,
50}
51
52impl<G: GroupElement + Serialize> MultiRecipientEncryption<G>
53where
54    <G as GroupElement>::ScalarType: FiatShamirChallenge + Zeroize,
55    G: HashToGroupElement,
56{
57    pub fn encrypt<R: AllowedRng>(
58        pk_and_msgs: &[(PublicKey<G>, Vec<u8>)],
59        encryption_random_oracle: &RandomOracle,
60        rng: &mut R,
61    ) -> MultiRecipientEncryption<G> {
62        let r = G::ScalarType::rand(rng);
63        let c = G::generator() * r;
64        let g_hat = G::hash_to_group_element(
65            &Self::g_hat_random_oracle(encryption_random_oracle).evaluate(&c),
66        );
67        let c_hat = g_hat * r;
68        let proof = DdhTupleNizk::<G>::create(
69            &r,
70            &g_hat,
71            &c,
72            &c_hat,
73            &Self::zk_random_oracle(encryption_random_oracle),
74            rng,
75        );
76
77        let encs_ro = Self::encs_random_oracle(encryption_random_oracle);
78        let encs = pk_and_msgs
79            .iter()
80            .enumerate()
81            .map(|(receiver_index, (receiver_pk, msg))| {
82                let pk_r = receiver_pk.0 * r;
83                let k = encs_ro.evaluate(&(receiver_index, pk_r));
84                let cipher = sym_cipher(&k);
85                // Since k is fresh per encryption, we can safely use a fixed nonce.
86                cipher.encrypt(&fixed_zero_nonce(), msg)
87            })
88            .collect::<Vec<_>>();
89
90        MultiRecipientEncryption {
91            c,
92            c_hat,
93            encs,
94            proof,
95        }
96    }
97
98    pub fn verify(&self, encryption_random_oracle: &RandomOracle) -> FastCryptoResult<()> {
99        let g_hat = G::hash_to_group_element(
100            &Self::g_hat_random_oracle(encryption_random_oracle).evaluate(&self.c),
101        );
102        self.proof.verify(
103            &g_hat,
104            &self.c,
105            &self.c_hat,
106            &Self::zk_random_oracle(encryption_random_oracle),
107        )?;
108        // Encryptions should not be empty.
109        self.encs
110            .iter()
111            .all(|e| !e.is_empty())
112            .then_some(())
113            .ok_or(FastCryptoError::InvalidInput)
114    }
115
116    /// Assumption: Verify is called before decrypt and do not call it again here to avoid redundant
117    /// checks.
118    pub fn decrypt(
119        &self,
120        sk: &PrivateKey<G>,
121        encryption_random_oracle: &RandomOracle,
122        receiver_index: usize,
123    ) -> Vec<u8> {
124        let enc_ro = Self::encs_random_oracle(encryption_random_oracle);
125        let ephemeral_key = self.c * sk.0;
126        let k = enc_ro.evaluate(&(receiver_index, ephemeral_key));
127        let cipher = sym_cipher(&k);
128        cipher
129            .decrypt(&fixed_zero_nonce(), &self.encs[receiver_index])
130            .expect("Decrypt should never fail for CTR mode")
131    }
132
133    /// Assumption: Verify is called before create_recovery_package and do not call it again here to
134    /// avoid redundant checks.
135    pub fn create_recovery_package<R: AllowedRng>(
136        &self,
137        sk: &PrivateKey<G>,
138        recovery_random_oracle: &RandomOracle,
139        rng: &mut R,
140    ) -> RecoveryPackage<G> {
141        let pk = G::generator() * sk.0;
142        let ephemeral_key = self.c * sk.0;
143
144        let proof = DdhTupleNizk::<G>::create(
145            &sk.0,
146            &self.c,
147            &pk,
148            &ephemeral_key,
149            recovery_random_oracle,
150            rng,
151        );
152
153        RecoveryPackage {
154            ephemeral_key,
155            proof,
156        }
157    }
158
159    pub fn decrypt_with_recovery_package(
160        &self,
161        pkg: &RecoveryPackage<G>,
162        recovery_random_oracle: &RandomOracle,
163        encryption_random_oracle: &RandomOracle,
164        receiver_pk: &PublicKey<G>,
165        receiver_index: usize,
166    ) -> FastCryptoResult<Vec<u8>> {
167        assert!(receiver_index < self.encs.len());
168        pkg.proof.verify(
169            &self.c,
170            &receiver_pk.0,
171            &pkg.ephemeral_key,
172            recovery_random_oracle,
173        )?;
174        let encs_ro = Self::encs_random_oracle(encryption_random_oracle);
175        let k = encs_ro.evaluate(&(receiver_index, pkg.ephemeral_key));
176        let cipher = sym_cipher(&k);
177        Ok(cipher
178            .decrypt(&fixed_zero_nonce(), &self.encs[receiver_index])
179            .expect("Decrypt should never fail for CTR mode"))
180    }
181
182    pub fn len(&self) -> usize {
183        self.encs.len()
184    }
185    pub fn is_empty(&self) -> bool {
186        self.encs.is_empty()
187    }
188
189    pub fn ephemeral_key(&self) -> &G {
190        &self.c
191    }
192
193    pub fn proof(&self) -> &DdhTupleNizk<G> {
194        &self.proof
195    }
196
197    fn encs_random_oracle(encryption_random_oracle: &RandomOracle) -> RandomOracle {
198        encryption_random_oracle.extend("encs")
199    }
200
201    fn zk_random_oracle(encryption_random_oracle: &RandomOracle) -> RandomOracle {
202        encryption_random_oracle.extend("zk")
203    }
204
205    fn g_hat_random_oracle(encryption_random_oracle: &RandomOracle) -> RandomOracle {
206        encryption_random_oracle.extend("g_hat")
207    }
208
209    #[cfg(test)]
210    pub fn modify_c_hat_for_testing(&mut self, c_hat: G) {
211        self.c_hat = c_hat;
212    }
213
214    #[cfg(test)]
215    pub fn copy_for_testing(&mut self, src: usize, dst: usize) {
216        self.encs[dst] = self.encs[src].clone();
217    }
218}
219
220fn fixed_zero_nonce() -> InitializationVector<U16> {
221    InitializationVector::<U16>::from_bytes(&[0u8; 16])
222        .expect("U16 could always be set from a 16 bytes array of zeros")
223}
224
225fn sym_cipher(k: &[u8; 64]) -> Aes256Ctr {
226    Aes256Ctr::new(
227        AesKey::<U32>::from_bytes(&k[0..U32::USIZE])
228            .expect("New shouldn't fail as use fixed size key is used"),
229    )
230}

232impl<G> PrivateKey<G>
233where
234    G: GroupElement + Serialize,
235    <G as GroupElement>::ScalarType: FiatShamirChallenge + Zeroize,
236{
237    pub fn new<R: AllowedRng>(rng: &mut R) -> Self {
238        Self(G::ScalarType::rand(rng))
239    }
240
241    pub fn from(sc: G::ScalarType) -> Self {
242        Self(sc)
243    }

246impl<G> PublicKey<G>
247where
248    G: GroupElement + Serialize + DeserializeOwned,
249    <G as GroupElement>::ScalarType: FiatShamirChallenge + Zeroize,
250{
251    pub fn from_private_key(sk: &PrivateKey<G>) -> Self {

253    }

255    pub fn as_element(&self) -> &G {
