24/// A polynomial that is using a scalar for the variable x and a generic
25/// element for the coefficients.
26#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
27pub struct Poly<C>(Vec<C>);

29pub type PrivatePoly<C> = Poly<<C as GroupElement>::ScalarType>;
30pub type PublicPoly<C> = Poly<C>;

32/// Vector related operations.
33
34impl<C: GroupElement> Poly<C> {
35    /// Returns an upper bound for the degree of the polynomial.
36    /// The returned number is equal to the size of the underlying coefficient vector - 1,
37    /// and in case some of the leading elements are zero, the actual degree will be smaller.
38    /// See also [Poly::degree].
39    pub fn degree_bound(&self) -> usize {
40        // e.g. c_0 + c_1 * x + c_2 * x^2 + c_3 * x^3
41        // ^ 4 coefficients correspond to a 3rd degree poly
42        self.0.len() - 1
43    }
44
45    /// Returns the degree of the polynomial.
46    pub fn degree(&self) -> usize {
47        self.0.iter().rposition(|&c| c != C::zero()).unwrap_or(0)
48    }
49
50    /// Removes leading zero coefficients.
51    pub(crate) fn into_reduced(mut self) -> Self {
52        self.0.truncate(self.degree() + 1);
53        self
54    }
55}

59        Self(c)
60    }

63impl<C: GroupElement> AddAssign<&Self> for Poly<C> {
64    fn add_assign(&mut self, other: &Self) {
65        self.0.iter_mut().zip(&other.0).for_each(|(a, b)| *a += *b);
66        // if we have a smaller degree we can copy the rest
67        if self.0.len() < other.0.len() {
68            self.0.extend_from_slice(&other.0[self.0.len()..]);
69        }
70    }
71}

81impl<C: Scalar> Mul<&Poly<C>> for &Poly<C> {
82    type Output = Poly<C>;
83
84    fn mul(self, rhs: &Poly<C>) -> Poly<C> {
85        if self.is_zero() || rhs.is_zero() {
86            return Poly::zero();
87        }
88        let mut result = vec![C::zero(); self.degree() + rhs.degree() + 1];
89        for (i, a) in self.0.iter().enumerate() {
90            for (j, b) in rhs.0.iter().enumerate() {
91                result[i + j] += *a * *b;
92            }
93        }
94        Poly::from(result)
95    }
96}

107impl<C: GroupElement> SubAssign<Poly<C>> for Poly<C> {
108    fn sub_assign(&mut self, rhs: Poly<C>) {
109        if self.0.len() < rhs.0.len() {
110            self.0.resize(rhs.0.len(), C::zero());
111        }
112        for (a, b) in self.0.iter_mut().zip(&rhs.0) {
113            *a -= *b;
114        }
115    }
116}

120impl<C: GroupElement> Poly<C> {
121    /// Returns a polynomial with the zero element.
122    pub fn zero() -> Self {
123        Self::from(vec![C::zero()])
124    }
125
126    pub(crate) fn is_zero(&self) -> bool {
127        self.0.iter().all(|&c| c == C::zero())
128    }
129
130    pub fn one() -> Self {
131        Self::from(vec![C::generator()])
132    }
133
134    // TODO: Some of the functions/steps below may be executed many times in practice thus cache can be
135    // used to improve efficiency (e.g., eval(i) may be called with the same index every time a partial
136    // signature from party i is verified).
137
138    /// Evaluates the polynomial at the specified value.
139    pub fn eval(&self, i: ShareIndex) -> Eval<C> {
140        // Use Horner's Method to evaluate the polynomial.
141        let xi: C::ScalarType = to_scalar(i);
142        let res = self
143            .0
144            .iter()
145            .rev()
146            .fold(C::zero(), |sum, coeff| sum * xi + coeff);
147
148        Eval {
149            index: i,
150            value: res,
151        }
152    }
153
154    /// Evaluate the polynomial for all x in the range [1,...,m].
155    /// If m is sufficiently larger than the degree, this is faster than just evaluating at each point.
156    /// If m == 0, the returned range will be empty.
157    ///
158    /// This is based on an algorithm in section 4.6.4 of Knuth's "The Art of Computer Programming".
159    pub fn eval_range(&self, m: u16) -> EvalRange<C> {
160        // The PolynomialEvaluator can't handle m = 0, m = u16::MAX or self.degree() >= u16::MAX,
161        // so in those extreme cases, we just evaluate everything using Polynomial::eval.
162        if m == 0 || m == u16::MAX || self.degree() >= u16::MAX as usize {
163            return EvalRange(
164                (1..m)
165                    .map(|i| self.eval(ShareIndex::new(i).unwrap()).value)
166                    .collect_vec(),
167            );
168        }
169        EvalRange(
170            PolynomialEvaluator::new(
171                self,
172                NonZeroU16::new(1).unwrap(),
173                NonZeroU16::new(1).unwrap(),
174            )
175            .expect("Checked above")
176            .take(m as usize)
177            .map(|e| e.value)
178            .collect_vec(),
179        )
180    }
181
182    /// Multiply x.1 with y using u128s if possible, otherwise convert x.1 to the group element and multiply.
183    /// Invariant: If res = fast_mult(x1, x2, y) then x.0 * x.1 * y = res.0 * res.1.
184    pub(crate) fn fast_mult(x: (C::ScalarType, u128), y: u128) -> (C::ScalarType, u128) {
185        if x.1.leading_zeros() >= (128 - y.leading_zeros()) {
186            (x.0, x.1 * y)
187        } else {
188            (x.0 * C::ScalarType::from(x.1), y)
189        }
190    }
191
192    /// Compute initial * \prod factors.
193    pub(crate) fn fast_product(
194        initial: C::ScalarType,
195        factors: impl Iterator<Item = u128>,
196    ) -> C::ScalarType {
197        let (result, remaining) = factors.fold((initial, 1), |acc, factor| {
198            debug_assert_ne!(factor, 0);
199            Self::fast_mult(acc, factor)
200        });
201        debug_assert_ne!(remaining, 0);
202        result * C::ScalarType::from(remaining)
203    }
204
205    pub(crate) fn get_lagrange_coefficients_for_c0(
206        t: u16,
207        shares: impl Iterator<Item = impl Borrow<Eval<C>>>,
208    ) -> FastCryptoResult<(C::ScalarType, Vec<C::ScalarType>)> {
209        Self::get_lagrange_coefficients_for(0, t, shares)
210    }
211
212    /// Expects exactly t unique shares.
213    /// Returns an error if x is one of the indices.
214    fn get_lagrange_coefficients_for(
215        x: u128,
216        t: u16,
217        shares: impl Iterator<Item = impl Borrow<Eval<C>>>,
218    ) -> FastCryptoResult<(C::ScalarType, Vec<C::ScalarType>)> {
219        let indices = shares.map(|s| s.borrow().index.get() as u128).collect_vec();
220        if !indices.iter().all_unique() || indices.len() != t as usize || indices.contains(&x) {
221            return Err(FastCryptoError::InvalidInput);
222        }
223
224        let x_as_scalar = C::ScalarType::from(x);
225        let full_numerator = C::ScalarType::product(
226            indices
227                .iter()
228                .map(|i| C::ScalarType::from(*i) - x_as_scalar),
229        );
230
231        Ok((
232            full_numerator,
233            indices
234                .iter()
235                .map(|i| {
236                    let mut negative = false;
237                    let mut denominator = Self::fast_product(
238                        C::ScalarType::from(*i) - x_as_scalar,
239                        indices.iter().filter(|j| *j != i).map(|j| {
240                            if i > j {
241                                negative = !negative;
242                                i - j
243                            } else {
244                                // i < j (but not equal)
245                                j - i
246                            }
247                        }),
248                    );
249                    if negative {
250                        denominator = -denominator;
251                    }
252                    denominator.inverse().expect("safe since i != j")
253                })
254                .collect(),
255        ))
256    }
257
258    /// Given exactly `t` polynomial evaluations, it will recover the polynomial's constant term.
259    /// For group elements better use recover_c0_msm.
260    pub fn recover_c0(
261        t: u16,
262        shares: impl Iterator<Item = impl Borrow<Eval<C>>> + Clone,
263    ) -> FastCryptoResult<C> {
264        let coeffs = Self::get_lagrange_coefficients_for_c0(t, shares.clone())?;
265        Ok(C::sum(
266            shares
267                .map(|s| s.borrow().value)
268                .zip(coeffs.1)
269                .map(|(c, s)| c * s),
270        ) * coeffs.0)
271    }
272
273    /// Checks if a given share is valid.
274    pub fn verify_share(&self, idx: ShareIndex, share: &C::ScalarType) -> FastCryptoResult<()> {
275        let e = C::generator() * share;
276        let pub_eval = self.eval(idx);
277        if pub_eval.value == e {
278            Ok(())
279        } else {
280            Err(FastCryptoError::InvalidInput)
281        }
282    }
283
284    /// Return the constant term of the polynomial.
285    pub fn c0(&self) -> &C {
286        &self.0[0]
287    }
288
289    /// Consume the polynomial and return the constant term.
290    pub fn into_c0(self) -> C {
291        self.0[0]
292    }
293
294    pub fn coefficient(&self, i: usize) -> &C {
295        if i >= self.0.len() {
296            panic!(
297                "Index out of bounds: requested {}, but polynomial has degree {}",
298                i,
299                self.degree_bound()
300            );
301        }
302        &self.0[i]
303    }
304
305    /// Returns the i'th coefficient for this polynomial.
306    /// If i is larger than the degree, this just returns a zero.
307    pub fn safe_coefficient(&self, i: usize) -> C {
308        if i >= self.0.len() {
309            return C::zero();
310        }
311        self.0[i]
312    }
313
314    /// Returns the coefficients of the polynomial.
315    pub fn as_vec(&self) -> &Vec<C> {
316        &self.0
317    }
318
319    pub fn to_vec(self) -> Vec<C> {
320        self.0
321    }
322
323    fn sum(terms: impl Iterator<Item = Poly<C>>) -> Poly<C> {
324        terms.fold(Poly::zero(), |sum, term| sum + &term)
325    }

330impl<C: Scalar> Poly<C> {
331    /// Returns a new polynomial of the given degree where each coefficients is
332    /// sampled at random from the given RNG.
333    /// In the context of secret sharing, the threshold is the degree + 1.
334    pub fn rand<R: AllowedRng>(degree: u16, rng: &mut R) -> Self {
335        let coeffs: Vec<C> = (0..=degree).map(|_| C::rand(rng)).collect();
336        Self::from(coeffs)
337    }
338
339    /// Returns a new polynomial of the given degree where the constant term is
340    /// fixed to `c0` and the rest of the coefficients are sampled at random.
341    pub fn rand_fixed_c0<R: AllowedRng>(degree: u16, c0: C, rng: &mut R) -> Self {
342        let mut coeffs = Self::rand(degree, rng).0;
343        coeffs[0] = c0;
344        Self::from(coeffs)
345    }
346
347    /// Commits the scalar polynomial to the group and returns a polynomial over
348    /// the group.
349    pub fn commit<P: GroupElement<ScalarType = C>>(&self) -> Poly<P> {
350        self.0
351            .iter()
352            .map(|c| P::generator() * c)
353            .collect_vec()
354            .into()
355    }
356
357    /// Given a set of shares with unique indices, compute what the value of the interpolated polynomial is at the given index.
358    /// Returns an error if the input is invalid (e.g., empty or duplicate indices).
359    ///
360    /// This is faster than first recovering the polynomial and then evaluating it at the given index.
361    pub fn recover_at(index: ShareIndex, points: &[Eval<C>]) -> FastCryptoResult<Eval<C>> {
362        // If the index we're looking for is already given, we can return that
363        if let Some(point) = points.iter().find(|p| p.index == index) {
364            return Ok(point.clone());
365        }
366        let lagrange_coefficients = Self::get_lagrange_coefficients_for(
367            index.get() as u128,
368            points.len() as u16,
369            points.iter(),
370        )?;
371        let value = C::sum(
372            lagrange_coefficients
373                .1
374                .iter()
375                .zip(points.iter().map(|p| p.value))
376                .map(|(c, s)| s * c),
377        ) * lagrange_coefficients.0;
378        Ok(Eval { index, value })
379    }
380
381    /// Given a set of shares with unique indices, compute the polynomial that
382    /// goes through all the points. The degree of the resulting polynomial is
383    /// at most `points.len() - 1`.
384    /// Returns an error if the input is invalid (e.g., empty or duplicate indices).
385    pub fn interpolate(points: &[Eval<C>]) -> FastCryptoResult<Poly<C>> {
386        if points.is_empty() || !points.iter().map(|p| p.index).all_unique() {
387            return Err(FastCryptoError::InvalidInput);
388        }
389
390        // Compute the full numerator polynomial: (x - x_1)(x - x_2)...(x - x_t)
391        let mut full_numerator = Poly::one();
392        for point in points {
393            full_numerator *= MonicLinear(-to_scalar::<C>(point.index));
394        }
395
396        Ok(Poly::sum(points.iter().enumerate().map(|(i, p_i)| {
397            let x_i = p_i.index.get() as u128;
398            let mut negative = false;
399            let mut denominator = Self::fast_product(
400                C::ScalarType::generator(),
401                points
402                    .iter()
403                    .enumerate()
404                    .filter(|(j, _)| *j != i)
405                    .map(|(_, p_j)| {
406                        let x_j = p_j.index.get() as u128;
407                        if x_i > x_j {
408                            x_i - x_j
409                        } else {
410                            negative = !negative;
411                            x_j - x_i
412                        }
413                    }),
414            );
415            if negative {
416                denominator = -denominator;
417            }
418            (&full_numerator / MonicLinear(-to_scalar::<C>(p_i.index)))
419                * &(p_i.value / denominator).unwrap()
420        }))
421        .into_reduced())
422    }
423
424    /// Returns the leading term of the polynomial.
425    /// If the polynomial is zero, returns a monomial with coefficient zero and degree zero.
426    fn lead(&self) -> Monomial<C> {
427        if self.is_zero() {
428            return Monomial::zero();
429        }
430        let degree = self.degree();
431        Monomial {
432            coefficient: self.0[degree],
433            degree,
434        }
435    }
436
437    /// Divide self by divisor, returning the quotient and remainder.
438    /// Returns an error if divisor is zero.
439    pub fn div_rem(&self, divisor: &Poly<C>) -> FastCryptoResult<(Poly<C>, Poly<C>)> {
440        if divisor.is_zero() {
441            return Err(FastCryptoError::InvalidInput);
442        }
443        let mut remainder = self.clone();
444        let mut quotient = Self::zero();
445
446        // Function to divide a term by the leading term of the divisor.
447        let divider = divisor.lead().divider();
448
449        while !remainder.is_zero() && remainder.degree() >= divisor.degree() {
450            let tmp = divider(&remainder.lead());
451            quotient += &tmp;
452            remainder -= divisor * &tmp;
453            remainder = remainder.into_reduced();
454        }
455        Ok((quotient, remainder))
456    }
457
458    /// Compute the extended GCD of two polynomials.
459    /// Returns (g, x, y, s, t) such that g = self * x + other * y.
460    /// The loop stops when the degree of g is less than degree_bound.
461    pub fn partial_extended_gcd(
462        &self,
463        other: &Poly<C>,
464        degree_bound: usize,
465    ) -> FastCryptoResult<(Poly<C>, Poly<C>, Poly<C>)> {
466        let mut r = (self.clone(), other.clone());
467        let mut s = (Poly::one(), Poly::zero());
468        let mut t = (Poly::zero(), Poly::one());
469
470        while r.0.degree() >= degree_bound && !r.1.is_zero() {
471            let (q, r_new) = r.0.div_rem(&r.1)?;
472            r = (r.1, r_new);
473            r.0 = r.0.into_reduced();
474
475            t.0 -= &q * &t.1;
476            s.0 -= &q * &s.1;
477
478            swap(&mut t.0, &mut t.1);
479            swap(&mut s.0, &mut s.1);
480        }
481        Ok((r.0, s.0, t.0))
482    }
483
484    pub fn extended_gcd(&self, other: &Poly<C>) -> FastCryptoResult<(Poly<C>, Poly<C>, Poly<C>)> {
485        self.partial_extended_gcd(other, 1)
486    }
487}

489impl<C: GroupElement + MultiScalarMul> Poly<C> {
490    /// Given exactly `t` polynomial evaluations, it will recover the polynomial's
491    /// constant term.
492    pub(crate) fn recover_c0_msm(
493        t: u16,
494        shares: impl Iterator<Item = impl Borrow<Eval<C>>> + Clone,
495    ) -> Result<C, FastCryptoError> {
496        let coeffs = Self::get_lagrange_coefficients_for_c0(t, shares.clone())?;
497        let plain_shares = shares.map(|s| s.borrow().value).collect::<Vec<_>>();
498        let res = C::multi_scalar_mul(&coeffs.1, &plain_shares).expect("sizes match") * coeffs.0;
499        Ok(res)
500    }
501
502    /// Scale each of the polynomials with the corresponding scalar and compute the sum.
503    /// Returns an error if the two slices does not have the same length.
504    pub fn multi_scalar_mul(
505        polynomials: &[Poly<C>],
506        scalars: &[C::ScalarType],
507    ) -> FastCryptoResult<Poly<C>> {
508        if polynomials.len() != scalars.len() {
509            return Err(FastCryptoError::InvalidInput);
510        }
511        if polynomials.is_empty() {
512            return Ok(Poly::zero());
513        }
514        let degree = polynomials
515            .iter()
516            .map(Poly::degree)
517            .max()
518            .expect("Is not empty");
519        Ok(Poly(
520            (0..=degree)
521                .map(|i| {
522                    C::multi_scalar_mul(
523                        scalars,
524                        &polynomials
525                            .iter()
526                            .map(|p| p.safe_coefficient(i))
527                            .collect_vec(),
528                    )
529                    .unwrap()
530                })
531                .collect_vec(),
532        ))
533    }
534}

536/// This represents a monomial, e.g., 3x^2, where 3 is the coefficient and 2 is the degree.
537struct Monomial<C> {
538    coefficient: C,
539    degree: usize,
540}
541
542impl<C: GroupElement> AddAssign<&Monomial<C>> for Poly<C> {
543    fn add_assign(&mut self, rhs: &Monomial<C>) {
544        if self.0.len() <= rhs.degree {
545            self.0.resize(rhs.degree + 1, C::zero());
546        }
547        self.0[rhs.degree] += rhs.coefficient;
548    }
549}
550
551impl<C: Scalar> Mul<&Monomial<C>> for &Poly<C> {
552    type Output = Poly<C>;
553
554    fn mul(self, rhs: &Monomial<C>) -> Poly<C> {
555        if rhs.coefficient == C::zero() {
556            return Poly::zero();
557        }
558        let mut result = vec![C::zero(); self.degree_bound() + rhs.degree + 1];
559        for (i, coefficient) in self.0.iter().enumerate() {
560            result[i + rhs.degree] = *coefficient * rhs.coefficient;
561        }
562        Poly::from(result)
563    }
564}

566impl<C: Scalar> Monomial<C> {
567    /// Returns a closure which on input `x` computes the division `x / self`.
568    /// Panics if the degree of `x` is smaller than `self` or if `self` is zero.
569    fn divider(self) -> impl Fn(&Monomial<C>) -> Monomial<C> {
570        let inverse = self.coefficient.inverse().unwrap();
571        move |p: &Monomial<C>| Monomial {
572            coefficient: p.coefficient * inverse,
573            degree: p.degree - self.degree,
574        }
575    }
576
577    fn zero() -> Self {
578        Monomial {
579            coefficient: C::zero(),
580            degree: 0,
581        }
582    }
583}

585/// Represents a monic linear polynomial of the form x + c.
586pub(crate) struct MonicLinear<C>(pub C);
587
588impl<C: Scalar> MulAssign<MonicLinear<C>> for Poly<C> {
589    fn mul_assign(&mut self, rhs: MonicLinear<C>) {
590        if rhs.0 == C::zero() || self.is_zero() {
591            *self = Poly::zero();
592            return;
593        }
594        self.0.push(*self.0.last().unwrap());
595        for i in (1..self.0.len() - 1).rev() {
596            self.0[i] = self.0[i] * rhs.0 + self.0[i - 1];
597        }
598        self.0[0] = self.0[0] * rhs.0;
599    }
600}
601
602impl<C: Scalar> Div<MonicLinear<C>> for &Poly<C> {
603    type Output = Poly<C>;
604
605    fn div(self, rhs: MonicLinear<C>) -> Self::Output {
606        let mut result = self.0[1..].to_vec();
607        for i in (0..result.len() - 1).rev() {
608            result[i] = result[i] - result[i + 1] * rhs.0;
609        }
610        Poly::from(result)
611    }
612}

619/// This can evaluate a polynomial at points in an arithmetic progression, e.g., x0, x0+h, x0+2h, ...
620/// This is generally faster when evaluating more points than the degree of the polynomial.
621/// The algorithm used is from section 4.6.4 in Knuth's "Art of Computer Programming".
622pub(crate) struct PolynomialEvaluator<C> {
623    state: Vec<C>,
624    first: bool,
625    index: NonZeroU16,
626    step: NonZeroU16,
627}
628
629impl<C: GroupElement> PolynomialEvaluator<C> {
630    /// Create a new evaluator.
631    /// Returns an [InvalidInput] error if `initial + step * polynomial.degree()` cannot be represented as an u16.
632    /// Once created, calling [Self::next] will return the evaluation for the next element in the arithmetic progression until the input cannot be represented as an u16.
633    fn new(polynomial: &Poly<C>, initial: NonZeroU16, step: NonZeroU16) -> FastCryptoResult<Self> {
634        // Compute initial values (see exercise 7 in 4.6.4 of TAOCP)
635        let points = (0..=polynomial.degree())
636            .map(|i| {
637                u16::try_from(i)
638                    .ok()
639                    .and_then(|i| i.checked_mul(step.get()))
640                    .and_then(|istep| istep.checked_add(initial.get()))
641                    .and_then(NonZeroU16::new)
642                    .map(|x| polynomial.eval(x).value)
643                    .ok_or(FastCryptoError::InvalidInput)
644            })
645            .collect::<FastCryptoResult<Vec<_>>>()?;
646
647        Ok(Self {
648            state: Self::compute_state(points),
649            first: true,
650            index: initial,
651            step,
652        })
653    }
654
655    /// Given evaluations on 0, 1, ..., degree, this returns an evaluator on points 1, 2, ...
656    pub(crate) fn simple_from_evaluations(points: Vec<C>) -> PolynomialEvaluator<C> {
657        let mut state = Self::compute_state(points);
658
659        // One iteration to skip zero
660        Self::iterate_state(&mut state);
661
662        Self {
663            state,
664            first: true,
665            index: NonZeroU16::new(1).unwrap(),
666            step: NonZeroU16::new(1).unwrap(),
667        }
668    }
669
670    fn compute_state(points: Vec<C>) -> Vec<C> {
671        let mut state = points;
672        for k in 1..state.len() {
673            for j in (k..state.len()).rev() {
674                state[j] = state[j] - state[j - 1];
675            }
676        }
677        state
678    }
679
680    fn iterate_state(state: &mut [C]) {
681        for j in 0..state.len() - 1 {
682            state[j] += state[j + 1]
683        }
684    }
685}
686
687impl<C: GroupElement> Iterator for PolynomialEvaluator<C> {
688    type Item = Eval<C>;
689
690    fn next(&mut self) -> Option<Self::Item> {
691        if self.first {
692            self.first = false;
693        } else {
694            self.index = match self.index.checked_add(self.step.get()) {
695                Some(new_index) => new_index,
696                None => return None,
697            };
698            Self::iterate_state(&mut self.state);
699        }
700        Some(Eval {
701            index: self.index,
702            value: self.state[0],
703        })
704    }
705}

707/// This holds the output of [Poly::eval_range].
708#[derive(Clone, Debug)]
709pub struct EvalRange<C>(Vec<C>);
710
711impl<C: Clone> EvalRange<C> {
712    /// Return all evaluations in this range as a vector, ordered by the indices.
713    pub fn to_vec(self) -> Vec<Eval<C>> {
714        self.into_iter().collect_vec()
715    }
716
717    pub fn take(self, n: u16) -> EvalRange<C> {
718        EvalRange(self.0.into_iter().take(n as usize).collect_vec())
719    }
720
721    #[cfg(test)]
722    #[allow(clippy::len_without_is_empty)]
723    pub fn len(&self) -> usize {
724        self.0.len()
725    }
726
727    /// Get the evaluation point for the given index. Panics if it is out of range.
728    pub fn get_eval(&self, index: ShareIndex) -> Eval<C>
729    where
730        C: Clone,
731    {
732        Eval {
733            index,
734            value: self[index].clone(),
735        }
736    }
737}
738
739impl<C> Index<ShareIndex> for EvalRange<C> {
740    type Output = C;
741
742    fn index(&self, index: ShareIndex) -> &Self::Output {
743        &self.0[index.get() as usize - 1]
744    }
745}
746
747impl<C> IntoIterator for EvalRange<C> {
748    type Item = Eval<C>;
749    type IntoIter = std::iter::Map<
750        core::iter::Enumerate<std::vec::IntoIter<C>>,
751        fn((usize, C)) -> IndexedValue<C>,
752    >;
753
754    fn into_iter(self) -> Self::IntoIter {
755        self.0.into_iter().enumerate().map(|(i, value)| Eval {
756            index: NonZeroU16::new(i as u16 + 1).unwrap(),
757            value,
758        })
759    }
760}

762impl<C: GroupElement> Add for EvalRange<C> {
763    type Output = Self;
764
765    fn add(self, rhs: Self) -> Self::Output {
766        EvalRange(
767            self.0
768                .iter()
769                .zip_eq(rhs.0)
770                .map(|(a, b)| b + a)
771                .collect_vec(),
772        )
773    }
774}

776impl<C: GroupElement> Mul<&C::ScalarType> for EvalRange<C> {
777    type Output = Self;
778
779    fn mul(self, rhs: &C::ScalarType) -> Self::Output {
780        Self(self.0.iter().map(|a| *a * rhs).collect_vec())
781    }
782}

784/// Create `n` shares for a given secret such that `t` shares can reconstruct the secret.
785/// Panics if t == 0 or t > n.
786pub(crate) fn create_secret_sharing<C: Scalar>(
787    rng: &mut impl AllowedRng,
788    secret: C,
789    t: u16,
790    n: u16,
791) -> EvalRange<C> {
792    assert!(t > 0 && t <= n);
793
794    // The first evaluation point (one zero) is given by the secret, and the remaining are random.
795    let evaluations_points = once(secret)
796        .chain(repeat_with(|| C::rand(rng)))
797        .take(t as usize)
798        .collect_vec();
799
800    // Compute evaluations of the polynomial for 1, 2, ..., n using a simple Evaluator
801    let evaluator = PolynomialEvaluator::simple_from_evaluations(evaluations_points);
802    EvalRange(evaluator.take(n as usize).map(|e| e.value).collect())
803}
