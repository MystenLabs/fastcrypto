18/// Types
19
20pub type Eval<A> = IndexedValue<A>;
21
22/// A polynomial that is using a scalar for the variable x and a generic
23/// element for the coefficients.
24#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
25pub struct Poly<C>(Vec<C>);
26
27pub type PrivatePoly<C> = Poly<<C as GroupElement>::ScalarType>;
28pub type PublicPoly<C> = Poly<C>;
29
30/// Vector related operations.
31
32impl<C> Poly<C> {
33    /// Returns the degree of the polynomial
34    pub fn degree(&self) -> usize {
35        // e.g. c_0 + c_1 * x + c_2 * x^2 + c_3 * x^3
36        // ^ 4 coefficients correspond to a 3rd degree poly
37        self.0.len() - 1
38    }
39}
40
41impl<C> From<Vec<C>> for Poly<C> {
42    fn from(c: Vec<C>) -> Self {
43        Self(c)
44    }
45}
46
47impl<C: GroupElement> AddAssign<&Self> for Poly<C> {
48    fn add_assign(&mut self, other: &Self) {
49        self.0.iter_mut().zip(&other.0).for_each(|(a, b)| *a += *b);
50        // if we have a smaller degree we can copy the rest
51        if self.0.len() < other.0.len() {
52            self.0.extend_from_slice(&other.0[self.0.len()..]);
53        }
54    }
55}
56
57/// GroupElement operations.
58
59impl<C: GroupElement> Poly<C> {
60    /// Returns a polynomial with the zero element.
61    pub fn zero() -> Self {
62        Self::from(vec![C::zero()])
63    }
64
65    // TODO: Some of the functions/steps below may be executed many times in practice thus cache can be
66    // used to improve efficiency (e.g., eval(i) may be called with the same index every time a partial
67    // signature from party i is verified).
68
69    /// Evaluates the polynomial at the specified value.
70    pub fn eval(&self, i: ShareIndex) -> Eval<C> {
71        // Use Horner's Method to evaluate the polynomial.
72        let xi = C::ScalarType::from(i.get().into());
73        let res = self
74            .0
75            .iter()
76            .rev()
77            .fold(C::zero(), |sum, coeff| sum * xi + coeff);
78
79        Eval {
80            index: i,
81            value: res,
82        }
83    }
84
85    // Multiply using u128 if possible, otherwise just convert one element to the group element and return the other.
86    pub(crate) fn fast_mult(x: u128, y: u128) -> Either<(C::ScalarType, u128), u128> {
87        if x.leading_zeros() >= (128 - y.leading_zeros()) {
88            Either::Right(x * y)
89        } else {
90            Either::Left((C::ScalarType::from(x), y))
91        }
92    }
93
94    // Expects exactly t unique shares.
95    fn get_lagrange_coefficients_for_c0(
96        t: u16,
97        mut shares: impl Iterator<Item = impl Borrow<Eval<C>>>,
98    ) -> FastCryptoResult<Vec<C::ScalarType>> {
99        let mut ids_set = HashSet::new();
100        let (shares_size_lower, shares_size_upper) = shares.size_hint();
101        let indices = shares.try_fold(
102            Vec::with_capacity(shares_size_upper.unwrap_or(shares_size_lower)),
103            |mut vec, s| {
104                // Check for duplicates.
105                if !ids_set.insert(s.borrow().index) {
106                    return Err(FastCryptoError::InvalidInput); // expected unique ids
107                }
108                vec.push(s.borrow().index.get() as u128);
109                Ok(vec)
110            },
111        )?;
112        if indices.len() != t as usize {
113            return Err(FastCryptoError::InvalidInput);
114        }
115
116        let full_numerator = indices.iter().fold(C::ScalarType::generator(), |acc, i| {
117            acc * C::ScalarType::from(*i)
118        });
119
120        let mut coeffs = Vec::new();
121        for i in &indices {
122            let mut negative = false;
123            let (mut denominator, remaining) = indices.iter().filter(|j| *j != i).fold(
124                (C::ScalarType::from(*i), 1u128),
125                |(prev_acc, remaining), j| {
126                    let diff = if i > j {
127                        negative = !negative;
128                        i - j
129                    } else {
130                        // i < j (but not equal)
131                        j - i
132                    };
133                    debug_assert_ne!(diff, 0);
134                    let either = Self::fast_mult(remaining, diff);
135                    match either {
136                        Either::Left((remaining_as_scalar, diff)) => {
137                            (prev_acc * remaining_as_scalar, diff)
138                        }
139                        Either::Right(new_remaining) => (prev_acc, new_remaining),
140                    }
141                },
142            );
143            debug_assert_ne!(remaining, 0);
144            denominator = denominator * C::ScalarType::from(remaining);
145            if negative {
146                denominator = -denominator;
147            }
148            // TODO: Consider returning full_numerator and dividing once outside instead of here per iteration.
149            let coeff = full_numerator / denominator;
150            coeffs.push(coeff.expect("safe since i != j"));
151        }
152        Ok(coeffs)
153    }
154
155    /// Given exactly `t` polynomial evaluations, it will recover the polynomial's constant term.
156    /// For group elements better use recover_c0_msm.
157    pub fn recover_c0(
158        t: u16,
159        shares: impl Iterator<Item = impl Borrow<Eval<C>>> + Clone,
160    ) -> FastCryptoResult<C> {
161        let coeffs = Self::get_lagrange_coefficients_for_c0(t, shares.clone())?;
162        let plain_shares = shares.map(|s| s.borrow().value);
163        let res = coeffs
164            .iter()
165            .zip(plain_shares)
166            .fold(C::zero(), |acc, (c, s)| acc + (s * *c));
167        Ok(res)
168    }
169
170    /// Checks if a given share is valid.
171    pub fn verify_share(&self, idx: ShareIndex, share: &C::ScalarType) -> FastCryptoResult<()> {
172        let e = C::generator() * share;
173        let pub_eval = self.eval(idx);
174        if pub_eval.value == e {
175            Ok(())
176        } else {
177            Err(FastCryptoError::InvalidInput)
178        }
179    }
180
181    /// Return the constant term of the polynomial.
182    pub fn c0(&self) -> &C {
183        &self.0[0]
184    }
185
186    pub fn coefficient(&self, i: usize) -> &C {
187        if i >= self.0.len() {
188            panic!(
189                "Index out of bounds: requested {}, but polynomial has degree {}",
190                i,
191                self.degree()
192            );
193        }
194        &self.0[i]
195    }
196
197    /// Returns the coefficients of the polynomial.
198    pub fn as_vec(&self) -> &Vec<C> {
199        &self.0
200    }
201}
202
203/// Scalar operations.
204
205impl<C: Scalar> Poly<C> {
206    /// Returns a new polynomial of the given degree where each coefficients is
207    /// sampled at random from the given RNG.
208    /// In the context of secret sharing, the threshold is the degree + 1.
209    pub fn rand<R: AllowedRng>(degree: u16, rng: &mut R) -> Self {
210        let coeffs: Vec<C> = (0..=degree).map(|_| C::rand(rng)).collect();
211        Self::from(coeffs)
212    }
213
214    /// Returns a new polynomial of the given degree where the constant term is
215    /// fixed to `c0` and the rest of the coefficients are sampled at random.
216    pub fn rand_fixed_c0<R: AllowedRng>(degree: u16, c0: C, rng: &mut R) -> Self {
217        let mut coeffs = Self::rand(degree, rng).0;
218        coeffs[0] = c0;
219        Self::from(coeffs)
220    }
221
222    /// Commits the scalar polynomial to the group and returns a polynomial over
223    /// the group.
224    pub fn commit<P: GroupElement<ScalarType = C>>(&self) -> Poly<P> {
225        let commits = self
226            .0
227            .iter()
228            .map(|c| P::generator() * c)
229            .collect::<Vec<P>>();
230
231        Poly::<P>::from(commits)
232    }
233}
234
235impl<C: GroupElement + MultiScalarMul> Poly<C> {
236    /// Given exactly `t` polynomial evaluations, it will recover the polynomial's
237    /// constant term.
238    pub(crate) fn recover_c0_msm(
239        t: u16,
240        shares: impl Iterator<Item = impl Borrow<Eval<C>>> + Clone,
241    ) -> Result<C, FastCryptoError> {
242        let coeffs = Self::get_lagrange_coefficients_for_c0(t, shares.clone())?;
243        let plain_shares = shares.map(|s| s.borrow().value).collect::<Vec<_>>();
244        let res = C::multi_scalar_mul(&coeffs, &plain_shares).expect("sizes match");
245        Ok(res)
246    }
247}
