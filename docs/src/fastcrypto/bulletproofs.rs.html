4//! Implementation of Pedersen Commitments and [Bulletproofs](https://crypto.stanford.edu/bulletproofs/), which are short non-interactive zero-knowledge proofs that require no trusted setup.
5//!
6//! # Example
7//! ```rust
8//! # use fastcrypto::bulletproofs::*;
9//! # use fastcrypto::bulletproofs::Range::Bits16;
10//! # use fastcrypto::groups::ristretto255::RistrettoScalar;
11//! # use fastcrypto::groups::Scalar;
12//! use fastcrypto::pedersen::{Blinding, PedersenCommitment};
13//! let value = 300;
14//! let range = Bits16;
15//! let mut rng = rand::thread_rng();
16//! let blinding = Blinding::rand(&mut rng);
17//! let proof =
18//!    RangeProof::prove(value, &blinding, &range, &mut rng).unwrap();
19//! let commitment = PedersenCommitment::new(&RistrettoScalar::from(value), &blinding);
20//! assert!(proof.verify(&commitment, &range, &mut rng).is_ok());
21//! ```
22
23use crate::error::FastCryptoError::{GeneralOpaqueError, InvalidInput, InvalidProof};
24use crate::error::FastCryptoResult;
25use crate::pedersen::{Blinding, PedersenCommitment, GENERATORS};
26use crate::traits::AllowedRng;
27use bulletproofs::{BulletproofGens, RangeProof as ExternalRangeProof};
28use merlin::Transcript;
29
30/// Bulletproof Range Proofs
31#[derive(Debug)]
32pub struct RangeProof(ExternalRangeProof);
33
34pub enum Range {
35    /// The range [0,...,2^8).
36    Bits8,
37
38    /// The range [0,...,2^16).
39    Bits16,
40
41    /// The range [0,...,2^32).
42    Bits32,
43
44    /// The range [0,...,2^64).
45    Bits64,
46}
47
48impl Range {
49    pub fn is_in_range(&self, value: u64) -> bool {
50        if value == 0 {
51            return true;
52        }
53        value.ilog2() < self.upper_bound_in_bits()
54    }
55
56    fn upper_bound_in_bits(&self) -> u32 {
57        match self {
58            Range::Bits8 => 8,
59            Range::Bits16 => 16,
60            Range::Bits32 => 32,
61            Range::Bits64 => 64,
62        }
63    }
64}
65
66impl RangeProof {
67    /// Prove that the `value` is in the given range using the given commitment blinding.
68    /// This enables creating proofs for an existing commitment.
69    /// Returns an `InvalidInput` error if the value is not in range.
70    pub fn prove(
71        value: u64,
72        blinding: &Blinding,
73        range: &Range,
74        rng: &mut impl AllowedRng,
75    ) -> FastCryptoResult<RangeProof> {
76        Self::prove_batch(&[value], std::slice::from_ref(blinding), range, rng)
77    }
78
79    /// Verifies a range proof: That the commitment is to a value in the given range.
80    pub fn verify(
81        &self,
82        commitment: &PedersenCommitment,
83        range: &Range,
84        rng: &mut impl AllowedRng,
85    ) -> FastCryptoResult<()> {
86        self.verify_batch(std::slice::from_ref(commitment), range, rng)
87    }
88
89    /// Create a proof that all the given `values` are in the range using the given commitment blindings.
90    /// This enables creating proofs for existing commitments.
91    ///
92    /// Fails if
93    /// * any of the `values` are <i>not</i> in the range.
94    /// * `values.len() != blindings.len()`,
95    /// * `values.len()` is not a power of 2,
96    pub fn prove_batch(
97        values: &[u64],
98        blindings: &[Blinding],
99        range: &Range,
100        rng: &mut impl AllowedRng,
101    ) -> FastCryptoResult<RangeProof> {
102        if values.iter().any(|&v| !range.is_in_range(v))
103            || blindings.len() != values.len()
104            || !values.len().is_power_of_two()
105        {
106            return Err(InvalidInput);
107        }
108
109        let bits = range.upper_bound_in_bits() as usize;
110        let bp_gens = BulletproofGens::new(bits, values.len());
111        let mut prover_transcript = Transcript::new(&[]);
112
113        // TODO: Can we avoid calculating the Pedersen commitments here?
114        ExternalRangeProof::prove_multiple_with_rng(
115            &bp_gens,
116            &GENERATORS,
117            &mut prover_transcript,
118            values,
119            &blindings.iter().map(|b| b.0 .0).collect::<Vec<_>>(),
120            bits,
121            rng,
122        )
123        .map(|(proof, _)| RangeProof(proof))
124        .map_err(|_| GeneralOpaqueError)
125    }
126
127    /// Verifies that a range proof that all commitments are to values in the given `range`.
128    pub fn verify_batch(
129        &self,
130        commitments: &[PedersenCommitment],
131        range: &Range,
132        rng: &mut impl AllowedRng,
133    ) -> FastCryptoResult<()> {
134        let bits = range.upper_bound_in_bits() as usize;
135        let bp_gens = BulletproofGens::new(bits, commitments.len());
136        let mut verifier_transcript = Transcript::new(&[]);
137
138        self.0
139            .verify_multiple_with_rng(
140                &bp_gens,
141                &GENERATORS,
142                &mut verifier_transcript,
143                &commitments
144                    .iter()
145                    .map(|c| c.0 .0.compress())
146                    .collect::<Vec<_>>(),
147                bits,
148                rng,
149            )
150            .map_err(|_| InvalidProof)
151    }
152
153    /// Serialize a range proof. The output will be serialized Risretto255 group elements and scalars.
154    /// It follows the format used in https://github.com/dalek-cryptography/bulletproofs/blob/be67b6d5f5ad1c1f54d5511b52e6d645a1313d07/src/range_proof/mod.rs#L59-L76.
155    pub fn to_bytes(&self) -> Vec<u8> {
156        self.0.to_bytes()
157    }
158
159    /// Deserialize a range proof. See also [Self::to_bytes].
160    pub fn from_bytes(bytes: &[u8]) -> FastCryptoResult<Self> {
161        ExternalRangeProof::from_bytes(bytes)
162            .map(RangeProof)
163            .map_err(|_| InvalidInput)
164    }
165}

167#[test]
168fn test_is_in_range() {
169    assert!(Range::Bits8.is_in_range(0));
170    assert!(Range::Bits8.is_in_range(u8::MAX as u64));
171    assert!(!Range::Bits8.is_in_range(1 << 8));
172    assert!(Range::Bits16.is_in_range(0));
173    assert!(Range::Bits16.is_in_range(u16::MAX as u64));
174    assert!(!Range::Bits16.is_in_range(1 << 16));
175    assert!(Range::Bits32.is_in_range(0));
176    assert!(Range::Bits32.is_in_range(u32::MAX as u64));
177    assert!(!Range::Bits32.is_in_range(1 << 32));
178    assert!(Range::Bits64.is_in_range(0));
179    assert!(Range::Bits64.is_in_range(u64::MAX));
180}

182#[test]
183fn test_range_proof_valid() {
184    use crate::groups::ristretto255::RistrettoScalar;
185
186    let range = Range::Bits32;
187    let mut rng = rand::thread_rng();
188    let blinding = Blinding::rand(&mut rng);
189
190    let value = 1u64;
191    let proof = RangeProof::prove(value, &blinding, &range, &mut rng).unwrap();
192    let commitment = PedersenCommitment::new(&RistrettoScalar::from(value), &blinding);
193    assert!(proof.verify(&commitment, &range, &mut rng).is_ok());
194}

196#[test]
197fn test_batch_range_proof_valid() {
198    use crate::groups::ristretto255::RistrettoScalar;
199    let range = Range::Bits32;
200    let mut rng = rand::thread_rng();
201    let values = (1u64..=8).collect::<Vec<_>>();
202    let (commitments, blindings) = values
203        .iter()
204        .map(|&v| PedersenCommitment::commit(&RistrettoScalar::from(v), &mut rng))
205        .unzip::<_, _, Vec<_>, Vec<_>>();
206    let proof = RangeProof::prove_batch(&values, &blindings, &range, &mut rng).unwrap();
207    assert!(proof.verify_batch(&commitments, &range, &mut rng).is_ok());
208}

210#[test]
211fn test_to_from_bytes() {
212    use crate::encoding::{Encoding, Hex};
213    use crate::groups::ristretto255::RistrettoScalar;
214    use crate::serde_helpers::ToFromByteArray;
215
216    let range = Range::Bits32;
217    let mut rng = rand::thread_rng();
218    let values = (1u64..=8).collect::<Vec<_>>();
219    let (commitments, blindings) = values
220        .iter()
221        .map(|&v| PedersenCommitment::commit(&RistrettoScalar::from(v), &mut rng))
222        .unzip::<_, _, Vec<_>, Vec<_>>();
223    commitments
224        .iter()
225        .for_each(|c| println!("{}", Hex::encode(c.0.to_byte_array())));
226    let proof = RangeProof::prove_batch(&values, &blindings, &range, &mut rng).unwrap();
227    let proof_bytes = proof.to_bytes();
228    println!("{}", Hex::encode(&proof_bytes));
229    let reconstructed_proof = RangeProof::from_bytes(&proof_bytes).unwrap();
230    assert!(reconstructed_proof
231        .verify_batch(&commitments, &range, &mut rng)
232        .is_ok());
233}
