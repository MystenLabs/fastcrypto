4//! Implementation of Pedersen Commitments and [Bulletproofs](https://crypto.stanford.edu/bulletproofs/), which are short non-interactive zero-knowledge proofs that require no trusted setup.
5//!
6//! # Example
7//! ```rust
8//! # use fastcrypto::bulletproofs::*;
9//! # use fastcrypto::bulletproofs::Range::Bits16;
10//! # use fastcrypto::groups::ristretto255::RistrettoScalar;
11//! # use fastcrypto::groups::Scalar;
12//! use fastcrypto::pedersen::{Blinding, PedersenCommitment};
13//! let mut rng = rand::thread_rng();
14//! let value = 300;
15//! let (commitment, blinding) = PedersenCommitment::commit_u64(value, &mut rng);
16//! let range = Bits16;
17//! let proof =
18//!    RangeProof::prove(value, &blinding, &range, b"dst", &mut rng).unwrap();
19//! assert!(proof.verify(&commitment, &range, b"dst", &mut rng).is_ok());
20//! ```
21
22use crate::error::FastCryptoError::{GeneralOpaqueError, InvalidInput, InvalidProof};
23use crate::error::FastCryptoResult;
24use crate::pedersen::{Blinding, PedersenCommitment, GENERATORS};
25use crate::traits::AllowedRng;
26use bulletproofs::{BulletproofGens, RangeProof as ExternalRangeProof};
27use itertools::Itertools;
28use merlin::Transcript;

30/// Bulletproof Range Proofs
31#[derive(Debug)]
32pub struct RangeProof(ExternalRangeProof);
33
34pub enum Range {
35    /// The range [0,...,2^8).
36    Bits8,
37
38    /// The range [0,...,2^16).
39    Bits16,
40
41    /// The range [0,...,2^32).
42    Bits32,
43
44    /// The range [0,...,2^64).
45    Bits64,
46}
47
48impl Range {
49    pub fn is_in_range(&self, value: u64) -> bool {
50        if value == 0 {
51            return true;
52        }
53        value.ilog2() < self.upper_bound_in_bits()
54    }
55
56    fn upper_bound_in_bits(&self) -> u32 {
57        match self {
58            Range::Bits8 => 8,
59            Range::Bits16 => 16,
60            Range::Bits32 => 32,
61            Range::Bits64 => 64,
62        }
63    }
64}
65
66impl RangeProof {
67    /// Prove that the `value` is in the given range using the given commitment blinding.
68    /// This enables creating proofs for an existing commitment.
69    /// Returns an `InvalidInput` error if the value is not in range.
70    pub fn prove(
71        value: u64,
72        blinding: &Blinding,
73        range: &Range,
74        dst: &[u8],
75        rng: &mut impl AllowedRng,
76    ) -> FastCryptoResult<RangeProof> {
77        Self::prove_batch(&[value], std::slice::from_ref(blinding), range, dst, rng)
78    }
79
80    /// Verifies a range proof: That the commitment is to a value in the given range.
81    pub fn verify(
82        &self,
83        commitment: &PedersenCommitment,
84        range: &Range,
85        dst: &[u8],
86        rng: &mut impl AllowedRng,
87    ) -> FastCryptoResult<()> {
88        self.verify_batch(std::slice::from_ref(commitment), range, dst, rng)
89    }
90
91    /// Create a proof that all the given `values` are in the range using the given commitment blindings.
92    /// This enables creating proofs for existing commitments.
93    ///
94    /// Fails if
95    /// * any of the `values` are <i>not</i> in the range.
96    /// * `values.len() != blindings.len()`,
97    /// * `values.len()` is not a power of 2,
98    pub fn prove_batch(
99        values: &[u64],
100        blindings: &[Blinding],
101        range: &Range,
102        dst: &[u8],
103        rng: &mut impl AllowedRng,
104    ) -> FastCryptoResult<RangeProof> {
105        if values.iter().any(|&v| !range.is_in_range(v))
106            || blindings.len() != values.len()
107            || !values.len().is_power_of_two()
108        {
109            return Err(InvalidInput);
110        }
111
112        let bits = range.upper_bound_in_bits() as usize;
113        ExternalRangeProof::prove_multiple_with_rng(
114            &BulletproofGens::new(bits, values.len()),
115            &GENERATORS,
116            &mut transcript(dst),
117            values,
118            &blindings.iter().map(Blinding::to_dalek).collect_vec(),
119            bits,
120            rng,
121        )
122        .map(|(proof, _)| RangeProof(proof))
123        .map_err(|_| GeneralOpaqueError)
124    }
125
126    /// Verifies that a range proof that all commitments are to values in the given `range`.
127    pub fn verify_batch(
128        &self,
129        commitments: &[PedersenCommitment],
130        range: &Range,
131        dst: &[u8],
132        rng: &mut impl AllowedRng,
133    ) -> FastCryptoResult<()> {
134        let bits = range.upper_bound_in_bits() as usize;
135        self.0
136            .verify_multiple_with_rng(
137                &BulletproofGens::new(bits, commitments.len()),
138                &GENERATORS,
139                &mut transcript(dst),
140                &commitments
141                    .iter()
142                    .map(PedersenCommitment::to_dalek)
143                    .collect_vec(),
144                bits,
145                rng,
146            )
147            .map_err(|_| InvalidProof)
148    }
149
150    /// Serialize a range proof. The output will be serialized Risretto255 group elements and scalars.
151    /// It follows the format used in https://github.com/dalek-cryptography/bulletproofs/blob/be67b6d5f5ad1c1f54d5511b52e6d645a1313d07/src/range_proof/mod.rs#L59-L76.
152    pub fn to_bytes(&self) -> Vec<u8> {
153        self.0.to_bytes()
154    }
155
156    /// Deserialize a range proof. See also [Self::to_bytes].
157    pub fn from_bytes(bytes: &[u8]) -> FastCryptoResult<Self> {
158        ExternalRangeProof::from_bytes(bytes)
159            .map(RangeProof)
160            .map_err(|_| InvalidInput)
161    }
162}
163
164/// Create a transcript bound to the given domain separation tag.
165fn transcript(dst: &[u8]) -> Transcript {
166    let mut transcript = Transcript::new(&[]);
167    transcript.append_message(b"DST", dst);
168    transcript
169}

171#[test]
172fn test_is_in_range() {
173    assert!(Range::Bits8.is_in_range(0));
174    assert!(Range::Bits8.is_in_range(u8::MAX as u64));
175    assert!(!Range::Bits8.is_in_range(1 << 8));
176    assert!(Range::Bits16.is_in_range(0));
177    assert!(Range::Bits16.is_in_range(u16::MAX as u64));
178    assert!(!Range::Bits16.is_in_range(1 << 16));
179    assert!(Range::Bits32.is_in_range(0));
180    assert!(Range::Bits32.is_in_range(u32::MAX as u64));
181    assert!(!Range::Bits32.is_in_range(1 << 32));
182    assert!(Range::Bits64.is_in_range(0));
183    assert!(Range::Bits64.is_in_range(u64::MAX));
184}

186#[test]
187fn test_range_proof() {
188    let range = Range::Bits16;
189    let mut rng = rand::thread_rng();
190    let value = 1234u64;
191    let (commitment, blinding) = PedersenCommitment::commit_u64(value, &mut rng);
192
193    let proof = RangeProof::prove(value, &blinding, &range, b"test", &mut rng).unwrap();
194    assert!(RangeProof::prove(value, &blinding, &Range::Bits8, b"test", &mut rng).is_err());
195
196    assert!(proof.verify(&commitment, &range, b"test", &mut rng).is_ok());
197    assert!(proof
198        .verify(&commitment, &range, b"other", &mut rng)
199        .is_err());
200}

202#[test]
203fn test_batch_range_proof_valid() {
204    let range = Range::Bits32;
205    let mut rng = rand::thread_rng();
206    let values = (1u64..=8).collect::<Vec<_>>();
207    let (commitments, blindings) = values
208        .iter()
209        .map(|&v| PedersenCommitment::commit_u64(v, &mut rng))
210        .unzip::<_, _, Vec<_>, Vec<_>>();
211    let proof =
212        RangeProof::prove_batch(&values, &blindings, &range, b"test_dst", &mut rng).unwrap();
213    assert!(proof
214        .verify_batch(&commitments, &range, b"test_dst", &mut rng)
215        .is_ok());
216}

218#[test]
219fn test_to_from_bytes() {
220    let range = Range::Bits32;
221    let mut rng = rand::thread_rng();
222    let values = (1u64..=8).collect::<Vec<_>>();
223    let (commitments, blindings) = values
224        .iter()
225        .map(|&v| PedersenCommitment::commit_u64(v, &mut rng))
226        .unzip::<_, _, Vec<_>, Vec<_>>();
227    let proof = RangeProof::prove_batch(&values, &blindings, &range, b"test", &mut rng).unwrap();
228    let proof_bytes = proof.to_bytes();
229    let reconstructed_proof = RangeProof::from_bytes(&proof_bytes).unwrap();
230    assert!(reconstructed_proof
231        .verify_batch(&commitments, &range, b"test", &mut rng)
232        .is_ok());
233}
