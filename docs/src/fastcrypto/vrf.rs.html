7/// Represents a public key of which is use to verify outputs for a verifiable random function (VRF).
8pub trait VRFPublicKey {
9    type PrivateKey: VRFPrivateKey<PublicKey = Self>;
10}
11
12/// Represents a private key used to compute outputs for a verifiable random function (VRF).
13pub trait VRFPrivateKey {
14    type PublicKey: VRFPublicKey<PrivateKey = Self>;
15}
16
17/// A keypair for a verifiable random function (VRF).
18pub trait VRFKeyPair<const OUTPUT_SIZE: usize> {
19    type Proof: VRFProof<OUTPUT_SIZE, PublicKey = Self::PublicKey>;
20    type PrivateKey: VRFPrivateKey<PublicKey = Self::PublicKey>;
21    type PublicKey: VRFPublicKey<PrivateKey = Self::PrivateKey>;
22
23    /// Generate a new keypair using the given RNG.
24    fn generate<R: AllowedRng>(rng: &mut R) -> Self;
25
26    /// Generate a proof for the given input.
27    fn prove(&self, input: &[u8]) -> Self::Proof;
28
29    /// Compute both hash and proof for the given input.
30    fn output(&self, input: &[u8]) -> ([u8; OUTPUT_SIZE], Self::Proof) {
31        let proof = self.prove(input);
32        let output = proof.to_hash();
33        (output, proof)
34    }
35}
36
37/// A proof that the output of a VRF was computed correctly.
38pub trait VRFProof<const OUTPUT_SIZE: usize> {
39    type PublicKey: VRFPublicKey;
40
41    /// Verify the correctness of this proof.
42    fn verify(&self, input: &[u8], public_key: &Self::PublicKey) -> Result<(), FastCryptoError>;
43
44    /// Verify the correctness of this proof and VRF output.
45    fn verify_output(
46        &self,
47        input: &[u8],
48        public_key: &Self::PublicKey,
49        output: &[u8; OUTPUT_SIZE],
50    ) -> Result<(), FastCryptoError> {
51        self.verify(input, public_key)?;
52        if &self.to_hash() != output {
53            return Err(FastCryptoError::GeneralOpaqueError);
54        }
55        Ok(())
56    }
57
58    /// Compute the output of the VRF with this proof.
59    fn to_hash(&self) -> [u8; OUTPUT_SIZE];
60}

62/// An implementation of an Elliptic Curve VRF (ECVRF) using the Ristretto255 group.
63/// The implementation follows the specifications in draft-irtf-cfrg-vrf-15
64/// (https://datatracker.ietf.org/doc/draft-irtf-cfrg-vrf/).
65pub mod ecvrf {
66    use crate::error::FastCryptoError;
67    use crate::groups::ristretto255::{RistrettoPoint, RistrettoScalar};
68    use crate::groups::{GroupElement, MultiScalarMul, Scalar};
69    use crate::hash::{HashFunction, ReverseWrapper, Sha512};
70    use crate::serde_helpers::ToFromByteArray;
71    use crate::traits::AllowedRng;
72    use crate::vrf::{VRFKeyPair, VRFPrivateKey, VRFProof, VRFPublicKey};
73    use elliptic_curve::hash2curve::{ExpandMsg, Expander};
74    use serde::{Deserialize, Serialize};
75    use zeroize::ZeroizeOnDrop;
76
77    /// draft-irtf-cfrg-vrf-15 specifies suites for suite-strings 0x00-0x04 and notes that future
78    /// designs should specify a different suite_string constant, so we use "sui_vrf" here.
79    const SUITE_STRING: &[u8; 7] = b"sui_vrf";
80
81    /// Length of challenges. Must not exceed the length of field elements which is 32 in this case.
82    /// We set C_LEN = 16 which is the same as the existing ECVRF suites in draft-irtf-cfrg-vrf-15.
83    const C_LEN: usize = 16;
84
85    /// Default hash function
86    type H = Sha512;
87
88    /// Domain separation tag used in ecvrf_encode_to_curve (see also draft-irtf-cfrg-hash-to-curve-16)
89    const DST: &[u8; 49] = b"ECVRF_ristretto255_XMD:SHA-512_R255MAP_RO_sui_vrf";
90
91    #[derive(Serialize, Deserialize, PartialEq, Eq, Debug)]
92    pub struct ECVRFPublicKey(RistrettoPoint);
93
94    impl VRFPublicKey for ECVRFPublicKey {
95        type PrivateKey = ECVRFPrivateKey;
96    }
97
98    impl ECVRFPublicKey {
99        /// Encode the given binary string as curve point. See section 5.4.1.2 of draft-irtf-cfrg-vrf-15.
100        fn ecvrf_encode_to_curve(&self, alpha_string: &[u8]) -> RistrettoPoint {
101            // This follows section 5.4.1.2 of draft-irtf-cfrg-vrf-15 for the ristretto255 group using
102            // SHA-512. The hash-to-curve for ristretto255 follows appendix B of draft-irtf-cfrg-hash-to-curve-16.
103
104            // Compute expand_message_xmd for the given message. Note that expand_message only returns
105            // and error if the len_in_bytes and output size of the hash function is out of bounds
106            // (https://github.com/mikelodder7/hash2field/blob/cdf56a2b722aeae25b8019945afe4cccec132f25/src/expand_msg_xmd.rs#L21),
107            // so we can safely unwrap since they are constants here.
108            let mut expanded_message = elliptic_curve::hash2curve::ExpandMsgXmd::<
109                <H as ReverseWrapper>::Variant,
110            >::expand_message(
111                &[&self.0.to_byte_array(), alpha_string],
112                &[DST],
113                H::OUTPUT_SIZE,
114            )
115            .unwrap();
116
117            let mut bytes = [0u8; H::OUTPUT_SIZE];
118            expanded_message.fill_bytes(&mut bytes);
119            RistrettoPoint::from_uniform_bytes(&bytes)
120        }
121
122        /// Implements ECVRF_validate_key which checks the validity of a public key. See section 5.4.5
123        /// of draft-irtf-cfrg-vrf-15.
124        fn valid(&self) -> bool {
125            self.0 != RistrettoPoint::zero()
126        }
127    }
128
129    #[derive(Serialize, Deserialize, PartialEq, Eq, Debug, ZeroizeOnDrop)]
130    pub struct ECVRFPrivateKey(RistrettoScalar);
131
132    impl VRFPrivateKey for ECVRFPrivateKey {
133        type PublicKey = ECVRFPublicKey;
134    }
135
136    impl ECVRFPrivateKey {
137        /// Generate scalar/nonce from binary string. See section 5.4.2.2. of draft-irtf-cfrg-vrf-15.
138        fn ecvrf_nonce_generation(&self, h_string: &[u8]) -> RistrettoScalar {
139            let hashed_sk_string = H::digest(self.0.to_byte_array());
140            let mut truncated_hashed_sk_string = [0u8; 32];
141            truncated_hashed_sk_string.copy_from_slice(&hashed_sk_string.digest[32..64]);
142
143            let mut hash_function = H::default();
144            hash_function.update(truncated_hashed_sk_string);
145            hash_function.update(h_string);
146            let k_string = hash_function.finalize();
147
148            RistrettoScalar::from_bytes_mod_order_wide(&k_string.digest)
149        }
150    }
151
152    #[derive(Serialize, Deserialize, PartialEq, Eq, Debug)]
153    pub struct ECVRFKeyPair {
154        pub pk: ECVRFPublicKey,
155        pub sk: ECVRFPrivateKey,
156    }
157
158    /// Generate challenge from five points. See section 5.4.3. of draft-irtf-cfrg-vrf-15.
159    fn ecvrf_challenge_generation(points: [&RistrettoPoint; 5]) -> Challenge {
160        let mut hash = H::default();
161        hash.update(SUITE_STRING);
162        hash.update([0x02]); //challenge_generation_domain_separator_front
163        points
164            .into_iter()
165            .for_each(|p| hash.update(p.to_byte_array()));
166        hash.update([0x00]); //challenge_generation_domain_separator_back
167        let digest = hash.finalize();
168
169        let mut challenge_bytes = [0u8; C_LEN];
170        challenge_bytes.copy_from_slice(&digest.digest[..C_LEN]);
171        Challenge(challenge_bytes)
172    }
173
174    /// Type representing a scalar of [C_LEN] bytes.
175    #[derive(Serialize, Deserialize, PartialEq, Eq, Debug)]
176    struct Challenge([u8; C_LEN]);
177
178    impl From<&Challenge> for RistrettoScalar {
179        fn from(c: &Challenge) -> Self {
180            let mut scalar = [0u8; 32];
181            scalar[..C_LEN].copy_from_slice(&c.0);
182            RistrettoScalar::from_bytes_mod_order(&scalar)
183        }
184    }

186    impl VRFKeyPair<64> for ECVRFKeyPair {
187        type Proof = ECVRFProof;
188        type PrivateKey = ECVRFPrivateKey;
189        type PublicKey = ECVRFPublicKey;
190
191        fn generate<R: AllowedRng>(rng: &mut R) -> Self {
192            let s = RistrettoScalar::rand(rng);
193            ECVRFKeyPair::from(ECVRFPrivateKey(s))
194        }
195
196        fn prove(&self, alpha_string: &[u8]) -> ECVRFProof {
197            // Follows section 5.1 of draft-irtf-cfrg-vrf-15.
198
199            let h = self.pk.ecvrf_encode_to_curve(alpha_string);
200            let h_string = h.to_byte_array();
201            let gamma = h * self.sk.0;
202            let k = self.sk.ecvrf_nonce_generation(&h_string);
203
204            let c = ecvrf_challenge_generation([
205                &self.pk.0,
206                &h,
207                &gamma,
208                &(RistrettoPoint::generator() * k),
209                &(h * k),
210            ]);
211            let s = k + RistrettoScalar::from(&c) * self.sk.0;
212
213            ECVRFProof { gamma, c, s }
214        }
215    }
216
217    impl From<ECVRFPrivateKey> for ECVRFKeyPair {
218        fn from(sk: ECVRFPrivateKey) -> Self {
219            let p = RistrettoPoint::generator() * sk.0;
220            ECVRFKeyPair {
221                pk: ECVRFPublicKey(p),
222                sk,
223            }
224        }
225    }

227    #[derive(Serialize, Deserialize, PartialEq, Eq, Debug)]
228    pub struct ECVRFProof {
229        gamma: RistrettoPoint,
230        c: Challenge,
231        s: RistrettoScalar,
232    }
233
234    impl VRFProof<64> for ECVRFProof {
235        type PublicKey = ECVRFPublicKey;
236
237        fn verify(
238            &self,
239            alpha_string: &[u8],
240            public_key: &Self::PublicKey,
241        ) -> Result<(), FastCryptoError> {
242            // Follows section 5.3 of draft-irtf-cfrg-vrf-15.
243
244            if !public_key.valid() {
245                return Err(FastCryptoError::InvalidInput);
246            }
247
248            let h = public_key.ecvrf_encode_to_curve(alpha_string);
249
250            let challenge = RistrettoScalar::from(&self.c);
251            let u = RistrettoPoint::multi_scalar_mul(
252                &[self.s, -challenge],
253                &[RistrettoPoint::generator(), public_key.0],
254            )?;
255            let v = RistrettoPoint::multi_scalar_mul(&[self.s, -challenge], &[h, self.gamma])?;
256
257            let c_prime = ecvrf_challenge_generation([&public_key.0, &h, &self.gamma, &u, &v]);
258
259            if c_prime != self.c {
260                return Err(FastCryptoError::GeneralOpaqueError);
261            }
262            Ok(())
263        }
264
265        fn to_hash(&self) -> [u8; 64] {
266            // Follows section 5.2 of draft-irtf-cfrg-vrf-15.
267            let mut hash = H::default();
268            hash.update(SUITE_STRING);
269            hash.update([0x03]); // proof_to_hash_domain_separator_front
270            hash.update(self.gamma.to_byte_array());
271            hash.update([0x00]); // proof_to_hash_domain_separator_back
272            hash.finalize().digest
273        }
274    }
