4use crate::hash::Digest;
5use crate::traits;
6use crate::traits::Signer;
7use tokio::sync::mpsc::{channel, Sender};
8use tokio::sync::oneshot;
9
10/// This service holds the node's private key. It takes digests as input and returns a signature
11/// over the digest (through a one-shot channel).
12#[derive(Clone)]
13pub struct SignatureService<Signature: traits::Authenticator, const DIGEST_LEN: usize> {
14    channel: Sender<(Digest<DIGEST_LEN>, oneshot::Sender<Signature>)>,
15}
16
17impl<Signature: traits::Authenticator, const DIGEST_LEN: usize>
18    SignatureService<Signature, DIGEST_LEN>
19{
20    pub fn new<S>(signer: S) -> Self
21    where
22        S: Signer<Signature> + Send + 'static,
23    {
24        let (tx, mut rx): (Sender<(Digest<DIGEST_LEN>, oneshot::Sender<_>)>, _) = channel(100);
25        tokio::spawn(async move {
26            while let Some((digest, sender)) = rx.recv().await {
27                let signature = signer.sign(digest.as_ref());
28                let _ = sender.send(signature);
29            }
30        });
31        Self { channel: tx }
32    }
33
34    pub async fn request_signature(&self, digest: Digest<DIGEST_LEN>) -> Signature {
35        let (sender, receiver): (oneshot::Sender<_>, oneshot::Receiver<_>) = oneshot::channel();
36        if let Err(e) = self.channel.send((digest, sender)).await {
37            panic!("Failed to send message Signature Service: {e}");
38        }
39        receiver
40            .await
41            .expect("Failed to receive signature from Signature Service")
42    }
43}
