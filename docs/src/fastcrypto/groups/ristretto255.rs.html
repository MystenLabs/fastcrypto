31pub const RISTRETTO_POINT_BYTE_LENGTH: usize = 32;
32pub const RISTRETTO_SCALAR_BYTE_LENGTH: usize = 32;

34/// Represents a point in the Ristretto group for Curve25519.
35#[derive(Default, Clone, Copy, Debug, PartialEq, Eq, Add, Sub, Neg, GroupOpsExtend)]
36pub struct RistrettoPoint(pub(crate) ExternalPoint);
37
38impl RistrettoPoint {
39    /// Construct a RistrettoPoint from the given data using a Ristretto-flavoured Elligator 2 map.
40    /// If the input bytes are uniformly distributed, the resulting point will be uniformly
41    /// distributed over the Ristretto group.
42    pub fn from_uniform_bytes(bytes: &[u8; 64]) -> Self {
43        RistrettoPoint(ExternalPoint::from_uniform_bytes(bytes))
44    }
45}
46
47impl Doubling for RistrettoPoint {
48    fn double(self) -> Self {
49        Self(self.0.add(self.0))
50    }
51}
52
53impl MultiScalarMul for RistrettoPoint {
54    fn multi_scalar_mul(scalars: &[Self::ScalarType], points: &[Self]) -> FastCryptoResult<Self> {
55        if scalars.len() != points.len() {
56            return Err(InvalidInput);
57        }
58
59        Ok(RistrettoPoint(ExternalPoint::vartime_multiscalar_mul(
60            scalars.iter().map(|s| s.0),
61            points.iter().map(|g| g.0),
62        )))
63    }
64}
65
66#[allow(clippy::suspicious_arithmetic_impl)]
67impl Div<RistrettoScalar> for RistrettoPoint {
68    type Output = Result<Self, FastCryptoError>;
69
70    fn div(self, rhs: RistrettoScalar) -> Self::Output {
71        let inv = rhs.inverse()?;
72        Ok(self * inv)
73    }
74}
75
76impl Mul<RistrettoScalar> for RistrettoPoint {
77    type Output = RistrettoPoint;
78
79    fn mul(self, rhs: RistrettoScalar) -> RistrettoPoint {
80        RistrettoPoint(self.0 * rhs.0)
81    }
82}
83
84impl GroupElement for RistrettoPoint {
85    type ScalarType = RistrettoScalar;
86
87    fn zero() -> RistrettoPoint {
88        RistrettoPoint(ExternalPoint::identity())
89    }
90
91    fn generator() -> Self {
92        RistrettoPoint(RISTRETTO_BASEPOINT_POINT)
93    }
94}
95
96impl HashToGroupElement for RistrettoPoint {
97    fn hash_to_group_element(msg: &[u8]) -> Self {
98        Self::from_uniform_bytes(&Sha512::digest(msg).digest)
99    }
100}
101
102impl ToFromByteArray<RISTRETTO_POINT_BYTE_LENGTH> for RistrettoPoint {
103    fn from_byte_array(bytes: &[u8; RISTRETTO_POINT_BYTE_LENGTH]) -> FastCryptoResult<Self> {
104        Option::from(ExternalPoint::from_bytes(bytes).map(RistrettoPoint)).ok_or(InvalidInput)
105    }
106
107    fn to_byte_array(&self) -> [u8; RISTRETTO_POINT_BYTE_LENGTH] {
108        self.0.compress().0
109    }
110}
111
112impl FromTrustedByteArray<RISTRETTO_POINT_BYTE_LENGTH> for RistrettoPoint {
113    fn from_trusted_byte_array(
114        bytes: &[u8; RISTRETTO_POINT_BYTE_LENGTH],
115    ) -> FastCryptoResult<Self> {
116        // Note that the external crate does not distinguish between from_bytes and from_bytes_unchecked:
117        // https://github.com/dalek-cryptography/curve25519-dalek/blob/11f5375375d3d52c153049f18bd8b1b7669c2565/curve25519-dalek/src/ristretto.rs#L1221-L1224
118        Option::from(ExternalPoint::from_bytes_unchecked(bytes).map(RistrettoPoint))
119            .ok_or(InvalidInput)
120    }
121}

125/// Represents a scalar.
126#[derive(Clone, Copy, Debug, PartialEq, Eq, Add, Sub, Neg, Div, GroupOpsExtend, Zeroize)]
127pub struct RistrettoScalar(pub(crate) ExternalScalar);
128
129impl RistrettoScalar {
130    /// Construct a [RistrettoScalar] by reducing a 64-byte little-endian integer modulo the group order.
131    pub fn from_bytes_mod_order_wide(bytes: &[u8; 64]) -> Self {
132        RistrettoScalar(ExternalScalar::from_bytes_mod_order_wide(bytes))
133    }
134
135    /// Construct a [RistrettoScalar] by reducing a 32-byte little-endian integer modulo the group order.
136    pub fn from_bytes_mod_order(bytes: &[u8; 32]) -> Self {
137        RistrettoScalar(ExternalScalar::from_bytes_mod_order(*bytes))
138    }
139}
140
141impl From<u128> for RistrettoScalar {
142    fn from(value: u128) -> RistrettoScalar {
143        RistrettoScalar(ExternalScalar::from(value))
144    }
145}
146
147impl From<u64> for RistrettoScalar {
148    fn from(value: u64) -> RistrettoScalar {
149        RistrettoScalar(ExternalScalar::from(value))
150    }
151}
152
153impl Mul<RistrettoScalar> for RistrettoScalar {
154    type Output = RistrettoScalar;
155
156    fn mul(self, rhs: RistrettoScalar) -> RistrettoScalar {
157        RistrettoScalar(self.0 * rhs.0)
158    }
159}
160
161#[allow(clippy::suspicious_arithmetic_impl)]
162impl Div<RistrettoScalar> for RistrettoScalar {
163    type Output = Result<RistrettoScalar, FastCryptoError>;
164
165    fn div(self, rhs: RistrettoScalar) -> Result<RistrettoScalar, FastCryptoError> {
166        let inv = rhs.inverse()?;
167        Ok(self * inv)
168    }
169}
170
171impl GroupElement for RistrettoScalar {
172    type ScalarType = Self;
173
174    fn zero() -> Self {
175        RistrettoScalar(ExternalScalar::ZERO)
176    }
177    fn generator() -> Self {
178        RistrettoScalar(ExternalScalar::ONE)
179    }
180}
181
182impl Scalar for RistrettoScalar {
183    fn rand<R: AllowedRng>(rng: &mut R) -> Self {
184        Self(ExternalScalar::random(rng))
185    }
186
187    fn inverse(&self) -> FastCryptoResult<Self> {
188        if self.0.is_zero().into() {
189            return Err(InvalidInput);
190        }
191        Ok(RistrettoScalar(self.0.invert()))
192    }
193}
194
195impl HashToGroupElement for RistrettoScalar {
196    fn hash_to_group_element(bytes: &[u8]) -> Self {
197        Self::from_bytes_mod_order_wide(&Sha512::digest(bytes).digest)
198    }
199}
200
201impl FiatShamirChallenge for RistrettoScalar {
202    fn fiat_shamir_reduction_to_group_element(msg: &[u8]) -> Self {
203        Self::hash_to_group_element(msg)
204    }
205}
206
207impl ToFromByteArray<RISTRETTO_SCALAR_BYTE_LENGTH> for RistrettoScalar {
208    fn from_byte_array(
209        bytes: &[u8; RISTRETTO_SCALAR_BYTE_LENGTH],
210    ) -> Result<Self, FastCryptoError> {
211        Ok(RistrettoScalar(
212            Option::from(ExternalScalar::from_canonical_bytes(*bytes)).ok_or(InvalidInput)?,
213        ))
214    }
215
216    fn to_byte_array(&self) -> [u8; RISTRETTO_SCALAR_BYTE_LENGTH] {
217        self.0.to_bytes()
218    }
219}
220
221impl FromTrustedByteArray<RISTRETTO_SCALAR_BYTE_LENGTH> for RistrettoScalar {
222    fn from_trusted_byte_array(
223        bytes: &[u8; RISTRETTO_SCALAR_BYTE_LENGTH],
224    ) -> FastCryptoResult<Self> {
225        Ok(Self::from_bytes_mod_order(bytes))
226    }
227}
