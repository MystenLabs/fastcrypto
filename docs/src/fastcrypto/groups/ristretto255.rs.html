31/// Represents a point in the Ristretto group for Curve25519.
32#[derive(Default, Clone, Copy, Debug, PartialEq, Eq, From, Add, Sub, Neg, GroupOpsExtend)]
33pub struct RistrettoPoint(ExternalRistrettoPoint);
34
35impl RistrettoPoint {
36    /// Construct a RistrettoPoint from the given data using an Ristretto-flavoured Elligator 2 map.
37    /// If the input bytes are uniformly distributed, the resulting point will be uniformly
38    /// distributed over the Ristretto group.
39    pub fn from_uniform_bytes(bytes: &[u8; 64]) -> Self {
40        RistrettoPoint::from(ExternalRistrettoPoint::from_uniform_bytes(bytes))
41    }
42
43    /// Construct a RistrettoPoint from the given data using a given hash function.
44    pub fn map_to_point<H: HashFunction<64>>(bytes: &[u8]) -> Self {
45        Self::from_uniform_bytes(&H::digest(bytes).digest)
46    }
47
48    /// Return this point in compressed form.
49    pub fn compress(&self) -> [u8; 32] {
50        self.0.compress().0
51    }
52
53    /// Return this point in compressed form.
54    pub fn decompress(bytes: &[u8; 32]) -> Result<Self, FastCryptoError> {
55        RistrettoPoint::try_from(bytes.as_slice())
56    }
57}
58
59impl Doubling for RistrettoPoint {
60    fn double(self) -> Self {
61        Self(self.0.add(self.0))
62    }
63}
64
65impl MultiScalarMul for RistrettoPoint {
66    fn multi_scalar_mul(scalars: &[Self::ScalarType], points: &[Self]) -> FastCryptoResult<Self> {
67        if scalars.len() != points.len() {
68            return Err(FastCryptoError::InvalidInput);
69        }
70
71        Ok(RistrettoPoint(
72            ExternalRistrettoPoint::vartime_multiscalar_mul(
73                scalars.iter().map(|s| s.0),
74                points.iter().map(|g| g.0),
75            ),
76        ))
77    }
78}

81impl Div<RistrettoScalar> for RistrettoPoint {
82    type Output = Result<Self, FastCryptoError>;
83
84    fn div(self, rhs: RistrettoScalar) -> Self::Output {
85        let inv = rhs.inverse()?;
86        Ok(self * inv)
87    }
88}

91    type Output = RistrettoPoint;

94        RistrettoPoint::from(self.0 * rhs.0)
95    }

98impl GroupElement for RistrettoPoint {
99    type ScalarType = RistrettoScalar;
100
101    fn zero() -> RistrettoPoint {
102        RistrettoPoint::from(ExternalRistrettoPoint::identity())
103    }
104
105    fn generator() -> Self {
106        RistrettoPoint::from(RISTRETTO_BASEPOINT_POINT)
107    }
108}
109
110impl TryFrom<&[u8]> for RistrettoPoint {
111    type Error = FastCryptoError;
112
113    /// Decode a ristretto point in compressed binary form.
114    fn try_from(bytes: &[u8]) -> Result<Self, FastCryptoError> {
115        let point = ExternalCompressedRistrettoPoint::from_slice(bytes);
116        let decompressed_point = point.decompress().ok_or(FastCryptoError::InvalidInput)?;
117        Ok(RistrettoPoint::from(decompressed_point))
118    }
119}
120
121impl HashToGroupElement for RistrettoPoint {
122    fn hash_to_group_element(msg: &[u8]) -> Self {
123        RistrettoPoint::map_to_point::<Sha512>(msg)
124    }
125}

127impl ToFromByteArray<RISTRETTO_POINT_BYTE_LENGTH> for RistrettoPoint {
128    fn from_byte_array(bytes: &[u8; RISTRETTO_POINT_BYTE_LENGTH]) -> Result<Self, FastCryptoError> {
129        Self::try_from(bytes.as_slice())
130    }
131
132    fn to_byte_array(&self) -> [u8; RISTRETTO_POINT_BYTE_LENGTH] {
133        self.compress()
134    }
135}

139/// Represents a scalar.
140#[derive(Clone, Copy, Debug, PartialEq, Eq, From, Add, Sub, Neg, Div, GroupOpsExtend, Zeroize)]
141pub struct RistrettoScalar(ExternalRistrettoScalar);
142
143impl RistrettoScalar {
144    /// The order of the base point.
145    pub fn group_order() -> RistrettoScalar {
146        RistrettoScalar(BASEPOINT_ORDER)
147    }
148
149    /// Construct a [RistrettoScalar] by reducing a 64-byte little-endian integer modulo the group order.
150    pub fn from_bytes_mod_order_wide(bytes: &[u8; 64]) -> Self {
151        RistrettoScalar(ExternalRistrettoScalar::from_bytes_mod_order_wide(bytes))
152    }
153
154    /// Construct a [RistrettoScalar] by reducing a 32-byte little-endian integer modulo the group order.
155    pub fn from_bytes_mod_order(bytes: &[u8; 32]) -> Self {
156        RistrettoScalar(ExternalRistrettoScalar::from_bytes_mod_order(*bytes))
157    }
158}
159
160impl From<u128> for RistrettoScalar {
161    fn from(value: u128) -> RistrettoScalar {
162        RistrettoScalar(ExternalRistrettoScalar::from(value))
163    }
164}
165
166impl Mul<RistrettoScalar> for RistrettoScalar {
167    type Output = RistrettoScalar;
168
169    fn mul(self, rhs: RistrettoScalar) -> RistrettoScalar {
170        RistrettoScalar::from(self.0 * rhs.0)
171    }
172}
173
174#[allow(clippy::suspicious_arithmetic_impl)]
175impl Div<RistrettoScalar> for RistrettoScalar {
176    type Output = Result<RistrettoScalar, FastCryptoError>;
177
178    fn div(self, rhs: RistrettoScalar) -> Result<RistrettoScalar, FastCryptoError> {
179        let inv = rhs.inverse()?;
180        Ok(self * inv)
181    }
182}
183
184impl GroupElement for RistrettoScalar {
185    type ScalarType = Self;
186
187    fn zero() -> Self {
188        RistrettoScalar::from(ExternalRistrettoScalar::zero())
189    }
190    fn generator() -> Self {
191        RistrettoScalar::from(ExternalRistrettoScalar::one())
192    }
193}
194
195impl Scalar for RistrettoScalar {
196    fn rand<R: AllowedRng>(rng: &mut R) -> Self {
197        Self(ExternalRistrettoScalar::random(rng))
198    }
199
200    fn inverse(&self) -> FastCryptoResult<Self> {
201        if self.0 == ExternalRistrettoScalar::zero() {
202            return Err(FastCryptoError::InvalidInput);
203        }
204        Ok(RistrettoScalar::from(self.0.invert()))
205    }
206}
207
208impl HashToGroupElement for RistrettoScalar {
209    fn hash_to_group_element(bytes: &[u8]) -> Self {
210        Self::from_bytes_mod_order_wide(&Sha512::digest(bytes).digest)
211    }
212}
213
214impl FiatShamirChallenge for RistrettoScalar {
215    fn fiat_shamir_reduction_to_group_element(msg: &[u8]) -> Self {
216        Self::hash_to_group_element(msg)
217    }
218}

220impl ToFromByteArray<RISTRETTO_SCALAR_BYTE_LENGTH> for RistrettoScalar {
221    fn from_byte_array(
222        bytes: &[u8; RISTRETTO_SCALAR_BYTE_LENGTH],
223    ) -> Result<Self, FastCryptoError> {
224        Ok(RistrettoScalar(

227        ))
228    }

232    }
