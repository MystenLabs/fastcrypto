25/// Creation of a keypair using the [RFC 5869](https://tools.ietf.org/html/rfc5869) HKDF specification.
26/// This requires choosing an HMAC function of the correct length (conservatively, the size of a private key for this curve).
27/// Despite the unsightly generics (which aim to ensure this works for a wide range of hash functions), this is straightforward to use.
28///
29/// Example:
30/// ```rust
31/// use fastcrypto::hash::Sha3_256;
32/// use fastcrypto::ed25519::Ed25519KeyPair;
33/// use fastcrypto::hmac::hkdf_generate_from_ikm;
34/// # fn main() {
35///     let ikm = b"some_ikm";
36///     let info = b"my_app";
37///     let salt = b"some_salt";
38///     let my_keypair = hkdf_generate_from_ikm::<Sha3_256, Ed25519KeyPair>(ikm, salt, info);
39///
40///     let my_keypair_default_info = hkdf_generate_from_ikm::<Sha3_256, Ed25519KeyPair>(ikm, salt, &[]);
41/// # }
42/// ```
43///
44/// Note: This HKDF function may not match the native library's deterministic key generation functions.
45/// For example, observe that in the blst library:
46/// ```rust
47/// use fastcrypto::bls12381::min_sig::BLS12381KeyPair;
48/// use fastcrypto::hash::Sha3_256;
49/// use fastcrypto::traits::{KeyPair, SigningKey, ToFromBytes};
50/// use fastcrypto::hmac::hkdf_generate_from_ikm;
51///
52/// # fn main() {
53///     let ikm = b"02345678001234567890123456789012";
54///     let info = b"my_app";
55///     let salt = b"some_salt";
56
57///     let my_keypair = hkdf_generate_from_ikm::<Sha3_256, BLS12381KeyPair>(ikm, salt, info).unwrap();
58///     let native_sk = blst::min_sig::SecretKey::key_gen_v4_5(ikm, salt, info).unwrap();
59
60///     assert_ne!(native_sk.to_bytes(), my_keypair.private().as_bytes());
61/// # }
62/// ```
63#[cfg(any(test, feature = "experimental"))]
64pub fn hkdf_generate_from_ikm<H, K>(
65    ikm: &[u8],  // IKM (32 bytes).
66    salt: &[u8], // Salt (can be empty).
67    info: &[u8], // Info (can be empty).
68) -> Result<K, FastCryptoError>
69where
70    // This is a tad tedious, because of HKDF's use of a sealed trait. But mostly harmless since the traits exposed (H and K) are both defined in fastcrypto.
71    H: ReverseWrapper,
72    <<H as ReverseWrapper>::Variant as CoreProxy>::Core: HashMarker
73        + UpdateCore
74        + FixedOutputCore
75        + BufferKindUser<BufferKind = Eager>
76        + Default
77        + Clone,
78    <<<H as ReverseWrapper>::Variant as CoreProxy>::Core as BlockSizeUser>::BlockSize: IsLess<U256>,
79    Le<<<<H as ReverseWrapper>::Variant as CoreProxy>::Core as BlockSizeUser>::BlockSize, U256>:
80        NonZero,
81    K: KeyPair,
82{
83    // When HMAC is applied, salt is padded with zeros if shorter than the H's block size
84    // (both for Some(&[]) and None as the value of salt).
85    let hk = hkdf::Hkdf::<H::Variant, Hmac<H::Variant>>::new(Some(salt), ikm);
86
87    let mut okm = vec![0u8; K::PrivKey::LENGTH];
88    hk.expand(info, &mut okm)
89        .map_err(|_| FastCryptoError::GeneralOpaqueError)?;
90
91    let secret_key = K::PrivKey::from_bytes(&okm[..]).unwrap();
92
93    let keypair = K::from(secret_key);
94    Ok(keypair)
95}

98/// HMAC-SHA3-256 based functions
99
100const HMAC_KEY_RECOMMENDED_LENGTH: usize = 32;
101const HKDF_KEY_RECOMMENDED_LENGTH: usize = 32;
102
103/// Type for key in [hmac_sha3_256].
104pub type HmacKey = PrivateSeed<HMAC_KEY_RECOMMENDED_LENGTH, false>;
105
106/// [Keyed-Hash Message Authentication Code](https://www.rfc-editor.org/rfc/rfc2104) (HMAC) using SHA3-256.
107pub fn hmac_sha3_256(key: &HmacKey, message: &[u8]) -> Digest<32> {
108    let mut hash = Hmac::<sha3::Sha3_256>::new_from_slice(key.as_bytes())
109        .expect("HMAC can take key of any size");
110    hash.update(message);
111    let output = hash.finalize();
112    Digest {
113        digest: output.into_bytes().into(),
114    }
115}
116
117/// Type for input keying material in [hkdf_sha3_256].
118pub type HkdfIkm = PrivateSeed<HKDF_KEY_RECOMMENDED_LENGTH, false>;
119
120/// [HMAC-based Extract-and-Expand Key Derivation Function](https://tools.ietf.org/html/rfc5869) (HKDF) using SHA3-256.
121pub fn hkdf_sha3_256(
122    ikm: &HkdfIkm,
123    salt: &[u8],
124    info: &[u8],
125    output_length: usize,
126) -> Result<Vec<u8>, FastCryptoError> {
127    if output_length > sha3::Sha3_256::output_size() * 255 {
128        return Err(FastCryptoError::InputTooLong(
129            sha3::Sha3_256::output_size() * 255,
130        ));
131    }
132    let hk = hkdf::Hkdf::<sha3::Sha3_256, Hmac<sha3::Sha3_256>>::new(Some(salt), ikm.as_bytes());
133    let mut output: Vec<u8> = vec![0; output_length];
134    hk.expand(info, output.as_mut_slice())
135        .map_err(|_| FastCryptoError::GeneralOpaqueError)?;
136    Ok(output)
