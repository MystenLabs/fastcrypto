4//! This module contains an implementation of the [ECDSA signature scheme](https://en.wikipedia.org/wiki/Elliptic_Curve_Digital_Signature_Algorithm) over the [secp256k1 curve](http://www.secg.org/sec2-v2.pdf).
5//!
6//! Messages can be signed and the public key can be recovered from the signature:
7//! # Example
8//! ```rust
9//! # use fastcrypto::secp256k1::recoverable::*;
10//! # use fastcrypto::traits::{KeyPair, Signer, VerifyingKey};
11//! # use fastcrypto::secp256k1::Secp256k1KeyPair;
12//! # use fastcrypto::traits::{RecoverableSignature, RecoverableSigner};
13//! use rand::thread_rng;
14//! let kp = Secp256k1KeyPair::generate(&mut thread_rng());
15//! let message: &[u8] = b"Hello, world!";
16//! let signature = kp.sign_recoverable(message);
17//! assert_eq!(&signature.recover(message).unwrap(), kp.public());
18//! ```
19

40/// Length of a compact signature followed by one extra byte for recovery id, used to recover the public key from a signature.
41pub const SECP256K1_RECOVERABLE_SIGNATURE_SIZE: usize = constants::COMPACT_SIGNATURE_SIZE + 1;
42
43/// Secp256k1 signature.
44#[readonly::make]
45#[derive(Debug, Clone)]
46pub struct Secp256k1RecoverableSignature {
47    pub sig: ExternalRecoverableSignature,
48    pub bytes: OnceCell<[u8; SECP256K1_RECOVERABLE_SIGNATURE_SIZE]>,
49}

51serialize_deserialize_with_to_from_bytes!(
52    Secp256k1RecoverableSignature,
53    SECP256K1_RECOVERABLE_SIGNATURE_SIZE
54);

56impl ToFromBytes for Secp256k1RecoverableSignature {
57    fn from_bytes(bytes: &[u8]) -> Result<Self, FastCryptoError> {
58        if bytes.len() != SECP256K1_RECOVERABLE_SIGNATURE_SIZE {
59            return Err(FastCryptoError::InputLengthWrong(
60                SECP256K1_RECOVERABLE_SIGNATURE_SIZE,
61            ));
62        }
63        RecoveryId::from_i32(bytes[SECP256K1_RECOVERABLE_SIGNATURE_SIZE - 1] as i32)
64            .and_then(|rec_id| {
65                ExternalRecoverableSignature::from_compact(
66                    &bytes[..(SECP256K1_RECOVERABLE_SIGNATURE_SIZE - 1)],
67                    rec_id,
68                )
69                .map(|sig| Secp256k1RecoverableSignature {
70                    sig,
71                    bytes: OnceCell::new(),
72                })
73            })
74            .map_err(|_| FastCryptoError::InvalidInput)
75    }
76}

78impl AsRef<[u8]> for Secp256k1RecoverableSignature {
79    fn as_ref(&self) -> &[u8] {
80        self.bytes.get_or_init::<_>(|| {
81            let mut bytes = [0u8; SECP256K1_RECOVERABLE_SIGNATURE_SIZE];
82            let (recovery_id, sig) = self.sig.serialize_compact();
83            bytes[..(SECP256K1_RECOVERABLE_SIGNATURE_SIZE - 1)].copy_from_slice(&sig);
84            bytes[SECP256K1_RECOVERABLE_SIGNATURE_SIZE - 1] = recovery_id.to_i32() as u8;
85            bytes
86        })
87    }
88}

106impl Secp256k1RecoverableSignature {
107    /// Convert a non-recoverable signature into a recoverable signature.
108    pub fn try_from_nonrecoverable(
109        signature: &Secp256k1Signature,
110        pk: &Secp256k1PublicKey,
111        message: &[u8],
112    ) -> Result<Self, FastCryptoError> {
113        // Secp256k1Signature::as_bytes is guaranteed to return SECP256K1_SIGNATURE_LENGTH = SECP256K1_RECOVERABLE_SIGNATURE_SIZE - 1 bytes.
114        let mut recoverable_signature_bytes = [0u8; SECP256K1_RECOVERABLE_SIGNATURE_SIZE];
115        recoverable_signature_bytes[0..SECP256K1_RECOVERABLE_SIGNATURE_SIZE - 1]
116            .copy_from_slice(signature.as_ref());
117
118        for recovery_id in 0..4 {
119            recoverable_signature_bytes[SECP256K1_RECOVERABLE_SIGNATURE_SIZE - 1] = recovery_id;
120            let recoverable_signature = <Secp256k1RecoverableSignature as ToFromBytes>::from_bytes(
121                &recoverable_signature_bytes,
122            )?;
123            if pk
124                .verify_recoverable(message, &recoverable_signature)
125                .is_ok()
126            {
127                return Ok(recoverable_signature);
128            }
129        }
130        Err(FastCryptoError::InvalidInput)
131    }
132}

134impl RecoverableSignature for Secp256k1RecoverableSignature {
135    type PubKey = Secp256k1PublicKey;
136    type Signer = Secp256k1KeyPair;
137    type DefaultHash = DefaultHash;
138
139    /// Recover public key from signature using the given hash function to hash the message.
140    fn recover_with_hash<H: HashFunction<32>>(
141        &self,
142        msg: &[u8],
143    ) -> Result<Secp256k1PublicKey, FastCryptoError> {
144        match Message::from_slice(&H::digest(msg).digest) {
145            Ok(message) => match self.sig.recover(&message) {
146                Ok(pubkey) => Secp256k1PublicKey::from_bytes(pubkey.serialize().as_slice()),
147                Err(_) => Err(FastCryptoError::GeneralOpaqueError),
148            },
149            Err(_) => Err(FastCryptoError::InvalidInput),
150        }
151    }
152}

154impl RecoverableSigner for Secp256k1KeyPair {
155    type PubKey = Secp256k1PublicKey;
156    type Sig = Secp256k1RecoverableSignature;
157
158    /// Create a new recoverable signature over the given message. The hash function `H` is used to hash the message.
159    fn sign_recoverable_with_hash<H: HashFunction<32>>(
160        &self,
161        msg: &[u8],
162    ) -> Secp256k1RecoverableSignature {
163        let secp = Secp256k1::signing_only();
164        let message = Message::from_slice(H::digest(msg).as_ref()).unwrap();
165
166        // Creates a 65-bytes signature of shape [r, s, v] where v can be 0 or 1.
167        // Pseudo-random deterministic nonce generation is used according to RFC6979.
168        Secp256k1RecoverableSignature {
169            sig: secp.sign_ecdsa_recoverable(&message, &self.secret.privkey),
170            bytes: OnceCell::new(),
171        }
172    }
173}
