11/// NIZKPoK for the DDH tuple [G, H=eG, xG, xH].
12/// - Prover selects a random r and sends A=rG, B=rH.
13/// - Prover computes challenge c and sends z=r+c*x.
14/// - Verifier checks that zG=A+c(xG) and zH=B+c(xH).
15///
16/// The NIZK is (A, B, z) where c is implicitly computed using a random oracle.
17#[derive(Debug, Clone, PartialEq, Eq, Serialize)]
18pub struct DdhTupleNizk<G: GroupElement>(G, G, G::ScalarType);
19
20impl<G: GroupElement> DdhTupleNizk<G>
21where
22    G: GroupElement + Serialize,
23    <G as GroupElement>::ScalarType: FiatShamirChallenge,
24{
25    /// Create a new NIZKPoK for the DDH tuple [G, H=eG, xG, xH] using the given RNG and random oracle.
26    pub fn create<R: AllowedRng>(
27        x: &G::ScalarType,
28        h: &G,
29        x_g: &G,
30        x_h: &G,
31        random_oracle: &RandomOracle,
32        rng: &mut R,
33    ) -> Self {
34        let r = G::ScalarType::rand(rng);
35        let a = G::generator() * r;
36        let b = *h * r;
37        let challenge = Self::fiat_shamir_challenge(h, x_g, x_h, &a, &b, random_oracle);
38        let z = challenge * x + r;
39        debug!("NIZK: Creating a proof for {h:?} {x_g:?} {x_h:?} with challenge {challenge:?}");
40        DdhTupleNizk(a, b, z)
41    }
42
43    /// Verify this NIZKPoK.
44    pub fn verify(
45        &self,
46        h: &G,
47        x_g: &G,
48        x_h: &G,
49        random_oracle: &RandomOracle,
50    ) -> FastCryptoResult<()> {
51        if *h == G::zero() || *x_g == G::zero() || *x_h == G::zero() {
52            // We should never see this, but just in case
53            return Err(FastCryptoError::InvalidProof);
54        }
55        let challenge = Self::fiat_shamir_challenge(h, x_g, x_h, &self.0, &self.1, random_oracle);
56        debug!("NIZK: Verifying a proof of {h:?} {x_g:?} {x_h:?} with challenge {challenge:?}");
57        if !is_valid_relation(
58            &self.0, // A
59            x_g,
60            &G::generator(),
61            &self.2, // z
62            &challenge,
63        ) || !is_valid_relation(
64            &self.1, // B
65            x_h, h, &self.2, // z
66            &challenge,
67        ) {
68            Err(FastCryptoError::InvalidProof)
69        } else {
70            Ok(())
71        }
72    }
73
74    /// Returns the challenge for Fiat-Shamir.
75    fn fiat_shamir_challenge(
76        h: &G,
77        x_g: &G,
78        x_h: &G,
79        a: &G,
80        b: &G,
81        random_oracle: &RandomOracle,
82    ) -> G::ScalarType {
83        let output = random_oracle.evaluate(&(G::generator(), h, x_g, x_h, a, b));
84        G::ScalarType::fiat_shamir_reduction_to_group_element(&output)
85    }
86}

88impl<'de, G> Deserialize<'de> for DdhTupleNizk<G>
89where
90    G: GroupElement + Deserialize<'de>,
91    G::ScalarType: Deserialize<'de>,
92{
93    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
94    where
95        D: Deserializer<'de>,
96    {
97        let tuple = <(G, G, G::ScalarType)>::deserialize(deserializer)?;
98        if tuple.0 == G::zero() || tuple.1 == G::zero() || tuple.2 == G::ScalarType::zero() {
99            return Err(serde::de::Error::custom(
100                "Invalid proof: one of the elements is inf/zero",
101            ));
102        }
103        Ok(DdhTupleNizk(tuple.0, tuple.1, tuple.2))
104    }
105}

107///   NIZKPoK for the DL [G, xG].
108/// - Prover selects a random r and sends A=rG.
109/// - Prover computes challenge c and sends z=r+c*x.
110/// - Verifier checks that zG=A+c(xG).
111///
112/// The NIZK is (A, z) where c is implicitly computed using a random oracle.
113#[derive(Debug, Clone, PartialEq, Eq, Serialize)]
114pub struct DLNizk<G: GroupElement>(G, G::ScalarType);
115
116impl<G: GroupElement> DLNizk<G>
117where
118    G: GroupElement + Serialize,
119    <G as GroupElement>::ScalarType: FiatShamirChallenge,
120{
121    /// Create a new NIZKPoK for the DL [G, xG] using the given RNG and random oracle.
122    pub fn create<R: AllowedRng>(
123        x: &G::ScalarType,
124        x_g: &G, // passed since probably already computed
125        random_oracle: &RandomOracle,
126        rng: &mut R,
127    ) -> Self {
128        let r = G::ScalarType::rand(rng);
129        let a = G::generator() * r;
130        let challenge = Self::fiat_shamir_challenge(x_g, &a, random_oracle);
131        let z = challenge * x + r;
132        debug!("NIZK: Creating a proof for {x_g:?} with challenge {challenge:?}");
133        DLNizk(a, z)
134    }
135
136    pub fn verify(&self, x_g: &G, random_oracle: &RandomOracle) -> FastCryptoResult<()> {
137        if *x_g == G::zero() {
138            // we should never see this, but just in case
139            return Err(FastCryptoError::InvalidProof);
140        }
141        let challenge = Self::fiat_shamir_challenge(x_g, &self.0, random_oracle);
142        debug!("NIZK: Verifying a proof of {x_g:?} with challenge {challenge:?}");
143        if !is_valid_relation(&self.0, x_g, &G::generator(), &self.1, &challenge) {
144            Err(FastCryptoError::InvalidProof)
145        } else {
146            Ok(())
147        }
148    }
149
150    /// Returns the challenge for Fiat-Shamir.
151    fn fiat_shamir_challenge(x_g: &G, a: &G, random_oracle: &RandomOracle) -> G::ScalarType {
152        let output = random_oracle.evaluate(&(G::generator(), x_g, a));
153        G::ScalarType::fiat_shamir_reduction_to_group_element(&output)
154    }
155}

158where
159    G: GroupElement + Deserialize<'de>,
160    G::ScalarType: Deserialize<'de>,
161{
162    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
163    where
164        D: Deserializer<'de>,
165    {
166        let tuple = <(G, G::ScalarType)>::deserialize(deserializer)?;
167        if tuple.0 == G::zero() || tuple.1 == G::ScalarType::zero() {
168            return Err(serde::de::Error::custom(
169                "Invalid proof: one of the elements is inf/zero",
170            ));
171        }
172        Ok(DLNizk(tuple.0, tuple.1))
173    }
174}

176/// Checks if e1 + c e2 = z e3
177fn is_valid_relation<G: GroupElement>(
178    e1: &G,
179    e2: &G,
180    e3: &G,
181    z: &G::ScalarType,
182    c: &G::ScalarType,
183) -> bool {
184    let left = *e1 + *e2 * c;
185    let right = *e3 * z;
186    left == right
