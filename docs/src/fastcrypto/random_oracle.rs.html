4use crate::groups::{FiatShamirChallenge, GroupElement};
5use digest::Digest;
6use serde::{Deserialize, Serialize};
7use sha3::Sha3_512;
8use std::fmt::Debug;
9
10/// Random Oracle from SHA3-512.
11/// - prefix should be globally unique.
12/// - evaluate serializes the given input and outputs SHA3-512(prefix_len as big-endian u32 | prefix | input).
13/// - Subprotocols may use a prefix that is extended from the prefix of its parent protocol, by
14///   deriving a new instance using extend, which simply concatenates the strings with the separator
15///   "-". E.g., RandomOracle::new("abc").extend("def") = RandomOracle::new("abc-def").
16///
17/// The caller must make sure to:
18/// - Choose distinct prefix & extension strings, preferably without "-" in them (asserted in debug
19///   mode).
20/// - Ensure that the length of prefix & extension is small enough to fit in u32.
21///   Violating this constraint will cause a panic.
22
23#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
24pub struct RandomOracle {
25    prefix: String,
26}
27
28impl RandomOracle {
29    /// Create a fresh random oracle with a given "session id"/prefix.
30    pub fn new(initial_prefix: &str) -> Self {
31        debug_assert!(!initial_prefix.contains('-'));
32        // Since we shouldn't get such long prefixes, it's safe to assert here.
33        assert!(initial_prefix.len() < u32::MAX as usize);
34        Self {
35            prefix: initial_prefix.into(),
36        }
37    }
38
39    /// Evaluate the random oracle on a given input.
40    pub fn evaluate<T: Serialize>(&self, obj: &T) -> [u8; 64] {
41        let mut hasher = Sha3_512::default();
42        let len: u32 = self
43            .prefix
44            .len()
45            .try_into()
46            .expect("prefix length should be less than u32::MAX, checked when set");
47        hasher.update(len.to_be_bytes());
48        hasher.update(&self.prefix);
49        let serialized = bcs::to_bytes(obj).expect("serialize should never fail");
50        hasher.update(&serialized);
51        hasher.finalize().into()
52    }
53
54    pub fn evaluate_to_group_element<G: GroupElement + FiatShamirChallenge, T: Serialize>(
55        &self,
56        obj: &T,
57    ) -> G {
58        G::fiat_shamir_reduction_to_group_element(&self.evaluate(obj))
59    }
60
61    /// Derive a new random oracle from the current one and additional string (can be done multiple times).
62    pub fn extend(&self, extension: &str) -> Self {
63        debug_assert!(!extension.contains('-'));
64        // Since we shouldn't get such long prefixes, it's safe to assert here.
65        assert!(self.prefix.len() + extension.len() + 1 < u32::MAX as usize);
66        Self {
67            prefix: self.prefix.clone() + "-" + extension,
68        }
69    }
70}
