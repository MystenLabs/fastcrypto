4//! This module contains conversion function between scalars (fr), field elements (fq) and elliptic curve
5//! points between the representations used by arkworks in ark-secp256r1 and RustCrypto's p256 crate.
6
7use ark_ec::AffineRepr;
8use ark_ff::{Field, PrimeField as ArkworksPrimeField, Zero};
9use ark_serialize::{CanonicalSerialize, CanonicalSerializeWithFlags, EmptyFlags};
10use elliptic_curve::bigint::ArrayEncoding;
11use elliptic_curve::scalar::FromUintUnchecked;
12use elliptic_curve::sec1::FromEncodedPoint;
13use elliptic_curve::sec1::ToEncodedPoint;
14use p256::{FieldBytes, U256};

16/// Convert a p256 scalar to an arkworks scalar.
17pub(crate) fn fr_p256_to_arkworks(scalar: &p256::Scalar) -> ark_secp256r1::Fr {
18    ark_secp256r1::Fr::from_be_bytes_mod_order(&scalar.to_bytes())
19}
20
21/// Convert an arkworks scalar to a p256 scalar.
22pub(crate) fn fr_arkworks_to_p256(scalar: &ark_secp256r1::Fr) -> p256::Scalar {
23    // This implementation is taken from bls_fr_to_blst_fr in fastcrypto-zkp.
24    let mut bytes = [0u8; 32];
25    scalar
26        .serialize_with_flags(&mut bytes[..], EmptyFlags)
27        .unwrap();
28    p256::Scalar::from_uint_unchecked(U256::from_le_byte_array(FieldBytes::from(bytes)))
29}
30
31/// Convert an arkworks field element to a p256 field element.
32pub(crate) fn fq_arkworks_to_p256(scalar: &ark_secp256r1::Fq) -> FieldBytes {
33    // This implementation is taken from bls_fq_to_blst_fp in fastcrypto-zkp.
34    let mut bytes = [0u8; 32];
35    scalar.serialize_uncompressed(&mut bytes[..]).unwrap();
36    bytes.reverse();
37    p256::FieldBytes::from(bytes)
38}

40/// Convert a p256 affine point to an arkworks affine point.
41pub(crate) fn affine_pt_p256_to_projective_arkworks(
42    point: &p256::AffinePoint,
43) -> ark_secp256r1::Projective {
44    if point.is_identity().into() {
45        return ark_secp256r1::Projective::zero();
46    }
47    let encoded_point = point.to_encoded_point(false);
48    ark_secp256r1::Projective::from(ark_secp256r1::Affine::new_unchecked(
49        ark_secp256r1::Fq::from_be_bytes_mod_order(encoded_point.x().unwrap()),
50        ark_secp256r1::Fq::from_be_bytes_mod_order(encoded_point.y().unwrap()),
51    ))
52}
53
54/// Reduce a big-endian integer representation modulo the subgroup order in arkworks representation.
55pub(crate) fn reduce_bytes(bytes: &[u8; 32]) -> ark_secp256r1::Fr {
56    ark_secp256r1::Fr::from_be_bytes_mod_order(bytes)
57}
58
59/// Reduce an arkworks field element (modulo field size) to a scalar (modulo subgroup order). This also
60/// returns a boolean indicating whether a modular reduction was performed.
61pub(crate) fn arkworks_fq_to_fr(scalar: &ark_secp256r1::Fq) -> (ark_secp256r1::Fr, bool) {
62    let mut bytes = [0u8; 32];
63    scalar.serialize_uncompressed(&mut bytes[..]).unwrap();
64    let output = ark_secp256r1::Fr::from_le_bytes_mod_order(&bytes);
65    (output, output.into_bigint() != scalar.into_bigint())
66}

68/// Converts an arkworks affine point to a p256 affine point.
69pub(crate) fn affine_pt_arkworks_to_p256(point: &ark_secp256r1::Affine) -> p256::AffinePoint {
70    if point.is_zero() {
71        return p256::AffinePoint::IDENTITY;
72    }
73    let encoded_point = p256::EncodedPoint::from_affine_coordinates(
74        &fq_arkworks_to_p256(point.x().expect("The point should not be zero")),
75        &fq_arkworks_to_p256(point.y().expect("The point should not be zero")),
76        false,
77    );
78    p256::AffinePoint::from_encoded_point(&encoded_point).unwrap()
79}

81/// Extract the affine x-coordinate from a projective point. Returns none if the point is the identity.
82pub(crate) fn get_affine_x_coordinate(
83    point: &ark_secp256r1::Projective,
84) -> Option<ark_secp256r1::Fq> {
85    if point.is_zero() {
86        return None;
87    }
88    let mut z_inv = point
89        .z
90        .inverse()
91        .expect("z is zero. This should never happen.");
92    z_inv.square_in_place();
93    Some(point.x * z_inv)
94}

96#[cfg(test)]
97mod tests {

105    #[test]
106    fn test_fr_p256_to_arkworks() {
107        let arkworks_seven = ark_secp256r1::Fr::from(7u32);
108        let p256_seven = p256::Scalar::from(7u32);
109
110        let actual_arkworks_seven = fr_p256_to_arkworks(&p256_seven);
111        assert_eq!(actual_arkworks_seven, arkworks_seven);
112
113        let actual_p256_seven = fr_arkworks_to_p256(&arkworks_seven);
114        assert_eq!(actual_p256_seven, p256_seven);
115    }
116
117    #[test]
118    fn test_pt_arkworks_to_p256() {
119        // 0
120        assert_eq!(
121            p256::AffinePoint::IDENTITY,
122            affine_pt_arkworks_to_p256(&ark_secp256r1::Affine::zero())
123        );
124
125        // G
126        assert_eq!(
127            p256::AffinePoint::generator(),
128            affine_pt_arkworks_to_p256(&ark_secp256r1::Affine::generator())
129        );
130
131        // 7G
132        assert_eq!(
133            (p256::AffinePoint::generator() * p256::Scalar::from(7u32)).to_affine(),
134            affine_pt_arkworks_to_p256(
135                &(ark_secp256r1::Projective::generator() * ark_secp256r1::Fr::from(7u32))
136                    .into_affine()
137            )
138        );
139
140        // sG, random s
141        let random_s = p256::Scalar::random(&mut rand::thread_rng());
142        assert_eq!(
143            (p256::AffinePoint::generator() * random_s).to_affine(),
144            affine_pt_arkworks_to_p256(
145                &(ark_secp256r1::Projective::generator() * fr_p256_to_arkworks(&random_s))
146                    .into_affine()
147            )
148        );
149    }

186        let s_fr = arkworks_fq_to_fr(&s).0;
