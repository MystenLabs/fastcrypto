4//! This module contains an implementation of the [ECDSA signature scheme](https://en.wikipedia.org/wiki/Elliptic_Curve_Digital_Signature_Algorithm) over the
5//! [secp256r1 NIST-P1 curve](https://www.secg.org/SEC2-Ver-1.0.pdf). The nonce is generated deterministically according to [RFC6979](https://www.rfc-editor.org/rfc/rfc6979).
6//!
7//! Messages can be signed and the signature can be verified again:
8//! # Example
9//! ```rust
10//! # use fastcrypto::secp256r1::*;
11//! # use fastcrypto::traits::{KeyPair, Signer, VerifyingKey};
12//! use rand::thread_rng;
13//! let kp = Secp256r1KeyPair::generate(&mut thread_rng());
14//! let message: &[u8] = b"Hello, world!";
15//! let signature = kp.sign(message);
16//! assert!(kp.public().verify(message, &signature).is_ok());
17//! ```

171impl Secp256r1PublicKey {
172    /// Verify the signature using the given hash function to hash the message.
173    pub fn verify_with_hash<H: HashFunction<32>>(
174        &self,
175        msg: &[u8],
176        signature: &Secp256r1Signature,
177    ) -> Result<(), FastCryptoError> {
178        if signature.sig.s().is_high().into() {
179            return Err(FastCryptoError::GeneralError(
180                "The s value of ECDSA signature must be low".to_string(),
181            ));
182        }
183
184        // The flow below is identical to verify_prehash from ecdsa-0.16.6/src/hazmat.rs, but using
185        // arkworks for the finite field and elliptic curve arithmetic.
186
187        // Split signature into scalars. Note that this panics if r or s are zero, which is handled
188        // in Secp256r1Signature::from_bytes.
189        let (r, s) = signature.sig.split_scalars();
190        let z = reduce_bytes(&H::digest(msg).digest);
191
192        // Convert scalars to arkworks representation
193        let r = fr_p256_to_arkworks(&r);
194        let s = fr_p256_to_arkworks(&s);
195        let q = affine_pt_p256_to_projective_arkworks(self.pubkey.as_affine());
196
197        // Compute inverse of s. This fails if s is zero which is checked in deserialization and in
198        // split_scalars above, but we avoid an unwrap here to be safe.
199        let s_inv = s.inverse().ok_or(FastCryptoError::InvalidSignature)?;
200
201        // Verify signature
202        let u1 = z * s_inv;
203        let u2 = r * s_inv;
204
205        // Do optimised double multiplication
206        let p = MULTIPLIER
207            .two_scalar_mul(
208                &secp256r1::Scalar(u1),
209                &ProjectivePoint(q),
210                &secp256r1::Scalar(u2),
211            )
212            .0;
213
214        // Note that x is none if and only if p is zero, in which case the signature is invalid. See
215        // step 5 in section 4.1.4 in "SEC 1: Elliptic Curve Cryptography".
216        let x = get_affine_x_coordinate(&p).ok_or(FastCryptoError::InvalidSignature)?;
217
218        let (x_reduced, _) = arkworks_fq_to_fr(&x);
219        if x_reduced == r {
220            return Ok(());
221        }
222        Err(FastCryptoError::InvalidSignature)
223    }
224}

233impl ToFromBytes for Secp256r1PublicKey {
234    fn from_bytes(bytes: &[u8]) -> Result<Self, FastCryptoError> {
235        match ExternalPublicKey::try_from(bytes) {
236            Ok(pubkey) => Ok(Secp256r1PublicKey {
237                pubkey,
238                // If the given bytes is in the right format (compressed), we keep them for next time to_bytes is called
239                bytes: match <[u8; SECP256R1_PUBLIC_KEY_LENGTH]>::try_from(bytes) {
240                    Ok(result) => OnceCell::with_value(result),
241                    Err(_) => OnceCell::new(),
242                },
243            }),
244            Err(_) => Err(FastCryptoError::InvalidInput),
245        }
246    }
247}

311impl ToFromBytes for Secp256r1Signature {
312    fn from_bytes(bytes: &[u8]) -> Result<Self, FastCryptoError> {
313        if bytes.len() != SECP256R1_SIGNATURE_LENTH {
314            return Err(FastCryptoError::InputLengthWrong(SECP256R1_SIGNATURE_LENTH));
315        }
316
317        // This fails if either r or s are zero: https://docs.rs/ecdsa/0.16.6/src/ecdsa/lib.rs.html#209-219.
318        let sig = ExternalSignature::try_from(bytes).map_err(|_| FastCryptoError::InvalidInput)?;
319
320        Ok(Secp256r1Signature {
321            sig,
322            bytes: OnceCell::new(),
323        })
324    }
325}

366/// Secp256r1 public/private key pair.
367#[derive(Debug, PartialEq, Eq)]
368pub struct Secp256r1KeyPair {
369    pub public: Secp256r1PublicKey,
370    pub secret: Secp256r1PrivateKey,
371}
372
373impl Secp256r1KeyPair {
374    /// Sign a message using the given hash function and return the signature and the elliptic curve
375    /// point R = kG where k is the ephemeral nonce generated according to RFC6979.
376    fn sign_common<H: HashFunction<32>>(&self, msg: &[u8]) -> (Signature, bool, bool) {
377        // Hash message
378        let z = reduce_bytes(&H::digest(msg).digest);
379
380        // Private key as scalar
381        let x = self.secret.privkey.as_nonzero_scalar();
382
383        // Generate nonce according to RFC6979. The unwrap is safe because k is generated smaller
384        // than the group size.
385        let k = fr_p256_to_arkworks(
386            &Scalar::from_repr(rfc6979::generate_k::<sha2::Sha256, _>(
387                &x.to_bytes(),
388                &NistP256::ORDER.encode_field_bytes(),
389                &fr_arkworks_to_p256(&z).to_bytes(),
390                &[],
391            ))
392            .unwrap(),
393        );
394
395        // Convert secret key and message to arkworks scalars.
396        let x = fr_p256_to_arkworks(x);
397
398        // Compute scalar inversion of k
399        let k_inv = k.inverse().expect("k should not be zero");
400
401        // Compute R = kG
402        let big_r = MULTIPLIER.mul(&secp256r1::Scalar(k)).0.into_affine();
403
404        // Lift x-coordinate of R and reduce it into an element of the scalar field
405        let (r, is_x_reduced) = arkworks_fq_to_fr(big_r.x().expect("R should not be zero"));
406
407        // Compute s as a signature over r and z.
408        let s = k_inv * (z + r * x);
409
410        // Convert to p256 format
411        let s = fr_arkworks_to_p256(&s).to_bytes();
412        let r = fr_arkworks_to_p256(&r).to_bytes();
413
414        // This can only fail if either ùíì or ùíî are zero (see ecdsa-0.15.0/src/lib.rs) which is negligible.
415        let signature = Signature::from_scalars(r, s).expect("r or s is zero");
416
417        // The parity of the y coordinate is needed for computing the recovery id.
418        let is_r_odd = big_r.y().expect("R is zero").into_bigint().is_odd();
419        let is_s_high: bool = signature.s().is_high().into();
420        let is_y_odd = is_r_odd ^ is_s_high;
421
422        // Normalize signature
423        let normalized_signature = signature.normalize_s().unwrap_or(signature);
424
425        (normalized_signature, is_y_odd, is_x_reduced)
426    }
427
428    /// Create a new signature using the given hash function to hash the message.
429    pub fn sign_with_hash<H: HashFunction<32>>(&self, msg: &[u8]) -> Secp256r1Signature {
430        let (signature, _, _) = self.sign_common::<H>(msg);
431
432        Secp256r1Signature {
433            sig: signature,
434            bytes: OnceCell::new(),
435        }
436    }
437}

454impl KeyPair for Secp256r1KeyPair {
455    type PubKey = Secp256r1PublicKey;
456    type PrivKey = Secp256r1PrivateKey;
457    type Sig = Secp256r1Signature;
458
459    fn public(&'_ self) -> &'_ Self::PubKey {
460        &self.public
461    }
462
463    fn private(self) -> Self::PrivKey {
464        Secp256r1PrivateKey::from_bytes(self.secret.as_ref()).unwrap()
465    }
466
467    #[cfg(feature = "copy_key")]
468    fn copy(&self) -> Self {
469        Secp256r1KeyPair {
470            public: self.public.clone(),
471            secret: Secp256r1PrivateKey::from_bytes(self.secret.as_ref()).unwrap(),
472        }
473    }
474
475    fn generate<R: AllowedRng>(rng: &mut R) -> Self {
476        let privkey = ExternalSecretKey::random(rng);
477        Secp256r1PrivateKey {
478            privkey,
479            bytes: OnceCell::new(),
480        }
481        .into()
482    }
483}
