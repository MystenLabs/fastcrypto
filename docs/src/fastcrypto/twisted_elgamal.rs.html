37/// A verifiable key encapsulation verifiably encrypts a private key to multiple recipients.
38/// A batch range proof shows that each 32-bit limb of the private key lies in the range [0, 2^32-1].
39/// A key consistency proof shows that the 32-bit key limbs have been encrypted to the correct recipient public keys
40/// and that they correspond to the provided sender public key.
41pub struct VerifiableKeyEncapsulation<const N: usize> {
42    pub ciphertexts: [MultiRecipientCiphertext; N],
43    pub range_proof: RangeProof,
44    pub consistency_proof: KeyConsistencyProof<N>,
45}
46
47/// A sigma protocol proof that `N` Twisted ElGamal ciphertexts encrypt the 32-bit limbs of a
48/// private key whose corresponding public key is known. Concretely, for each limb `i` and
49/// recipient `j` it proves: (1) the decryption handle `D_ij = r_i * S_j` was formed with the
50/// same blinding `r_i` used in the Pedersen commitment `C_i = r_i * G + u_i * H`, and (2) the
51/// same limb values `u_i` that open those commitments reconstruct to the private key, i.e.
52/// `(\sum_i u_i * 2^{32i}) * G == U` where `U` is the sender's public key. Crucially, the proof
53/// binds (1) and (2) together, so the verifier is assured that the values inside the commitments
54/// are exactly the limbs of the private key for `U`. The proof is made non-interactive via the
55/// Fiat-Shamir transform and supports multiple recipients sharing the same commitment per limb.
56pub struct KeyConsistencyProof<const N: usize> {
57    a1: Vec<RistrettoPoint>,
58    a2: [RistrettoPoint; N],
59    a3: [RistrettoPoint; N],
60    z1: [RistrettoScalar; N],
61    z2: [RistrettoScalar; N],
62}

92impl Ciphertext {
93    pub fn encrypt(
94        encryption_key: &PublicKey,
95        message: u32,
96        rng: &mut impl AllowedRng,
97    ) -> (Self, Blinding) {
98        let blinding = Blinding::rand(rng);
99        (
100            Self {
101                decryption_handle: encryption_key.0 * blinding.0,
102                commitment: PedersenCommitment::new(
103                    &RistrettoScalar::from(message as u64),
104                    &blinding,
105                ),
106            },
107            blinding,
108        )
109    }
110
111    pub fn encrypt_with_consistency_proof(
112        encryption_key: &PublicKey,
113        message: u32,
114        rng: &mut impl AllowedRng,
115    ) -> FastCryptoResult<(Self, Blinding, ConsistencyProof)> {
116        let (ciphertext, blinding) = Self::encrypt(encryption_key, message, rng);
117        let proof = ConsistencyProof::prove(
118            &RistrettoScalar::from(message as u64),
119            &ciphertext,
120            &blinding,
121            encryption_key,
122            rng,
123        )?;
124        Ok((ciphertext, blinding, proof))
125    }
126
127    pub fn decrypt(
128        &self,
129        private_key: &PrivateKey,
130        table: &HashMap<[u8; RISTRETTO_POINT_BYTE_LENGTH], u16>,
131    ) -> FastCryptoResult<u32> {
132        let mut c = self.commitment.0 - (self.decryption_handle / private_key.0)?;
133        for x_low in 0..1 << 16 {
134            if let Some(&x_high) = table.get(&c.to_byte_array()) {
135                return Ok(x_low + ((x_high as u32) << 16));
136            }
137            c -= *H;
138        }
139        Err(InvalidInput)
140    }
141
142    /// Create a PoK of a private key such that the given encryption is of the message 0.
143    pub fn zero_proof(&self, private_key: &PrivateKey, rng: &mut impl AllowedRng) -> ZeroProof {
144        ZeroProof::create(self, private_key, rng)
145    }
146}

148impl ZeroProof {
149    pub fn create<R: AllowedRng>(
150        encryption: &Ciphertext,
151        private_key: &PrivateKey,
152        rng: &mut R,
153    ) -> Self {
154        let r = RistrettoScalar::rand(rng);
155        let a1 = RistrettoPoint::generator() * r;
156        let a2 = encryption.commitment.0 * r;
157        let pk = PublicKey::from(private_key);
158        let challenge = fiat_shamir_challenge(&(&a1, &a2, &pk, &encryption));
159        let z = challenge * private_key.0 + r;
160        ZeroProof { a1, a2, z }
161    }
162
163    pub fn verify(&self, encryption: &Ciphertext, pk: &PublicKey) -> FastCryptoResult<()> {
164        let c = fiat_shamir_challenge(&(&self.a1, &self.a2, &pk, &encryption));
165        if self.a1
166            != RistrettoPoint::multi_scalar_mul(&[-c, self.z], &[pk.0, *G])
167                .expect("Constant lengths")
168            || self.a2
169                != RistrettoPoint::multi_scalar_mul(
170                    &[-c, self.z],
171                    &[encryption.decryption_handle, encryption.commitment.0],
172                )
173                .expect("Constant lengths")
174        {
175            Err(InvalidProof)
176        } else {
177            Ok(())
178        }
179    }
180}

182impl ConsistencyProof {
183    pub fn prove(
184        message: &RistrettoScalar,
185        ciphertext: &Ciphertext,
186        blinding: &Blinding,
187        encryption_key: &PublicKey,
188        rng: &mut impl AllowedRng,
189    ) -> FastCryptoResult<Self> {
190        let r1 = RistrettoScalar::rand(rng);
191        let r2 = RistrettoScalar::rand(rng);
192        let a1 = encryption_key.0 * r1;
193        let a2 = RistrettoPoint::multi_scalar_mul(&[r1, r2], &[*G, *H]).expect("Constant length");
194
195        let c = Self::challenge(&a1, &a2, ciphertext, encryption_key);
196        let z1 = r1 + c * blinding.0;
197        let z2 = r2 + c * message;
198
199        Ok(Self { a1, a2, z1, z2 })
200    }
201
202    pub fn challenge(
203        a: &RistrettoPoint,
204        b: &RistrettoPoint,
205        ciphertext: &Ciphertext,
206        encryption_key: &PublicKey,
207    ) -> RistrettoScalar {
208        fiat_shamir_challenge(&(
209            &*G,
210            &*H,
211            a,
212            b,
213            &ciphertext.commitment,
214            &ciphertext.decryption_handle,
215            encryption_key,
216        ))
217    }
218
219    pub fn verify(
220        &self,
221        ciphertext: &Ciphertext,
222        encryption_key: &PublicKey,
223    ) -> FastCryptoResult<()> {
224        let c = Self::challenge(&self.a1, &self.a2, ciphertext, encryption_key);
225        if self.a1
226            != RistrettoPoint::multi_scalar_mul(
227                &[-c, self.z1],
228                &[ciphertext.decryption_handle, encryption_key.0],
229            )
230            .expect("Constant lengths")
231            || self.a2
232                != RistrettoPoint::multi_scalar_mul(
233                    &[-c, self.z1, self.z2],
234                    &[ciphertext.commitment.0, *G, *H],
235                )
236                .expect("Constant lengths")
237        {
238            return Err(InvalidProof);
239        }
240        Ok(())
241    }
242}

244impl MultiRecipientCiphertext {
245    // Encrypt a 32-bit ciphertext to multiple recipients where the decryption handles share the same blinding factor.
246    pub fn encrypt(
247        encryption_keys: &[PublicKey],
248        message: u32,
249        rng: &mut impl AllowedRng,
250    ) -> (Self, Blinding) {
251        let blinding = Blinding::rand(rng);
252        (
253            Self {
254                decryption_handles: encryption_keys.iter().map(|pk| pk.0 * blinding.0).collect(),
255                commitment: PedersenCommitment::new(
256                    &RistrettoScalar::from(message as u64),
257                    &blinding,
258                ),
259            },
260            blinding,
261        )
262    }
263
264    /// Decrypt the ciphertext of a single recipient identified by the provided index.
265    pub fn decrypt(
266        &self,
267        index: usize,
268        decryption_key: &PrivateKey,
269        table: &HashMap<[u8; RISTRETTO_POINT_BYTE_LENGTH], u16>,
270    ) -> FastCryptoResult<u32> {
271        self.ciphertext(index)?.decrypt(decryption_key, table)
272    }
273
274    pub fn ciphertext(&self, index: usize) -> FastCryptoResult<Ciphertext> {
275        if index >= self.decryption_handles.len() {
276            return Err(InvalidInput);
277        }
278        Ok(Ciphertext {
279            commitment: self.commitment.clone(),
280            decryption_handle: self.decryption_handles[index],
281        })
282    }

285impl<const N: usize> KeyConsistencyProof<N> {
286    pub fn prove(
287        sender_private_key_limbs: &[u32; N],
288        sender_public_key: &PublicKey,
289        recipient_encryption_keys: &[PublicKey],
290        ciphertexts: &[MultiRecipientCiphertext; N],
291        blindings: &[Blinding; N],
292        rng: &mut impl AllowedRng,
293    ) -> Self {
294        // Sample N random a_i and b_i
295        let a: [_; N] = from_fn(|_| RistrettoScalar::rand(rng));
296        let b: [_; N] = from_fn(|_| RistrettoScalar::rand(rng));
297
298        // A_1ij = a_i * pk_j for all (i, j) — N*m elements, ordered by limb then recipient
299        let a1 = a
300            .iter()
301            .flat_map(|ai| recipient_encryption_keys.iter().map(move |pk| pk.0 * ai))
302            .collect_vec();
303
304        // A_2i = a_i * G + b_i * H for all i
305        let a2 = from_fn(|i| *G * a[i] + *H * b[i]);
306
307        // A_3i = b_i * G for all i
308        let a3 = from_fn(|i| *G * b[i]);
309
310        // c = Hash(G, H, sender_public_key, recipient_encryption_keys, ciphertexts, a1, a2, a3)
311        let c = Self::challenge(
312            sender_public_key,
313            recipient_encryption_keys,
314            ciphertexts,
315            &a1,
316            &a2,
317            &a3,
318        );
319
320        // z_1i = a_i + c * r_i
321        let z1 = from_fn(|i| a[i] + c * blindings[i].0);
322
323        // z_2i = b_i + c * u_i
324        let z2 = from_fn(|i| b[i] + c * RistrettoScalar::from(sender_private_key_limbs[i] as u64));
325
326        Self { a1, a2, a3, z1, z2 }
327    }
328
329    /// Verify checks the provided consistency proof. To do so, it batches all three groups of verification equations
330    /// into a single MSM using hash-derived scalars. The three groups of equations that must hold for a valid proof are:
331    ///
332    ///   Check 1 (decryption handle consistency): Verifies that each decryption handle was formed with the same
333    ///   blinding r_i as the commitment via
334    ///     A1_ij + c * D_ij == z_1i * S_j
335    ///   for all limbs i and recipients j where D_ij = r_i * S_j is the decryption handle and S_j is recipient j's public key.
336    ///   Combined equations using scalars mu_ij = Hash("mu", c, i, j):
337    ///     \sum_j (\sum_i mu_ij * z_1i) * S_j - \sum_{i,j} mu_ij * A1_ij - \sum_{i,j} (c * mu_ij) * D_ij == 0
338    ///
339    ///   Check 2 (commitment consistency): Verifies knowledge of the blinding r_i and message u_i opening the
340    ///   commitment via
341    ///     A2_i + c * C_i == z_1i * G + z_2i * H
342    ///   for all limbs i where C_i = r_i * G + u_i * H is the Pedersen commitment.
343    ///   Combined equations using scalars rho_i = Hash("rho", c, i):
344    ///     (\sum_i rho_i * z_1i) * G + (\sum_i rho_i * z_2i) * H - \sum_i rho_i * A2_i - \sum_i (c * rho_i) * C_i == 0
345    ///
346    ///   Check 3 (public key consistency): Verifies that the encrypted 32-bit key limbs u_i reconstruct to the
347    ///   private key corresponding to U via
348    ///     (\sum_i z_2i * 2^{32i}) * G == (\sum_i A3_i * 2^{32i}) + c * U
349    ///   where U is the sender's public key.
350    ///
351    ///   We combine the individual checks as
352    ///     (check 1) + alpha * (check 2) + beta * (check 3) == 0
353    ///   using hash-derived outer scalars alpha = Hash("alpha", c) and beta = Hash("beta", c) to ensure soundness.
354    pub fn verify(
355        &self,
356        sender_public_key: &PublicKey,
357        recipient_encryption_keys: &[PublicKey],
358        ciphertexts: &[MultiRecipientCiphertext; N],
359    ) -> FastCryptoResult<()> {
360        // Fiat-Shamir challenge
361        let c = Self::challenge(
362            sender_public_key,
363            recipient_encryption_keys,
364            ciphertexts,
365            &self.a1,
366            &self.a2,
367            &self.a3,
368        );
369
370        // Number of recipients
371        let m = recipient_encryption_keys.len();
372
373        // Compute inner scalars mu_ij = Hash("mu", c, i, j) for all i and j used in check 1
374        let mu: Vec<RistrettoScalar> = (0..N)
375            .flat_map(|i| (0..m).map(move |j| fiat_shamir_challenge(&("mu", &c, i, j))))
376            .collect();
377
378        // Compute inner scalars rho_i = Hash("rho", c, i) for all i used in check 2
379        let rho: [RistrettoScalar; N] = from_fn(|i| fiat_shamir_challenge(&("rho", &c, i)));
380
381        // Compute outer scalars alpha = Hash("alpha", c) and beta = Hash("beta", c) combining the three zero-expressions:
382        //   (check 1) + alpha * (check 2) + beta * (check 3) == 0
383        let alpha = fiat_shamir_challenge(&("alpha", &c));
384        let beta = fiat_shamir_challenge(&("beta", &c));
385
386        // Check 2: compute sum_i(rho_i * z_1i) and sum_i(rho_i * z_2i)
387        let rho_z1 = RistrettoScalar::inner_product(rho, self.z1);
388        let rho_z2 = RistrettoScalar::inner_product(rho, self.z2);
389
390        // Check 3: compute z = \sum_i z_2i * 2^{32i}
391        let b = RistrettoScalar::from(1u64 << 32);
392        let z = RistrettoScalar::inner_product(
393            iterate(RistrettoScalar::generator(), |e| e * b),
394            self.z2,
395        );
396
397        let mut scalars: Vec<RistrettoScalar> = vec![alpha * rho_z1 + beta * z, alpha * rho_z2];
398        let mut points: Vec<RistrettoPoint> = vec![*G, *H];
399
400        // Check 1: Append (\sum_i mu_ij * z_1i, S_j) terms for each recipient j
401        for j in 0..m {
402            scalars.push(RistrettoScalar::inner_product(
403                (0..N).map(|i| mu[i * m + j]),
404                self.z1,
405            ));
406            points.push(recipient_encryption_keys[j].0);
407        }
408
409        // Check 1: Append (-mu_ij, A1_ij) and (-c * mu_ij, D_ij) terms
410        for (i, (a1_chunk, ci)) in self.a1.chunks(m).zip(ciphertexts).enumerate() {
411            for (j, (a1ij, dij)) in a1_chunk.iter().zip(&ci.decryption_handles).enumerate() {
412                scalars.push(-mu[i * m + j]);
413                points.push(*a1ij);
414                scalars.push(-(c * mu[i * m + j]));
415                points.push(*dij);
416            }
417        }
418
419        // Check 2: Append (-alpha * rho_i, A2_i) and (-c * alpha * rho_i, C_i) terms
420        for ((rhoi, a2i), ci) in rho.iter().zip(self.a2).zip(ciphertexts) {
421            scalars.push(-(alpha * *rhoi));
422            points.push(a2i);
423            scalars.push(-(c * alpha * *rhoi));
424            points.push(ci.commitment.0);
425        }
426
427        // Check 3: Append (-beta * c, U) and (-beta * 2^{32i}, A3_i) terms
428        scalars.push(-(beta * c));
429        points.push(sender_public_key.0);
430        let mut exp = RistrettoScalar::generator();
431        for a3i in self.a3 {
432            scalars.push(-(beta * exp));
433            points.push(a3i);
434            exp *= b;
435        }
436
437        if RistrettoPoint::multi_scalar_mul(&scalars, &points).expect("Consistent lengths")
438            != RistrettoPoint::zero()
439        {
440            return Err(InvalidProof);
441        }
442
443        Ok(())
444    }
445
446    pub fn challenge(
447        sender_public_key: &PublicKey,
448        recipient_encryption_keys: &[PublicKey],
449        ciphertexts: &[MultiRecipientCiphertext; N],
450        a1: &[RistrettoPoint],
451        a2: &[RistrettoPoint],
452        a3: &[RistrettoPoint],
453    ) -> RistrettoScalar {
454        fiat_shamir_challenge(&(
455            &*G,
456            &*H,
457            sender_public_key,
458            recipient_encryption_keys,
459            ciphertexts.as_slice(),
460            a1,
461            a2,
462            a3,
463        ))
464    }
465}

467impl<const N: usize> VerifiableKeyEncapsulation<N> {
468    /// Verifiably encrypt a private key to multiple recipient public keys.
469    pub fn batch_seal(
470        sender_private_key: &PrivateKey,
471        recipient_encryption_keys: &[PublicKey],
472        rng: &mut impl AllowedRng,
473    ) -> VerifiableKeyEncapsulation<N> {
474        // Re-arrange private key into N 32-bit limbs
475        let limbs: [u32; N] = sender_private_key
476            .0
477            .to_byte_array()
478            .chunks(4)
479            .map(|c| u32::from_le_bytes(c.try_into().unwrap()))
480            .collect_vec()
481            .try_into()
482            .unwrap();
483
484        // Encrypt all N 32-bit key limbs with Twisted ElGamal to the recipient public keys
485        let (ciphertexts, blindings): (Vec<_>, Vec<_>) = limbs
486            .iter()
487            .map(|&li| MultiRecipientCiphertext::encrypt(recipient_encryption_keys, li, rng))
488            .unzip();
489        let ciphertexts: [MultiRecipientCiphertext; N] = ciphertexts.try_into().unwrap();
490
491        // Create range proof
492        let range_proof =
493            RangeProof::prove_batch(&limbs.map(|m| m as u64), &blindings, &Range::Bits32, rng)
494                .unwrap();
495
496        // Create consistency proof
497        let consistency_proof = KeyConsistencyProof::prove(
498            &limbs,
499            &PublicKey::from(sender_private_key),
500            recipient_encryption_keys,
501            &ciphertexts,
502            &blindings.try_into().unwrap(),
503            rng,
504        );
505
506        VerifiableKeyEncapsulation {
507            ciphertexts,
508            range_proof,
509            consistency_proof,
510        }
511    }
512
513    /// Verifiably encrypt a private key to a single recipient public key.
514    pub fn seal(
515        sender_private_key: &PrivateKey,
516        recipient_encryption_key: &PublicKey,
517        rng: &mut impl AllowedRng,
518    ) -> VerifiableKeyEncapsulation<N> {
519        Self::batch_seal(
520            sender_private_key,
521            std::slice::from_ref(recipient_encryption_key),
522            rng,
523        )
524    }
525
526    /// Verify the key consistency proofs.
527    pub fn verify(
528        &self,
529        sender_public_key: &PublicKey,
530        recipient_encryption_keys: &[PublicKey],
531        rng: &mut impl AllowedRng,
532    ) -> FastCryptoResult<()> {
533        // Verify range proof over the Pedersen commitments of all limb ciphertexts
534        let commitments = self
535            .ciphertexts
536            .iter()
537            .map(|c| c.commitment.clone())
538            .collect::<Vec<_>>();
539        self.range_proof
540            .verify_batch(&commitments, &Range::Bits32, rng)?;
541        self.consistency_proof.verify(
542            sender_public_key,
543            recipient_encryption_keys,
544            &self.ciphertexts,
545        )
546    }
547
548    /// Open the key encapsulation for a single recipient decryption key identified by the provided index using the
549    /// provided 16-bit discrete log decryption table. All recipient public keys must be provided to verify the
550    /// consistency proof, which is bound to the full set of recipients.
551    pub fn open(
552        &self,
553        index: usize,
554        recipient_decryption_key: &PrivateKey,
555        recipient_public_keys: &[PublicKey],
556        sender_public_key: &PublicKey,
557        table: &HashMap<[u8; RISTRETTO_POINT_BYTE_LENGTH], u16>,
558        rng: &mut impl AllowedRng,
559    ) -> FastCryptoResult<PrivateKey> {
560        // Verify consistency proofs against all recipient keys
561        self.verify(sender_public_key, recipient_public_keys, rng)?;
562
563        // Decrypt each limb ciphertext using the recipient's decryption key
564        let limbs = self
565            .ciphertexts
566            .iter()
567            .map(|c| c.decrypt(index, recipient_decryption_key, table))
568            .collect::<FastCryptoResult<Vec<_>>>()?;
569
570        // Reconstruct the private key from its 32-bit limbs
571        let private_key_bytes = limbs.iter().flat_map(|l| l.to_le_bytes()).collect_vec();
572        let private_key = RistrettoScalar::from_byte_array(&private_key_bytes.try_into().unwrap())?;
573
574        // Verify the recovered key corresponds to the sender's public key
575        if PublicKey::from(&PrivateKey(private_key)) != *sender_public_key {
576            return Err(InvalidInput);
577        }
578        Ok(PrivateKey(private_key))
579    }
580}

582/// Precompute discrete log table for use in decryption. This only needs to be computed once.
583///
584/// The table contains a mapping from Ristretto points <i>(2<sup>16</sup> x) G<i> to <i>x</i> for all <i>x</i> in the range <i>0, .., 2<sup>16</sup>-1</i>.
585pub fn precompute_table() -> HashMap<[u8; RISTRETTO_POINT_BYTE_LENGTH], u16> {
586    let step = H.repeated_doubling(16);
587    iterate(RistrettoPoint::zero(), |p| p + step)
588        .enumerate()
589        .map(|(i, p)| (p.to_byte_array(), i as u16))
590        .take(1 << 16)
591        .collect()
592}
593
594/// Derive a Fiat-Shamir challenge scalar from any serializable message.
595///
596/// TODO: Add domain seperation
597fn fiat_shamir_challenge<T: Serialize>(msg: &T) -> RistrettoScalar {
598    let mut digest = Blake2b256::digest(bcs::to_bytes(msg).unwrap()).digest;
599    digest[31] = 0;
600    RistrettoScalar::from_byte_array(&digest).expect("Always in field")
601}

697#[test]
698fn test_key_consistency_proof() {
699    const N: usize = 8;
700    let mut rng = rand::thread_rng();
701
702    // Generate sender and recipient key pairs
703    let (pk_snd, sk_snd) = generate_keypair(&mut rng);
704    let (pk_rcv, _sk_rcv) = generate_keypair(&mut rng);
705
706    // Split sender private key into 8 x 32-bit limbs
707    let sk_bytes = sk_snd.0.to_byte_array();
708    let limbs: [u32; N] = std::array::from_fn(|i| {
709        u32::from_le_bytes(sk_bytes[4 * i..4 * (i + 1)].try_into().unwrap())
710    });
711
712    // Encrypt each limb as a multi-recipient Twisted ElGamal ciphertext towards the recipient
713    let (ciphertexts, blindings): (Vec<_>, Vec<_>) = limbs
714        .iter()
715        .map(|&limb| {
716            MultiRecipientCiphertext::encrypt(std::slice::from_ref(&pk_rcv), limb, &mut rng)
717        })
718        .unzip();
719    let ciphertexts: [MultiRecipientCiphertext; N] = ciphertexts.try_into().unwrap();
720    let blindings: [Blinding; N] = blindings.try_into().unwrap();
721
722    // Prove
723    let proof = KeyConsistencyProof::<N>::prove(
724        &limbs,
725        &pk_snd,
726        std::slice::from_ref(&pk_rcv),
727        &ciphertexts,
728        &blindings,
729        &mut rng,
730    );
731
732    // Verification passes with correct sender public key
733    assert!(proof
734        .verify(&pk_snd, std::slice::from_ref(&pk_rcv), &ciphertexts)
735        .is_ok());
736
737    // Verification fails with a different sender public key
738    let (other_pk_snd, _) = generate_keypair(&mut rng);
739    assert!(proof
740        .verify(&other_pk_snd, &[pk_rcv], &ciphertexts)
741        .is_err());
742}

744#[test]
745fn test_verifiable_key_encapsulation() {
746    const N: usize = 8;
747    let mut rng = rand::thread_rng();
748    let table = precompute_table();
749
750    // Sender key pair; the private key will be encapsulated
751    let (pk_snd, sk_snd) = generate_keypair(&mut rng);
752
753    // Three recipient key pairs
754    let (pk_rcv_0, sk_rcv_0) = generate_keypair(&mut rng);
755    let (pk_rcv_1, sk_rcv_1) = generate_keypair(&mut rng);
756    let (pk_rcv_2, sk_rcv_2) = generate_keypair(&mut rng);
757
758    let recipient_keys = [pk_rcv_0.clone(), pk_rcv_1.clone(), pk_rcv_2.clone()];
759
760    // Seal the sender's private key to all three recipients
761    let encapsulation =
762        VerifiableKeyEncapsulation::<N>::batch_seal(&sk_snd, &recipient_keys, &mut rng);
763
764    // Verification passes for the correct sender public key and recipient keys
765    assert!(encapsulation
766        .verify(&pk_snd, &recipient_keys, &mut rng)
767        .is_ok());
768
769    // Verification fails with a wrong sender public key
770    let (other_pk, _) = generate_keypair(&mut rng);
771    assert!(encapsulation
772        .verify(&other_pk, &recipient_keys, &mut rng)
773        .is_err());
774
775    // Each recipient can independently recover the sender's private key
776    let recovered_0 = encapsulation
777        .open(0, &sk_rcv_0, &recipient_keys, &pk_snd, &table, &mut rng)
778        .unwrap();
779    let recovered_1 = encapsulation
780        .open(1, &sk_rcv_1, &recipient_keys, &pk_snd, &table, &mut rng)
781        .unwrap();
782    let recovered_2 = encapsulation
783        .open(2, &sk_rcv_2, &recipient_keys, &pk_snd, &table, &mut rng)
784        .unwrap();
785
786    assert_eq!(recovered_0.0, sk_snd.0);
787    assert_eq!(recovered_1.0, sk_snd.0);
788    assert_eq!(recovered_2.0, sk_snd.0);
789
790    // A recipient cannot open another recipient's slot with their own key
791    assert!(encapsulation
792        .open(1, &sk_rcv_0, &recipient_keys, &pk_snd, &table, &mut rng)
793        .is_err());
794}
